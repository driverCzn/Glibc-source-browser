19#include <argz.h>
20#include <errno.h>
21#include <stdlib.h>
22#include <string.h>
23
24
25error_t
26__argz_add_sep (char **argz, size_t *argz_len, const char *string, int delim)
27{
28 size_t nlen = strlen (string) + 1;
29
30 if (nlen > 1)
31 {
32 const char *rp;
33 char *wp;
34
35 *argz = (char *) realloc (*argz, *argz_len + nlen);
36 if (*argz == NULL)
37 return ENOMEM;
38
39 wp = *argz + *argz_len;
40 rp = string;
41 do
42 if (*rp == delim)
43 {
44 if (wp > *argz && wp[-1] != '\0')
45 *wp++ = '\0';
46 else
47 --nlen;
48 }
49 else
50 *wp++ = *rp;
51 while (*rp++ != '\0');
52
53 *argz_len += nlen;
54 }
55
56 return 0;
57}
58weak_alias (__argz_add_sep, argz_add_sep)
