24#include <utmp.h>

27/* The structure describing the functions in a backend. */
28struct utfuncs
29{
30 int (*setutent) (void);
31 int (*getutent_r) (struct utmp *, struct utmp **);
32 int (*getutid_r) (const struct utmp *, struct utmp *, struct utmp **);
33 int (*getutline_r) (const struct utmp *, struct utmp *, struct utmp **);
34 struct utmp *(*pututline) (const struct utmp *);
35 void (*endutent) (void);
36 int (*updwtmp) (const char *, const struct utmp *);
37};
38
39/* The tables from the services. */
40extern const struct utfuncs __libc_utmp_file_functions attribute_hidden;
41extern const struct utfuncs __libc_utmp_unknown_functions attribute_hidden;
42
43/* Currently selected backend. */
44extern const struct utfuncs *__libc_utmp_jump_table attribute_hidden;
45
46/* Current file name. */
47extern const char *__libc_utmp_file_name attribute_hidden;
48
49/* Locks access to the global data. */
50__libc_lock_define (extern, __libc_utmp_lock attribute_hidden)
51
52
53#endif /* utmp-private.h */
