19#include <sys/types.h>
20#include <termios.h>
21#include <unistd.h>
22#include <utmp.h>
23#include <pty.h>
24
25int
26forkpty (int *amaster, char *name, const struct termios *termp,
27 const struct winsize *winp)
28{
29 int master, slave, pid;
30
31 if (openpty (&master, &slave, name, termp, winp) == -1)
32 return -1;
33
34 switch (pid = fork ())
35 {
36 case -1:
37 close (master);
38 close (slave);
39 return -1;
40 case 0:
41 /* Child. */
42 close (master);
43 if (login_tty (slave))
44 _exit (1);
45
46 return 0;
47 default:
48 /* Parent. */
49 *amaster = master;
50 close (slave);
51
52 return pid;
53 }
54}
