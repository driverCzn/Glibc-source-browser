27/* We need to disable the redirect for __ftello64 for the alias
28 definitions below to work. */
29#define __ftello64 __ftello64_disable
30
31#include <stdio.h>
32#include <stdlib.h>
33#include <libioP.h>
34#include <errno.h>
35
36off_t
37__ftello (FILE *fp)
38{
39 off64_t pos;
40 CHECK_FILE (fp, -1L);
41 _IO_acquire_lock (fp);
42 pos = _IO_seekoff_unlocked (fp, 0, _IO_seek_cur, 0);
43 if (_IO_in_backup (fp) && pos != _IO_pos_BAD)
44 {
45 if (fp->_mode <= 0)
46 pos -= fp->_IO_save_end - fp->_IO_save_base;
47 }
48 _IO_release_lock (fp);
49 if (pos == _IO_pos_BAD)
50 {
51 if (errno == 0)
52 __set_errno (EIO);
53 return -1L;
54 }
55 if ((off64_t) (off_t) pos != pos)
56 {
57 __set_errno (EOVERFLOW);
58 return -1L;
59 }
60 return pos;
61}
