19#include <wchar.h>
20
21#ifdef WCSCAT
22# define __wcscat WCSCAT
23#endif
24
25/* Append SRC on the end of DEST. */
26wchar_t *
27__wcscat (wchar_t *dest, const wchar_t *src)
28{
29 wchar_t *s1 = dest;
30 const wchar_t *s2 = src;
31 wchar_t c;
32
33 /* Find the end of the string. */
34 do
35 c = *s1++;
36 while (c != L'\0');
37
38 /* Make S1 point before the next character, so we can increment
39 it while memory is read (wins on pipelined cpus). */
40 s1 -= 2;
41
42 do
43 {
44 c = *s2++;
45 *++s1 = c;
46 }
47 while (c != L'\0');
48
49 return dest;
50}
51#ifndef WCSCAT
52weak_alias (__wcscat, wcscat)
53#endif
