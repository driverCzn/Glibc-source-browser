19#include <wchar.h>
20
21#define __need_ptrdiff_t
22#include <stddef.h>
23
24#ifdef WCPCPY
25# define __wcpcpy WCPCPY
26#endif
27/* Copy SRC to DEST, returning the address of the terminating L'\0' in
28 DEST. */
29wchar_t *
30__wcpcpy (wchar_t *dest, const wchar_t *src)
31{
32 wchar_t *wcp = (wchar_t *) dest - 1;
33 wint_t c;
34 const ptrdiff_t off = src - dest + 1;
35
36 do
37 {
38 c = wcp[off];
39 *++wcp = c;
40 }
41 while (c != L'\0');
42
43 return wcp;
44}
45
46#ifndef WCPCPY
47weak_alias (__wcpcpy, wcpcpy)
48#endif
