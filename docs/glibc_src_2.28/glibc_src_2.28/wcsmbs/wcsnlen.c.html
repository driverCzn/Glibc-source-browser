25/* Return length of string S at most maxlen. */
26size_t
27__wcsnlen (const wchar_t *s, size_t maxlen)
28{
29 size_t len = 0;
30
31 while (maxlen > 0 && s[len] != L'\0')
32 {
33 ++len;
34 if (--maxlen == 0 || s[len] == L'\0')
35 return len;
36 ++len;
37 if (--maxlen == 0 || s[len] == L'\0')
38 return len;
39 ++len;
40 if (--maxlen == 0 || s[len] == L'\0')
41 return len;
42 ++len;
43 --maxlen;
44 }
45
46 return len;
47}
48#ifndef WCSNLEN
49weak_alias (__wcsnlen, wcsnlen)
50#endif
