18#include <stdio.h>

20/* Generate a unique filename in P_tmpdir. If S is NULL return NULL.
21 This makes this function thread safe. */
22char *
23tmpnam_r (char *s)
24{
25 if (s == NULL)
26 return NULL;
27
28 if (__path_search (s, L_tmpnam, NULL, NULL, 0))
29 return NULL;
30 if (__gen_tempname (s, 0, 0, __GT_NOCREATE))
31 return NULL;
32
33 return s;
34}
35
36link_warning (tmpnam_r,
37 "the use of `tmpnam_r' is dangerous, better use `mkstemp'")
