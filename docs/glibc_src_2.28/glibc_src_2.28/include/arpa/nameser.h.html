18# undef NS_GET16
19# define NS_GET16(s, cp) \
20 do { \
21 const uint16_t *t_cp = (const uint16_t *) (cp); \
22 (s) = ntohs (*t_cp); \
23 (cp) += NS_INT16SZ; \
24 } while (0)
25
26# undef NS_GET32
27# define NS_GET32(l, cp) \
28 do { \
29 const uint32_t *t_cp = (const uint32_t *) (cp); \
30 (l) = ntohl (*t_cp); \
31 (cp) += NS_INT32SZ; \
32 } while (0)
33
34# undef NS_PUT16
35# define NS_PUT16(s, cp) \
36 do { \
37 uint16_t *t_cp = (uint16_t *) (cp); \
38 *t_cp = htons (s); \
39 (cp) += NS_INT16SZ; \
40 } while (0)

42# undef NS_PUT32
