18#include <unistd.h>
19#include <sys/param.h>
20#ifdef HAVE_INLINED_SYSCALLS
21# include <errno.h>
22# include <sysdep.h>
23#endif
24
25
26ssize_t
27__readlink_chk (const char *path, void *buf, size_t len, size_t buflen)
28{
29 if (len > buflen)
30 __chk_fail ();
31
32#ifdef HAVE_INLINED_SYSCALLS
33 return INLINE_SYSCALL (readlink, 3, path, buf, len);
34#else
35 return __readlink (path, buf, len);
36#endif
37}
