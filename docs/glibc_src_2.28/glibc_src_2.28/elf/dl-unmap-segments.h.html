19#ifndef _DL_UNMAP_SEGMENTS_H
20#define _DL_UNMAP_SEGMENTS_H 1
21
22#include <link.h>
23#include <sys/mman.h>
24
25/* _dl_map_segments ensures that any whole pages in gaps between segments
26 are filled in with PROT_NONE mappings. So we can just unmap the whole
27 range in one fell swoop. */
28
29static __always_inline void
30_dl_unmap_segments (struct link_map *l)
31{
32 __munmap ((void *) l->l_map_start, l->l_map_end - l->l_map_start);
33}
34
35#endif /* dl-unmap-segments.h */
