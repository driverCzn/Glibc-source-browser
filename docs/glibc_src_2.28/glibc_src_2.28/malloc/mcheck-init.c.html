18/* The object of this file should be installed as libmcheck.a,
19 so one can do -lmcheck to turn on mcheck. */
20
21#include <malloc.h>
22#include <mcheck.h>
23#include <shlib-compat.h>
24
25static void
26turn_on_mcheck (void)
27{
28 mcheck (NULL);
29}
30
31void (*__malloc_initialize_hook) (void) = turn_on_mcheck;
32compat_symbol_reference (libc, __malloc_initialize_hook,
33 __malloc_initialize_hook, GLIBC_2_0);
