19#include <dynarray.h>
20#include <stdlib.h>
21#include <string.h>
22
23bool
24__libc_dynarray_resize_clear (struct dynarray_header *list, size_t size,
25 void *scratch, size_t element_size)
26{
27 size_t old_size = list->used;
28 if (!__libc_dynarray_resize (list, size, scratch, element_size))
29 return false;
30 /* __libc_dynarray_resize already checked for overflow. */
31 memset (list->array + (old_size * element_size), 0,
32 (size - old_size) * element_size);
33 return true;
34}
