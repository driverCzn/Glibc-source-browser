24void
25xfclose (FILE *fp)
26{
27 if (ferror (fp))
28 FAIL_EXIT1 ("stdio stream closed with pending errors");
29 if (fflush (fp) != 0)
30 FAIL_EXIT1 ("fflush: %m");
31 if (fclose (fp) != 0)
32 FAIL_EXIT1 ("fclose: %m");
33}
