21#include <fcntl.h>
22#include <paths.h>
23#include <stdlib.h>
24#include <unistd.h>

26void
27ignore_stderr (void)
28{
29 int fd = open (_PATH_DEVNULL, O_WRONLY);
30 if (fd == -1)
31 close (STDERR_FILENO);
32 else
33 {
34 dup2 (fd, STDERR_FILENO);
35 close (fd);
36 }
37 setenv ("LIBC_FATAL_STDERR_", "1", 1);
38}
