18#include <netinet/in.h>
19
20#undef htons
21#undef ntohs
22
23uint16_t
24htons (uint16_t x)
25{
26#if BYTE_ORDER == BIG_ENDIAN
27 return x;
28#elif BYTE_ORDER == LITTLE_ENDIAN
29 return __bswap_16 (x);
30#else
31# error "What kind of system is this?"
32#endif
33}
34weak_alias (htons, ntohs)
