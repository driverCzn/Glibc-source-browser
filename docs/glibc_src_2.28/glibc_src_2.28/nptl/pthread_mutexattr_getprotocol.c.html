23int
24pthread_mutexattr_getprotocol (const pthread_mutexattr_t *attr, int *protocol)
25{
26 const struct pthread_mutexattr *iattr;
27
28 iattr = (const struct pthread_mutexattr *) attr;
29
30 *protocol = ((iattr->mutexkind & PTHREAD_MUTEXATTR_PROTOCOL_MASK)
31 >> PTHREAD_MUTEXATTR_PROTOCOL_SHIFT);
32
33 return 0;
34}
