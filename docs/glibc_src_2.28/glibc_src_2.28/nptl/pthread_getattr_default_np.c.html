23int
24pthread_getattr_default_np (pthread_attr_t *out)
25{
26 struct pthread_attr *real_out;
27
28 real_out = (struct pthread_attr *) out;
29
30 lll_lock (__default_pthread_attr_lock, LLL_PRIVATE);
31 *real_out = __default_pthread_attr;
32 lll_unlock (__default_pthread_attr_lock, LLL_PRIVATE);
33
34 return 0;
35}
