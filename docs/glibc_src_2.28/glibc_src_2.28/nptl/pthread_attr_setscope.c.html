19#include <errno.h>
20#include "pthreadP.h"
21
22
23int
24__pthread_attr_setscope (pthread_attr_t *attr, int scope)
25{
26 struct pthread_attr *iattr;
27
28 iattr = (struct pthread_attr *) attr;
29
30 /* Catch invalid values. */
31 switch (scope)
32 {
33 case PTHREAD_SCOPE_SYSTEM:
34 iattr->flags &= ~ATTR_FLAG_SCOPEPROCESS;
35 break;
36
37 case PTHREAD_SCOPE_PROCESS:
38 return ENOTSUP;
39
40 default:
41 return EINVAL;
42 }
43
44 return 0;
45}
46strong_alias (__pthread_attr_setscope, pthread_attr_setscope)
