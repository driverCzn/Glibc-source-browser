22int
23pthread_condattr_getclock (const pthread_condattr_t *attr, clockid_t *clock_id)
24{
25 *clock_id = (((((const struct pthread_condattr *) attr)->value) >> 1)
26 & ((1 << COND_CLOCK_BITS) - 1));
27 return 0;
28}
