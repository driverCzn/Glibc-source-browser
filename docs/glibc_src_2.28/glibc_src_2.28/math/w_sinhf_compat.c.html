25#if LIBM_SVID_COMPAT
26float
27__sinhf (float x)
28{
29 float z = __ieee754_sinhf (x);
30 if (__builtin_expect (!isfinite (z), 0) && isfinite (x)
31 && _LIB_VERSION != _IEEE_)
32 return __kernel_standard_f (x, x, 125); /* sinhf overflow */
33
34 return z;
35}
36libm_alias_float (__sinh, sinh)
