26#if LIBM_SVID_COMPAT
27/* wrapper acosl */
28long double
29__acosl (long double x)
30{
31 if (__builtin_expect (isgreater (fabsl (x), 1.0L), 0)
32 && _LIB_VERSION != _IEEE_)
33 {
34 /* acos(|x|>1) */
35 feraiseexcept (FE_INVALID);
36 return __kernel_standard_l (x, x, 201);
37 }
38
39 return __ieee754_acosl (x);
40}
41libm_alias_ldouble (__acos, acos)
