25#if LIBM_SVID_COMPAT
26/* wrapper acosh */
27double
28__acosh (double x)
29{
30 if (__builtin_expect (isless (x, 1.0), 0) && _LIB_VERSION != _IEEE_)
31 /* acosh(x<1) */
32 return __kernel_standard (x, x, 29);
33
34 return __ieee754_acosh (x);
35}
36libm_alias_double (__acosh, acosh)
