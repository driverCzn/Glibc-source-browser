70/* Read one entry from the given stream. */
71int
72__fgetpwent_r (FILE *stream, struct passwd *resbuf, char *buffer,
73 size_t buflen, struct passwd **result)
74{
75 char *p;
76
77 flockfile (stream);
78 do
79 {
80 buffer[buflen - 1] = '\xff';
81 p = fgets_unlocked (buffer, buflen, stream);
82 if (p == NULL && feof_unlocked (stream))
83 {
84 funlockfile (stream);
85 *result = NULL;
86 __set_errno (ENOENT);
87 return errno;
88 }
89 if (p == NULL || buffer[buflen - 1] != '\xff')
90 {
91 funlockfile (stream);
92 *result = NULL;
93 __set_errno (ERANGE);
94 return errno;
95 }
96
97 /* Skip leading blanks. */
98 while (isspace (*p))
99 ++p;
100 } while (*p == '\0' || *p == '#' || /* Ignore empty and comment lines. */
101 /* Parse the line. If it is invalid, loop to
102 get the next line of the file to parse. */
103 ! parse_line (p, resbuf, (void *) buffer, buflen, &errno));
104
105 funlockfile (stream);
106
107 *result = resbuf;
108 return 0;
109}
