18#include <errno.h>
19#include <libc-lock.h>
20#include <pwd.h>
21#include <stdio.h>
22#include <stdlib.h>

25/* We need to protect the dynamic buffer handling. */
26__libc_lock_define_initialized (static, lock);
27
28libc_freeres_ptr (static char *buffer);
29
30/* Read one entry from the given stream. */
31struct passwd *
32fgetpwent (FILE *stream)
33{
34 static size_t buffer_size;
35 static struct passwd resbuf;
36 fpos_t pos;
37 struct passwd *result;
38 int save;
39
40 if (fgetpos (stream, &pos) != 0)
41 return NULL;
42
43 /* Get lock. */
44 __libc_lock_lock (lock);
45
46 /* Allocate buffer if not yet available. */
47 if (buffer == NULL)
48 {
49 buffer_size = NSS_BUFLEN_PASSWD;
50 buffer = malloc (buffer_size);
51 }
52
53 while (buffer != NULL
54 && (__fgetpwent_r (stream, &resbuf, buffer, buffer_size, &result)
55 == ERANGE))
56 {
57 char *new_buf;
58 buffer_size += NSS_BUFLEN_PASSWD;
59 new_buf = realloc (buffer, buffer_size);
60 if (new_buf == NULL)
61 {
62 /* We are out of memory. Free the current buffer so that the
63 process gets a chance for a normal termination. */
64 save = errno;
65 free (buffer);
66 __set_errno (save);
67 }
68 buffer = new_buf;
69
70 /* Reset the stream. */
71 if (fsetpos (stream, &pos) != 0)
72 buffer = NULL;
73 }
74
75 if (buffer == NULL)
76 result = NULL;
77
78 /* Release lock. Preserve error value. */
79 save = errno;
80 __libc_lock_unlock (lock);
81 __set_errno (save);
82
83 return result;
84}
