19#ifndef _RES_USE_INET6_H
20#define _RES_USE_INET6_H
21
22#include <resolv/resolv_context.h>
23#include <resolv/resolv-internal.h>
24
25/* Ensure that RES_USE_INET6 is disabled in *CTX. Return true if
26 __resolv_context_enable_inet6 below should enable RES_USE_INET6
27 again. */
28static inline bool
29__resolv_context_disable_inet6 (struct resolv_context *ctx)
30{
31 if (ctx != NULL && ctx->resp->options & DEPRECATED_RES_USE_INET6)
32 {
33 ctx->resp->options &= ~DEPRECATED_RES_USE_INET6;
34 return true;
35 }
36 else
37 return false;
38}
39
40/* If ENABLE, re-enable RES_USE_INET6 in *CTX. To be paired with
41 __resolv_context_disable_inet6. */
42static inline void
43__resolv_context_enable_inet6 (struct resolv_context *ctx, bool enable)
44{
45 if (ctx != NULL && enable)
46 ctx->resp->options |= DEPRECATED_RES_USE_INET6;
47}

49#endif
