28/* See concurrency notes in stdlib/exit.h where this lock is declared. */
29__libc_lock_define_initialized (, __exit_funcs_lock)
30
31
32int
33attribute_hidden
34__internal_atexit (void (*func) (void *), void *arg, void *d,
35 struct exit_function_list **listp)
36{
37 struct exit_function *new;
38
39 __libc_lock_lock (__exit_funcs_lock);
40 new = __new_exitfn (listp);
41
42 if (new == NULL)
43 {
44 __libc_lock_unlock (__exit_funcs_lock);
45 return -1;
46 }
47
48#ifdef PTR_MANGLE
49 PTR_MANGLE (func);
50#endif
51 new->func.cxa.fn = (void (*) (void *, int)) func;
52 new->func.cxa.arg = arg;
53 new->func.cxa.dso_handle = d;
54 new->flavor = ef_cxa;
55 __libc_lock_unlock (__exit_funcs_lock);
56 return 0;
57}

60/* Register a function to be called by exit or when a shared library
61 is unloaded. This function is only called from code generated by
62 the C++ compiler. */
63int
64__cxa_atexit (void (*func) (void *), void *arg, void *d)
65{
66 return __internal_atexit (func, arg, d, &__exit_funcs);
67}
68libc_hidden_def (__cxa_atexit)

71static struct exit_function_list initial;
72struct exit_function_list *__exit_funcs = &initial;
73uint64_t __new_exitfn_called;
74
75/* Must be called with __exit_funcs_lock held. */
76struct exit_function *
77__new_exitfn (struct exit_function_list **listp)
78{
79 struct exit_function_list *p = NULL;
80 struct exit_function_list *l;
81 struct exit_function *r = NULL;
82 size_t i = 0;
83
84 if (__exit_funcs_done)
85 /* Exit code is finished processing all registered exit functions,
86 therefore we fail this registration. */
87 return NULL;
88
89 for (l = *listp; l != NULL; p = l, l = l->next)
90 {
91 for (i = l->idx; i > 0; --i)
92 if (l->fns[i - 1].flavor != ef_free)
93 break;
94
95 if (i > 0)
96 break;
97
98 /* This block is completely unused. */
99 l->idx = 0;
100 }
101
102 if (l == NULL || i == sizeof (l->fns) / sizeof (l->fns[0]))
103 {
104 /* The last entry in a block is used. Use the first entry in
105 the previous block if it exists. Otherwise create a new one. */
106 if (p == NULL)
107 {
108 assert (l != NULL);
109 p = (struct exit_function_list *)
110 calloc (1, sizeof (struct exit_function_list));
111 if (p != NULL)
112 {
113 p->next = *listp;
114 *listp = p;
115 }
116 }
117
118 if (p != NULL)
119 {
120 r = &p->fns[0];
121 p->idx = 1;
122 }
123 }
124 else
125 {
126 /* There is more room in the block. */
127 r = &l->fns[i];
128 l->idx = i + 1;
129 }
130
131 /* Mark entry as used, but we don't know the flavor now. */
132 if (r != NULL)
133 {
134 r->flavor = ef_us;
135 ++__new_exitfn_called;
136 }
137
138 return r;
139}
