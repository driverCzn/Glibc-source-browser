20/* The actual implementation for all floating point sizes is in strtod.c.
21 These macros tell it to produce the `float' version, `strtof'. */
22
23#include <bits/floatn.h>
24
25#if __HAVE_FLOAT32 && !__HAVE_DISTINCT_FLOAT32
26# define strtof32 __hide_strtof32
27# define wcstof32 __hide_wcstof32
28#endif
29
30#define FLOAT float
31#define FLT FLT
32#ifdef USE_WIDE_CHAR
33#define STRTOF wcstof
34#define STRTOF_L __wcstof_l
35#else
36# define STRTOF strtof
37# define STRTOF_L __strtof_l
38#endif
39
40
41#include "strtod.c"
42
43#if __HAVE_FLOAT32 && !__HAVE_DISTINCT_FLOAT32
44# undef strtof32
45# undef wcstof32
46# ifdef USE_WIDE_CHAR
47weak_alias (wcstof, wcstof32)
48# else
49weak_alias (strtof, strtof32)
50# endif
51#endif
