21int
22__jrand48_r (unsigned short int xsubi[3], struct drand48_data *buffer,
23 long int *result)
24{
25 /* Compute next state. */
26 if (__drand48_iterate (xsubi, buffer) < 0)
27 return -1;
28
29 /* Store the result. */
30 *result = (int32_t) ((xsubi[2] << 16) | xsubi[1]);
31
32 return 0;
33}
