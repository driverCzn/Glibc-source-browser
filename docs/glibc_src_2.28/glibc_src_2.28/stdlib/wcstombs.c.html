23/* Convert the `wchar_t' string in PWCS to a multibyte character string
24 in S, writing no more than N characters. Return the number of bytes
25 written, or (size_t) -1 if an invalid `wchar_t' was found. */
26size_t
27wcstombs (char *s, const wchar_t *pwcs, size_t n)
28{
29 mbstate_t state;
30
31 memset (&state, '\0', sizeof state);
32
33 /* Return how many we wrote (or maybe an error). */
34 return __wcsrtombs (s, &pwcs, n, &state);
35}
