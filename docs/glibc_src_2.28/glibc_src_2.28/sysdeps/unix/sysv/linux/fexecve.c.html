30/* Execute the file FD refers to, overlaying the running program image.
31 ARGV and ENVP are passed to the new program, as for `execve'. */
32int
33fexecve (int fd, char *const argv[], char *const envp[])
34{
35 if (fd < 0 || argv == NULL || envp == NULL)
36 {
37 __set_errno (EINVAL);
38 return -1;
39 }
40
41#ifdef __NR_execveat
42 /* Avoid implicit array coercion in syscall macros. */
43 INLINE_SYSCALL (execveat, 5, fd, "", &argv[0], &envp[0], AT_EMPTY_PATH);
44# ifndef __ASSUME_EXECVEAT
45 if (errno != ENOSYS)
46 return -1;
47# endif
48#endif
49
50#ifndef __ASSUME_EXECVEAT
51 /* We use the /proc filesystem to get the information. If it is not
52 mounted we fail. */
53 char buf[sizeof "/proc/self/fd/" + sizeof (int) * 3];
54 __snprintf (buf, sizeof (buf), "/proc/self/fd/%d", fd);
55
56 /* We do not need the return value. */
57 __execve (buf, argv, envp);
58
59 int save = errno;
60
61 /* We come here only if the 'execve' call fails. Determine whether
62 /proc is mounted. If not we return ENOSYS. */
63 struct stat st;
64 if (stat ("/proc/self/fd", &st) != 0 && errno == ENOENT)
65 save = ENOSYS;
66
67 __set_errno (save);
68#endif
69
70 return -1;
71}
