28int
29accept4 (int fd, __SOCKADDR_ARG addr, socklen_t *addr_len, int flags)
30{
31#ifdef __ASSUME_ACCEPT4_SYSCALL
32 return SYSCALL_CANCEL (accept4, fd, addr.__sockaddr__, addr_len, flags);
33#else
34 return SOCKETCALL_CANCEL (accept4, fd, addr.__sockaddr__, addr_len, flags);
35#endif
36}
