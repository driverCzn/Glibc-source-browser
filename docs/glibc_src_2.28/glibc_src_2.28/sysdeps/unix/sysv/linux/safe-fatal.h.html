19#ifndef _SAFE_FATAL_H
20#define _SAFE_FATAL_H 1
21
22#include <sysdep.h>
23#include <unistd.h>
24
25static inline void
26__safe_fatal (void)
27{
28 INTERNAL_SYSCALL_DECL (err);
29 pid_t self = INTERNAL_SYSCALL (getpid, err, 0);
30 INTERNAL_SYSCALL (kill, err, 2, self, SIGKILL);
31}
