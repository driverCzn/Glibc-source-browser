18#include <sys/uio.h>
19#include <sysdep-cancel.h>
20
21#ifdef __ASSUME_PREADV
22
23# ifndef __NR_preadv64
24# define __NR_preadv64 __NR_preadv
25# endif
26
27ssize_t
28preadv64 (int fd, const struct iovec *vector, int count, off64_t offset)
29{
30 return SYSCALL_CANCEL (preadv64, fd, vector, count, LO_HI_LONG (offset));
31}
32#else
33static ssize_t __atomic_preadv64_replacement (int, const struct iovec *,
34 int, off64_t);
35ssize_t
36preadv64 (int fd, const struct iovec *vector, int count, off64_t offset)
37{
38#ifdef __NR_preadv64
39 ssize_t result = SYSCALL_CANCEL (preadv64, fd, vector, count,
40 LO_HI_LONG (offset));
41 if (result >= 0 || errno != ENOSYS)
42 return result;
43#endif
44 return __atomic_preadv64_replacement (fd, vector, count, offset);
45}
46# define PREADV static __atomic_preadv64_replacement
47# define PREAD __pread64
48# define OFF_T off64_t
49# include <sysdeps/posix/preadv_common.c>
50#endif
51libc_hidden_def (preadv64)
