18#include <sys/mman.h>
19#include <errno.h>
20#include <sysdep.h>
21
22int
23mlock2 (const void *addr, size_t length, unsigned int flags)
24{
25#ifdef __ASSUME_MLOCK2
26 return INLINE_SYSCALL_CALL (mlock2, addr, length, flags);
27#else
28 if (flags == 0)
29 return INLINE_SYSCALL_CALL (mlock, addr, length);
30# ifdef __NR_mlock2
31 int ret = INLINE_SYSCALL_CALL (mlock2, addr, length, flags);
32 if (ret == 0 || errno != ENOSYS)
33 return ret;
34# endif /* __NR_mlock2 */
35 /* Treat the missing system call as an invalid (non-zero) flag
36 argument. */
37 __set_errno (EINVAL);
38 return -1;
39#endif /* __ASSUME_MLOCK2 */
40}
