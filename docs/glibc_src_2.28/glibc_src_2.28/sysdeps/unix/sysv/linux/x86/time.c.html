19#include <time.h>
20
21#ifdef SHARED
22
23#include <dl-vdso.h>
24#include <errno.h>
25
26static time_t
27__time_syscall (time_t *t)
28{
29 INTERNAL_SYSCALL_DECL (err);
30 return INTERNAL_SYSCALL (time, err, 1, t);
31}
32
33# ifndef time_type
34/* The i386 time.c includes this file with a defined time_type macro.
35 For x86_64 we have to define it to time as the internal symbol is the
36 ifunc'ed one. */
37# define time_type time
38# endif
39
40#undef INIT_ARCH
41#define INIT_ARCH() PREPARE_VERSION_KNOWN (linux26, LINUX_2_6);
42/* If the vDSO is not available we fall back on the syscall. */
43libc_ifunc_hidden (time_type, time,
44 (_dl_vdso_vsym ("__vdso_time", &linux26)
45 ?: &__time_syscall))
46libc_hidden_def (time)
47
48#else
49
50# include <sysdep.h>
51
52time_t
53time (time_t *t)
54{
55 INTERNAL_SYSCALL_DECL (err);
56 return INTERNAL_SYSCALL (time, err, 1, t);
57}
58
59#endif
