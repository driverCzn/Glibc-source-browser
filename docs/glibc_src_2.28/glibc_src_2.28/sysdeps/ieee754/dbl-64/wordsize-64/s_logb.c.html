20#include <math.h>
21
22#include <math_private.h>
23#include <libm-alias-double.h>
24
25
26double
27__logb (double x)
28{
29 int64_t ix, ex;
30
31 EXTRACT_WORDS64 (ix, x);
32 ix &= UINT64_C(0x7fffffffffffffff);
33 if (ix == 0)
34 return -1.0 / fabs (x);
35 ex = ix >> 52;
36 if (ex == 0x7ff)
37 return x * x;
38 if (__glibc_unlikely (ex == 0))
39 {
40 int m = __builtin_clzll (ix);
41 ex -= m - 12;
42 }
43 return (double) (ex - 1023);
44}
45#ifndef __logb
46libm_alias_double (__logb, logb)
47#endif
