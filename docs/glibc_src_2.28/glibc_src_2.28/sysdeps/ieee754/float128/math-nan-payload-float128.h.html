21#define SET_NAN_PAYLOAD(flt, mant) \
22 do \
23 { \
24 union ieee854_float128 u; \
25 u.d = (flt); \
26 u.ieee_nan.mantissa0 = 0; \
27 u.ieee_nan.mantissa1 = 0; \
28 u.ieee_nan.mantissa2 = (mant) >> 32; \
29 u.ieee_nan.mantissa3 = (mant); \
30 if ((u.ieee.mantissa0 | u.ieee.mantissa1 \
31 | u.ieee.mantissa2 | u.ieee.mantissa3) != 0) \
32 (flt) = u.d; \
33 } \
34 while (0)
