27void
28__sincosl (long double x, long double *sinx, long double *cosx)
29{
30 int32_t se, i0, i1 __attribute__ ((unused));
31
32 /* High word of x. */
33 GET_LDOUBLE_WORDS (se, i0, i1, x);
34
35 /* |x| ~< pi/4 */
36 se &= 0x7fff;
37 if (se < 0x3ffe || (se == 0x3ffe && i0 <= 0xc90fdaa2))
38 {
39 *sinx = __kernel_sinl (x, 0.0, 0);
40 *cosx = __kernel_cosl (x, 0.0);
41 }
42 else if (se == 0x7fff)
43 {
44 /* sin(Inf or NaN) is NaN */
45 *sinx = *cosx = x - x;
46 if (isinf (x))
47 __set_errno (EDOM);
48 }
49 else
50 {
51 /* Argument reduction needed. */
52 long double y[2];
53 int n;
54
55 n = __ieee754_rem_pio2l (x, y);
56 switch (n & 3)
57 {
58 case 0:
59 *sinx = __kernel_sinl (y[0], y[1], 1);
60 *cosx = __kernel_cosl (y[0], y[1]);
61 break;
62 case 1:
63 *sinx = __kernel_cosl (y[0], y[1]);
64 *cosx = -__kernel_sinl (y[0], y[1], 1);
65 break;
66 case 2:
67 *sinx = -__kernel_sinl (y[0], y[1], 1);
68 *cosx = -__kernel_cosl (y[0], y[1]);
69 break;
70 default:
71 *sinx = -__kernel_cosl (y[0], y[1]);
72 *cosx = __kernel_sinl (y[0], y[1], 1);
73 break;
74 }
75 }
76}
