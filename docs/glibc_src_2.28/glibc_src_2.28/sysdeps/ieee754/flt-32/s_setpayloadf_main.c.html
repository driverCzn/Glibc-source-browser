19#include <math.h>
20#include <math_private.h>
21#include <libm-alias-float.h>
22#include <nan-high-order-bit.h>
23#include <stdint.h>
24
25#define SET_HIGH_BIT (HIGH_ORDER_BIT_IS_SET_FOR_SNAN ? SIG : !SIG)
26#define BIAS 0x7f
27#define PAYLOAD_DIG 22
28#define EXPLICIT_MANT_DIG 23
29
30int
31FUNC (float *x, float payload)
32{
33 uint32_t ix;
34 GET_FLOAT_WORD (ix, payload);
35 int exponent = ix >> EXPLICIT_MANT_DIG;
36 /* Test if argument is (a) negative or too large; (b) too small,
37 except for 0 when allowed; (c) not an integer. */
38 if (exponent >= BIAS + PAYLOAD_DIG
39 || (exponent < BIAS && !(SET_HIGH_BIT && ix == 0))
40 || (ix & ((1U << (BIAS + EXPLICIT_MANT_DIG - exponent)) - 1)) != 0)
41 {
42 SET_FLOAT_WORD (*x, 0);
43 return 1;
44 }
45 if (ix != 0)
46 {
47 ix &= (1U << EXPLICIT_MANT_DIG) - 1;
48 ix |= 1U << EXPLICIT_MANT_DIG;
49 ix >>= BIAS + EXPLICIT_MANT_DIG - exponent;
50 }
51 ix |= 0x7f800000 | (SET_HIGH_BIT ? 0x400000 : 0);
52 SET_FLOAT_WORD (*x, ix);
53 return 0;
54}
