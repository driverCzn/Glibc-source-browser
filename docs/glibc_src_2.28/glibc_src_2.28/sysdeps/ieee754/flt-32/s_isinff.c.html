10/*
11 * isinff(x) returns 1 if x is inf, -1 if x is -inf, else 0;
12 * no branching!
13 */
14
15#include <math.h>
16#include <math_private.h>
17
18int
19__isinff (float x)
20{
21 int32_t ix,t;
22 GET_FLOAT_WORD(ix,x);
23 t = ix & 0x7fffffff;
24 t ^= 0x7f800000;
25 t |= -t;
26 return ~(t >> 31) & (ix >> 30);
27}
28hidden_def (__isinff)
29weak_alias (__isinff, isinff)
