20#include <errno.h>
21#include <math.h>
22#include <math_private.h>
23#include <libm-alias-float.h>
24
25float __tanf(float x)
26{
27 float y[2],z=0.0;
28 int32_t n, ix;
29
30 GET_FLOAT_WORD(ix,x);
31
32 /* |x| ~< pi/4 */
33 ix &= 0x7fffffff;
34 if(ix <= 0x3f490fda) return __kernel_tanf(x,z,1);
35
36 /* tan(Inf or NaN) is NaN */
37 else if (ix>=0x7f800000) {
38 if (ix==0x7f800000)
39 __set_errno (EDOM);
40 return x-x; /* NaN */
41 }
42
43 /* argument reduction needed */
44 else {
45 n = __ieee754_rem_pio2f(x,y);
46 return __kernel_tanf(y[0],y[1],1-((n&1)<<1)); /* 1 -- n even
47 -1 -- n odd */
48 }
49}
