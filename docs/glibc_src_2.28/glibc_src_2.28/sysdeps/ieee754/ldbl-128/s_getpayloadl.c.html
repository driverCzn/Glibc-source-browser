19#include <math.h>
20#include <math_private.h>
21#include <libm-alias-ldouble.h>
22#include <stdint.h>
23
24_Float128
25__getpayloadl (const _Float128 *x)
26{
27 uint64_t hx, lx;
28 GET_LDOUBLE_WORDS64 (hx, lx, *x);
29 hx &= 0x7fffffffffffULL;
30 /* Construct the representation of the return value directly, since
31 128-bit integers may not be available. */
32 int lz;
33 if (hx == 0)
34 {
35 if (lx == 0)
36 return 0.0L;
37 else
38 lz = __builtin_clzll (lx) + 64;
39 }
40 else
41 lz = __builtin_clzll (hx);
42 int shift = lz - 15;
43 if (shift >= 64)
44 {
45 hx = lx << (shift - 64);
46 lx = 0;
47 }
48 else
49 {
50 /* 2 <= SHIFT <= 63. */
51 hx = (hx << shift) | (lx >> (64 - shift));
52 lx <<= shift;
53 }
54 hx = (hx & 0xffffffffffffULL) | ((0x3fffULL + 127 - lz) << 48);
55 _Float128 ret;
56 SET_LDOUBLE_WORDS64 (ret, hx, lx);
57 return ret;
58}
