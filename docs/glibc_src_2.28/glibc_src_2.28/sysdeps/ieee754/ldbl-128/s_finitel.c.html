20/*
21 * finitel(x) returns 1 is x is finite, else 0;
22 * no branching!
23 */
24
25#include <math.h>
26#include <math_private.h>
27
28int __finitel(_Float128 x)
29{
30 int64_t hx;
31 GET_LDOUBLE_MSW64(hx,x);
32 return (int)((uint64_t)((hx&0x7fff000000000000LL)
33 -0x7fff000000000000LL)>>63);
34}
35mathx_hidden_def (__finitel)
36weak_alias (__finitel, finitel)
