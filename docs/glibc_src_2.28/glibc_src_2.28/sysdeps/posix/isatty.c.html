18#include <unistd.h>
19#include <termios.h>
20
21/* Return 1 if FD is a terminal, 0 if not. */
22int
23__isatty (int fd)
24{
25 struct termios term;
26
27 return __tcgetattr (fd, &term) == 0;
28}
29
