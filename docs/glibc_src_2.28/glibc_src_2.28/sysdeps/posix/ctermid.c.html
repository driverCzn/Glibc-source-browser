22/* Return the name of the controlling terminal. If S is not NULL, the
23 name is copied into it (it should be at least L_ctermid bytes
24 long), otherwise we return a pointer to a non-const but read-only
25 string literal, that POSIX states the caller must not modify. */
26char *
27ctermid (char *s)
28{
29 char *name = (char /*drop const*/ *) "/dev/tty";
30
31 if (s == NULL)
32 return name;
33
34 return strcpy (s, name);
35}
