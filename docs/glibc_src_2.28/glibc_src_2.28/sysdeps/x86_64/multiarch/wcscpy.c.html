20/* Define multiple versions only for the definition in libc. */
21#if IS_IN (libc)
22# define wcscpy __redirect_wcscpy
23# include <wchar.h>
24# undef wcscpy
25
26# define SYMBOL_NAME wcscpy
27# include <init-arch.h>
28
29extern __typeof (REDIRECT_NAME) OPTIMIZE (sse2) attribute_hidden;
30extern __typeof (REDIRECT_NAME) OPTIMIZE (ssse3) attribute_hidden;
31
32static inline void *
33IFUNC_SELECTOR (void)
34{
35 const struct cpu_features* cpu_features = __get_cpu_features ();
36
37 if (CPU_FEATURES_CPU_P (cpu_features, SSSE3))
38 return OPTIMIZE (ssse3);
39
40 return OPTIMIZE (sse2);
41}
42
43libc_ifunc_redirected (__redirect_wcscpy, wcscpy, IFUNC_SELECTOR ());
44#endif
