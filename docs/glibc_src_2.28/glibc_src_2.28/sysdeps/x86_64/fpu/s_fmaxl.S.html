20#include <sysdep.h>
21#include <libm-alias-ldouble.h>
22
23 .text
24ENTRY(__fmaxl)
25 fldt 8(%rsp) // x
26 fldt 24(%rsp) // x : y
27
28 fucomi %st(1), %st
29 jp 2f
30 fcmovb %st(1), %st
31
32 fstp %st(1)
33
34 ret
35
362: // Unordered.
37 fucomi %st(0), %st
38 jp 3f
39 // st(1) is a NaN; st(0) is not. Test if st(1) is signaling.
40 testb $0x40, 15(%rsp)
41 jz 4f
42 fstp %st(1)
43 ret
44
453: // st(0) is a NaN; st(1) may or may not be.
46 fxch
47 fucomi %st(0), %st
48 jp 4f
49 // st(1) is a NaN; st(0) is not. Test if st(1) is signaling.
50 testb $0x40, 31(%rsp)
51 jz 4f
52 fstp %st(1)
53 ret
54
554: // Both arguments are NaNs, or one is a signaling NaN.
56 faddp
57 ret
58END(__fmaxl)
