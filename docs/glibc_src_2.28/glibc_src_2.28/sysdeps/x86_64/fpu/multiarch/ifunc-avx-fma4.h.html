20#include <init-arch.h>
21
22extern __typeof (REDIRECT_NAME) OPTIMIZE (sse2) attribute_hidden;
23extern __typeof (REDIRECT_NAME) OPTIMIZE (avx) attribute_hidden;
24extern __typeof (REDIRECT_NAME) OPTIMIZE (fma) attribute_hidden;
25extern __typeof (REDIRECT_NAME) OPTIMIZE (fma4) attribute_hidden;
26
27static inline void *
28IFUNC_SELECTOR (void)
29{
30 const struct cpu_features* cpu_features = __get_cpu_features ();
31
32 if (CPU_FEATURES_ARCH_P (cpu_features, FMA_Usable)
33 && CPU_FEATURES_ARCH_P (cpu_features, AVX2_Usable))
34 return OPTIMIZE (fma);
35
36 if (CPU_FEATURES_ARCH_P (cpu_features, FMA4_Usable))
37 return OPTIMIZE (fma4);
38
39 if (CPU_FEATURES_ARCH_P (cpu_features, AVX_Usable))
40 return OPTIMIZE (avx);
41
42 return OPTIMIZE (sse2);
43}
