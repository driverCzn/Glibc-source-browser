13ENTRY(__copysignl)
14 movl 32(%rsp),%edx
15 movl 16(%rsp),%eax
16 andl $0x8000,%edx
17 andl $0x7fff,%eax
18 orl %edx,%eax
19 movl %eax,16(%rsp)
20 fldt 8(%rsp)
21 ret
22END (__copysignl)
