25ENTRY (__wcscmp)
26/*
27 * This implementation uses SSE to compare up to 16 bytes at a time.
28*/
29 mov %esi, %eax
30 mov %edi, %edx
31 pxor %xmm0, %xmm0 /* clear %xmm0 for null char checks */
