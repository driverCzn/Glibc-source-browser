19#include <string.h>
20#include <rpc/rpc.h>
21#include <rpcsvc/nis.h>

23#include "nis_xdr.h"

25nis_result *
26nis_clone_result (const nis_result *src, nis_result *dest)
27{
28 char *addr;
29 unsigned int size;
30 XDR xdrs;
31
32 if (src == NULL)
33 return (NULL);
34
35 size = xdr_sizeof ((xdrproc_t)_xdr_nis_result, (char *)src);
36 if ((addr = calloc(1, size)) == NULL)
37 return NULL;
38
39 xdrmem_create (&xdrs, addr, size, XDR_ENCODE);
40 if (!_xdr_nis_result (&xdrs, (nis_result *)src))
41 {
42 xdr_destroy (&xdrs);
43 free (addr);
44 return NULL;
45 }
46 xdr_destroy (&xdrs);
47
48 nis_result *res;
49 if (dest == NULL)
50 {
51 if ((res = calloc (1, sizeof (nis_result))) == NULL)
52 {
53 free (addr);
54 return NULL;
55 }
56 }
57 else
58 res = dest;
59
60 xdrmem_create (&xdrs, addr, size, XDR_DECODE);
61 if (!_xdr_nis_result (&xdrs, res))
62 {
63 xdr_destroy (&xdrs);
64 if (res != dest)
65 free (res);
66 free (addr);
67 return NULL;
68 }
69 xdr_destroy (&xdrs);
70 free (addr);
71
72 return res;
73}
