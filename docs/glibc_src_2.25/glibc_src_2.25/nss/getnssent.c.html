18#include <errno.h>
19#include <stdlib.h>
20#include <netdb.h>
21#include "nsswitch.h"
22
23void *
24__nss_getent (getent_r_function func, void **resbuf, char **buffer,
25 size_t buflen, size_t *buffer_size, int *h_errnop)
26{
27 void *result;
28
29 if (*buffer == NULL)
30 {
31 *buffer_size = buflen;
32 *buffer = malloc (*buffer_size);
33 }
34
35 while (*buffer != NULL
36 && func (resbuf, *buffer, *buffer_size, &result, h_errnop) == ERANGE
37 && (h_errnop == NULL || *h_errnop == NETDB_INTERNAL))
38 {
39 char *new_buf;
40 *buffer_size *= 2;
41 new_buf = realloc (*buffer, *buffer_size);
42 if (new_buf == NULL)
43 {
44 /* We are out of memory. Free the current buffer so that the
45 process gets a chance for a normal termination. */
46 int save = errno;
47 free (*buffer);
48 __set_errno (save);
49 }
50 *buffer = new_buf;
51 }
52
53 if (*buffer == NULL)
54 result = NULL;
55
56 return result;
57}
