17/*
18 * significand(x) computes just
19 * scalb(x, (double) -ilogb(x)),
20 * for exercising the fraction-part(F) IEEE 754-1985 test vector.
21 */
22
23#include <math.h>
24#include <math_private.h>
25
26double __significand(double x)
27{
28 return __ieee754_scalb(x,(double) -__ilogb(x));
29}
30weak_alias (__significand, significand)
31#ifdef NO_LONG_DOUBLE
32strong_alias (__significand, __significandl)
33weak_alias (__significand, significandl)
34#endif
