24/* wrapper acosf */
25float
26__acosf (float x)
27{
28 if (__builtin_expect (isgreater (fabsf (x), 1.0f), 0)
29 && _LIB_VERSION != _IEEE_)
30 {
31 /* acos(|x|>1) */
32 feraiseexcept (FE_INVALID);
33 return __kernel_standard_f (x, x, 101);
34 }
35
36 return __ieee754_acosf (x);
37}
38weak_alias (__acosf, acosf)
