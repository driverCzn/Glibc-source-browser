19#include <errno.h>
20#include <math.h>
21#include <math_private.h>
22
23
24static double
25__attribute__ ((noinline))
26sysv_scalb (double x, double fn)
27{
28 double z = __ieee754_scalb (x, fn);
29
30 if (__glibc_unlikely (isinf (z)))
31 {
32 if (isfinite (x))
33 return __kernel_standard (x, fn, 32); /* scalb overflow */
34 else
35 __set_errno (ERANGE);
36 }
37 else if (__builtin_expect (z == 0.0, 0) && z != x)
38 return __kernel_standard (x, fn, 33); /* scalb underflow */
39
40 return z;
41}
42
43
44/* Wrapper scalb */
45double
46__scalb (double x, double fn)
47{
48 if (__glibc_unlikely (_LIB_VERSION == _SVID_))
49 return sysv_scalb (x, fn);
50 else
51 {
52 double z = __ieee754_scalb (x, fn);
53
54 if (__glibc_unlikely (!isfinite (z) || z == 0.0))
55 {
56 if (isnan (z))
57 {
58 if (!isnan (x) && !isnan (fn))
59 __set_errno (EDOM);
60 }
61 else if (isinf (z))
62 {
63 if (!isinf (x) && !isinf (fn))
64 __set_errno (ERANGE);
65 }
66 else
67 {
68 /* z == 0. */
69 if (x != 0.0 && !isinf (fn))
70 __set_errno (ERANGE);
71 }
72 }
73 return z;
74 }
75}
76weak_alias (__scalb, scalb)

78strong_alias (__scalb, __scalbl)
79weak_alias (__scalb, scalbl)
80#endif
