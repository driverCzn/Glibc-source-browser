23/* This function works on a copy of the buffer object, so that it can
24 remain non-addressable in the caller. */
25struct alloc_buffer
26__libc_alloc_buffer_copy_bytes (struct alloc_buffer buf,
27 const void *src, size_t len)
28{
29 void *ptr = alloc_buffer_alloc_bytes (&buf, len);
30 if (ptr != NULL)
31 memcpy (ptr, src, len);
32 return buf;
33}
