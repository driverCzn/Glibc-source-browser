25td_err_e
26td_ta_delete (td_thragent_t *ta)
27{
28 LOG ("td_ta_delete");
29
30 /* Safety check. Note that the test will also fail for TA == NULL. */
31 if (!ta_ok (ta))
32 return TD_BADTA;
33
34 /* Remove the handle from the list. */
35 list_del (&ta->list);
36
37 /* The handle was allocated in `td_ta_new'. */
38 free (ta);
39
40 return TD_OK;
41}
