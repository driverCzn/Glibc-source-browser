27/* Copy no more than N bytes of SRC to DEST, stopping when C is found.
28 Return the position in DEST one byte past where C was copied, or
29 NULL if C was not found in the first N bytes of SRC. */
30void *
31__memccpy (void *dest, const void *src, int c, size_t n)
32{
33 void *p = memchr (src, c, n);
34
35 if (p != NULL)
36 return __mempcpy (dest, src, p - src + 1);
37
38 memcpy (dest, src, n);
39 return NULL;
40}
