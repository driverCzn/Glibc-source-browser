18#include <string.h>
19
20#ifndef STRNCAT
21# undef strncat
22# define STRNCAT strncat
23#endif
24
25char *
26STRNCAT (char *s1, const char *s2, size_t n)
27{
28 char *s = s1;
29
30 /* Find the end of S1. */
31 s1 += strlen (s1);
32
33 size_t ss = __strnlen (s2, n);
34
35 s1[ss] = '\0';
36 memcpy (s1, s2, ss);
37
38 return s;
39}
