18#include <string.h>
19
20#undef strncpy
21
22#ifndef STRNCPY
23 #define STRNCPY strncpy
24#endif
25
26char *
27STRNCPY (char *s1, const char *s2, size_t n)
28{
29 size_t size = __strnlen (s2, n);
30 if (size != n)
31 memset (s1 + size, '\0', n - size);
32 return memcpy (s1, s2, size);
33}
