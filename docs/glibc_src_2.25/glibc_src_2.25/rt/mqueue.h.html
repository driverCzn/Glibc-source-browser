18#ifndef _MQUEUE_H
19#define _MQUEUE_H 1
20
21#include <features.h>
22#include <sys/types.h>
23#include <fcntl.h>
24#define __need_sigevent_t
25#include <bits/siginfo.h>
26#include <bits/types/struct_timespec.h>
27/* Get the definition of mqd_t and struct mq_attr. */
28#include <bits/mqueue.h>
29
30__BEGIN_DECLS
31
32/* Establish connection between a process and a message queue NAME and
33 return message queue descriptor or (mqd_t) -1 on error. OFLAG determines
34 the type of access used. If O_CREAT is on OFLAG, the third argument is
35 taken as a `mode_t', the mode of the created message queue, and the fourth
36 argument is taken as `struct mq_attr *', pointer to message queue
37 attributes. If the fourth argument is NULL, default attributes are
38 used. */
39extern mqd_t mq_open (const char *__name, int __oflag, ...)
40 __THROW __nonnull ((1));
41
42/* Removes the association between message queue descriptor MQDES and its
43 message queue. */
44extern int mq_close (mqd_t __mqdes) __THROW;
45
46/* Query status and attributes of message queue MQDES. */
47extern int mq_getattr (mqd_t __mqdes, struct mq_attr *__mqstat)
48 __THROW __nonnull ((2));
49
50/* Set attributes associated with message queue MQDES and if OMQSTAT is
51 not NULL also query its old attributes. */
52extern int mq_setattr (mqd_t __mqdes,
53 const struct mq_attr *__restrict __mqstat,
54 struct mq_attr *__restrict __omqstat)
55 __THROW __nonnull ((2));
56
57/* Remove message queue named NAME. */
58extern int mq_unlink (const char *__name) __THROW __nonnull ((1));
59
60/* Register notification issued upon message arrival to an empty
61 message queue MQDES. */
62extern int mq_notify (mqd_t __mqdes, const struct sigevent *__notification)
63 __THROW;
64
65/* Receive the oldest from highest priority messages in message queue
66 MQDES. */
67extern ssize_t mq_receive (mqd_t __mqdes, char *__msg_ptr, size_t __msg_len,
68 unsigned int *__msg_prio) __nonnull ((2));
69
70/* Add message pointed by MSG_PTR to message queue MQDES. */
71extern int mq_send (mqd_t __mqdes, const char *__msg_ptr, size_t __msg_len,
72 unsigned int __msg_prio) __nonnull ((2));
73
74#ifdef __USE_XOPEN2K
75/* Receive the oldest from highest priority messages in message queue
76 MQDES, stop waiting if ABS_TIMEOUT expires. */
77extern ssize_t mq_timedreceive (mqd_t __mqdes, char *__restrict __msg_ptr,
78 size_t __msg_len,
79 unsigned int *__restrict __msg_prio,
80 const struct timespec *__restrict __abs_timeout)
81 __nonnull ((2, 5));
82
83/* Add message pointed by MSG_PTR to message queue MQDES, stop blocking
84 on full message queue if ABS_TIMEOUT expires. */
85extern int mq_timedsend (mqd_t __mqdes, const char *__msg_ptr,
86 size_t __msg_len, unsigned int __msg_prio,
87 const struct timespec *__abs_timeout)
88 __nonnull ((2, 5));
89#endif
90
91/* Define some inlines helping to catch common problems. */
92#if __USE_FORTIFY_LEVEL > 0 && defined __fortify_function \
93 && defined __va_arg_pack_len
94# include <bits/mqueue2.h>
95#endif
96
97__END_DECLS
98
99#endif /* mqueue.h */
