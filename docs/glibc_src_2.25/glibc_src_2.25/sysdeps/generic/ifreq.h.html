26static inline struct ifreq *
27__if_nextreq (struct ifreq *ifr)
28{
29#ifdef _HAVE_SA_LEN
30 if (ifr->ifr_addr.sa_len > sizeof ifr->ifr_addr)
31 return (struct ifreq *) ((char *) &ifr->ifr_addr + ifr->ifr_addr.sa_len);
32#endif
33 return ifr + 1;
34}

39static inline void
40__if_freereq (struct ifreq *ifreqs, int num_ifs)
41{
42 free (ifreqs);
43}
