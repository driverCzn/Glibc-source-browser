21#include <string.h>
22#include <utmp.h>
23
24#include "utmp-private.h"

26/* Test whether two entries match. */
27static int
28__utmp_equal (const struct utmp *entry, const struct utmp *match)
29{
30 return
31 (
32#if _HAVE_UT_TYPE - 0
33 (entry->ut_type == INIT_PROCESS
34 || entry->ut_type == LOGIN_PROCESS
35 || entry->ut_type == USER_PROCESS
36 || entry->ut_type == DEAD_PROCESS)
37 &&
38 (match->ut_type == INIT_PROCESS
39 || match->ut_type == LOGIN_PROCESS
40 || match->ut_type == USER_PROCESS
41 || match->ut_type == DEAD_PROCESS)
42 &&
43#endif
44#if _HAVE_UT_ID - 0
45 (entry->ut_id[0] && match->ut_id[0]
46 ? strncmp (entry->ut_id, match->ut_id, sizeof match->ut_id) == 0
47 : strncmp (entry->ut_line, match->ut_line, sizeof match->ut_line) == 0)
48#else
49 strncmp (entry->ut_line, match->ut_line, sizeof match->ut_line) == 0
50#endif
51 );
52}
