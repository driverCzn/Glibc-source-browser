20#include <errno.h>
21#define __need_NULL
22#include <stddef.h>
23#include <signal.h>
24#include <string.h> /* For the real memset prototype. */

26
27int
28sigignore (int sig)
29{
30 struct sigaction act;
31
32 act.sa_handler = SIG_IGN;
33 if (__sigemptyset (&act.sa_mask) < 0)
34 return -1;
35 act.sa_flags = 0;
36
37 return __sigaction (sig, &act, NULL);
38}
