22/* wrapper exp */
23double
24__exp (double x)
25{
26 double z = __ieee754_exp (x);
27 if (__builtin_expect (!isfinite (z) || z == 0, 0)
28 && isfinite (x) && _LIB_VERSION != _IEEE_)
29 return __kernel_standard (x, x, 6 + !!signbit (x));
30
31 return z;
32}
33hidden_def (__exp)
34weak_alias (__exp, exp)
35#ifdef NO_LONG_DOUBLE
36strong_alias (__exp, __expl)
37weak_alias (__exp, expl)
38#endif
