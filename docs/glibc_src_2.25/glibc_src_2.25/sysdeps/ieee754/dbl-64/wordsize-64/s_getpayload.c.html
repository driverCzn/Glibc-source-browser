19#include <math.h>
20#include <math_private.h>
21#include <stdint.h>
22
23double
24getpayload (const double *x)
25{
26 uint64_t ix;
27 EXTRACT_WORDS64 (ix, *x);
28 ix &= 0x7ffffffffffffULL;
29 return (double) ix;
30}
31#ifdef NO_LONG_DOUBLE
32weak_alias (getpayload, getpayloadl)
33#endif
