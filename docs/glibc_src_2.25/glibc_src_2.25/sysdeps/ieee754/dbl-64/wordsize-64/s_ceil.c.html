13/*
14 * ceil(x)
15 * Return x rounded toward -inf to integral value
16 * Method:
17 * Bit twiddling.
18 */
19
20#include <math.h>
21#include <math_private.h>
22
23double
24__ceil(double x)
25{
26 int64_t i0,i;
27 int32_t j0;
28 EXTRACT_WORDS64(i0,x);
29 j0 = ((i0>>52)&0x7ff)-0x3ff;
30 if(j0<=51) {
31 if(j0<0) {
32 /* return 0*sign(x) if |x|<1 */
33 if(i0<0) {i0=INT64_C(0x8000000000000000);}
34 else if(i0!=0) { i0=INT64_C(0x3ff0000000000000);}
35 } else {
36 i = INT64_C(0x000fffffffffffff)>>j0;
37 if((i0&i)==0) return x; /* x is integral */
38 if(i0>0) i0 += UINT64_C(0x0010000000000000)>>j0;
39 i0 &= (~i);
40 }
41 } else {
42 if(j0==0x400) return x+x; /* inf or NaN */
43 else return x; /* x is integral */
44 }
45 INSERT_WORDS64(x,i0);
46 return x;
47}

50# ifdef NO_LONG_DOUBLE
51strong_alias (__ceil, __ceill)
52weak_alias (__ceil, ceill)
53# endif
54#endif
