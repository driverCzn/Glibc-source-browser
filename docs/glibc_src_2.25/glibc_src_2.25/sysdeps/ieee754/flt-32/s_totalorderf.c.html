19#include <math.h>
20#include <math_private.h>
21#include <nan-high-order-bit.h>
22#include <stdint.h>
23
24int
25totalorderf (float x, float y)
26{
27 int32_t ix, iy;
28 GET_FLOAT_WORD (ix, x);
29 GET_FLOAT_WORD (iy, y);
30#if HIGH_ORDER_BIT_IS_SET_FOR_SNAN
31 /* For the preferred quiet NaN convention, this operation is a
32 comparison of the representations of the arguments interpreted as
33 sign-magnitude integers. If both arguments are NaNs, invert the
34 quiet/signaling bit so comparing that way works. */
35 if ((ix & 0x7fffffff) > 0x7f800000 && (iy & 0x7fffffff) > 0x7f800000)
36 {
37 ix ^= 0x00400000;
38 iy ^= 0x00400000;
39 }
40#endif
41 uint32_t ix_sign = ix >> 31;
42 uint32_t iy_sign = iy >> 31;
43 ix ^= ix_sign >> 1;
44 iy ^= iy_sign >> 1;
45 return ix <= iy;
46}
