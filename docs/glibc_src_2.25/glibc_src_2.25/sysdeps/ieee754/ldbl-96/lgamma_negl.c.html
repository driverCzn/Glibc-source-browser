280/* Compute sin (pi * X) for -0.25 <= X <= 0.5. */
281
282static long double
283lg_sinpi (long double x)
284{
285 if (x <= 0.25L)
286 return __sinl (M_PIl * x);
287 else
288 return __cosl (M_PIl * (0.5L - x));
289}
290
291/* Compute cos (pi * X) for -0.25 <= X <= 0.5. */
292
293static long double
294lg_cospi (long double x)
295{
296 if (x <= 0.25L)
297 return __cosl (M_PIl * x);
298 else
299 return __sinl (M_PIl * (0.5L - x));
300}
301
302/* Compute cot (pi * X) for -0.25 <= X <= 0.5. */
303
304static long double
305lg_cotpi (long double x)
306{
307 return lg_cospi (x) / lg_sinpi (x);
308}

310/* Compute lgamma of a negative argument -33 < X < -2, setting
311 *SIGNGAMP accordingly. */
312
313long double
314__lgamma_negl (long double x, int *signgamp)
315{
316 /* Determine the half-integer region X lies in, handle exact
317 integers and determine the sign of the result. */
318 int i = __floorl (-2 * x);
319 if ((i & 1) == 0 && i == -2 * x)
320 return 1.0L / 0.0L;
321 long double xn = ((i & 1) == 0 ? -i / 2 : (-i - 1) / 2);
322 i -= 4;
323 *signgamp = ((i & 2) == 0 ? -1 : 1);
324
325 SET_RESTORE_ROUNDL (FE_TONEAREST);
326
327 /* Expand around the zero X0 = X0_HI + X0_LO. */
328 long double x0_hi = lgamma_zeros[i][0], x0_lo = lgamma_zeros[i][1];
329 long double xdiff = x - x0_hi - x0_lo;
330
331 /* For arguments in the range -3 to -2, use polynomial
332 approximations to an adjusted version of the gamma function. */
333 if (i < 2)
334 {
335 int j = __floorl (-8 * x) - 16;
336 long double xm = (-33 - 2 * j) * 0.0625L;
337 long double x_adj = x - xm;
338 size_t deg = poly_deg[j];
339 size_t end = poly_end[j];
340 long double g = poly_coeff[end];
341 for (size_t j = 1; j <= deg; j++)
342 g = g * x_adj + poly_coeff[end - j];
343 return __log1pl (g * xdiff / (x - xn));
344 }
345
346 /* The result we want is log (sinpi (X0) / sinpi (X))
347 + log (gamma (1 - X0) / gamma (1 - X)). */
348 long double x_idiff = fabsl (xn - x), x0_idiff = fabsl (xn - x0_hi - x0_lo);
349 long double log_sinpi_ratio;
350 if (x0_idiff < x_idiff * 0.5L)
351 /* Use log not log1p to avoid inaccuracy from log1p of arguments
352 close to -1. */
353 log_sinpi_ratio = __ieee754_logl (lg_sinpi (x0_idiff)
354 / lg_sinpi (x_idiff));
355 else
356 {
357 /* Use log1p not log to avoid inaccuracy from log of arguments
358 close to 1. X0DIFF2 has positive sign if X0 is further from
359 XN than X is from XN, negative sign otherwise. */
360 long double x0diff2 = ((i & 1) == 0 ? xdiff : -xdiff) * 0.5L;
361 long double sx0d2 = lg_sinpi (x0diff2);
362 long double cx0d2 = lg_cospi (x0diff2);
363 log_sinpi_ratio = __log1pl (2 * sx0d2
364 * (-sx0d2 + cx0d2 * lg_cotpi (x_idiff)));
365 }
366
367 long double log_gamma_ratio;
368 long double y0 = 1 - x0_hi;
369 long double y0_eps = -x0_hi + (1 - y0) - x0_lo;
370 long double y = 1 - x;
371 long double y_eps = -x + (1 - y);
372 /* We now wish to compute LOG_GAMMA_RATIO
373 = log (gamma (Y0 + Y0_EPS) / gamma (Y + Y_EPS)). XDIFF
374 accurately approximates the difference Y0 + Y0_EPS - Y -
375 Y_EPS. Use Stirling's approximation. First, we may need to
376 adjust into the range where Stirling's approximation is
377 sufficiently accurate. */
378 long double log_gamma_adj = 0;
379 if (i < 8)
380 {
381 int n_up = (9 - i) / 2;
382 long double ny0, ny0_eps, ny, ny_eps;
383 ny0 = y0 + n_up;
384 ny0_eps = y0 - (ny0 - n_up) + y0_eps;
385 y0 = ny0;
386 y0_eps = ny0_eps;
387 ny = y + n_up;
388 ny_eps = y - (ny - n_up) + y_eps;
389 y = ny;
390 y_eps = ny_eps;
391 long double prodm1 = __lgamma_productl (xdiff, y - n_up, y_eps, n_up);
392 log_gamma_adj = -__log1pl (prodm1);
393 }
394 long double log_gamma_high
395 = (xdiff * __log1pl ((y0 - e_hi - e_lo + y0_eps) / e_hi)
396 + (y - 0.5L + y_eps) * __log1pl (xdiff / y) + log_gamma_adj);
397 /* Compute the sum of (B_2k / 2k(2k-1))(Y0^-(2k-1) - Y^-(2k-1)). */
398 long double y0r = 1 / y0, yr = 1 / y;
399 long double y0r2 = y0r * y0r, yr2 = yr * yr;
400 long double rdiff = -xdiff / (y * y0);
401 long double bterm[NCOEFF];
402 long double dlast = rdiff, elast = rdiff * yr * (yr + y0r);
403 bterm[0] = dlast * lgamma_coeff[0];
404 for (size_t j = 1; j < NCOEFF; j++)
405 {
406 long double dnext = dlast * y0r2 + elast;
407 long double enext = elast * yr2;
408 bterm[j] = dnext * lgamma_coeff[j];
409 dlast = dnext;
410 elast = enext;
411 }
412 long double log_gamma_low = 0;
413 for (size_t j = 0; j < NCOEFF; j++)
414 log_gamma_low += bterm[NCOEFF - 1 - j];
415 log_gamma_ratio = log_gamma_high + log_gamma_low;
416
417 return log_sinpi_ratio + log_gamma_ratio;
418}
