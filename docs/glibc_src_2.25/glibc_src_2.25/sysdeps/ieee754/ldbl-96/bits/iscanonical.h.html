23extern int __iscanonicall (long double __x)
24 __THROW __attribute__ ((__const__));
25#define __iscanonicalf(x) ((void) (__typeof (x)) (x), 1)
26#define __iscanonical(x) ((void) (__typeof (x)) (x), 1)
27
28/* Return nonzero value if X is canonical. In IEEE interchange binary
29 formats, all values are canonical, but the argument must still be
30 converted to its semantic type for any exceptions arising from the
31 conversion, before being discarded; in extended precision, there
32 are encodings that are not consistently handled as corresponding to
33 any particular value of the type, and we return 0 for those. */
34#define iscanonical(x) __MATH_TG ((x), __iscanonical, (x))
