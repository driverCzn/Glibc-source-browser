20#include <machine/asm.h>
21
22 .section .rodata
23
24 .align ALIGNARG(4)
25 .type mask,@object
26mask:
27 .byte 0xff, 0xff, 0xff, 0x7f
28 ASM_SIZE_DIRECTIVE(mask)
29
30#ifdef PIC
31#define MO(op) op##(%rip)
32#else
33#define MO(op) op
34#endif
35
36 .text
37ENTRY(__copysignf)
38 movss MO(mask),%xmm3
39 andps %xmm3,%xmm0
40 andnps %xmm1,%xmm3
41 orps %xmm3,%xmm0
42 retq
43END (__copysignf)
44
45weak_alias (__copysignf, copysignf)
