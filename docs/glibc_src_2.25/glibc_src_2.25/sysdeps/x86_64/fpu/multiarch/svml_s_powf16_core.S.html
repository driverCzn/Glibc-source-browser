23ENTRY (_ZGVeN16vv_powf)
24 .type _ZGVeN16vv_powf, @gnu_indirect_function
25 LOAD_RTLD_GLOBAL_RO_RDX
26 leaq _ZGVeN16vv_powf_skx(%rip), %rax
27 HAS_ARCH_FEATURE (AVX512DQ_Usable)
28 jnz 2f
29 leaq _ZGVeN16vv_powf_knl(%rip), %rax
30 HAS_ARCH_FEATURE (AVX512F_Usable)
31 jnz 2f
32 leaq _ZGVeN16vv_powf_avx2_wrapper(%rip), %rax
332: ret
34END (_ZGVeN16vv_powf)
