27#define POPCNT(l) \
28 ({ __cpu_mask r; \
29 asm ("popcnt %1, %0" : "=r" (r) : "0" (l));\
30 r; })
31#define __sched_cpucount static popcount_cpucount
32#include <posix/sched_cpucount.c>
