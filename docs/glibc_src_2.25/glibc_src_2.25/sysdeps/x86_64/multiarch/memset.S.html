24/* Define multiple versions only for the definition in lib. */
25#if IS_IN (libc)
26ENTRY(memset)
27 .type memset, @gnu_indirect_function
28 LOAD_RTLD_GLOBAL_RO_RDX
29 lea __memset_erms(%rip), %RAX_LP
30 HAS_ARCH_FEATURE (Prefer_ERMS)
31 jnz 2f
32 lea __memset_sse2_unaligned_erms(%rip), %RAX_LP
33 HAS_CPU_FEATURE (ERMS)
34 jnz 1f
35 lea __memset_sse2_unaligned(%rip), %RAX_LP
361:
37 HAS_ARCH_FEATURE (AVX2_Usable)
38 jz 2f
39 lea __memset_avx2_unaligned_erms(%rip), %RAX_LP
40 HAS_CPU_FEATURE (ERMS)
41 jnz L(AVX512F)
42 lea __memset_avx2_unaligned(%rip), %RAX_LP
43L(AVX512F):
44 HAS_ARCH_FEATURE (Prefer_No_AVX512)
45 jnz 2f
46 HAS_ARCH_FEATURE (AVX512F_Usable)
47 jz 2f
48 lea __memset_avx512_no_vzeroupper(%rip), %RAX_LP
49 HAS_ARCH_FEATURE (Prefer_No_VZEROUPPER)
50 jnz 2f
51 lea __memset_avx512_unaligned_erms(%rip), %RAX_LP
52 HAS_CPU_FEATURE (ERMS)
53 jnz 2f
54 lea __memset_avx512_unaligned(%rip), %RAX_LP
552: ret
56END(memset)
57#endif

59#if IS_IN (libc)
60# define MEMSET_SYMBOL(p,s) p##_sse2_##s

63# undef libc_hidden_builtin_def
64/* It doesn't make sense to send libc-internal memset calls through a PLT.
65 The speedup we get from using SSE2 instructions is likely eaten away
66 by the indirect call in the PLT. */
67# define libc_hidden_builtin_def(name) \
68 .globl __GI_memset; __GI_memset = __memset_sse2_unaligned
69# endif
