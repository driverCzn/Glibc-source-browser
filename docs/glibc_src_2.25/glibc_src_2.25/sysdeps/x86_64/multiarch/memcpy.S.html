21#include <sysdep.h>
22#include <init-arch.h>

24/* Define multiple versions only for the definition in lib and for
25 DSO. In static binaries we need memcpy before the initialization
26 happened. */
27#if defined SHARED && IS_IN (libc)
28 .text
29ENTRY(__new_memcpy)
30 .type __new_memcpy, @gnu_indirect_function
31 LOAD_RTLD_GLOBAL_RO_RDX
32 lea __memcpy_erms(%rip), %RAX_LP
33 HAS_ARCH_FEATURE (Prefer_ERMS)
34 jnz 2f
35 HAS_ARCH_FEATURE (Prefer_No_AVX512)
36 jnz 1f
37 HAS_ARCH_FEATURE (AVX512F_Usable)
38 jz 1f
39 lea __memcpy_avx512_no_vzeroupper(%rip), %RAX_LP
40 HAS_ARCH_FEATURE (Prefer_No_VZEROUPPER)
41 jnz 2f
42 lea __memcpy_avx512_unaligned_erms(%rip), %RAX_LP
43 HAS_CPU_FEATURE (ERMS)
44 jnz 2f
45 lea __memcpy_avx512_unaligned(%rip), %RAX_LP
46 ret
471: lea __memcpy_avx_unaligned(%rip), %RAX_LP
48 HAS_ARCH_FEATURE (AVX_Fast_Unaligned_Load)
49 jz L(Fast_Unaligned_Load)
50 HAS_CPU_FEATURE (ERMS)
51 jz 2f
52 lea __memcpy_avx_unaligned_erms(%rip), %RAX_LP
53 ret
54L(Fast_Unaligned_Load):
55 lea __memcpy_sse2_unaligned(%rip), %RAX_LP
56 HAS_ARCH_FEATURE (Fast_Unaligned_Copy)
57 jz L(SSSE3)
58 HAS_CPU_FEATURE (ERMS)
59 jz 2f
60 lea __memcpy_sse2_unaligned_erms(%rip), %RAX_LP
61 ret
62L(SSSE3):
63 HAS_CPU_FEATURE (SSSE3)
64 jz 2f
65 lea __memcpy_ssse3_back(%rip), %RAX_LP
66 HAS_ARCH_FEATURE (Fast_Copy_Backward)
67 jnz 2f
68 lea __memcpy_ssse3(%rip), %RAX_LP
692: ret
70END(__new_memcpy)
71
72# undef memcpy
73# include <shlib-compat.h>
74versioned_symbol (libc, __new_memcpy, memcpy, GLIBC_2_14);
75#endif
