25int
26__fpclassifyl (long double x)
27{
28 u_int32_t ex, hx, lx;
29 int retval = FP_NORMAL;
30
31 GET_LDOUBLE_WORDS (ex, hx, lx, x);
32 ex &= 0x7fff;
33 if ((ex | lx | hx) == 0)
34 retval = FP_ZERO;
35 else if (ex == 0 && (hx & 0x80000000) == 0)
36 retval = FP_SUBNORMAL;
37 else if (ex == 0x7fff)
38 retval = ((hx & 0x7fffffff) | lx) != 0 ? FP_NAN : FP_INFINITE;
39
40 return retval;
41}
