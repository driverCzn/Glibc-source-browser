25/* Reasonable initial tuning values, may be revised in the future.
26 This is a conservative initial value. */
27
28struct elision_config __elision_aconf =
29 {
30 /* How often to not attempt to use elision if a transaction aborted
31 because the lock is already acquired. Expressed in number of lock
32 acquisition attempts. */
33 .skip_lock_busy = 3,
34 /* How often to not attempt to use elision if a transaction aborted due
35 to reasons other than other threads' memory accesses. Expressed in
36 number of lock acquisition attempts. */
37 .skip_lock_internal_abort = 3,
38 /* How often we retry using elision if there is chance for the transaction
39 to finish execution (e.g., it wasn't aborted due to the lock being
40 already acquired. */
41 .retry_try_xbegin = 3,
42 /* Same as SKIP_LOCK_INTERNAL_ABORT but for trylock. */
43 .skip_trylock_internal_abort = 3,
44 };
45
46/* Force elision for all new locks. This is used to decide whether existing
47 DEFAULT locks should be automatically upgraded to elision in
48 pthread_mutex_lock(). Disabled for suid programs. Only used when elision
49 is available. */
50
51int __pthread_force_elision attribute_hidden;
52
53/* Initialize elison. */
54
55static void
56elision_init (int argc __attribute__ ((unused)),
57 char **argv __attribute__ ((unused)),
58 char **environ)
59{
60 int elision_available = HAS_CPU_FEATURE (RTM);
61#ifdef ENABLE_LOCK_ELISION
62 __pthread_force_elision = __libc_enable_secure ? 0 : elision_available;
63#endif
64 if (!elision_available)
65 __elision_aconf.retry_try_xbegin = 0; /* Disable elision on rwlocks */
66}
67
68#ifdef SHARED
69# define INIT_SECTION ".init_array"
70#else
71# define INIT_SECTION ".preinit_array"
72#endif
73
74void (*const __pthread_init_array []) (int, char **, char **)
75 __attribute__ ((section (INIT_SECTION), aligned (sizeof (void *)))) =
76{
77 &elision_init
78};
