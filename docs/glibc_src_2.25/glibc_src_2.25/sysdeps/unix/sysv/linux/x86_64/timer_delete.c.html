30#if SHLIB_COMPAT (librt, GLIBC_2_2, GLIBC_2_3_3)
31int
32__timer_delete_old (int timerid)
33{
34 int res = __timer_delete_new (__compat_timer_list[timerid]);
35
36 if (res == 0)
37 /* Successful timer deletion, now free the index. We only need to
38 store a word and that better be atomic. */
39 __compat_timer_list[timerid] = NULL;
40
41 return res;
42}
43compat_symbol (librt, __timer_delete_old, timer_delete, GLIBC_2_2);
44#endif
