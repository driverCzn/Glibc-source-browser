22/* If the Prefer_MAP_32BIT_EXEC bit is set, try to map executable pages
23 with MAP_32BIT first. */
24#define MMAP_PREPARE(addr, len, prot, flags, fd, offset) \
25 if ((addr) == NULL \
26 && ((prot) & PROT_EXEC) != 0 \
27 && HAS_ARCH_FEATURE (Prefer_MAP_32BIT_EXEC)) \
28 { \
29 __ptr_t ret = (__ptr_t) INLINE_SYSCALL (mmap, 6, (addr), (len), \
30 (prot), \
31 (flags) | MAP_32BIT, \
32 (fd), (offset)); \
33 if (ret != MAP_FAILED) \
34 return ret; \
35 }
