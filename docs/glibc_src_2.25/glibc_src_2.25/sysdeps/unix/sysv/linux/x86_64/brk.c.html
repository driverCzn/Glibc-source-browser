19#include <errno.h>
20#include <unistd.h>
21#include <sysdep.h>
22
23/* This must be initialized data because commons can't have aliases. */
24void *__curbrk = 0;
25
26int
27__brk (void *addr)
28{
29 void *newbrk;
30
31 __curbrk = newbrk = (void *) INLINE_SYSCALL (brk, 1, addr);
32
33 if (newbrk < addr)
34 {
35 __set_errno (ENOMEM);
36 return -1;
37 }
38
39 return 0;
40}
41weak_alias (__brk, brk)
