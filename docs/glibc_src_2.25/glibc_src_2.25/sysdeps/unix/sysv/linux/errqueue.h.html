20#ifndef _BITS_ERRQUEUE_H
21#define _BITS_ERRQUEUE_H 1
22
23#include <sys/types.h>
24#include <sys/socket.h>
25
26struct sock_extended_err
27 {
28 u_int32_t ee_errno;
29 u_int8_t ee_origin;
30 u_int8_t ee_type;
31 u_int8_t ee_code;
32 u_int8_t ee_pad;
33 u_int32_t ee_info;
34 u_int32_t ee_data;
35 };
36
37#define SO_EE_ORIGIN_NONE 0
38#define SO_EE_ORIGIN_LOCAL 1
39#define SO_EE_ORIGIN_ICMP 2
40#define SO_EE_ORIGIN_ICMP6 3
41
42#define SO_EE_OFFENDER(see) \
43 ((struct sockaddr *)(((struct sock_extended_err)(see))+1))
44
45#endif /* bits/errqueue.h */
