18#include <fcntl.h>
19#include <sysdep.h>

21extern int __posix_fallocate64_l64 (int fd, __off64_t offset, __off64_t len);
22#define __posix_fallocate64_l64 static internal_fallocate64
23#include <sysdeps/posix/posix_fallocate64.c>
24#undef __posix_fallocate64_l64
25
26/* Reserve storage for the data of the file associated with FD. */
27int
28__posix_fallocate64_l64 (int fd, __off64_t offset, __off64_t len)
29{
30 INTERNAL_SYSCALL_DECL (err);
31#ifdef INTERNAL_SYSCALL_TYPES
32 int res = INTERNAL_SYSCALL_TYPES (fallocate, err, 4, int, fd,
33 int, 0, off_t, offset,
34 off_t, len);
35#else
36 int res = INTERNAL_SYSCALL_CALL (fallocate, err, fd, 0,
37 SYSCALL_LL64 (offset), SYSCALL_LL64 (len));
38#endif
39 if (! INTERNAL_SYSCALL_ERROR_P (res, err))
40 return 0;
41 if (INTERNAL_SYSCALL_ERRNO (res, err) != EOPNOTSUPP)
42 return INTERNAL_SYSCALL_ERRNO (res, err);
43 return internal_fallocate64 (fd, offset, len);
44}
