26int
27__sysctl (int *name, int nlen, void *oldval, size_t *oldlenp,
28 void *newval, size_t newlen)
29{
30 struct __sysctl_args args =
31 {
32 .name = name,
33 .nlen = nlen,
34 .oldval = oldval,
35 .oldlenp = oldlenp,
36 .newval = newval,
37 .newlen = newlen
38 };
39
40 return INLINE_SYSCALL (_sysctl, 1, &args);
41}
