18#include <errno.h>
19#include <nss.h>
20#include <stdio.h>
21#include <shadow.h>
22
23#define flockfile(s) _IO_flockfile (s)
24#define funlockfile(s) _IO_funlockfile (s)
25
26#define _S(x) x ? x : ""
27
28
29/* Write an entry to the given stream.
30 This must know the format of the password file. */
31int
32putspent (const struct spwd *p, FILE *stream)
33{
34 int errors = 0;
35
36 if (p->sp_namp == NULL || !__nss_valid_field (p->sp_namp)
37 || !__nss_valid_field (p->sp_pwdp))
38 {
39 __set_errno (EINVAL);
40 return -1;
41 }
42
43 flockfile (stream);
44
45 if (fprintf (stream, "%s:%s:", p->sp_namp, _S (p->sp_pwdp)) < 0)
46 ++errors;
47
48 if ((p->sp_lstchg != (long int) -1
49 && fprintf (stream, "%ld:", p->sp_lstchg) < 0)
50 || (p->sp_lstchg == (long int) -1
51 && putc_unlocked (':', stream) == EOF))
52 ++errors;
53
54 if ((p->sp_min != (long int) -1
55 && fprintf (stream, "%ld:", p->sp_min) < 0)
56 || (p->sp_min == (long int) -1
57 && putc_unlocked (':', stream) == EOF))
58 ++errors;
59
60 if ((p->sp_max != (long int) -1
61 && fprintf (stream, "%ld:", p->sp_max) < 0)
62 || (p->sp_max == (long int) -1
63 && putc_unlocked (':', stream) == EOF))
64 ++errors;
65
66 if ((p->sp_warn != (long int) -1
67 && fprintf (stream, "%ld:", p->sp_warn) < 0)
68 || (p->sp_warn == (long int) -1
69 && putc_unlocked (':', stream) == EOF))
70 ++errors;
71
72 if ((p->sp_inact != (long int) -1
73 && fprintf (stream, "%ld:", p->sp_inact) < 0)
74 || (p->sp_inact == (long int) -1
75 && putc_unlocked (':', stream) == EOF))
76 ++errors;
77
78 if ((p->sp_expire != (long int) -1
79 && fprintf (stream, "%ld:", p->sp_expire) < 0)
80 || (p->sp_expire == (long int) -1
81 && putc_unlocked (':', stream) == EOF))
82 ++errors;
83
84 if (p->sp_flag != ~0ul
85 && fprintf (stream, "%ld", p->sp_flag) < 0)
86 ++errors;
87
88 if (putc_unlocked ('\n', stream) == EOF)
89 ++errors;
90
91 funlockfile (stream);
92
93 return errors ? -1 : 0;
94}
