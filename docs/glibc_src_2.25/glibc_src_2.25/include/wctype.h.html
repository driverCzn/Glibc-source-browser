89/* The spec says that isdigit must only match the decimal digits. We
90 can check this without a memory access. */
91# if IS_IN (libc)
92# undef iswdigit
93# define iswdigit(c) ({ wint_t __c = (c); __c >= L'0' && __c <= L'9'; })
