42/* The `stat', `fstat', `lstat' functions have to be handled special since
43 even while not compiling the library with optimization calls to these
44 functions in the shared library must reference the `xstat' etc functions.
45 We have to use macros but we cannot define them in the normal headers
46 since on user level we must use real functions. */
47#define stat(fname, buf) __xstat (_STAT_VER, fname, buf)
48#define lstat(fname, buf) __lxstat (_STAT_VER, fname, buf)
49#define __lstat(fname, buf) __lxstat (_STAT_VER, fname, buf)
