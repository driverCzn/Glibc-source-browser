24libc_freeres_ptr (static struct utmp *buffer);

27struct utmp *
28__getutent (void)
29{
30 struct utmp *result;
31
32 if (buffer == NULL)
33 {
34 buffer = (struct utmp *) malloc (sizeof (struct utmp));
35 if (buffer == NULL)
36 return NULL;
37 }
38
39 if (__getutent_r (buffer, &result) < 0)
40 return NULL;
41
42 return result;
43}
44weak_alias (__getutent, getutent)
