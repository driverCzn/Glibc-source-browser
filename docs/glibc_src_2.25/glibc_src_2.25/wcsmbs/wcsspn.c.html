25/* Return the length of the maximum initial segment
26 of WCS which contains only wide-characters in ACCEPT. */
27size_t
28wcsspn (const wchar_t *wcs, const wchar_t *accept)
29{
30 const wchar_t *p;
31 const wchar_t *a;
32 size_t count = 0;
33
34 for (p = wcs; *p != L'\0'; ++p)
35 {
36 for (a = accept; *a != L'\0'; ++a)
37 if (*p == *a)
38 break;
39 if (*a == L'\0')
40 return count;
41 else
42 ++count;
43 }
44
45 return count;
46}
