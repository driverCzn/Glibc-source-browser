81/*
82 * Ascii internet address interpretation routine.
83 * The value returned is in network order.
84 */
85in_addr_t
86__inet_addr(const char *cp) {
87 struct in_addr val;
88
89 if (__inet_aton(cp, &val))
90 return (val.s_addr);
91 return (INADDR_NONE);
92}

95/*
96 * Check whether "cp" is a valid ascii representation
97 * of an Internet address and convert to a binary address.
98 * Returns 1 if the address is valid, 0 if not.
99 * This replaces inet_addr, the return value from which
100 * cannot distinguish between failure and a local broadcast address.
101 */
102int
103__inet_aton(const char *cp, struct in_addr *addr)
104{
105 static const in_addr_t max[4] = { 0xffffffff, 0xffffff, 0xffff, 0xff };
106 in_addr_t val;
107 char c;
108 union iaddr {
109 uint8_t bytes[4];
110 uint32_t word;
111 } res;
112 uint8_t *pp = res.bytes;
113 int digit;
114
115 int saved_errno = errno;
116 __set_errno (0);
117
118 res.word = 0;
119
120 c = *cp;
121 for (;;) {
122 /*
123 * Collect number up to ``.''.
124 * Values are specified as for C:
125 * 0x=hex, 0=octal, isdigit=decimal.
126 */
127 if (!isdigit(c))
128 goto ret_0;
129 {
130 char *endp;
131 unsigned long ul = strtoul (cp, (char **) &endp, 0);
132 if (ul == ULONG_MAX && errno == ERANGE)
133 goto ret_0;
134 if (ul > 0xfffffffful)
135 goto ret_0;
136 val = ul;
137 digit = cp != endp;
138 cp = endp;
139 }
140 c = *cp;
141 if (c == '.') {
142 /*
143 * Internet format:
144 * a.b.c.d
145 * a.b.c (with c treated as 16 bits)
146 * a.b (with b treated as 24 bits)
147 */
148 if (pp > res.bytes + 2 || val > 0xff)
149 goto ret_0;
150 *pp++ = val;
151 c = *++cp;
152 } else
153 break;
154 }
155 /*
156 * Check for trailing characters.
157 */
158 if (c != '\0' && (!isascii(c) || !isspace(c)))
159 goto ret_0;
160 /*
161 * Did we get a valid digit?
162 */
163 if (!digit)
164 goto ret_0;
165
166 /* Check whether the last part is in its limits depending on
167 the number of parts in total. */
168 if (val > max[pp - res.bytes])
169 goto ret_0;
170
171 if (addr != NULL)
172 addr->s_addr = res.word | htonl (val);
173
174 __set_errno (saved_errno);
175 return (1);
176
177ret_0:
178 __set_errno (saved_errno);
179 return (0);
180}
