99/*
100 * Formulate a normal query, send, and await answer.
101 * Returned answer is placed in supplied buffer "answer".
102 * Perform preliminary check of answer, returning success only
103 * if no error is indicated and the answer count is nonzero.
104 * Return the size of the response on success, -1 on error.
105 * Error number is left in H_ERRNO.
106 *
107 * Caller must parse answer and determine whether it answers the question.
108 */
109int
110__libc_res_nquery(res_state statp,
111 const char *name, /* domain name */
112 int class, int type, /* class and type of query */
113 u_char *answer, /* buffer to put answer */
114 int anslen, /* size of answer buffer */
115 u_char **answerp, /* if buffer needs to be enlarged */
116 u_char **answerp2,
117 int *nanswerp2,
118 int *resplen2,
119 int *answerp2_malloced)
120{
121 HEADER *hp = (HEADER *) answer;
122 HEADER *hp2;
123 int n, use_malloc = 0;
124 u_int oflags = statp->_flags;
125
126 size_t bufsize = (type == T_QUERY_A_AND_AAAA ? 2 : 1) * QUERYSIZE;
127 u_char *buf = alloca (bufsize);
128 u_char *query1 = buf;
129 int nquery1 = -1;
130 u_char *query2 = NULL;
131 int nquery2 = 0;
132
133 again:
134 hp->rcode = NOERROR; /* default */
135
136#ifdef DEBUG
137 if (statp->options & RES_DEBUG)
138 printf(";; res_query(%s, %d, %d)\n", name, class, type);
139#endif
140
141 if (type == T_QUERY_A_AND_AAAA)
142 {
143 n = res_nmkquery(statp, QUERY, name, class, T_A, NULL, 0, NULL,
144 query1, bufsize);
145 if (n > 0)
146 {
147 if ((oflags & RES_F_EDNS0ERR) == 0
148 && (statp->options & (RES_USE_EDNS0|RES_USE_DNSSEC)) != 0)
149 {
150 /* Use RESOLV_EDNS_BUFFER_SIZE because the receive
151 buffer can be reallocated. */
152 n = __res_nopt (statp, n, query1, bufsize,
153 RESOLV_EDNS_BUFFER_SIZE);
154 if (n < 0)
155 goto unspec_nomem;
156 }
157
158 nquery1 = n;
159 /* Align the buffer. */
160 int npad = ((nquery1 + __alignof__ (HEADER) - 1)
161 & ~(__alignof__ (HEADER) - 1)) - nquery1;
162 if (n > bufsize - npad)
163 {
164 n = -1;
165 goto unspec_nomem;
166 }
167 int nused = n + npad;
168 query2 = buf + nused;
169 n = res_nmkquery(statp, QUERY, name, class, T_AAAA, NULL, 0,
170 NULL, query2, bufsize - nused);
171 if (n > 0
172 && (oflags & RES_F_EDNS0ERR) == 0
173 && (statp->options & (RES_USE_EDNS0|RES_USE_DNSSEC)) != 0)
174 /* Use RESOLV_EDNS_BUFFER_SIZE because the receive
175 buffer can be reallocated. */
176 n = __res_nopt (statp, n, query2, bufsize,
177 RESOLV_EDNS_BUFFER_SIZE);
178 nquery2 = n;
179 }
180
181 unspec_nomem:;
182 }
183 else
184 {
185 n = res_nmkquery(statp, QUERY, name, class, type, NULL, 0, NULL,
186 query1, bufsize);
187
188 if (n > 0
189 && (oflags & RES_F_EDNS0ERR) == 0
190 && (statp->options & (RES_USE_EDNS0|RES_USE_DNSSEC)) != 0)
191 {
192 /* Use RESOLV_EDNS_BUFFER_SIZE if the receive buffer
193 can be reallocated. */
194 size_t advertise;
195 if (answerp == NULL)
196 advertise = anslen;
197 else
198 advertise = RESOLV_EDNS_BUFFER_SIZE;
199 n = __res_nopt (statp, n, query1, bufsize, advertise);
200 }
201
202 nquery1 = n;
203 }
204
205 if (__builtin_expect (n <= 0, 0) && !use_malloc) {
206 /* Retry just in case res_nmkquery failed because of too
207 short buffer. Shouldn't happen. */
208 bufsize = (type == T_QUERY_A_AND_AAAA ? 2 : 1) * MAXPACKET;
209 buf = malloc (bufsize);
210 if (buf != NULL) {
211 query1 = buf;
212 use_malloc = 1;
213 goto again;
214 }
215 }
216 if (__glibc_unlikely (n <= 0)) {
217 /* If the query choked with EDNS0, retry without EDNS0. */
218 if ((statp->options & (RES_USE_EDNS0|RES_USE_DNSSEC)) != 0
219 && ((oflags ^ statp->_flags) & RES_F_EDNS0ERR) != 0) {
220 statp->_flags |= RES_F_EDNS0ERR;
221#ifdef DEBUG
222 if (statp->options & RES_DEBUG)
223 printf(";; res_nquery: retry without EDNS0\n");
224#endif
225 goto again;
226 }
227#ifdef DEBUG
228 if (statp->options & RES_DEBUG)
229 printf(";; res_query: mkquery failed\n");
230#endif
231 RES_SET_H_ERRNO(statp, NO_RECOVERY);
232 if (use_malloc)
233 free (buf);
234 return (n);
235 }
236 assert (answerp == NULL || (void *) *answerp == (void *) answer);
237 n = __libc_res_nsend(statp, query1, nquery1, query2, nquery2, answer,
238 anslen, answerp, answerp2, nanswerp2, resplen2,
239 answerp2_malloced);
240 if (use_malloc)
241 free (buf);
242 if (n < 0) {
243#ifdef DEBUG
244 if (statp->options & RES_DEBUG)
245 printf(";; res_query: send error\n");
246#endif
247 RES_SET_H_ERRNO(statp, TRY_AGAIN);
248 return (n);
249 }
250
251 if (answerp != NULL)
252 /* __libc_res_nsend might have reallocated the buffer. */
253 hp = (HEADER *) *answerp;
254
255 /* We simplify the following tests by assigning HP to HP2 or
256 vice versa. It is easy to verify that this is the same as
257 ignoring all tests of HP or HP2. */
258 if (answerp2 == NULL || *resplen2 < (int) sizeof (HEADER))
259 {
260 hp2 = hp;
261 }
262 else
263 {
264 hp2 = (HEADER *) *answerp2;
265 if (n < (int) sizeof (HEADER))
266 {
267 hp = hp2;
268 }
269 }
270
271 /* Make sure both hp and hp2 are defined */
272 assert((hp != NULL) && (hp2 != NULL));
273
274 if ((hp->rcode != NOERROR || ntohs(hp->ancount) == 0)
275 && (hp2->rcode != NOERROR || ntohs(hp2->ancount) == 0)) {
276#ifdef DEBUG
277 if (statp->options & RES_DEBUG) {
278 printf(";; rcode = %d, ancount=%d\n", hp->rcode,
279 ntohs(hp->ancount));
280 if (hp != hp2)
281 printf(";; rcode2 = %d, ancount2=%d\n", hp2->rcode,
282 ntohs(hp2->ancount));
283 }
284#endif
285 switch (hp->rcode == NOERROR ? hp2->rcode : hp->rcode) {
286 case NXDOMAIN:
287 if ((hp->rcode == NOERROR && ntohs (hp->ancount) != 0)
288 || (hp2->rcode == NOERROR
289 && ntohs (hp2->ancount) != 0))
290 goto success;
291 RES_SET_H_ERRNO(statp, HOST_NOT_FOUND);
292 break;
293 case SERVFAIL:
294 RES_SET_H_ERRNO(statp, TRY_AGAIN);
295 break;
296 case NOERROR:
297 if (ntohs (hp->ancount) != 0
298 || ntohs (hp2->ancount) != 0)
299 goto success;
300 RES_SET_H_ERRNO(statp, NO_DATA);
301 break;
302 case FORMERR:
303 case NOTIMP:
304 /* Servers must not reply to AAAA queries with
305 NOTIMP etc but some of them do. */
306 if ((hp->rcode == NOERROR && ntohs (hp->ancount) != 0)
307 || (hp2->rcode == NOERROR
308 && ntohs (hp2->ancount) != 0))
309 goto success;
310 /* FALLTHROUGH */
311 case REFUSED:
312 default:
313 RES_SET_H_ERRNO(statp, NO_RECOVERY);
314 break;
315 }
316 return (-1);
317 }
318 success:
319 return (n);
320}

335/*
336 * Formulate a normal query, send, and retrieve answer in supplied buffer.
337 * Return the size of the response on success, -1 on error.
338 * If enabled, implement search rules until answer or unrecoverable failure
339 * is detected. Error code, if any, is left in H_ERRNO.
340 */
341int
342__libc_res_nsearch(res_state statp,
343 const char *name, /* domain name */
344 int class, int type, /* class and type of query */
345 u_char *answer, /* buffer to put answer */
346 int anslen, /* size of answer */
347 u_char **answerp,
348 u_char **answerp2,
349 int *nanswerp2,
350 int *resplen2,
351 int *answerp2_malloced)
352{
353 const char *cp, * const *domain;
354 HEADER *hp = (HEADER *) answer;
355 char tmp[NS_MAXDNAME];
356 u_int dots;
357 int trailing_dot, ret, saved_herrno;
358 int got_nodata = 0, got_servfail = 0, root_on_list = 0;
359 int tried_as_is = 0;
360 int searched = 0;
361
362 __set_errno (0);
363 RES_SET_H_ERRNO(statp, HOST_NOT_FOUND); /* True if we never query. */
364
365 dots = 0;
366 for (cp = name; *cp != '\0'; cp++)
367 dots += (*cp == '.');
368 trailing_dot = 0;
369 if (cp > name && *--cp == '.')
370 trailing_dot++;
371
372 /* If there aren't any dots, it could be a user-level alias. */
373 if (!dots && (cp = res_hostalias(statp, name, tmp, sizeof tmp))!= NULL)
374 return (__libc_res_nquery(statp, cp, class, type, answer,
375 anslen, answerp, answerp2,
376 nanswerp2, resplen2, answerp2_malloced));
377
378#ifdef DEBUG
379 if (statp->options & RES_DEBUG)
380 printf("dots=%d, statp->ndots=%d, trailing_dot=%d, name=%s\n",
381 (int)dots,(int)statp->ndots,(int)trailing_dot,name);
382#endif
383
384 /*
385 * If there are enough dots in the name, let's just give it a
386 * try 'as is'. The threshold can be set with the "ndots" option.
387 * Also, query 'as is', if there is a trailing dot in the name.
388 */
389 saved_herrno = -1;
390 if (dots >= statp->ndots || trailing_dot) {
391 ret = __libc_res_nquerydomain(statp, name, NULL, class, type,
392 answer, anslen, answerp,
393 answerp2, nanswerp2, resplen2,
394 answerp2_malloced);
395 if (ret > 0 || trailing_dot
396 /* If the second response is valid then we use that. */
397 || (ret == 0 && resplen2 != NULL && *resplen2 > 0))
398 return (ret);
399 saved_herrno = h_errno;
400 tried_as_is++;
401 if (answerp && *answerp != answer) {
402 answer = *answerp;
403 anslen = MAXPACKET;
404 }
405 if (answerp2 && *answerp2_malloced)
406 {
407 free (*answerp2);
408 *answerp2 = NULL;
409 *nanswerp2 = 0;
410 *answerp2_malloced = 0;
411 }
412 }
413
414 /*
415 * We do at least one level of search if
416 * - there is no dot and RES_DEFNAME is set, or
417 * - there is at least one dot, there is no trailing dot,
418 * and RES_DNSRCH is set.
419 */
420 if ((!dots && (statp->options & RES_DEFNAMES) != 0) ||
421 (dots && !trailing_dot && (statp->options & RES_DNSRCH) != 0)) {
422 int done = 0;
423
424 for (domain = (const char * const *)statp->dnsrch;
425 *domain && !done;
426 domain++) {
427 const char *dname = domain[0];
428 searched = 1;
429
430 /* __libc_res_nquerydoman concatenates name
431 with dname with a "." in between. If we
432 pass it in dname the "." we got from the
433 configured default search path, we'll end
434 up with "name..", which won't resolve.
435 OTOH, passing it "" will result in "name.",
436 which has the intended effect for both
437 possible representations of the root
438 domain. */
439 if (dname[0] == '.')
440 dname++;
441 if (dname[0] == '\0')
442 root_on_list++;
443
444 ret = __libc_res_nquerydomain(statp, name, dname,
445 class, type,
446 answer, anslen, answerp,
447 answerp2, nanswerp2,
448 resplen2, answerp2_malloced);
449 if (ret > 0 || (ret == 0 && resplen2 != NULL
450 && *resplen2 > 0))
451 return (ret);
452
453 if (answerp && *answerp != answer) {
454 answer = *answerp;
455 anslen = MAXPACKET;
456 }
457 if (answerp2 && *answerp2_malloced)
458 {
459 free (*answerp2);
460 *answerp2 = NULL;
461 *nanswerp2 = 0;
462 *answerp2_malloced = 0;
463 }
464
465 /*
466 * If no server present, give up.
467 * If name isn't found in this domain,
468 * keep trying higher domains in the search list
469 * (if that's enabled).
470 * On a NO_DATA error, keep trying, otherwise
471 * a wildcard entry of another type could keep us
472 * from finding this entry higher in the domain.
473 * If we get some other error (negative answer or
474 * server failure), then stop searching up,
475 * but try the input name below in case it's
476 * fully-qualified.
477 */
478 if (errno == ECONNREFUSED) {
479 RES_SET_H_ERRNO(statp, TRY_AGAIN);
480 return (-1);
481 }
482
483 switch (statp->res_h_errno) {
484 case NO_DATA:
485 got_nodata++;
486 /* FALLTHROUGH */
487 case HOST_NOT_FOUND:
488 /* keep trying */
489 break;
490 case TRY_AGAIN:
491 if (hp->rcode == SERVFAIL) {
492 /* try next search element, if any */
493 got_servfail++;
494 break;
495 }
496 /* FALLTHROUGH */
497 default:
498 /* anything else implies that we're done */
499 done++;
500 }
501
502 /* if we got here for some reason other than DNSRCH,
503 * we only wanted one iteration of the loop, so stop.
504 */
505 if ((statp->options & RES_DNSRCH) == 0)
506 done++;
507 }
508 }
509
510 /*
511 * If the query has not already been tried as is then try it
512 * unless RES_NOTLDQUERY is set and there were no dots.
513 */
514 if ((dots || !searched || (statp->options & RES_NOTLDQUERY) == 0)
515 && !(tried_as_is || root_on_list)) {
516 ret = __libc_res_nquerydomain(statp, name, NULL, class, type,
517 answer, anslen, answerp,
518 answerp2, nanswerp2, resplen2,
519 answerp2_malloced);
520 if (ret > 0 || (ret == 0 && resplen2 != NULL
521 && *resplen2 > 0))
522 return (ret);
523 }
524
525 /* if we got here, we didn't satisfy the search.
526 * if we did an initial full query, return that query's H_ERRNO
527 * (note that we wouldn't be here if that query had succeeded).
528 * else if we ever got a nodata, send that back as the reason.
529 * else send back meaningless H_ERRNO, that being the one from
530 * the last DNSRCH we did.
531 */
532 if (answerp2 && *answerp2_malloced)
533 {
534 free (*answerp2);
535 *answerp2 = NULL;
536 *nanswerp2 = 0;
537 *answerp2_malloced = 0;
538 }
539 if (saved_herrno != -1)
540 RES_SET_H_ERRNO(statp, saved_herrno);
541 else if (got_nodata)
542 RES_SET_H_ERRNO(statp, NO_DATA);
543 else if (got_servfail)
544 RES_SET_H_ERRNO(statp, TRY_AGAIN);
545 return (-1);
546}

561/*
562 * Perform a call on res_query on the concatenation of name and domain.
563 */
564static int
565__libc_res_nquerydomain(res_state statp,
566 const char *name,
567 const char *domain,
568 int class, int type, /* class and type of query */
569 u_char *answer, /* buffer to put answer */
570 int anslen, /* size of answer */
571 u_char **answerp,
572 u_char **answerp2,
573 int *nanswerp2,
574 int *resplen2,
575 int *answerp2_malloced)
576{
577 char nbuf[MAXDNAME];
578 const char *longname = nbuf;
579 size_t n, d;
580
581#ifdef DEBUG
582 if (statp->options & RES_DEBUG)
583 printf(";; res_nquerydomain(%s, %s, %d, %d)\n",
584 name, domain?domain:"<Nil>", class, type);
585#endif
586 if (domain == NULL) {
587 n = strlen(name);
588
589 /* Decrement N prior to checking it against MAXDNAME
590 so that we detect a wrap to SIZE_MAX and return
591 a reasonable error. */
592 n--;
593 if (n >= MAXDNAME - 1) {
594 RES_SET_H_ERRNO(statp, NO_RECOVERY);
595 return (-1);
596 }
597 longname = name;
598 } else {
599 n = strlen(name);
600 d = strlen(domain);
601 if (n + d + 1 >= MAXDNAME) {
602 RES_SET_H_ERRNO(statp, NO_RECOVERY);
603 return (-1);
604 }
605 sprintf(nbuf, "%s.%s", name, domain);
606 }
607 return (__libc_res_nquery(statp, longname, class, type, answer,
608 anslen, answerp, answerp2, nanswerp2,
609 resplen2, answerp2_malloced));
610}

626const char *
627res_hostalias(const res_state statp, const char *name, char *dst, size_t siz) {
628 char *file, *cp1, *cp2;
629 char buf[BUFSIZ];
630 FILE *fp;
631
632 if (statp->options & RES_NOALIASES)
633 return (NULL);
634 file = getenv("HOSTALIASES");
635 if (file == NULL || (fp = fopen(file, "rce")) == NULL)
636 return (NULL);
637 setbuf(fp, NULL);
638 buf[sizeof(buf) - 1] = '\0';
639 while (fgets(buf, sizeof(buf), fp)) {
640 for (cp1 = buf; *cp1 && !isspace(*cp1); ++cp1)
641 ;
642 if (!*cp1)
643 break;
644 *cp1 = '\0';
645 if (ns_samename(buf, name) == 1) {
646 while (isspace(*++cp1))
647 ;
648 if (!*cp1)
649 break;
650 for (cp2 = cp1 + 1; *cp2 && !isspace(*cp2); ++cp2)
651 ;
652 *cp2 = '\0';
653 strncpy(dst, cp1, siz - 1);
654 dst[siz - 1] = '\0';
655 fclose(fp);
656 return (dst);
657 }
658 }
659 fclose(fp);
660 return (NULL);
661}
