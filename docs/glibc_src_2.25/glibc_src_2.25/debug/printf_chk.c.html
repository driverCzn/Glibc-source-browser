23/* Write formatted output to stdout from the format string FORMAT. */
24int
25___printf_chk (int flag, const char *format, ...)
26{
27 va_list ap;
28 int done;
29
30 _IO_acquire_lock_clear_flags2 (stdout);
31 if (flag > 0)
32 stdout->_flags2 |= _IO_FLAGS2_FORTIFY;
33
34 va_start (ap, format);
35 done = vfprintf (stdout, format, ap);
36 va_end (ap);
37
38 if (flag > 0)
39 stdout->_flags2 &= ~_IO_FLAGS2_FORTIFY;
40 _IO_release_lock (stdout);
41
42 return done;
43}
44ldbl_strong_alias (___printf_chk, __printf_chk)
