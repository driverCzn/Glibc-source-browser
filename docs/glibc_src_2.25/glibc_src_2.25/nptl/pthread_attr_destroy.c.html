19#include <assert.h>
20#include <errno.h>
21#include <string.h>
22#include <unistd.h>
23#include "pthreadP.h"
24#include <shlib-compat.h>
25
26int
27__pthread_attr_destroy (pthread_attr_t *attr)
28{
29 struct pthread_attr *iattr;
30
31 assert (sizeof (*attr) >= sizeof (struct pthread_attr));
32 iattr = (struct pthread_attr *) attr;
33
34#if SHLIB_COMPAT(libpthread, GLIBC_2_0, GLIBC_2_1)
35 /* In old struct pthread_attr, neither next nor cpuset are
36 present. */
37 if (__builtin_expect ((iattr->flags & ATTR_FLAG_OLDATTR), 0) == 0)
38#endif
39 /* The affinity CPU set might be allocated dynamically. */
40 free (iattr->cpuset);
41
42 return 0;
43}
44strong_alias (__pthread_attr_destroy, pthread_attr_destroy)
