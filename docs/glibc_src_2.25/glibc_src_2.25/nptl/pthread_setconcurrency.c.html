23/* Global definition. Needed in pthread_getconcurrency as well. */
24int __concurrency_level;

27int
28pthread_setconcurrency (int level)
29{
30 if (level < 0)
31 return EINVAL;
32
33 __concurrency_level = level;
34
35 /* XXX For ports which actually need to handle the concurrency level
36 some more code is probably needed here. */
37
38 return 0;
39}
