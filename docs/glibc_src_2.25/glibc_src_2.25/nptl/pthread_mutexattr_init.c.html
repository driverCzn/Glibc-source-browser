19#include <string.h>
20#include <pthreadP.h>
21
22
23int
24__pthread_mutexattr_init (pthread_mutexattr_t *attr)
25{
26 if (sizeof (struct pthread_mutexattr) != sizeof (pthread_mutexattr_t))
27 memset (attr, '\0', sizeof (*attr));
28
29 /* We use bit 31 to signal whether the mutex is going to be
30 process-shared or not. By default it is zero, i.e., the mutex is
31 not process-shared. */
32 ((struct pthread_mutexattr *) attr)->mutexkind = PTHREAD_MUTEX_NORMAL;
33
34 return 0;
35}
36strong_alias (__pthread_mutexattr_init, pthread_mutexattr_init)
