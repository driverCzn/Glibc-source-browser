19#include <sys/ipc.h>
20#include <sys/stat.h>
21
22key_t
23ftok (const char *pathname, int proj_id)
24{
25 struct stat64 st;
26 key_t key;
27
28 if (__xstat64 (_STAT_VER, pathname, &st) < 0)
29 return (key_t) -1;
30
31 key = ((st.st_ino & 0xffff) | ((st.st_dev & 0xff) << 16)
32 | ((proj_id & 0xff) << 24));
33
34 return key;
35}
