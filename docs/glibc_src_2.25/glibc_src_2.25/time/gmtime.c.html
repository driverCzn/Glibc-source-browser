21/* Return the `struct tm' representation of *T in UTC,
22 using *TP to store the result. */
23struct tm *
24__gmtime_r (const time_t *t, struct tm *tp)
25{
26 return __tz_convert (t, 0, tp);
27}
28libc_hidden_def (__gmtime_r)
29weak_alias (__gmtime_r, gmtime_r)
30
31
32/* Return the `struct tm' representation of *T in UTC. */
33struct tm *
34gmtime (const time_t *t)
35{
36 return __tz_convert (t, 0, &_tmbuf);
37}
