21void
22__scandir_cancel_handler (void *arg)
23{
24 struct scandir_cancel_struct *cp = arg;
25 void **v = cp->v;
26
27 for (size_t i = 0; i < cp->cnt; ++i)
28 free (v[i]);
29 free (v);
30 (void) __closedir (cp->dp);
31}
