25/* Convert VALUE into ASCII in base BASE (2..36).
26 Write backwards starting the character just before BUFLIM.
27 Return the address of the first (left-to-right) character in the number.
28 Use upper case letters iff UPPER_CASE is nonzero. */
29
30extern wchar_t *_itowa (unsigned long long int value, wchar_t *buflim,
31 unsigned int base, int upper_case);
32
33static inline wchar_t *
34__attribute__ ((unused, always_inline))
35_itowa_word (_ITOA_WORD_TYPE value, wchar_t *buflim,
36 unsigned int base, int upper_case)
37{
38 extern const wchar_t _itowa_upper_digits[] attribute_hidden;
39 extern const wchar_t _itowa_lower_digits[] attribute_hidden;
40 const wchar_t *digits = (upper_case
41 ? _itowa_upper_digits : _itowa_lower_digits);
42 wchar_t *bp = buflim;
43
44 switch (base)
45 {
46#define SPECIAL(Base) \
47 case Base: \
48 do \
49 *--bp = digits[value % Base]; \
50 while ((value /= Base) != 0); \
51 break
52
53 SPECIAL (10);
54 SPECIAL (16);
55 SPECIAL (8);
56 default:
57 do
58 *--bp = digits[value % base];
59 while ((value /= base) != 0);
60 }
61 return bp;
62}
63#undef SPECIAL
64
65#if !_ITOA_NEEDED
66/* No need for special long long versions. */
67# define _itowa(value, buf, base, upper_case) \
68 _itowa_word (value, buf, base, upper_case)
69#endif
70
71#endif /* itowa.h */
