22/* If O_LARGEFILE is zero, mkostemp.c defines mkostemp64 as an alias. */
23#if defined O_LARGEFILE && O_LARGEFILE != 0
24
25/* Generate a unique temporary file name from TEMPLATE.
26 The last six characters of TEMPLATE must be "XXXXXX";
27 they are replaced with a string that makes the filename unique.
28 Then open the file and return a fd. */
29int
30mkostemp64 (char *template, int flags)
31{
32 return __gen_tempname (template, 0, flags | O_LARGEFILE, __GT_FILE);
33}
34
35#endif
