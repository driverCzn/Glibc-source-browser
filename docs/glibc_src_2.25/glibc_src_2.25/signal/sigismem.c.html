18#include "sigsetops.h"
19
20/* Return 1 if SIGNO is in SET, 0 if not. */
21int
22sigismember (const sigset_t *set, int signo)
23{
24 if (set == NULL || signo <= 0 || signo >= NSIG)
25 {
26 __set_errno (EINVAL);
27 return -1;
28 }
29
30 return __sigismember (set, signo);
31}
