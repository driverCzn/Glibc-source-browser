18#include <errno.h>
19#include <signal.h>
20#define __need_NULL
21#include <stddef.h>
22
23/* Combine sets LEFT and RIGHT by logical AND and place result in DEST. */
24int
25sigandset (sigset_t *dest, const sigset_t *left, const sigset_t *right)
26{
27 if (dest == NULL || left == NULL || right == NULL)
28 {
29 __set_errno (EINVAL);
30 return -1;
31 }
32
33 return __sigandset (dest, left, right);
34}
