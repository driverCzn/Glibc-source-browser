21/* Define the lookup function. */
22#include "wchar-lookup.h"
23
24wint_t
25__towctrans (wint_t wc, wctrans_t desc)
26{
27 /* If the user passes in an invalid DESC valid (the one returned from
28 `wctrans' in case of an error) simply return the value. */
29 if (desc == (wctrans_t) 0)
30 return wc;
31
32 return wctrans_table_lookup ((const char *) desc, wc);
33}
