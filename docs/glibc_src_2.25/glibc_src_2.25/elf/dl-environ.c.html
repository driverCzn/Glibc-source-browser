24/* Walk through the environment of the process and return all entries
25 starting with `LD_'. */
26char *
27internal_function
28_dl_next_ld_env_entry (char ***position)
29{
30 char **current = *position;
31 char *result = NULL;
32
33 while (*current != NULL)
34 {
35 if (__builtin_expect ((*current)[0] == 'L', 0)
36 && (*current)[1] == 'D' && (*current)[2] == '_')
37 {
38 result = &(*current)[3];
39
40 /* Save current position for next visit. */
41 *position = ++current;
42
43 break;
44 }
45
46 ++current;
47 }
48
49 return result;
50}

53/* In ld.so __environ is not exported. */
54extern char **__environ attribute_hidden;
55
56int
57unsetenv (const char *name)
58{
59 char **ep;
60
61 ep = __environ;
62 while (*ep != NULL)
63 {
64 size_t cnt = 0;
65
66 while ((*ep)[cnt] == name[cnt] && name[cnt] != '\0')
67 ++cnt;
68
69 if (name[cnt] == '\0' && (*ep)[cnt] == '=')
70 {
71 /* Found it. Remove this pointer by moving later ones to
72 the front. */
73 char **dp = ep;
74
75 do
76 dp[0] = dp[1];
77 while (*dp++);
78 /* Continue the loop in case NAME appears again. */
79 }
80 else
81 ++ep;
82 }
83
84 return 0;
85}
