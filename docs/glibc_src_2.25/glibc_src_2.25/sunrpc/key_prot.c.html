89bool_t
90xdr_cryptkeyres (XDR * xdrs, cryptkeyres * objp)
91{
92 if (!xdr_keystatus (xdrs, &objp->status))
93 return FALSE;
94 switch (objp->status)
95 {
96 case KEY_SUCCESS:
97 if (!xdr_des_block (xdrs, &objp->cryptkeyres_u.deskey))
98 return FALSE;
99 break;
100 default:
101 break;
102 }
103 return TRUE;
104}

107bool_t
108xdr_unixcred (XDR * xdrs, unixcred * objp)
109{
110 if (!xdr_u_int (xdrs, &objp->uid))
111 return FALSE;
112 if (!xdr_u_int (xdrs, &objp->gid))
113 return FALSE;
114 if (!xdr_array (xdrs, (void *) &objp->gids.gids_val,
115 (u_int *) & objp->gids.gids_len, MAXGIDS,
116 sizeof (u_int), (xdrproc_t) xdr_u_int))
117 return FALSE;
118 return TRUE;
119}

122bool_t
123xdr_getcredres (XDR * xdrs, getcredres * objp)
124{
125 if (!xdr_keystatus (xdrs, &objp->status))
126 return FALSE;
127 switch (objp->status)
128 {
129 case KEY_SUCCESS:
130 if (!xdr_unixcred (xdrs, &objp->getcredres_u.cred))
131 return FALSE;
132 break;
133 default:
134 break;
135 }
136 return TRUE;
137}

140bool_t
141xdr_key_netstarg (XDR * xdrs, key_netstarg * objp)
142{
143 if (!xdr_keybuf (xdrs, objp->st_priv_key))
144 return FALSE;
145 if (!xdr_keybuf (xdrs, objp->st_pub_key))

149 return TRUE;
150}

153bool_t
154xdr_key_netstres (XDR * xdrs, key_netstres * objp)
155{
156 if (!xdr_keystatus (xdrs, &objp->status))
157 return FALSE;
158 switch (objp->status)
159 {
160 case KEY_SUCCESS:
161 if (!xdr_key_netstarg (xdrs, &objp->key_netstres_u.knet))
162 return FALSE;
163 break;
164 default:
165 break;
166 }
167 return TRUE;
168}
