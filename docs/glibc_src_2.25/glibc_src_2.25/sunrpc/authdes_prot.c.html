31/*
32 * authdes_prot.c, XDR routines for DES authentication
33 */
34
35#include <rpc/types.h>
36#include <rpc/xdr.h>
37#include <rpc/auth.h>
38#include <rpc/auth_des.h>
39
40#define ATTEMPT(xdr_op) if (!(xdr_op)) return (FALSE)
41
42bool_t
43xdr_authdes_cred (XDR *xdrs, struct authdes_cred *cred)
44{
45 /*
46 * Unrolled xdr
47 */
48 ATTEMPT (xdr_enum (xdrs, (enum_t *) & cred->adc_namekind));
49 switch (cred->adc_namekind)
50 {
51 case ADN_FULLNAME:
52 ATTEMPT (xdr_string (xdrs, &cred->adc_fullname.name, MAXNETNAMELEN));
53 ATTEMPT (xdr_opaque (xdrs, (caddr_t) & cred->adc_fullname.key,
54 sizeof (des_block)));
55 ATTEMPT (xdr_opaque (xdrs, (caddr_t) & cred->adc_fullname.window,
56 sizeof (cred->adc_fullname.window)));
57 return (TRUE);
58 case ADN_NICKNAME:
59 ATTEMPT (xdr_opaque (xdrs, (caddr_t) & cred->adc_nickname,
60 sizeof (cred->adc_nickname)));
61 return TRUE;
62 default:
63 return FALSE;
64 }
65}
66libc_hidden_nolink_sunrpc (xdr_authdes_cred, GLIBC_2_1)
67

69bool_t
70xdr_authdes_verf (register XDR *xdrs, register struct authdes_verf *verf)
71{
72 /*
73 * Unrolled xdr
74 */
75 ATTEMPT (xdr_opaque (xdrs, (caddr_t) & verf->adv_xtimestamp,
76 sizeof (des_block)));
77 ATTEMPT (xdr_opaque (xdrs, (caddr_t) & verf->adv_int_u,
78 sizeof (verf->adv_int_u)));
79 return TRUE;
80}
