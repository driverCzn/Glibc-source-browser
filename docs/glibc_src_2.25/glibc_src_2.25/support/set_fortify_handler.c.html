21#include <signal.h>

23void
24set_fortify_handler (void (*handler) (int sig))
25{
26 struct sigaction sa;
27
28 sa.sa_handler = handler;
29 sa.sa_flags = 0;
30 sigemptyset (&sa.sa_mask);
31
32 sigaction (SIGABRT, &sa, NULL);
33 ignore_stderr ();
34}
