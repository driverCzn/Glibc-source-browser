23char *
24support_format_address_family (int family)
25{
26 switch (family)
27 {
28 case AF_INET:
29 return xstrdup ("INET");
30 case AF_INET6:
31 return xstrdup ("INET6");
32 default:
33 return xasprintf ("<unknown address family %d>", family);
34 }
35}
