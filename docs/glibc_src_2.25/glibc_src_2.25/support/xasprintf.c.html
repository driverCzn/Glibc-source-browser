26char *
27xasprintf (const char *format, ...)
28{
29 va_list ap;
30 va_start (ap, format);
31 char *result;
32 if (vasprintf (&result, format, ap) < 0)
33 FAIL_EXIT1 ("asprintf: %m");
34 va_end (ap);
35 return result;
36}
