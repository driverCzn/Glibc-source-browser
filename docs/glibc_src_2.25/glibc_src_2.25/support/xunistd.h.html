19/* These wrapper functions use POSIX types and therefore cannot be
20 declared in <support/support.h>. */
21
22#ifndef SUPPORT_XUNISTD_H
23#define SUPPORT_XUNISTD_H
24
25#include <sys/cdefs.h>
26#include <sys/types.h>
27#include <unistd.h>
28
29__BEGIN_DECLS
30
31struct stat64;
32
33pid_t xfork (void);
34pid_t xwaitpid (pid_t, int *status, int flags);
35void xpipe (int[2]);
36void xdup2 (int, int);
37int xopen (const char *path, int flags, mode_t);
38void xstat (const char *path, struct stat64 *);
39void xfstat (int fd, struct stat64 *);
40void xmkdir (const char *path, mode_t);
41void xchroot (const char *path);
42void xunlink (const char *path);
43long xsysconf (int name);
44long long xlseek (int fd, long long offset, int whence);
45void xftruncate (int fd, long long length);
46
47/* Read the link at PATH. The caller should free the returned string
48 with free. */
49char *xreadlink (const char *path);
50
51/* Close the file descriptor. Ignore EINTR errors, but terminate the
52 process on other errors. */
53void xclose (int);
54
55/* Write the buffer. Retry on short writes. */
56void xwrite (int, const void *, size_t);
57
58/* Invoke mmap with a zero file offset. */
59void *xmmap (void *addr, size_t length, int prot, int flags, int fd);
60void xmprotect (void *addr, size_t length, int prot);
61void xmunmap (void *addr, size_t length);
62
63__END_DECLS
64
65#endif /* SUPPORT_XUNISTD_H */
