19#include <fcntl.h>
20#include <string.h>
21#include <support/check.h>
22#include <support/xunistd.h>
23
24void
25support_write_file_string (const char *path, const char *contents)
26{
27 int fd = xopen (path, O_CREAT | O_TRUNC | O_WRONLY, 0666);
28 const char *end = contents + strlen (contents);
29 for (const char *p = contents; p < end; )
30 {
31 ssize_t ret = write (fd, p, end - p);
32 if (ret < 0)
33 FAIL_EXIT1 ("cannot write to \"%s\": %m", path);
34 if (ret == 0)
35 FAIL_EXIT1 ("zero-length write to \"%s\"", path);
36 p += ret;
37 }
38 xclose (fd);
39}
