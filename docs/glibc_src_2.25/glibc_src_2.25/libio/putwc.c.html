18#include "libioP.h"
19#include <wchar.h>
20
21wint_t
22putwc (wchar_t wc, _IO_FILE *fp)
23{
24 wint_t result;
25 CHECK_FILE (fp, WEOF);
26 _IO_acquire_lock (fp);
27 result = _IO_putwc_unlocked (wc, fp);
28 _IO_release_lock (fp);
29 return result;
30}
