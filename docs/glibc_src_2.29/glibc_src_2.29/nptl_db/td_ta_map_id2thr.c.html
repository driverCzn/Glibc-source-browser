23td_err_e
24td_ta_map_id2thr (const td_thragent_t *ta, pthread_t pt, td_thrhandle_t *th)
25{
26 LOG ("td_ta_map_id2thr");
27
28 /* Test whether the TA parameter is ok. */
29 if (! ta_ok (ta))
30 return TD_BADTA;
31
32 /* Create the `td_thrhandle_t' object. */
33 th->th_ta_p = (td_thragent_t *) ta;
34 th->th_unique = (psaddr_t) pt;
35
36 return TD_OK;
37}
