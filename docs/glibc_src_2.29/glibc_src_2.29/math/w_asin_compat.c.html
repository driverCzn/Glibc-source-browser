26#if LIBM_SVID_COMPAT
27/* wrapper asin */
28double
29__asin (double x)
30{
31 if (__builtin_expect (isgreater (fabs (x), 1.0), 0)
32 && _LIB_VERSION != _IEEE_)
33 {
34 /* asin(|x|>1) */
35 __feraiseexcept (FE_INVALID);
36 return __kernel_standard (x, x, 2);
37 }
38
39 return __ieee754_asin (x);
40}
41libm_alias_double (__asin, asin)
