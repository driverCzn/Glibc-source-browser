23static float
24__attribute__ ((noinline))
25invalid_fn (float x, float fn)
26{
27 if (rintf (fn) != fn)
28 return (fn - fn) / (fn - fn);
29 else if (fn > 65000.0f)
30 return __scalbnf (x, 65000);
31 else
32 return __scalbnf (x,-65000);
33}

35
36float
37__ieee754_scalbf (float x, float fn)
38{
39 if (__glibc_unlikely (isnan (x)))
40 return x * fn;
41 if (__glibc_unlikely (!isfinite (fn)))
42 {
43 if (isnan (fn) || fn > 0.0f)
44 return x * fn;
45 if (x == 0.0f)
46 return x;
47 return x / -fn;
48 }
49 if (__glibc_unlikely (fabsf (fn) >= 0x1p31f || (float) (int) fn != fn))
50 return invalid_fn (x, fn);
51
52 return __scalbnf (x, (int) fn);
53}
