25nis_error
26nis_rmdir (const_nis_name dir, const nis_server *server)
27{
28 nis_error res, res2;
29
30 if (server == NULL)
31 return NIS_SYSTEMERROR;
32
33 res2 = __do_niscall2 (server, 1, NIS_RMDIR,
34 (xdrproc_t) _xdr_nis_name,
35 (caddr_t) &dir, (xdrproc_t) _xdr_nis_error,
36 (caddr_t) &res, 0, NULL);
37 if (res2 != NIS_SUCCESS)
38 return res2;
39
40 return res;
41}
