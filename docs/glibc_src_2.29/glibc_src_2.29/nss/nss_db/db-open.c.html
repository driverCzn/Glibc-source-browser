30/* Open the database stored in FILE. If succesful, store either a
31 pointer to the mapped file or a file handle for the file in H and
32 return NSS_STATUS_SUCCESS. On failure, return the appropriate
33 lookup status. */
34enum nss_status
35internal_setent (const char *file, struct nss_db_map *mapping)
36{
37 enum nss_status status = NSS_STATUS_UNAVAIL;
38
39 int fd = __open_nocancel (file, O_RDONLY | O_LARGEFILE | O_CLOEXEC);
40 if (fd != -1)
41 {
42 struct nss_db_header header;
43
44 if (read (fd, &header, sizeof (header)) == sizeof (header))
45 {
46 mapping->header = mmap (NULL, header.allocate, PROT_READ,
47 MAP_PRIVATE, fd, 0);
48 mapping->len = header.allocate;
49 if (mapping->header != MAP_FAILED)
50 status = NSS_STATUS_SUCCESS;
51 else if (errno == ENOMEM)
52 status = NSS_STATUS_TRYAGAIN;
53 }
54
55 __close_nocancel_nostatus (fd);
56 }
57
58 return status;
59}
60
61
62/* Close the database. */
63void
64internal_endent (struct nss_db_map *mapping)
65{
66 if (mapping->header != NULL)
67 {
68 munmap (mapping->header, mapping->len);
69 mapping->header = NULL;
70 }
71}
