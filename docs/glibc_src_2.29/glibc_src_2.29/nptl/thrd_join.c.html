21int
22thrd_join (thrd_t thr, int *res)
23{
24 void *pthread_res;
25 int err_code = __pthread_timedjoin_ex (thr, &pthread_res, NULL, true);
26 if (res)
27 *res = (int) (uintptr_t) pthread_res;
28
29 return thrd_err_map (err_code);
30}
