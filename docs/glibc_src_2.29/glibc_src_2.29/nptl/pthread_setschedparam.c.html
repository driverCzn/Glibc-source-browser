19#include <errno.h>
20#include <sched.h>
21#include <string.h>
22#include "pthreadP.h"
23#include <lowlevellock.h>

25
26int
27__pthread_setschedparam (pthread_t threadid, int policy,
28 const struct sched_param *param)
29{
30 struct pthread *pd = (struct pthread *) threadid;
31
32 /* Make sure the descriptor is valid. */
33 if (INVALID_TD_P (pd))
34 /* Not a valid thread handle. */
35 return ESRCH;
36
37 int result = 0;
38
39 /* See CREATE THREAD NOTES in nptl/pthread_create.c. */
40 lll_lock (pd->lock, LLL_PRIVATE);
41
42 struct sched_param p;
43 const struct sched_param *orig_param = param;
44
45 /* If the thread should have higher priority because of some
46 PTHREAD_PRIO_PROTECT mutexes it holds, adjust the priority. */
47 if (__builtin_expect (pd->tpp != NULL, 0)
48 && pd->tpp->priomax > param->sched_priority)
49 {
50 p = *param;
51 p.sched_priority = pd->tpp->priomax;
52 param = &p;
53 }
54
55 /* Try to set the scheduler information. */
56 if (__builtin_expect (__sched_setscheduler (pd->tid, policy,
57 param) == -1, 0))
58 result = errno;
59 else
60 {
61 /* We succeeded changing the kernel information. Reflect this
62 change in the thread descriptor. */
63 pd->schedpolicy = policy;
64 memcpy (&pd->schedparam, orig_param, sizeof (struct sched_param));
65 pd->flags |= ATTR_FLAG_SCHED_SET | ATTR_FLAG_POLICY_SET;
66 }
67
68 lll_unlock (pd->lock, LLL_PRIVATE);
69
70 return result;
71}
72strong_alias (__pthread_setschedparam, pthread_setschedparam)
