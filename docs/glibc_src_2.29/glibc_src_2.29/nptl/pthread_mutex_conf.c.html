19#if HAVE_TUNABLES
20# define TUNABLE_NAMESPACE pthread
21#include <pthread_mutex_conf.h>
22#include <stdint.h>
23#include <stdbool.h>
24#include <unistd.h> /* Get STDOUT_FILENO for _dl_printf. */
25#include <elf/dl-tunables.h>
26
27struct mutex_config __mutex_aconf =
28{
29 /* The maximum number of times a thread should spin on the lock before
30 calling into kernel to block. */
31 .spin_count = DEFAULT_ADAPTIVE_COUNT,
32};
33
34static void
35TUNABLE_CALLBACK (set_mutex_spin_count) (tunable_val_t *valp)
36{
37 __mutex_aconf.spin_count = (int32_t) (valp)->numval;
38}
39
40void
41__pthread_tunables_init (void)
42{
43 TUNABLE_GET (mutex_spin_count, int32_t,
44 TUNABLE_CALLBACK (set_mutex_spin_count));
45}
46#endif
