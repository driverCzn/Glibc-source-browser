21#ifdef WCPNCPY
22# define __wcpncpy WCPNCPY
23#endif

25/* Copy no more than N wide-characters of SRC to DEST, returning the
26 address of the last character written into DEST. */
27wchar_t *
28__wcpncpy (wchar_t *dest, const wchar_t *src, size_t n)
29{
30 size_t size = __wcsnlen (src, n);
31 __wmemcpy (dest, src, size);
32 dest += size;
33 if (size == n)
34 return dest;
35 return wmemset (dest, L'\0', (n - size));
36}
37
38#ifndef WCPNCPY
39weak_alias (__wcpncpy, wcpncpy)
40#endif
