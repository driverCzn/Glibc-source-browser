67/* Magic number and bits for the _flags field. The magic number is
68 mostly vestigial, but preserved for compatibility. It occupies the
69 high 16 bits of _flags; the low 16 bits are actual flag bits. */
70
71#define _IO_MAGIC 0xFBAD0000 /* Magic number */
72#define _IO_MAGIC_MASK 0xFFFF0000
73#define _IO_USER_BUF 0x0001 /* Don't deallocate buffer on close. */
74#define _IO_UNBUFFERED 0x0002
75#define _IO_NO_READS 0x0004 /* Reading not allowed. */
76#define _IO_NO_WRITES 0x0008 /* Writing not allowed. */
77#define _IO_EOF_SEEN 0x0010
78#define _IO_ERR_SEEN 0x0020
79#define _IO_DELETE_DONT_CLOSE 0x0040 /* Don't call close(_fileno) on close. */
80#define _IO_LINKED 0x0080 /* In the list of all open files. */
81#define _IO_IN_BACKUP 0x0100
82#define _IO_LINE_BUF 0x0200
83#define _IO_TIED_PUT_GET 0x0400 /* Put and get pointer move in unison. */
84#define _IO_CURRENTLY_PUTTING 0x0800
85#define _IO_IS_APPENDING 0x1000
86#define _IO_IS_FILEBUF 0x2000
87 /* 0x4000 No longer used, reserved for compat. */
88#define _IO_USER_LOCK 0x8000
89

157/* Extra data for wide character streams. */
158struct _IO_wide_data
159{
160 wchar_t *_IO_read_ptr; /* Current read pointer */
161 wchar_t *_IO_read_end; /* End of get area. */
162 wchar_t *_IO_read_base; /* Start of putback+get area. */
163 wchar_t *_IO_write_base; /* Start of put area. */
164 wchar_t *_IO_write_ptr; /* Current put pointer. */
165 wchar_t *_IO_write_end; /* End of put area. */
166 wchar_t *_IO_buf_base; /* Start of reserve area. */
167 wchar_t *_IO_buf_end; /* End of reserve area. */
168 /* The following fields are used to support backing up and undo. */
169 wchar_t *_IO_save_base; /* Pointer to start of non-current get area. */
170 wchar_t *_IO_backup_base; /* Pointer to first valid character of
171 backup area */
172 wchar_t *_IO_save_end; /* Pointer to end of non-current get area. */
173
174 __mbstate_t _IO_state;
175 __mbstate_t _IO_last_state;
176 struct _IO_codecvt _codecvt;
177
178 wchar_t _shortbuf[1];
179
180 const struct _IO_jump_t *_wide_vtable;
181};

203#define _IO_getc_unlocked(_fp) __getc_unlocked_body (_fp)
204#define _IO_peekc_unlocked(_fp) \
205 (__glibc_unlikely ((_fp)->_IO_read_ptr >= (_fp)->_IO_read_end) \
206 && __underflow (_fp) == EOF \
207 ? EOF \
208 : *(unsigned char *) (_fp)->_IO_read_ptr)
209#define _IO_putc_unlocked(_ch, _fp) __putc_unlocked_body (_ch, _fp)
210
211# define _IO_getwc_unlocked(_fp) \
212 (__glibc_unlikely ((_fp)->_wide_data == NULL \
213 || ((_fp)->_wide_data->_IO_read_ptr \
214 >= (_fp)->_wide_data->_IO_read_end)) \
215 ? __wuflow (_fp) : (wint_t) *(_fp)->_wide_data->_IO_read_ptr++)
216# define _IO_putwc_unlocked(_wch, _fp) \
217 (__glibc_unlikely ((_fp)->_wide_data == NULL \
218 || ((_fp)->_wide_data->_IO_write_ptr \
219 >= (_fp)->_wide_data->_IO_write_end)) \
220 ? __woverflow (_fp, _wch) \
221 : (wint_t) (*(_fp)->_wide_data->_IO_write_ptr++ = (_wch)))

279/* A special optimized version of the function above. It optimizes the
280 case of initializing an unoriented byte stream. */
281#define _IO_fwide(__fp, __mode) \
282 ({ int __result = (__mode); \
283 if (__result < 0 && ! _IO_fwide_maybe_incompatible) \
284 { \
285 if ((__fp)->_mode == 0) \
286 /* We know that all we have to do is to set the flag. */ \
287 (__fp)->_mode = -1; \
288 __result = (__fp)->_mode; \
289 } \
290 else if (__builtin_constant_p (__mode) && (__mode) == 0) \
291 __result = _IO_fwide_maybe_incompatible ? -1 : (__fp)->_mode; \
292 else \
293 __result = _IO_fwide (__fp, __result); \
294 __result; })
