18#include <assert.h>
19#include <stdlib.h>
20#include "exit.h"
21#include <sysdep.h>
22
23/* Register a function to be called by exit. */
24int
25__on_exit (void (*func) (int status, void *arg), void *arg)
26{
27 struct exit_function *new;
28
29 /* As a QoI issue we detect NULL early with an assertion instead
30 of a SIGSEGV at program exit when the handler is run (bug 20544). */
31 assert (func != NULL);
32
33 __libc_lock_lock (__exit_funcs_lock);
34 new = __new_exitfn (&__exit_funcs);
35
36 if (new == NULL)
37 {
38 __libc_lock_unlock (__exit_funcs_lock);
39 return -1;
40 }
41
42#ifdef PTR_MANGLE
43 PTR_MANGLE (func);
44#endif
45 new->func.on.fn = func;
46 new->func.on.arg = arg;
47 new->flavor = ef_on;
48 __libc_lock_unlock (__exit_funcs_lock);
49 return 0;
50}
51weak_alias (__on_exit, on_exit)
