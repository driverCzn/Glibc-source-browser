18#include <string.h>
19#include <stdlib.h>
20#include <time.h>
21#include <unistd.h>
22
23char *
24strfry (char *string)
25{
26 static int init;
27 static struct random_data rdata;
28
29 if (!init)
30 {
31 static char state[32];
32 rdata.state = NULL;
33 __initstate_r (time ((time_t *) NULL) ^ getpid (),
34 state, sizeof (state), &rdata);
35 init = 1;
36 }
37
38 size_t len = strlen (string);
39 if (len > 0)
40 for (size_t i = 0; i < len - 1; ++i)
41 {
42 int32_t j;
43 __random_r (&rdata, &j);
44 j = j % (len - i) + i;
45
46 char c = string[i];
47 string[i] = string[j];
48 string[j] = c;
49 }
50
51 return string;
52}
