18#include <stdarg.h>
19#undef __OPTIMIZE__ /* Avoid inline `vprintf' function. */
20#include <stdio.h>
21#include <libioP.h>
22
23#undef vprintf

25/* Write formatted output to stdout according to the
26 format string FORMAT, using the argument list in ARG. */
27int
28__vprintf (const char *format, va_list ap)
29{
30 return __vfprintf_internal (stdout, format, ap, 0);
31}
