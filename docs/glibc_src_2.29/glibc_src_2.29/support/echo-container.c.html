21int
22main (int argc, const char **argv)
23{
24 int i;
25
26 for (i = 1; i < argc; i++)
27 {
28 if (i > 1)
29 putchar (' ');
30 fputs (argv[i], stdout);
31 }
32 putchar ('\n');
