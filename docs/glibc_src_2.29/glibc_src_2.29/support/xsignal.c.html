22sighandler_t
23xsignal (int sig, sighandler_t handler)
24{
25 sighandler_t result = signal (sig, handler);
26 if (result == SIG_ERR)
27 FAIL_EXIT1 ("signal (%d, %p): %m", sig, handler);
28 return result;
29}
