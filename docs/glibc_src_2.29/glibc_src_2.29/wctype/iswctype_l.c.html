19#include <ctype.h>
20#include <wctype.h>
21
22#define USE_IN_EXTENDED_LOCALE_MODEL 1
23#include "wchar-lookup.h"
24
25
26int
27__iswctype_l (wint_t wc, wctype_t desc, locale_t locale)
28{
29 /* If the user passes in an invalid DESC valid (the one returned from
30 `__wctype_l' in case of an error) simply return 0. */
31 if (desc == (wctype_t) 0)
32 return 0;
33
34 return wctype_table_lookup ((const char *) desc, wc);
35}
