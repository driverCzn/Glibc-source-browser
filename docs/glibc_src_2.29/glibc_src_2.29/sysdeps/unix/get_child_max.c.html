19#include <limits.h>
20#include <sys/sysinfo.h>
21#include <sys/resource.h>
22
23#ifndef CHILD_MAX
24long int
25__get_child_max (void)
26{
27# ifdef RLIMIT_NPROC
28 struct rlimit limit;
29 if (__getrlimit (RLIMIT_NPROC, &limit) == 0
30 && limit.rlim_cur != RLIM_INFINITY)
31 return limit.rlim_cur;
32# endif
33
34 return -1;
35}
36#endif
