23ssize_t
24copy_file_range (int infd, __off64_t *pinoff,
25 int outfd, __off64_t *poutoff,
26 size_t length, unsigned int flags)
27{
28#ifdef __NR_copy_file_range
29 return SYSCALL_CANCEL (copy_file_range, infd, pinoff, outfd, poutoff,
30 length, flags);

32 __set_errno (ENOSYS);

34#endif
35}
