24/* Return descriptor for message queue associated with KEY. The MSGFLG
25 parameter describes how to proceed with clashing of key values. */
26
27int
28msgget (key_t key, int msgflg)
29{
30#ifdef __ASSUME_DIRECT_SYSVIPC_SYSCALLS
31 return INLINE_SYSCALL_CALL (msgget, key, msgflg);
32#else
33 return INLINE_SYSCALL_CALL (ipc, IPCOP_msgget, key, msgflg, 0, NULL);
34#endif
35}
