22/* Add message pointed by MSG_PTR to message queue MQDES, stop blocking
23 on full message queue if ABS_TIMEOUT expires. */
24int
25__mq_timedsend (mqd_t mqdes, const char *msg_ptr, size_t msg_len,
26 unsigned int msg_prio, const struct timespec *abs_timeout)
27{
28 return SYSCALL_CANCEL (mq_timedsend, mqdes, msg_ptr, msg_len, msg_prio,
29 abs_timeout);
30}
