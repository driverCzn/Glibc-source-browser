18#include <errno.h>
19#include <sched.h>
20#include <sysdep.h>
21
22#ifdef HAVE_GETCPU_VSYSCALL
23# define HAVE_VSYSCALL
24#endif
25#include <sysdep-vdso.h>
26
27int
28sched_getcpu (void)
29{
30#ifdef __NR_getcpu
31 unsigned int cpu;
32 int r = INLINE_VSYSCALL (getcpu, 3, &cpu, NULL, NULL);
33
34 return r == -1 ? r : cpu;
35#else
36 __set_errno (ENOSYS);
37 return -1;
38#endif
39}
