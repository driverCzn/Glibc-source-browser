19#ifndef _LOWLEVELLOCK_FUTEX_H
20#define _LOWLEVELLOCK_FUTEX_H 1

22#ifndef __ASSEMBLER__
23#include <sysdep.h>
24#include <tls.h>

28#define FUTEX_WAIT 0
29#define FUTEX_WAKE 1
30#define FUTEX_REQUEUE 3
31#define FUTEX_CMP_REQUEUE 4
32#define FUTEX_WAKE_OP 5
33#define FUTEX_OP_CLEAR_WAKE_IF_GT_ONE ((4 << 24) | 1)
34#define FUTEX_LOCK_PI 6
35#define FUTEX_UNLOCK_PI 7
36#define FUTEX_TRYLOCK_PI 8
37#define FUTEX_WAIT_BITSET 9
38#define FUTEX_WAKE_BITSET 10
39#define FUTEX_WAIT_REQUEUE_PI 11
40#define FUTEX_CMP_REQUEUE_PI 12
41#define FUTEX_PRIVATE_FLAG 128
42#define FUTEX_CLOCK_REALTIME 256
43
44#define FUTEX_BITSET_MATCH_ANY 0xffffffff
45
46/* Values for 'private' parameter of locking macros. Yes, the
47 definition seems to be backwards. But it is not. The bit will be
48 reversed before passing to the system call. */
49#define LLL_PRIVATE 0
50#define LLL_SHARED FUTEX_PRIVATE_FLAG
51
52#ifndef __ASSEMBLER__
53
54#if IS_IN (libc) || IS_IN (rtld)
55/* In libc.so or ld.so all futexes are private. */
56# define __lll_private_flag(fl, private) \
57 ({ \
58 /* Prevent warnings in callers of this macro. */ \
59 int __lll_private_flag_priv __attribute__ ((unused)); \
60 __lll_private_flag_priv = (private); \
61 ((fl) | FUTEX_PRIVATE_FLAG); \
62 })
63#else
64# define __lll_private_flag(fl, private) \
65 (((fl) | FUTEX_PRIVATE_FLAG) ^ (private))
66#endif
67
68#define lll_futex_syscall(nargs, futexp, op, ...) \
69 ({ \
70 INTERNAL_SYSCALL_DECL (__err); \
71 long int __ret = INTERNAL_SYSCALL (futex, __err, nargs, futexp, op, \
72 __VA_ARGS__); \
73 (__glibc_unlikely (INTERNAL_SYSCALL_ERROR_P (__ret, __err)) \
74 ? -INTERNAL_SYSCALL_ERRNO (__ret, __err) : 0); \
75 })
76
77#define lll_futex_wait(futexp, val, private) \
78 lll_futex_timed_wait (futexp, val, NULL, private)
79
80#define lll_futex_timed_wait(futexp, val, timeout, private) \
81 lll_futex_syscall (4, futexp, \
82 __lll_private_flag (FUTEX_WAIT, private), \
83 val, timeout)
84
85#define lll_futex_timed_wait_bitset(futexp, val, timeout, clockbit, private) \
86 lll_futex_syscall (6, futexp, \
87 __lll_private_flag (FUTEX_WAIT_BITSET | (clockbit), \
88 private), \
89 val, timeout, NULL /* Unused. */, \
90 FUTEX_BITSET_MATCH_ANY)
91
92#define lll_futex_wake(futexp, nr, private) \
93 lll_futex_syscall (4, futexp, \
94 __lll_private_flag (FUTEX_WAKE, private), nr, 0)
95
96/* Returns non-zero if error happened, zero if success. */
97#define lll_futex_requeue(futexp, nr_wake, nr_move, mutex, val, private) \
98 lll_futex_syscall (6, futexp, \
99 __lll_private_flag (FUTEX_CMP_REQUEUE, private), \
100 nr_wake, nr_move, mutex, val)
101
102/* Returns non-zero if error happened, zero if success. */
103#define lll_futex_wake_unlock(futexp, nr_wake, nr_wake2, futexp2, private) \
104 lll_futex_syscall (6, futexp, \
105 __lll_private_flag (FUTEX_WAKE_OP, private), \
106 nr_wake, nr_wake2, futexp2, \
107 FUTEX_OP_CLEAR_WAKE_IF_GT_ONE)
108
109/* Priority Inheritance support. */
110#define lll_futex_wait_requeue_pi(futexp, val, mutex, private) \
111 lll_futex_timed_wait_requeue_pi (futexp, val, NULL, 0, mutex, private)
112
113#define lll_futex_timed_wait_requeue_pi(futexp, val, timeout, clockbit, \
114 mutex, private) \
115 lll_futex_syscall (5, futexp, \
116 __lll_private_flag (FUTEX_WAIT_REQUEUE_PI \
117 | (clockbit), private), \
118 val, timeout, mutex)
119
120
121#define lll_futex_cmp_requeue_pi(futexp, nr_wake, nr_move, mutex, \
122 val, private) \
123 lll_futex_syscall (6, futexp, \
124 __lll_private_flag (FUTEX_CMP_REQUEUE_PI, \
125 private), \
126 nr_wake, nr_move, mutex, val)
127
128
129/* Cancellable futex macros. */
130#define lll_futex_wait_cancel(futexp, val, private) \
131 ({ \
132 int __oldtype = CANCEL_ASYNC (); \
133 long int __err = lll_futex_wait (futexp, val, LLL_SHARED); \
134 CANCEL_RESET (__oldtype); \
135 __err; \
136 })
137
138
139#endif /* !__ASSEMBLER__ */
