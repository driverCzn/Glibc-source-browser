30/* Create a device file named PATH relative to FD, with permission and
31 special bits MODE and device number DEV (which can be constructed
32 from major and minor device numbers with the `makedev' macro above). */
33int
34__xmknodat (int vers, int fd, const char *file, mode_t mode, dev_t *dev)
35{
36 if (vers != _MKNOD_VER)
37 return INLINE_SYSCALL_ERROR_RETURN_VALUE (EINVAL);
38
39 /* We must convert the value to dev_t type used by the kernel. */
40 unsigned long long int k_dev = (*dev) & ((1ULL << 32) - 1);
41 if (k_dev != *dev)
42 return INLINE_SYSCALL_ERROR_RETURN_VALUE (EINVAL);
43
44 return INLINE_SYSCALL (mknodat, 4, fd, file, mode, (unsigned int) k_dev);
45}
