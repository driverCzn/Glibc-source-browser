18#include <errno.h>
19#include <sysdep.h>
20#include <time.h>
21
22#include "kernel-posix-cpu-timers.h"

24/* Set CLOCK to value TP. */
25int
26__clock_settime (clockid_t clock_id, const struct timespec *tp)
27{
28 /* Make sure the time cvalue is OK. */
29 if (tp->tv_nsec < 0 || tp->tv_nsec >= 1000000000)
30 {
31 __set_errno (EINVAL);
32 return -1;
33 }
34
35 return INLINE_SYSCALL_CALL (clock_settime, clock_id, tp);
36}
37weak_alias (__clock_settime, clock_settime)
