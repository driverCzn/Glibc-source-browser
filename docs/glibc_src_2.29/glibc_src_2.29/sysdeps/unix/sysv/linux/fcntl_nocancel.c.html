25#ifndef __NR_fcntl64
26# define __NR_fcntl64 __NR_fcntl
27#endif
28
29#ifndef FCNTL_ADJUST_CMD
30# define FCNTL_ADJUST_CMD(__cmd) __cmd
31#endif
32
33int
34__fcntl64_nocancel (int fd, int cmd, ...)
35{
36 va_list ap;
37 void *arg;
38
39 va_start (ap, cmd);
40 arg = va_arg (ap, void *);
41 va_end (ap);
42
43 cmd = FCNTL_ADJUST_CMD (cmd);
44
45 return __fcntl64_nocancel_adjusted (fd, cmd, arg);
46}
47hidden_def (__fcntl64_nocancel)
48
49int
50__fcntl64_nocancel_adjusted (int fd, int cmd, void *arg)
51{
52 if (cmd == F_GETOWN)
53 {
54 INTERNAL_SYSCALL_DECL (err);
55 struct f_owner_ex fex;
56 int res = INTERNAL_SYSCALL_CALL (fcntl64, err, fd, F_GETOWN_EX, &fex);
57 if (!INTERNAL_SYSCALL_ERROR_P (res, err))
58 return fex.type == F_OWNER_GID ? -fex.pid : fex.pid;
59
60 return INLINE_SYSCALL_ERROR_RETURN_VALUE
61 (INTERNAL_SYSCALL_ERRNO (res, err));
62 }
63
64 return INLINE_SYSCALL_CALL (fcntl64, fd, cmd, (void *) arg);
65}
