19#include <arch-pkey.h>
20#include <errno.h>
21
22int
23pkey_set (int key, unsigned int rights)
24{
25 if (key < 0 || key > 15 || rights > 3)
26 {
27 __set_errno (EINVAL);
28 return -1;
29 }
30 unsigned int mask = 3 << (2 * key);
31 unsigned int pkru = pkey_read ();
32 pkru = (pkru & ~mask) | (rights << (2 * key));
33 pkey_write (pkru);
34 return 0;
35}
