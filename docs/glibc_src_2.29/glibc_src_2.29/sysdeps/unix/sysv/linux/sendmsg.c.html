24ssize_t
25__libc_sendmsg (int fd, const struct msghdr *msg, int flags)
26{
27# ifdef __ASSUME_SENDMSG_SYSCALL
28 return SYSCALL_CANCEL (sendmsg, fd, msg, flags);
29# else
30 return SOCKETCALL_CANCEL (sendmsg, fd, msg, flags);
31# endif
32}
