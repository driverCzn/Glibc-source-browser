22#include <utmp.h>
23
24#include "utmp-private.h"

26/* Test whether two entries match. */
27static int
28__utmp_equal (const struct utmp *entry, const struct utmp *match)
29{
30 return (entry->ut_type == INIT_PROCESS
31 || entry->ut_type == LOGIN_PROCESS
32 || entry->ut_type == USER_PROCESS
33 || entry->ut_type == DEAD_PROCESS)
34 && (match->ut_type == INIT_PROCESS
35 || match->ut_type == LOGIN_PROCESS
36 || match->ut_type == USER_PROCESS
37 || match->ut_type == DEAD_PROCESS)
38 && (entry->ut_id[0] && match->ut_id[0]
39 ? strncmp (entry->ut_id, match->ut_id, sizeof match->ut_id) == 0
40 : (strncmp (entry->ut_line, match->ut_line, sizeof match->ut_line)
41 == 0));
42}
