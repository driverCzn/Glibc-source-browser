16#include <float.h>
17#include <math.h>
18#include <math-narrow-eval.h>
19#include <math_private.h>
20#include <math-underflow.h>
21
22static const float one = 1.0, shuge = 1.0e37;
23
24float
25__ieee754_sinhf(float x)
26{
27 float t,w,h;
28 int32_t ix,jx;
29
30 GET_FLOAT_WORD(jx,x);
31 ix = jx&0x7fffffff;
32
33 /* x is INF or NaN */
34 if(__builtin_expect(ix>=0x7f800000, 0)) return x+x;
35
36 h = 0.5;
37 if (jx<0) h = -h;
38 /* |x| in [0,22], return sign(x)*0.5*(E+E/(E+1))) */
39 if (ix < 0x41b00000) { /* |x|<22 */
40 if (__builtin_expect(ix<0x31800000, 0)) { /* |x|<2**-28 */
41 math_check_force_underflow (x);
42 if(shuge+x>one) return x;/* sinh(tiny) = tiny with inexact */
43 }
44 t = __expm1f(fabsf(x));
45 if(ix<0x3f800000) return h*((float)2.0*t-t*t/(t+one));
46 return h*(t+t/(t+one));
47 }
48
49 /* |x| in [22, log(maxdouble)] return 0.5*exp(|x|) */
50 if (ix < 0x42b17180) return h*__ieee754_expf(fabsf(x));
51
52 /* |x| in [log(maxdouble), overflowthresold] */
53 if (ix<=0x42b2d4fc) {
54 w = __ieee754_expf((float)0.5*fabsf(x));
55 t = h*w;
56 return t*w;
57 }
58
59 /* |x| > overflowthresold, sinh(x) overflow */
60 return math_narrow_eval (x*shuge);
61}
62strong_alias (__ieee754_sinhf, __sinhf_finite)
