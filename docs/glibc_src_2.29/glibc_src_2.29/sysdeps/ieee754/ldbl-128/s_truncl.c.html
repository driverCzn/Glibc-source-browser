28_Float128
29__truncl (_Float128 x)
30{
31 int32_t j0;
32 uint64_t i0, i1, sx;
33
34 GET_LDOUBLE_WORDS64 (i0, i1, x);
35 sx = i0 & 0x8000000000000000ULL;
36 j0 = ((i0 >> 48) & 0x7fff) - 0x3fff;
37 if (j0 < 48)
38 {
39 if (j0 < 0)
40 /* The magnitude of the number is < 1 so the result is +-0. */
41 SET_LDOUBLE_WORDS64 (x, sx, 0);
42 else
43 SET_LDOUBLE_WORDS64 (x, i0 & ~(0x0000ffffffffffffLL >> j0), 0);
44 }
45 else if (j0 > 111)
46 {
47 if (j0 == 0x4000)
48 /* x is inf or NaN. */
49 return x + x;
50 }
51 else
52 {
53 SET_LDOUBLE_WORDS64 (x, i0, i1 & ~(0xffffffffffffffffULL >> (j0 - 48)));
54 }
55
56 return x;
57}
