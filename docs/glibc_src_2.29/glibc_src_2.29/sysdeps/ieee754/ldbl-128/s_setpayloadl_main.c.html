19#include <math.h>
20#include <math_private.h>
21#include <libm-alias-ldouble.h>
22#include <nan-high-order-bit.h>
23#include <stdint.h>
24
25#define SET_HIGH_BIT (HIGH_ORDER_BIT_IS_SET_FOR_SNAN ? SIG : !SIG)
26#define BIAS 0x3fff
27#define PAYLOAD_DIG 111
28#define EXPLICIT_MANT_DIG 112
29
30int
31FUNC (_Float128 *x, _Float128 payload)
32{
33 uint64_t hx, lx;
34 GET_LDOUBLE_WORDS64 (hx, lx, payload);
35 int exponent = hx >> (EXPLICIT_MANT_DIG - 64);
36 /* Test if argument is (a) negative or too large; (b) too small,
37 except for 0 when allowed; (c) not an integer. */
38 if (exponent >= BIAS + PAYLOAD_DIG
39 || (exponent < BIAS && !(SET_HIGH_BIT && hx == 0 && lx == 0)))
40 {
41 SET_LDOUBLE_WORDS64 (*x, 0, 0);
42 return 1;
43 }
44 int shift = BIAS + EXPLICIT_MANT_DIG - exponent;
45 if (shift < 64
46 ? (lx & ((1ULL << shift) - 1)) != 0
47 : (lx != 0 || (hx & ((1ULL << (shift - 64)) - 1)) != 0))
48 {
49 SET_LDOUBLE_WORDS64 (*x, 0, 0);
50 return 1;
51 }
52 if (exponent != 0)
53 {
54 hx &= (1ULL << (EXPLICIT_MANT_DIG - 64)) - 1;
55 hx |= 1ULL << (EXPLICIT_MANT_DIG - 64);
56 if (shift >= 64)
57 {
58 lx = hx >> (shift - 64);
59 hx = 0;
60 }
61 else if (shift != 0)
62 {
63 lx = (lx >> shift) | (hx << (64 - shift));
64 hx >>= shift;
65 }
66 }
67 hx |= 0x7fff000000000000ULL | (SET_HIGH_BIT ? 0x800000000000ULL : 0);
68 SET_LDOUBLE_WORDS64 (*x, hx, lx);
69 return 0;
70}
