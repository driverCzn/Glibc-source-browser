16/*
17 * nearbyintl(x)
18 * Return x rounded to integral value according to the prevailing
19 * rounding mode.
20 * Method:
21 * Using floating addition.
22 * Exception:
23 * Inexact flag raised if x not equal to rintl(x).
24 */
25
26#include <fenv.h>
27#include <math.h>
28#include <math-barriers.h>
29#include <math_private.h>
30#include <libm-alias-ldouble.h>
31
32static const _Float128
33TWO112[2]={
34 L(5.19229685853482762853049632922009600E+33), /* 0x406F000000000000, 0 */
35 L(-5.19229685853482762853049632922009600E+33) /* 0xC06F000000000000, 0 */
36};
37
38_Float128 __nearbyintl(_Float128 x)
39{
40 fenv_t env;
41 int64_t i0,j0,sx;
42 uint64_t i1 __attribute__ ((unused));
43 _Float128 w,t;
44 GET_LDOUBLE_WORDS64(i0,i1,x);
45 sx = (((uint64_t)i0)>>63);
46 j0 = ((i0>>48)&0x7fff)-0x3fff;
47 if(j0<112) {
48 if(j0<0) {
49 feholdexcept (&env);
50 w = TWO112[sx] + math_opt_barrier (x);
51 t = w-TWO112[sx];
52 math_force_eval (t);
53 fesetenv (&env);
54 GET_LDOUBLE_MSW64(i0,t);
55 SET_LDOUBLE_MSW64(t,(i0&0x7fffffffffffffffLL)|(sx<<63));
56 return t;
57 }
58 } else {
59 if(j0==0x4000) return x+x; /* inf or NaN */
60 else return x; /* x is integral */
61 }
62 feholdexcept (&env);
63 w = TWO112[sx] + math_opt_barrier (x);
64 t = w-TWO112[sx];
65 math_force_eval (t);
66 fesetenv (&env);
67 return t;
68}
