26int
27__fpclassifyl (_Float128 x)
28{
29 uint64_t hx, lx;
30 int retval = FP_NORMAL;
31
32 GET_LDOUBLE_WORDS64 (hx, lx, x);
33 lx |= (hx & 0x0000ffffffffffffLL);
34 hx &= 0x7fff000000000000LL;
35 if ((hx | lx) == 0)
36 retval = FP_ZERO;
37 else if (hx == 0)
38 retval = FP_SUBNORMAL;
39 else if (hx == 0x7fff000000000000LL)
40 retval = lx != 0 ? FP_NAN : FP_INFINITE;
41
42 return retval;
43}
