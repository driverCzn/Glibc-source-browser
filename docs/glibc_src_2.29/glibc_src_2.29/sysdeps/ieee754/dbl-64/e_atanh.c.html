20/* __ieee754_atanh(x)
21 Method :
22 1.Reduced x to positive by atanh(-x) = -atanh(x)
23 2.For x>=0.5
24 1 2x x
25 atanh(x) = --- * log(1 + -------) = 0.5 * log1p(2 * --------)
26 2 1 - x 1 - x
27
28 For x<0.5
29 atanh(x) = 0.5*log1p(2x+2x*x/(1-x))
30
31 Special cases:
32 atanh(x) is NaN if |x| > 1 with signal;
33 atanh(NaN) is that NaN with no signal;
34 atanh(+-1) is +-INF with signal.
35
36 */
37
38#include <float.h>
39#include <inttypes.h>
40#include <math.h>
41#include <math-barriers.h>
42#include <math_private.h>
43#include <math-underflow.h>
44
45static const double huge = 1e300;
46
47double
48__ieee754_atanh (double x)
49{
50 double xa = fabs (x);
51 double t;
52 if (isless (xa, 0.5))
53 {
54 if (__glibc_unlikely (xa < 0x1.0p-28))
55 {
56 math_force_eval (huge + x);
57 math_check_force_underflow (x);
58 return x;
59 }
60
61 t = xa + xa;
62 t = 0.5 * __log1p (t + t * xa / (1.0 - xa));
63 }
64 else if (__glibc_likely (isless (xa, 1.0)))
65 t = 0.5 * __log1p ((xa + xa) / (1.0 - xa));
66 else
67 {
68 if (isgreater (xa, 1.0))
69 return (x - x) / (x - x);
70
71 return x / 0.0;
72 }
73
74 return copysign (t, x);
75}
76strong_alias (__ieee754_atanh, __atanh_finite)
