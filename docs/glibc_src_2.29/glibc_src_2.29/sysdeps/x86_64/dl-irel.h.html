20#ifndef _DL_IREL_H
21#define _DL_IREL_H
22
23#include <stdio.h>
24#include <unistd.h>
25
26#define ELF_MACHINE_IRELA 1
27
28static inline ElfW(Addr)
29__attribute ((always_inline))
30elf_ifunc_invoke (ElfW(Addr) addr)
31{
32 return ((ElfW(Addr) (*) (void)) (addr)) ();
33}
34
35static inline void
36__attribute ((always_inline))
37elf_irela (const ElfW(Rela) *reloc)
38{
39 ElfW(Addr) *const reloc_addr = (void *) reloc->r_offset;
40 const unsigned long int r_type = ELFW(R_TYPE) (reloc->r_info);
41
42 if (__glibc_likely (r_type == R_X86_64_IRELATIVE))
43 {
44 ElfW(Addr) value = elf_ifunc_invoke(reloc->r_addend);
45 *reloc_addr = value;
46 }
47 else
48 __libc_fatal ("Unexpected reloc type in static binary.\n");
49}
50
51#endif /* dl-irel.h */
