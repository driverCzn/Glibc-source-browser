18#include <dirent.h>
19#include <errno.h>
20#include <fcntl.h>
21#include <stddef.h>
22#include <sys/stat.h>
23
24#include <not-cancel.h>
25
26
27DIR *
28__fdopendir (int fd)
29{
30 struct stat64 statbuf;
31
32 if (__builtin_expect (__fxstat64 (_STAT_VER, fd, &statbuf), 0) < 0)
33 return NULL;
34 if (__glibc_unlikely (! S_ISDIR (statbuf.st_mode)))
35 {
36 __set_errno (ENOTDIR);
37 return NULL;
38 }
39
40 /* Make sure the descriptor allows for reading. */
41 int flags = __fcntl64_nocancel (fd, F_GETFL);
42 if (__glibc_unlikely (flags == -1))
43 return NULL;
44 if (__glibc_unlikely ((flags & O_ACCMODE) == O_WRONLY))
45 {
46 __set_errno (EINVAL);
47 return NULL;
48 }
49
50 return __alloc_dir (fd, false, flags, &statbuf);
51}
52weak_alias (__fdopendir, fdopendir)
