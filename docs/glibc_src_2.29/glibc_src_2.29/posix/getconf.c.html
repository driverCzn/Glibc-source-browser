425static void
426print_all (const char *path)
427{
428 const struct conf *c;
429 size_t clen;
430 long int value;
431 char *cvalue;
432 for (c = vars; c->name != NULL; ++c) {
433 printf("%-35s", c->name);
434 switch (c->call) {
435 case PATHCONF:
436 value = pathconf (path, c->call_name);
437 if (value != -1) {
438 printf("%ld", value);
439 }
440 printf("\n");
441 break;
442 case SYSCONF:
443 value = sysconf (c->call_name);
444 if (value == -1l) {
445 if (c->call_name == _SC_UINT_MAX
446 || c->call_name == _SC_ULONG_MAX)
447 printf ("%lu", value);
448 }
449 else {
450 printf ("%ld", value);
451 }
452 printf ("\n");
453 break;
454 case CONFSTR:
455 clen = confstr (c->call_name, (char *) NULL, 0);
456 cvalue = (char *) malloc (clen);
457 if (cvalue == NULL)
458 error (3, 0, _("memory exhausted"));
459 if (confstr (c->call_name, cvalue, clen) != clen)
460 error (3, errno, "confstr");
461 printf ("%.*s\n", (int) clen, cvalue);
462 free (cvalue);
463 break;
464 }
465 }
466 exit (0);
467}

469int
470main (int argc, char *argv[])
471{
472 const struct conf *c;
473
474 /* Set locale. Do not set LC_ALL because the other categories must
475 not be affected (according to POSIX.2). */
476 setlocale (LC_CTYPE, "");
477 setlocale (LC_MESSAGES, "");
478
479 /* Initialize the message catalog. */
480 textdomain (PACKAGE);
481
482 if (argc > 1 && strcmp (argv[1], "--version") == 0)
483 {
484 printf ("getconf %s%s\n", PKGVERSION, VERSION);
485 printf (gettext ("\
486Copyright (C) %s Free Software Foundation, Inc.\n\
487This is free software; see the source for copying conditions. There is NO\n\
488warranty; not even for MERCHANTABILITY or FITNESS FOR A PARTICULAR PURPOSE.\n\
489"), "2019");
490 printf (gettext ("Written by %s.\n"), "Roland McGrath");
491 return 0;
492 }
493
494 if (argc > 1 && strcmp (argv[1], "--help") == 0)
495 {
496 printf (gettext ("\
497Usage: getconf [-v SPEC] VAR\n\
498 or: getconf [-v SPEC] PATH_VAR PATH\n\
499\n\
500Get the configuration value for variable VAR, or for variable PATH_VAR\n\
501for path PATH. If SPEC is given, give values for compilation\n\
502environment SPEC.\n\n"));
503 printf (gettext ("For bug reporting instructions, please see:\n\
504%s.\n"), REPORT_BUGS_TO);
505 return 0;
506 }
507
508#ifdef ALL_ENVIRONMENTS_DEFINED
509 if (argc > 1 && strncmp (argv[1], "-v", 2) == 0)
510 {
511 if (argv[1][2] == '\0')
512 {
513 if (argc < 3)
514 usage ();
515
516 argv += 2;
517 argc -= 2;
518 }
519 else
520 {
521 argv += 1;
522 argc += 1;
523 }
524 }
525#else
526 const char *getconf_dir = getenv ("GETCONF_DIR") ?: GETCONF_DIR;
527 size_t getconf_dirlen = strlen (getconf_dir);
528
529 const char *spec = NULL;
530 char buf[sizeof "POSIX_V6_LPBIG_OFFBIG"];
531 char *argv0 = argv[0];
532 if (argc > 1 && strncmp (argv[1], "-v", 2) == 0)
533 {
534 if (argv[1][2] == '\0')
535 {
536 if (argc < 3)
537 usage ();
538
539 spec = argv[2];
540 argv += 2;
541 argc -= 2;
542 }
543 else
544 {
545 spec = &argv[1][2];
546 argv += 1;
547 argc += 1;
548 }
549 }
550 else
551 {
552 char default_name[getconf_dirlen + sizeof "/default"];
553 memcpy (mempcpy (default_name, getconf_dir, getconf_dirlen),
554 "/default", sizeof "/default");
555 int len = readlink (default_name, buf, sizeof buf - 1);
556 if (len > 0)
557 {
558 buf[len] = '\0';
559 spec = buf;
560 }
561 }
562
563 /* Check for the specifications we know. */
564 if (spec != NULL)
565 {
566 size_t i;
567 for (i = 0; i < nspecs; ++i)
568 if (strcmp (spec, specs[i].name) == 0)
569 break;
570
571 if (i == nspecs)
572 error (2, 0, _("unknown specification \"%s\""), spec);
573
574 switch (specs[i].num)
575 {
576# ifndef _XBS5_ILP32_OFF32
577 case _SC_XBS5_ILP32_OFF32:
578# endif
579# ifndef _XBS5_ILP32_OFFBIG
580 case _SC_XBS5_ILP32_OFFBIG:
581# endif
582# ifndef _XBS5_LP64_OFF64
583 case _SC_XBS5_LP64_OFF64:
584# endif
585# ifndef _XBS5_LPBIG_OFFBIG
586 case _SC_XBS5_LPBIG_OFFBIG:
587# endif
588# ifndef _POSIX_V6_ILP32_OFF32
589 case _SC_V6_ILP32_OFF32:
590# endif
591# ifndef _POSIX_V6_ILP32_OFFBIG
592 case _SC_V6_ILP32_OFFBIG:
593# endif
594# ifndef _POSIX_V6_LP64_OFF64
595 case _SC_V6_LP64_OFF64:
596# endif
597# ifndef _POSIX_V6_LPBIG_OFFBIG
598 case _SC_V6_LPBIG_OFFBIG:
599# endif
600# ifndef _POSIX_V7_ILP32_OFF32
601 case _SC_V7_ILP32_OFF32:
602# endif
603# ifndef _POSIX_V7_ILP32_OFFBIG
604 case _SC_V7_ILP32_OFFBIG:
605# endif
606# ifndef _POSIX_V7_LP64_OFF64
607 case _SC_V7_LP64_OFF64:
608# endif
609# ifndef _POSIX_V7_LPBIG_OFFBIG
610 case _SC_V7_LPBIG_OFFBIG:
611# endif
612 {
613 const char *args[argc + 3];
614 size_t spec_len = strlen (spec);
615 char getconf_name[getconf_dirlen + 1 + spec_len + 1];
616 memcpy (mempcpy (mempcpy (getconf_name, getconf_dir,
617 getconf_dirlen),
618 "/", 1), spec, spec_len + 1);
619 args[0] = argv0;
620 args[1] = "-v";
621 args[2] = spec;
622 memcpy (&args[3], &argv[1], argc * sizeof (argv[1]));
623 execv (getconf_name, (char * const *) args);
624 error (4, errno, _("Couldn't execute %s"), getconf_name);
625 }
626 default:
627 break;
628 }
629 }
630#endif
631
632 if (argc > 1 && strcmp (argv[1], "-a") == 0)
633 {
634 if (argc == 2)
635 print_all ("/");
636 else if (argc == 3)
637 print_all (argv[2]);
638 else
639 usage ();
640 }
641
642 int ai = 1;
643 if (argc > ai && strcmp (argv[ai], "--") == 0)
644 ++ai;
645
646 if (argc - ai < 1 || argc - ai > 2)
647 usage ();
648
649 for (c = vars; c->name != NULL; ++c)
650 if (strcmp (c->name, argv[ai]) == 0
651 || (strncmp (c->name, "_POSIX_", 7) == 0
652 && strcmp (c->name + 7, argv[ai]) == 0))
653 {
654 long int value;
655 size_t clen;
656 char *cvalue;
657 switch (c->call)
658 {
659 case PATHCONF:
660 if (argc - ai < 2)
661 usage ();
662 errno = 0;
663 value = pathconf (argv[ai + 1], c->call_name);
664 if (value == -1)
665 {
666 if (errno)
667 error (3, errno, "pathconf: %s", argv[ai + 1]);
668 else
669 puts (_("undefined"));
670 }
671 else
672 printf ("%ld\n", value);
673 exit (0);
674
675 case SYSCONF:
676 if (argc - ai > 1)
677 usage ();
678 value = sysconf (c->call_name);
679 if (value == -1l)
680 {
681 if (c->call_name == _SC_UINT_MAX
682 || c->call_name == _SC_ULONG_MAX)
683 printf ("%lu\n", value);
684 else
685 puts (_("undefined"));
686 }
687 else
688 printf ("%ld\n", value);
689 exit (0);
690
691 case CONFSTR:
692 if (argc - ai > 1)
693 usage ();
694 clen = confstr (c->call_name, (char *) NULL, 0);
695 cvalue = (char *) malloc (clen);
696 if (cvalue == NULL)
697 error (3, 0, _("memory exhausted"));
698
699 if (confstr (c->call_name, cvalue, clen) != clen)
700 error (3, errno, "confstr");
701
702 printf ("%.*s\n", (int) clen, cvalue);
703 exit (0);
704 }
705 }
706
707 error (2, 0, _("Unrecognized variable `%s'"), argv[ai]);
708 /* NOTREACHED */
709 return 2;
710}
