27#include "../libio/libioP.h"
28#include <limits.h>

30char *
31__gets_chk (char *buf, size_t size)
32{
33 size_t count;
34 int ch;
35 char *retval;
36
37 if (size == 0)
38 __chk_fail ();
39
40 _IO_acquire_lock (_IO_stdin);
41 ch = _IO_getc_unlocked (_IO_stdin);
42 if (ch == EOF)
43 {
44 retval = NULL;
45 goto unlock_return;
46 }
47 if (ch == '\n')
48 count = 0;
49 else
50 {
51 /* This is very tricky since a file descriptor may be in the
52 non-blocking mode. The error flag doesn't mean much in this
53 case. We return an error only when there is a new error. */
54 int old_error = _IO_stdin->_flags & _IO_ERR_SEEN;
55 _IO_stdin->_flags &= ~_IO_ERR_SEEN;
56 buf[0] = (char) ch;
57 count = _IO_getline (_IO_stdin, buf + 1, size - 1, '\n', 0) + 1;
58 if (_IO_stdin->_flags & _IO_ERR_SEEN)
59 {
60 retval = NULL;
61 goto unlock_return;
62 }
63 else
64 _IO_stdin->_flags |= old_error;
65 }
66 if (count >= size)
67 __chk_fail ();
68 buf[count] = 0;
69 retval = buf;
70unlock_return:
71 _IO_release_lock (_IO_stdin);
72 return retval;
73}
74
75link_warning (__gets_chk, "the `gets' function is dangerous and should not be used.")
