21/* Write formatted output to FP from the format string FORMAT. */
22int
23__vfwprintf_chk (FILE *fp, int flag, const wchar_t *format, va_list ap)
24{
25 /* For flag > 0 (i.e. __USE_FORTIFY_LEVEL > 1) request that %n
26 can only come from read-only format strings. */
27 unsigned int mode = (flag > 0) ? PRINTF_FORTIFY : 0;
28
29 return __vfwprintf_internal (fp, format, ap, mode);
30}
