28/* We have to use the lock in getutent_r.c. */
29__libc_lock_define (extern, __libc_utmp_lock attribute_hidden)
30
31
32int
33__getutid_r (const struct utmp *id, struct utmp *buffer, struct utmp **result)
34{
35 int retval;
36
37 /* Test whether ID has any of the legal types. */
38 if (id->ut_type != RUN_LVL && id->ut_type != BOOT_TIME
39 && id->ut_type != OLD_TIME && id->ut_type != NEW_TIME
40 && id->ut_type != INIT_PROCESS && id->ut_type != LOGIN_PROCESS
41 && id->ut_type != USER_PROCESS && id->ut_type != DEAD_PROCESS)
42 /* No, using '<' and '>' for the test is not possible. */
43 {
44 __set_errno (EINVAL);
45 *result = NULL;
46 return -1;
47 }
48
49 __libc_lock_lock (__libc_utmp_lock);
50
51 retval = __libc_getutid_r (id, buffer, result);
52
53 __libc_lock_unlock (__libc_utmp_lock);
54
55 return retval;
56}
57libc_hidden_def (__getutid_r)
58weak_alias (__getutid_r, getutid_r)
