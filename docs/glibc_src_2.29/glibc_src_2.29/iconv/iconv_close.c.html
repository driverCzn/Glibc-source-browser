20#include <errno.h>
21#include <iconv.h>
22
23#include <gconv_int.h>
24
25
26int
27iconv_close (iconv_t cd)
28{
29 if (__glibc_unlikely (cd == (iconv_t *) -1L))
30 {
31 __set_errno (EBADF);
32 return -1;
33 }
34
35 return __gconv_close ((__gconv_t) cd) ? -1 : 0;
36}
