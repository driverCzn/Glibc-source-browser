26#if !defined SHARED && IS_IN (libdl)

28void *
29dlmopen (Lmid_t nsid, const char *file, int mode)
30{
31 return __dlmopen (nsid, file, mode, RETURN_ADDRESS (0));
32}
33static_link_warning (dlmopen)

35#else
36
37struct dlmopen_args
38{
39 /* Namespace ID. */
40 Lmid_t nsid;
41 /* The arguments for dlopen_doit. */
42 const char *file;
43 int mode;
44 /* The return value of dlopen_doit. */
45 void *new;
46 /* Address of the caller. */
47 const void *caller;
48};
49
50static void
51dlmopen_doit (void *a)
52{
53 struct dlmopen_args *args = (struct dlmopen_args *) a;
54
55 /* Non-shared code has no support for multiple namespaces. */
56 if (args->nsid != LM_ID_BASE)
57 {
58# ifdef SHARED
59 /* If trying to open the link map for the main executable the namespace
60 must be the main one. */
61 if (args->file == NULL)
62# endif
63 _dl_signal_error (EINVAL, NULL, NULL, N_("invalid namespace"));
64
65 /* It makes no sense to use RTLD_GLOBAL when loading a DSO into
66 a namespace other than the base namespace. */
67 if (__glibc_unlikely (args->mode & RTLD_GLOBAL))
68 _dl_signal_error (EINVAL, NULL, NULL, N_("invalid mode"));
69 }
70
71 args->new = GLRO(dl_open) (args->file ?: "", args->mode | __RTLD_DLOPEN,
72 args->caller,
73 args->nsid, __dlfcn_argc, __dlfcn_argv,
74 __environ);
75}
76
77
78void *
79__dlmopen (Lmid_t nsid, const char *file, int mode DL_CALLER_DECL)
80{
81# ifdef SHARED
82 if (!rtld_active ())
83 return _dlfcn_hook->dlmopen (nsid, file, mode, RETURN_ADDRESS (0));
84# endif
85
86 struct dlmopen_args args;
87 args.nsid = nsid;
88 args.file = file;
89 args.mode = mode;
90 args.caller = DL_CALLER;
91
92# ifdef SHARED
93 return _dlerror_run (dlmopen_doit, &args) ? NULL : args.new;
94# else
95 if (_dlerror_run (dlmopen_doit, &args))
96 return NULL;
97
98 __libc_register_dl_open_hook ((struct link_map *) args.new);
99 __libc_register_dlfcn_hook ((struct link_map *) args.new);
100
101 return args.new;
102# endif
103}
104# ifdef SHARED
105strong_alias (__dlmopen, dlmopen)
106# endif
107#endif
