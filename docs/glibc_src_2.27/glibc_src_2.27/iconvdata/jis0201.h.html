20#ifndef _JIS0201_H
21#define _JIS0201_H 1
22
23#include <stdint.h>
24
25/* Conversion table. */
26extern const uint32_t __jisx0201_to_ucs4[];
27
28
29static inline uint32_t
30__attribute ((always_inline))
31jisx0201_to_ucs4 (char ch)
32{
33 uint32_t val = __jisx0201_to_ucs4[(unsigned char) ch];
34
35 if (val == 0 && ch != '\0')
36 val = __UNKNOWN_10646_CHAR;
37
38 return val;
39}
40
41
42static inline size_t
43__attribute ((always_inline))
44ucs4_to_jisx0201 (uint32_t wch, unsigned char *s)
45{
46 unsigned char ch;
47
48 if (wch == 0xa5)
49 ch = '\x5c';
50 else if (wch == 0x203e)
51 ch = '\x7e';
52 else if (wch < 0x7e && wch != 0x5c)
53 ch = wch;
54 else if (wch >= 0xff61 && wch <= 0xff9f)
55 ch = wch - 0xfec0;
56 else
57 return __UNKNOWN_10646_CHAR;
58
59 s[0] = ch;
60 return 1;
61}
