31int
32__pthread_attr_init_2_1 (pthread_attr_t *attr)
33{
34 struct pthread_attr *iattr;
35
36 ASSERT_TYPE_SIZE (pthread_attr_t, __SIZEOF_PTHREAD_ATTR_T);
37 ASSERT_PTHREAD_INTERNAL_SIZE (pthread_attr_t, struct pthread_attr);
38
39 /* Many elements are initialized to zero so let us do it all at
40 once. This also takes care of clearing the bytes which are not
41 internally used. */
42 memset (attr, '\0', __SIZEOF_PTHREAD_ATTR_T);
43
44 iattr = (struct pthread_attr *) attr;
45
46 /* Default guard size specified by the standard. */
47 iattr->guardsize = __getpagesize ();
48
49 return 0;
50}
51versioned_symbol (libpthread, __pthread_attr_init_2_1, pthread_attr_init,
52 GLIBC_2_1);

55#if SHLIB_COMPAT(libpthread, GLIBC_2_0, GLIBC_2_1)
56int
57__pthread_attr_init_2_0 (pthread_attr_t *attr)
58{
59 /* This code is specific to the old LinuxThread code which has a too
60 small pthread_attr_t definition. The struct looked like
61 this: */
62 struct old_attr
63 {
64 int detachstate;
65 int schedpolicy;
66 struct sched_param schedparam;
67 int inheritsched;
68 int scope;
69 };
70 struct pthread_attr *iattr;
71
72 /* Many elements are initialized to zero so let us do it all at
73 once. This also takes care of clearing the bytes which are not
74 internally used. */
75 memset (attr, '\0', sizeof (struct old_attr));
76
77 iattr = (struct pthread_attr *) attr;
78 iattr->flags |= ATTR_FLAG_OLDATTR;
79
80 /* We cannot enqueue the attribute because that member is not in the
81 old attribute structure. */
82 return 0;
83}
84compat_symbol (libpthread, __pthread_attr_init_2_0, pthread_attr_init,
85 GLIBC_2_0);
