19#include <errno.h>
20#include "pthreadP.h"
21#include <shlib-compat.h>
22
23
24#if SHLIB_COMPAT(libpthread, GLIBC_2_0, GLIBC_2_3_2)
25int
26__pthread_cond_init_2_0 (pthread_cond_2_0_t *cond,
27 const pthread_condattr_t *cond_attr)
28{
29 struct pthread_condattr *icond_attr = (struct pthread_condattr *) cond_attr;
30
31 /* The type of the first argument is actually that of the old, too
32 small pthread_cond_t. We use only the first word of it, as a
33 pointer. */
34 cond->cond = NULL;
35
36 /* We can't support PSHARED condvars in the old pthread_cond_*
37 functions and neither clocks other than CLOCK_REALTIME. */
38 if (icond_attr != NULL && icond_attr->value)
39 return EINVAL;
40
41 return 0;
42}
43compat_symbol (libpthread, __pthread_cond_init_2_0, pthread_cond_init,
44 GLIBC_2_0);
45#endif
