20#include <setjmp.h>
21#include <shlib-compat.h>

23/* libpthread once had its own longjmp (and siglongjmp alias), though there
24 was no apparent reason for it. There is no use in having a separate
25 symbol in libpthread, but the historical ABI requires it. For static
26 linking, there is no need to provide anything here--the libc version
27 will be linked in. For shared library ABI compatibility, there must be
28 longjmp and siglongjmp symbols in libpthread.so.
29
30 With an IFUNC resolver, it would be possible to avoid the indirection,
31 but the IFUNC resolver might run before the __libc_longjmp symbol has
32 been relocated, in which case the IFUNC resolver would not be able to
33 provide the correct address. */
34
35#if SHLIB_COMPAT (libpthread, GLIBC_2_0, GLIBC_2_22)
36
37static void __attribute__ ((noreturn, used))
38longjmp_compat (jmp_buf env, int val)
39{
40 __libc_longjmp (env, val);
41}
42
43strong_alias (longjmp_compat, longjmp_alias)
44compat_symbol (libpthread, longjmp_alias, longjmp, GLIBC_2_0);
45
46strong_alias (longjmp_alias, siglongjmp_alias)
47compat_symbol (libpthread, siglongjmp_alias, siglongjmp, GLIBC_2_0);
48
49#endif
