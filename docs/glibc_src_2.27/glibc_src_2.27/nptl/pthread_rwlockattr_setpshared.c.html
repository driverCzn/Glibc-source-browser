19#include <errno.h>
20#include "pthreadP.h"
21#include <futex-internal.h>
22
23
24int
25pthread_rwlockattr_setpshared (pthread_rwlockattr_t *attr, int pshared)
26{
27 struct pthread_rwlockattr *iattr;
28
29 int err = futex_supports_pshared (pshared);
30 if (err != 0)
31 return err;
32
33 iattr = (struct pthread_rwlockattr *) attr;
34
35 iattr->pshared = pshared;
36
37 return 0;
38}
