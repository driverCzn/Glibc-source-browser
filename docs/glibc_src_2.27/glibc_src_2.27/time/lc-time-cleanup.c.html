19#include "../locale/localeinfo.h"
20#include <stdlib.h>
21
22void
23_nl_cleanup_time (struct __locale_data *locale)
24{
25 struct lc_time_data *const data = locale->private.time;
26 if (data != NULL)
27 {
28 locale->private.time = NULL;
29 locale->private.cleanup = NULL;
30
31 free (data->eras);
32 free (data->alt_digits);
33 free (data->walt_digits);
34
35 free (data);
36 }
37}
