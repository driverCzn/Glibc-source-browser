22/* Return a string as returned by asctime which is the representation
23 of *T in that form. Reentrant version. */
24char *
25ctime_r (const time_t *t, char *buf)
26{
27 struct tm tm;
28 return __asctime_r (__localtime_r (t, &tm), buf);
29}
