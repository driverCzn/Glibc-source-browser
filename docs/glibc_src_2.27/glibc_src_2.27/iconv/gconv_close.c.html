25int
26__gconv_close (__gconv_t cd)
27{
28 struct __gconv_step *srunp;
29 struct __gconv_step_data *drunp;
30 size_t nsteps;
31
32 /* Free all resources by calling destructor functions and release
33 the implementations. */
34 srunp = cd->__steps;
35 nsteps = cd->__nsteps;
36 drunp = cd->__data;
37 do
38 {
39 if (!(drunp->__flags & __GCONV_IS_LAST) && drunp->__outbuf != NULL)
40 free (drunp->__outbuf);
41 }
42 while (!((drunp++)->__flags & __GCONV_IS_LAST));
43
44 /* Free the data allocated for the descriptor. */
45 free (cd);
46
47 /* Close the participating modules. */
48 return __gconv_close_transform (srunp, nsteps);
49}
