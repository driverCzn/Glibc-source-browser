24static void
25strip (char *wp, const char *s)
26{
27 int slash_count = 0;
28
29 while (*s != '\0')
30 {
31 if (__isalnum_l (*s, _nl_C_locobj_ptr)
32 || *s == '_' || *s == '-' || *s == '.' || *s == ',' || *s == ':')
33 *wp++ = __toupper_l (*s, _nl_C_locobj_ptr);
34 else if (*s == '/')
35 {
36 if (++slash_count == 3)
37 break;
38 *wp++ = '/';
39 }
40 ++s;
41 }
42
43 while (slash_count++ < 2)
44 *wp++ = '/';
45
46 *wp = '\0';
47}

50static inline char * __attribute__ ((unused, always_inline))
51upstr (char *dst, const char *str)
52{
53 char *cp = dst;
54 while ((*cp++ = __toupper_l (*str++, _nl_C_locobj_ptr)) != '\0')
55 /* nothing */;
56 return dst;
57}
