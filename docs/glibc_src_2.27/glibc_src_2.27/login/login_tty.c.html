34#include <errno.h>
35#include <sys/param.h>
36#include <sys/ioctl.h>
37#include <unistd.h>
38#include <fcntl.h>
39#include <utmp.h>
40
41int
42login_tty (int fd)
43{
44 (void) setsid();
45#ifdef TIOCSCTTY
46 if (ioctl(fd, TIOCSCTTY, (char *)NULL) == -1)
47 return (-1);
48#else
49 {
50 /* This might work. */
51 char *fdname = ttyname (fd);
52 int newfd;
53 if (fdname)
54 {
55 if (fd != 0)
56 (void) close (0);
57 if (fd != 1)
58 (void) close (1);
59 if (fd != 2)
60 (void) close (2);
61 newfd = open (fdname, O_RDWR);
62 (void) close (newfd);
63 }
64 }
65#endif
66 while (dup2(fd, 0) == -1 && errno == EBUSY)
67 ;
68 while (dup2(fd, 1) == -1 && errno == EBUSY)
69 ;
70 while (dup2(fd, 2) == -1 && errno == EBUSY)
71 ;
72 if (fd > 2)
73 (void) close(fd);
74 return (0);
75}
