19#include <errno.h>
20#include <string.h>
21#include <utmp.h>
22#include <sys/time.h>
23
24int
25logout (const char *line)
26{
27 struct utmp tmp, utbuf;
28 struct utmp *ut;
29 int result = 0;
30
31 /* Tell that we want to use the UTMP file. */
32 if (utmpname (_PATH_UTMP) == -1)
33 return 0;
34
35 /* Open UTMP file. */
36 setutent ();
37
38 /* Fill in search information. */
39#if _HAVE_UT_TYPE - 0
40 tmp.ut_type = USER_PROCESS;
41#endif
42 strncpy (tmp.ut_line, line, sizeof tmp.ut_line);
43
44 /* Read the record. */
45 if (getutline_r (&tmp, &utbuf, &ut) >= 0)
46 {
47 /* Clear information about who & from where. */
48 memset (ut->ut_name, '\0', sizeof ut->ut_name);
49#if _HAVE_UT_HOST - 0
50 memset (ut->ut_host, '\0', sizeof ut->ut_host);
51#endif
52#if _HAVE_UT_TV - 0
53 struct timeval tv;
54 __gettimeofday (&tv, NULL);
55 ut->ut_tv.tv_sec = tv.tv_sec;
56 ut->ut_tv.tv_usec = tv.tv_usec;
57#else
58 ut->ut_time = time (NULL);
59#endif
60#if _HAVE_UT_TYPE - 0
61 ut->ut_type = DEAD_PROCESS;
62#endif
63
64 if (pututline (ut) != NULL)
65 result = 1;
66 }
67
68 /* Close UTMP file. */
69 endutent ();
70
71 return result;
72}
