28/* We have to use the lock in getutent_r.c. */
29__libc_lock_define (extern, __libc_utmp_lock attribute_hidden)
30
31
32int
33__getutid_r (const struct utmp *id, struct utmp *buffer, struct utmp **result)
34{
35#if (_HAVE_UT_ID - 0) && (_HAVE_UT_TYPE - 0)
36 int retval;
37
38 /* Test whether ID has any of the legal types. */
39 if (id->ut_type != RUN_LVL && id->ut_type != BOOT_TIME
40 && id->ut_type != OLD_TIME && id->ut_type != NEW_TIME
41 && id->ut_type != INIT_PROCESS && id->ut_type != LOGIN_PROCESS
42 && id->ut_type != USER_PROCESS && id->ut_type != DEAD_PROCESS)
43 /* No, using '<' and '>' for the test is not possible. */
44 {
45 __set_errno (EINVAL);
46 *result = NULL;
47 return -1;
48 }
49
50 __libc_lock_lock (__libc_utmp_lock);
51
52 retval = (*__libc_utmp_jump_table->getutid_r) (id, buffer, result);
53
54 __libc_lock_unlock (__libc_utmp_lock);
55
56 return retval;
57#else /* !_HAVE_UT_ID && !_HAVE_UT_TYPE */
58 __set_errno (ENOSYS);
59 return -1;
60#endif
61}
62libc_hidden_def (__getutid_r)
63weak_alias (__getutid_r, getutid_r)
