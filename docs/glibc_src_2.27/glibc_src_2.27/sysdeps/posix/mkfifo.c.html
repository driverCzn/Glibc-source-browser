18#include <errno.h>
19#include <stddef.h>
20#include <sys/stat.h>

23/* Create a named pipe (FIFO) named PATH with protections MODE. */
24int
25mkfifo (const char *path, mode_t mode)
26{
27 dev_t dev = 0;
28 return __xmknod (_MKNOD_VER, path, mode | S_IFIFO, &dev);
29}
