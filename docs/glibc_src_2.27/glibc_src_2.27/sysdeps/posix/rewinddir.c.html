18#include <stddef.h>
19#include <dirent.h>
20#include <sys/types.h>
21#include <unistd.h>
22#include <dirstream.h>
23
24/* Rewind DIRP to the beginning of the directory. */
25void
26__rewinddir (DIR *dirp)
27{
28#if IS_IN (libc)
29 __libc_lock_lock (dirp->lock);
30#endif
31 (void) __lseek (dirp->fd, (off_t) 0, SEEK_SET);
32 dirp->filepos = 0;
33 dirp->offset = 0;
34 dirp->size = 0;
35 dirp->errcode = 0;
36#if IS_IN (libc)
37 __libc_lock_unlock (dirp->lock);
38#endif
39}
40libc_hidden_def (__rewinddir)
41weak_alias (__rewinddir, rewinddir)
