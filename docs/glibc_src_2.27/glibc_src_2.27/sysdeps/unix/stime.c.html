18#include <errno.h>
19#include <stddef.h> /* For NULL. */
20#include <sys/time.h>
21#include <time.h>
22
23/* Set the system clock to *WHEN. */
24
25int
26stime (const time_t *when)
27{
28 struct timeval tv;
29
30 if (when == NULL)
31 {
32 __set_errno (EINVAL);
33 return -1;
34 }
35
36 tv.tv_sec = *when;
37 tv.tv_usec = 0;
38 return __settimeofday (&tv, (struct timezone *) 0);
39}
