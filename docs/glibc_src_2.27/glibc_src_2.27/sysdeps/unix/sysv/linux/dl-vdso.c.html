20#include <ldsodefs.h>

22
23void *
24_dl_vdso_vsym (const char *name, const struct r_found_version *vers)
25{
26 struct link_map *map = GLRO (dl_sysinfo_map);
27 void *value = NULL;
28
29
30 if (map != NULL)
31 {
32 /* Use a WEAK REF so we don't error out if the symbol is not found. */
33 ElfW (Sym) wsym;
34 memset (&wsym, 0, sizeof (ElfW (Sym)));
35 wsym.st_info = (unsigned char) ELFW (ST_INFO (STB_WEAK, STT_NOTYPE));
36
37 /* Search the scope of the vdso map. */
38 const ElfW (Sym) *ref = &wsym;
39 lookup_t result = GLRO (dl_lookup_symbol_x) (name, map, &ref,
40 map->l_local_scope,
41 vers, 0, 0, NULL);
42
43 if (ref != NULL)
44 value = DL_SYMBOL_ADDRESS (result, ref);
45 }
46
47 return value;
48}
