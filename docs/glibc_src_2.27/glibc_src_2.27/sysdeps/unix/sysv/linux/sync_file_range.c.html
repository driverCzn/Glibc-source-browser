19#include <fcntl.h>
20#include <sysdep-cancel.h>
21
22int
23sync_file_range (int fd, __off64_t offset, __off64_t len, unsigned int flags)
24{
25#if defined (__NR_sync_file_range2)
26 return SYSCALL_CANCEL (sync_file_range2, fd, flags, SYSCALL_LL64 (offset),
27 SYSCALL_LL64 (len));
28#elif defined (__NR_sync_file_range)
29 return SYSCALL_CANCEL (sync_file_range, fd,
30 __ALIGNMENT_ARG SYSCALL_LL64 (offset),
31 SYSCALL_LL64 (len), flags);
32#endif
33}
