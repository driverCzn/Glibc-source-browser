18#include <signal.h>
19#include <sysdep-cancel.h>
20
21/* Change the set of blocked signals to SET,
22 wait until a signal arrives, and restore the set of blocked signals. */
23int
24__sigsuspend (const sigset_t *set)
25{
26 return SYSCALL_CANCEL (rt_sigsuspend, set, _NSIG / 8);
27}

29weak_alias (__sigsuspend, sigsuspend)
30strong_alias (__sigsuspend, __libc_sigsuspend)
