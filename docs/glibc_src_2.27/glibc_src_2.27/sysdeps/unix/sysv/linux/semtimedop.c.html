19#include <sys/sem.h>
20#include <ipc_priv.h>
21#include <sysdep.h>
22#include <errno.h>
23
24/* Perform user-defined atomical operation of array of semaphores. */
25
26int
27semtimedop (int semid, struct sembuf *sops, size_t nsops,
28 const struct timespec *timeout)
29{
30#ifdef __ASSUME_DIRECT_SYSVIPC_SYSCALLS
31 return INLINE_SYSCALL_CALL (semtimedop, semid, sops, nsops, timeout);
32#else
33 return INLINE_SYSCALL_CALL (ipc, IPCOP_semtimedop, semid, nsops, 0, sops,
34 timeout);
35#endif
36}
