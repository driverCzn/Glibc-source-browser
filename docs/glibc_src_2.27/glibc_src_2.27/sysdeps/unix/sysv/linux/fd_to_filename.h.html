19#include <stdlib.h>
20#include <string.h>
21#include <sys/stat.h>
22#include <_itoa.h>
23
24static inline const char *
25fd_to_filename (int fd)
26{
27 char *ret = malloc (30);
28
29 if (ret != NULL)
30 {
31 struct stat64 st;
32
33 *_fitoa_word (fd, __stpcpy (ret, "/proc/self/fd/"), 10, 0) = '\0';
34
35 /* We must make sure the file exists. */
36 if (__lxstat64 (_STAT_VER, ret, &st) < 0)
37 {
38 /* /proc is not mounted or something else happened. Don't
39 return the file name. */
40 free (ret);
41 ret = NULL;
42 }
43 }
44 return ret;
45}
