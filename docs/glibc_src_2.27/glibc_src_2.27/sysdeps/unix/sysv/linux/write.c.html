23/* Write NBYTES of BUF to FD. Return the number written, or -1. */
24ssize_t
25__libc_write (int fd, const void *buf, size_t nbytes)
26{
27 return SYSCALL_CANCEL (write, fd, buf, nbytes);
28}
29libc_hidden_def (__libc_write)
