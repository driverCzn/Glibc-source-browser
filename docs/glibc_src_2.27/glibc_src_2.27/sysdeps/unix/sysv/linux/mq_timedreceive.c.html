22/* Receive the oldest from highest priority messages in message queue
23 MQDES, stop waiting if ABS_TIMEOUT expires. */
24ssize_t
25__mq_timedreceive (mqd_t mqdes, char *__restrict msg_ptr, size_t msg_len,
26 unsigned int *__restrict msg_prio,
27 const struct timespec *__restrict abs_timeout)
28{
29 return SYSCALL_CANCEL (mq_timedreceive, mqdes, msg_ptr, msg_len, msg_prio,
30 abs_timeout);
31}
