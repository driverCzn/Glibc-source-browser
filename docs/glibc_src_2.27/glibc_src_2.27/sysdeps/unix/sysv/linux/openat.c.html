26/* Open FILE with access OFLAG. Interpret relative paths relative to
27 the directory associated with FD. If OFLAG includes O_CREAT or
28 O_TMPFILE, a fourth argument is the file protection. */
29int
30__libc_openat (int fd, const char *file, int oflag, ...)
31{
32 mode_t mode = 0;
33 if (__OPEN_NEEDS_MODE (oflag))
34 {
35 va_list arg;
36 va_start (arg, oflag);
37 mode = va_arg (arg, mode_t);
38 va_end (arg);
39 }
40
41 return SYSCALL_CANCEL (openat, fd, file, oflag, mode);
42}
