25static long int linux_sysconf (int name);
26extern long int __cache_sysconf (int) attribute_hidden;

28
29/* Get the value of the system variable NAME. */
30long int
31__sysconf (int name)
32{
33 if (name >= _SC_LEVEL1_ICACHE_SIZE && name <= _SC_LEVEL4_CACHE_LINESIZE)
34 return __cache_sysconf (name);
35
36 return linux_sysconf (name);
37}

39/* Now the generic Linux version. */
