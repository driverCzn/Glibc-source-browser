18#ifndef _SYS_SIGNALFD_H
19#define _SYS_SIGNALFD_H 1
20
21#include <stdint.h>
22#include <bits/types/sigset_t.h>
23
24/* Get the platform-dependent flags. */
25#include <bits/signalfd.h>
26
27struct signalfd_siginfo
28{
29 uint32_t ssi_signo;
30 int32_t ssi_errno;
31 int32_t ssi_code;
32 uint32_t ssi_pid;
33 uint32_t ssi_uid;
34 int32_t ssi_fd;
35 uint32_t ssi_tid;
36 uint32_t ssi_band;
37 uint32_t ssi_overrun;
38 uint32_t ssi_trapno;
39 int32_t ssi_status;
40 int32_t ssi_int;
41 uint64_t ssi_ptr;
42 uint64_t ssi_utime;
43 uint64_t ssi_stime;
44 uint64_t ssi_addr;
45 uint8_t __pad[48];
46};
47
48__BEGIN_DECLS
49
50/* Request notification for delivery of signals in MASK to be
51 performed using descriptor FD.*/
52extern int signalfd (int __fd, const sigset_t *__mask, int __flags)
53 __THROW __nonnull ((2));
54
55__END_DECLS
56
57#endif /* sys/signalfd.h */
