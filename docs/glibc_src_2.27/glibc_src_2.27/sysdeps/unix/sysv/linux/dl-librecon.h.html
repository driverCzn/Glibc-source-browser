20#ifndef _DL_LIBRECON_H
21#define _DL_LIBRECON_H 1
22
23static inline void __attribute__ ((unused, always_inline))
24_dl_osversion_init (char *assume_kernel)
25{
26 unsigned long int i, j, osversion = 0;
27 char *p = assume_kernel, *q;
28
29 for (i = 0; i < 3; i++, p = q + 1)
30 {
31 j = _dl_strtoul (p, &q);
32 if (j >= 255 || p == q || (i < 2 && *q && *q != '.'))
33 {
34 osversion = 0;
35 break;
36 }
37 osversion |= j << (16 - 8 * i);
38 if (!*q)
39 break;
40 }
41 if (osversion)
42 GLRO(dl_osversion) = osversion;
43}
44
45/* Recognizing extra environment variables. */
46#define EXTRA_LD_ENVVARS_13 \
47 if (memcmp (envline, "ASSUME_KERNEL", 13) == 0) \
48 { \
49 _dl_osversion_init (&envline[14]); \
50 break; \
51 }
52
53#define DL_OSVERSION_INIT \
54 do { \
55 char *assume_kernel = getenv ("LD_ASSUME_KERNEL"); \
56 if (assume_kernel) \
57 _dl_osversion_init (assume_kernel); \
58 } while (0)
59
60#endif /* dl-librecon.h */
