19/* This function is used on maybe_enable_malloc_check (elf/dl-tunables.c)
20 and to avoid having to build/use multiple versions if stack protection
21 in enabled it is defined as inline. */
22static inline int
23__access_noerrno (const char *pathname, int mode)
24{
25 int res;
26 INTERNAL_SYSCALL_DECL (err);
27#ifdef __NR_access
28 res = INTERNAL_SYSCALL_CALL (access, err, pathname, mode);
29#else
30 res = INTERNAL_SYSCALL_CALL (faccessat, err, AT_FDCWD, pathname, mode);
31#endif
32 if (INTERNAL_SYSCALL_ERROR_P (res, err))
33 return INTERNAL_SYSCALL_ERRNO (res, err);
34 return 0;
35}
