19#include <errno.h>
20#include <sys/ustat.h>
21#include <sys/sysmacros.h>
22
23#include <sysdep.h>
24#include <sys/syscall.h>
25
26int
27ustat (dev_t dev, struct ustat *ubuf)
28{
29 unsigned long long int k_dev;
30
31 /* We must convert the value to dev_t type used by the kernel. */
32 k_dev = dev & ((1ULL << 32) - 1);
33 if (k_dev != dev)
34 return INLINE_SYSCALL_ERROR_RETURN_VALUE (EINVAL);
35
36 return INLINE_SYSCALL (ustat, 2, (unsigned int) k_dev, ubuf);
37}
