19#include <errno.h>
20#include <time.h>
21#include <unistd.h>
22#include "kernel-posix-cpu-timers.h"
23
24int
25__clock_getcpuclockid (pid_t pid, clockid_t *clock_id)
26{
27 /* The clockid_t value is a simple computation from the PID.
28 But we do a clock_getres call to validate it. */
29
30 const clockid_t pidclock = MAKE_PROCESS_CPUCLOCK (pid, CPUCLOCK_SCHED);
31
32 INTERNAL_SYSCALL_DECL (err);
33 int r = INTERNAL_SYSCALL (clock_getres, err, 2, pidclock, NULL);
34 if (!INTERNAL_SYSCALL_ERROR_P (r, err))
35 {
36 *clock_id = pidclock;
37 return 0;
38 }
39
40 if (INTERNAL_SYSCALL_ERRNO (r, err) == EINVAL)
41 {
42 /* The clock_getres system call checked the PID for us. */
43 return ESRCH;
44 }
45 else
46 return INTERNAL_SYSCALL_ERRNO (r, err);
47}
48weak_alias (__clock_getcpuclockid, clock_getcpuclockid)
