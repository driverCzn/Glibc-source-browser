19#include <errno.h>
20#include <fenv.h>
21#include <math.h>
22#include <math_private.h>
23#include <libm-alias-double.h>
24#include <stdbool.h>
25#include <stdint.h>
26
27#define BIAS 0x3ff
28#define MANT_DIG 53
29
30#if UNSIGNED
31# define RET_TYPE uintmax_t
32#else
33# define RET_TYPE intmax_t
34#endif
35
36#include <fromfp.h>
37
38RET_TYPE
39FUNC (double x, int round, unsigned int width)
40{
41 if (width > INTMAX_WIDTH)
42 width = INTMAX_WIDTH;
43 uint64_t ix;
44 EXTRACT_WORDS64 (ix, x);
45 bool negative = (ix & 0x8000000000000000ULL) != 0;
46 if (width == 0)
47 return fromfp_domain_error (negative, width);
48 ix &= 0x7fffffffffffffffULL;
49 if (ix == 0)
50 return 0;
51 int exponent = ix >> (MANT_DIG - 1);
52 exponent -= BIAS;
53 int max_exponent = fromfp_max_exponent (negative, width);
54 if (exponent > max_exponent)
55 return fromfp_domain_error (negative, width);
56
57 ix &= ((1ULL << (MANT_DIG - 1)) - 1);
58 ix |= 1ULL << (MANT_DIG - 1);
59 uintmax_t uret;
60 bool half_bit, more_bits;
61 if (exponent >= MANT_DIG - 1)
62 {
63 uret = ix;
64 uret <<= exponent - (MANT_DIG - 1);
65 half_bit = false;
66 more_bits = false;
67 }
68 else if (exponent >= -1)
69 {
70 uint64_t h = 1ULL << (MANT_DIG - 2 - exponent);
71 half_bit = (ix & h) != 0;
72 more_bits = (ix & (h - 1)) != 0;
73 uret = ix >> (MANT_DIG - 1 - exponent);
74 }
75 else
76 {
77 uret = 0;
78 half_bit = false;
79 more_bits = true;
80 }
81 return fromfp_round_and_return (negative, uret, half_bit, more_bits, round,
82 exponent, max_exponent, width);
83}
