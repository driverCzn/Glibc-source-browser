19#include <fix-int-fp-convert-zero.h>
20#include <math.h>
21#include <math_private.h>
22#include <libm-alias-float.h>
23#include <stdint.h>
24
25float
26__getpayloadf (const float *x)
27{
28 uint32_t ix;
29 GET_FLOAT_WORD (ix, *x);
30 ix &= 0x3fffff;
31 if (FIX_INT_FP_CONVERT_ZERO && ix == 0)
32 return 0.0f;
33 return (float) ix;
34}
