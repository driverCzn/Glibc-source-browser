20/*
21 * finitef(x) returns 1 is x is finite, else 0;
22 * no branching!
23 */
24
25#include <math.h>
26#include <math_private.h>
27
28#undef __finitef
29
30#ifndef FINITEF
31# define FINITEF __finitef
32#endif
33
34int FINITEF(float x)
35{
36 int32_t ix;
37 GET_FLOAT_WORD(ix,x);
38 return (int)((uint32_t)((ix&0x7f800000)-0x7f800000)>>31);
39}
40hidden_def (__finitef)
41weak_alias (__finitef, finitef)
