41float
42__ieee754_j1f(float x)
43{
44 float z, s,c,ss,cc,r,u,v,y;
45 int32_t hx,ix;
46
47 GET_FLOAT_WORD(hx,x);
48 ix = hx&0x7fffffff;
49 if(__builtin_expect(ix>=0x7f800000, 0)) return one/x;
50 y = fabsf(x);
51 if(ix >= 0x40000000) { /* |x| >= 2.0 */
52 __sincosf (y, &s, &c);
53 ss = -s-c;
54 cc = s-c;
55 if(ix<0x7f000000) { /* make sure y+y not overflow */
56 z = __cosf(y+y);
57 if ((s*c)>zero) cc = z/ss;
58 else ss = z/cc;
59 }
60 /*
61 * j1(x) = 1/sqrt(pi) * (P(1,x)*cc - Q(1,x)*ss) / sqrt(x)
62 * y1(x) = 1/sqrt(pi) * (P(1,x)*ss + Q(1,x)*cc) / sqrt(x)
63 */
64 if(ix>0x48000000) z = (invsqrtpi*cc)/__ieee754_sqrtf(y);
65 else {
66 u = ponef(y); v = qonef(y);
67 z = invsqrtpi*(u*cc-v*ss)/__ieee754_sqrtf(y);
68 }
69 if(hx<0) return -z;
70 else return z;
71 }
72 if(__builtin_expect(ix<0x32000000, 0)) { /* |x|<2**-27 */
73 if(huge+x>one) { /* inexact if x!=0 necessary */
74 float ret = math_narrow_eval ((float) 0.5 * x);
75 math_check_force_underflow (ret);
76 if (ret == 0 && x != 0)
77 __set_errno (ERANGE);
78 return ret;
79 }
80 }
81 z = x*x;
82 r = z*(r00+z*(r01+z*(r02+z*r03)));
83 s = one+z*(s01+z*(s02+z*(s03+z*(s04+z*s05))));
84 r *= x;
85 return(x*(float)0.5+r/s);
86}

104float
105__ieee754_y1f(float x)
106{
107 float z, s,c,ss,cc,u,v;
108 int32_t hx,ix;
109
110 GET_FLOAT_WORD(hx,x);
111 ix = 0x7fffffff&hx;
112 /* if Y1(NaN) is NaN, Y1(-inf) is NaN, Y1(inf) is 0 */
113 if(__builtin_expect(ix>=0x7f800000, 0)) return one/(x+x*x);
114 if(__builtin_expect(ix==0, 0))
115 return -1/zero; /* -inf and divide by zero exception. */
116 if(__builtin_expect(hx<0, 0)) return zero/(zero*x);
117 if(ix >= 0x40000000) { /* |x| >= 2.0 */
118 SET_RESTORE_ROUNDF (FE_TONEAREST);
119 __sincosf (x, &s, &c);
120 ss = -s-c;
121 cc = s-c;
122 if(ix<0x7f000000) { /* make sure x+x not overflow */
123 z = __cosf(x+x);
124 if ((s*c)>zero) cc = z/ss;
125 else ss = z/cc;
126 }
127 /* y1(x) = sqrt(2/(pi*x))*(p1(x)*sin(x0)+q1(x)*cos(x0))
128 * where x0 = x-3pi/4
129 * Better formula:
130 * cos(x0) = cos(x)cos(3pi/4)+sin(x)sin(3pi/4)
131 * = 1/sqrt(2) * (sin(x) - cos(x))
132 * sin(x0) = sin(x)cos(3pi/4)-cos(x)sin(3pi/4)
133 * = -1/sqrt(2) * (cos(x) + sin(x))
134 * To avoid cancellation, use
135 * sin(x) +- cos(x) = -cos(2x)/(sin(x) -+ cos(x))
136 * to compute the worse one.
137 */
138 if(ix>0x48000000) z = (invsqrtpi*ss)/__ieee754_sqrtf(x);
139 else {
140 u = ponef(x); v = qonef(x);
141 z = invsqrtpi*(u*ss+v*cc)/__ieee754_sqrtf(x);
142 }
143 return z;
144 }
145 if(__builtin_expect(ix<=0x33000000, 0)) { /* x < 2**-25 */
146 z = -tpi / x;
147 if (isinf (z))
148 __set_errno (ERANGE);
149 return z;
150 }
151 z = x*x;
152 u = U0[0]+z*(U0[1]+z*(U0[2]+z*(U0[3]+z*U0[4])));
153 v = one+z*(V0[0]+z*(V0[1]+z*(V0[2]+z*(V0[3]+z*V0[4]))));
154 return(x*(u/v) + tpi*(__ieee754_j1f(x)*__ieee754_logf(x)-one/x));
155}

232static float
233ponef(float x)
234{
235 const float *p,*q;
236 float z,r,s;
237 int32_t ix;
238 GET_FLOAT_WORD(ix,x);
239 ix &= 0x7fffffff;
240 /* ix >= 0x40000000 for all calls to this function. */
241 if(ix>=0x41000000) {p = pr8; q= ps8;}
242 else if(ix>=0x40f71c58){p = pr5; q= ps5;}
243 else if(ix>=0x4036db68){p = pr3; q= ps3;}
244 else {p = pr2; q= ps2;}
245 z = one/(x*x);
246 r = p[0]+z*(p[1]+z*(p[2]+z*(p[3]+z*(p[4]+z*p[5]))));
247 s = one+z*(q[0]+z*(q[1]+z*(q[2]+z*(q[3]+z*q[4]))));
248 return one+ r/s;
249}

330static float
331qonef(float x)
332{
333 const float *p,*q;
334 float s,r,z;
335 int32_t ix;
336 GET_FLOAT_WORD(ix,x);
337 ix &= 0x7fffffff;
338 /* ix >= 0x40000000 for all calls to this function. */
339 if(ix>=0x40200000) {p = qr8; q= qs8;}
340 else if(ix>=0x40f71c58){p = qr5; q= qs5;}
341 else if(ix>=0x4036db68){p = qr3; q= qs3;}
342 else {p = qr2; q= qs2;}
343 z = one/(x*x);
344 r = p[0]+z*(p[1]+z*(p[2]+z*(p[3]+z*(p[4]+z*p[5]))));
345 s = one+z*(q[0]+z*(q[1]+z*(q[2]+z*(q[3]+z*(q[4]+z*q[5])))));
346 return ((float).375 + r/s)/x;
347}
