19#include <math.h>
20#include <math_private.h>
21#include <nan-high-order-bit.h>
22
23int
24__issignalingl (_Float128 x)
25{
26 uint64_t hxi, lxi __attribute__ ((unused));
27 GET_LDOUBLE_WORDS64 (hxi, lxi, x);
28#if HIGH_ORDER_BIT_IS_SET_FOR_SNAN
29 /* We only have to care about the high-order bit of x's significand, because
30 having it set (sNaN) already makes the significand different from that
31 used to designate infinity. */
32 return ((hxi & UINT64_C (0x7fff800000000000))
33 == UINT64_C (0x7fff800000000000));
34#else
35 /* To keep the following comparison simple, toggle the quiet/signaling bit,
36 so that it is set for sNaNs. This is inverse to IEEE 754-2008 (as well as
37 common practice for IEEE 754-1985). */
38 hxi ^= UINT64_C (0x0000800000000000);
39 /* If lxi != 0, then set any suitable bit of the significand in hxi. */
40 hxi |= (lxi | -lxi) >> 63;
41 /* We have to compare for greater (instead of greater or equal), because x's
42 significand being all-zero designates infinity not NaN. */
43 return (hxi & UINT64_C (0x7fffffffffffffff)) > UINT64_C (0x7fff800000000000);
44#endif
45}
