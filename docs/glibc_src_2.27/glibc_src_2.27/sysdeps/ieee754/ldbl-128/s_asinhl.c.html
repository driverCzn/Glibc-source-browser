21/* asinhl(x)
22 * Method :
23 * Based on
24 * asinhl(x) = signl(x) * logl [ |x| + sqrtl(x*x+1) ]
25 * we have
26 * asinhl(x) := x if 1+x*x=1,
27 * := signl(x)*(logl(x)+ln2)) for large |x|, else
28 * := signl(x)*logl(2|x|+1/(|x|+sqrtl(x*x+1))) if|x|>2, else
29 * := signl(x)*log1pl(|x| + x^2/(1 + sqrtl(1+x^2)))
30 */
31
32#include <float.h>
33#include <math.h>
34#include <math_private.h>
35#include <libm-alias-ldouble.h>
36
37static const _Float128
38 one = 1,
39 ln2 = L(6.931471805599453094172321214581765681e-1),
40 huge = L(1.0e+4900);
41
42_Float128
43__asinhl (_Float128 x)
44{
45 _Float128 t, w;
46 int32_t ix, sign;
47 ieee854_long_double_shape_type u;
48
49 u.value = x;
50 sign = u.parts32.w0;
51 ix = sign & 0x7fffffff;
52 if (ix == 0x7fff0000)
53 return x + x; /* x is inf or NaN */
54 if (ix < 0x3fc70000)
55 { /* |x| < 2^ -56 */
56 math_check_force_underflow (x);
57 if (huge + x > one)
58 return x; /* return x inexact except 0 */
59 }
60 u.parts32.w0 = ix;
61 if (ix > 0x40350000)
62 { /* |x| > 2 ^ 54 */
63 w = __ieee754_logl (u.value) + ln2;
64 }
65 else if (ix >0x40000000)
66 { /* 2^ 54 > |x| > 2.0 */
67 t = u.value;
68 w = __ieee754_logl (2.0 * t + one / (__ieee754_sqrtl (x * x + one) + t));
69 }
70 else
71 { /* 2.0 > |x| > 2 ^ -56 */
72 t = x * x;
73 w = __log1pl (u.value + t / (one + __ieee754_sqrtl (one + t)));
74 }
75 if (sign & 0x80000000)
76 return -w;
77 else
78 return w;
79}
