27_Float128
28__roundl (_Float128 x)
29{
30 int32_t j0;
31 uint64_t i1, i0;
32
33 GET_LDOUBLE_WORDS64 (i0, i1, x);
34 j0 = ((i0 >> 48) & 0x7fff) - 0x3fff;
35 if (j0 < 48)
36 {
37 if (j0 < 0)
38 {
39 i0 &= 0x8000000000000000ULL;
40 if (j0 == -1)
41 i0 |= 0x3fff000000000000LL;
42 i1 = 0;
43 }
44 else
45 {
46 uint64_t i = 0x0000ffffffffffffLL >> j0;
47 if (((i0 & i) | i1) == 0)
48 /* X is integral. */
49 return x;
50
51 i0 += 0x0000800000000000LL >> j0;
52 i0 &= ~i;
53 i1 = 0;
54 }
55 }
56 else if (j0 > 111)
57 {
58 if (j0 == 0x4000)
59 /* Inf or NaN. */
60 return x + x;
61 else
62 return x;
63 }
64 else
65 {
66 uint64_t i = -1ULL >> (j0 - 48);
67 if ((i1 & i) == 0)
68 /* X is integral. */
69 return x;
70
71 uint64_t j = i1 + (1LL << (111 - j0));
72 if (j < i1)
73 i0 += 1;
74 i1 = j;
75 i1 &= ~i;
76 }
77
78 SET_LDOUBLE_WORDS64 (x, i0, i1);
79 return x;
80}
