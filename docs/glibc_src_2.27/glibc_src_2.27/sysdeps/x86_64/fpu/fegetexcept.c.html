21
22int
23fegetexcept (void)
24{
25 unsigned short int exc;
26
27 /* Get the current control word. */
28 __asm__ ("fstcw %0" : "=m" (*&exc));
29
30 return (~exc) & FE_ALL_EXCEPT;
31}
