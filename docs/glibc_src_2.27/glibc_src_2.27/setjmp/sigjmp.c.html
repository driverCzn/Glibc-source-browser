18#include <stddef.h>
19#include <setjmpP.h>
20#include <signal.h>
21
22/* This function is called by the `sigsetjmp' macro
23 before doing a `__setjmp' on ENV[0].__jmpbuf.
24 Always return zero. */
25
26int
27__sigjmp_save (sigjmp_buf env, int savemask)
28{
29 env[0].__mask_was_saved = (savemask &&
30 __sigprocmask (SIG_BLOCK, (sigset_t *) NULL,
31 (sigset_t *) &env[0].__saved_mask) == 0);
32
33 return 0;
34}
