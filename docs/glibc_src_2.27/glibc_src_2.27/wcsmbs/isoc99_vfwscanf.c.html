18#include <libioP.h>
19#include <stdio.h>
20#include <wchar.h>
21
22/* Read formatted input from STREAM according to the format string FORMAT. */
23/* VARARGS2 */
24int
25__isoc99_vfwscanf (FILE *stream, const wchar_t *format, _IO_va_list args)
26{
27 int done;
28
29 _IO_acquire_lock_clear_flags2 (stream);
30 stream->_flags2 |= _IO_FLAGS2_SCANF_STD;
31 done = _IO_vfwscanf (stream, format, args, NULL);
32 _IO_release_lock (stream);
33 return done;
34}
