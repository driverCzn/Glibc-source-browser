21/* Determine number of column positions required for first N wide
22 characters (or fewer if S ends before this) in S. */
23int
24wcswidth (const wchar_t *s, size_t n)
25{
26 int result = 0;
27
28 while (n-- > 0 && *s != L'\0')
29 {
30 int now = internal_wcwidth (*s);
31 if (now == -1)
32 return -1;
33 result += now;
34 ++s;
35 }
36
37 return result;
38}
