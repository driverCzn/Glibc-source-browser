25/* Data structure to contain the action information. */
26struct __spawn_action
27{
28 enum
29 {
30 spawn_do_close,
31 spawn_do_dup2,
32 spawn_do_open
33 } tag;
34
35 union
36 {
37 struct
38 {
39 int fd;
40 } close_action;
41 struct
42 {
43 int fd;
44 int newfd;
45 } dup2_action;
46 struct
47 {
48 int fd;
49 char *path;
50 int oflag;
51 mode_t mode;
52 } open_action;
53 } action;
54};
