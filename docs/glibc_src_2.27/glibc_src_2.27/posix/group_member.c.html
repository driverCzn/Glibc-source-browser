19#include <sys/types.h>
20#include <unistd.h>
21#include <stdlib.h>
22#include <limits.h>
23
24#ifndef NGROUPS_MAX
25#define NGROUPS_MAX 16 /* First guess. */
26#endif
27
28int
29__group_member (gid_t gid)
30{
31 int n, size;
32 gid_t *groups;
33
34 size = NGROUPS_MAX;
35 do
36 {
37 groups = __alloca (size * sizeof *groups);
38 n = __getgroups (size, groups);
39 size *= 2;
40 }
41 while (n == size / 2);
42
43 while (n-- > 0)
44 if (groups[n] == gid)
45 return 1;
46
47 return 0;
48}
49weak_alias (__group_member, group_member)
