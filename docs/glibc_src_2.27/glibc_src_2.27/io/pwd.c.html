18#include <mcheck.h>
19#include <stdio.h>
20#include <stdlib.h>
21#include <unistd.h>
22
23int
24main (void)
25{
26 char *dir;
27
28 /* Let this program be used for debugging. */
29 mtrace ();
30
31 dir = getcwd ((char *) NULL, 0);
32
33 if (dir == NULL)
34 perror ("getcwd");
35 else
36 {
37 puts (dir);
38 free (dir);
39 }
40
41 return (dir == NULL ? EXIT_FAILURE : EXIT_SUCCESS);
42}
