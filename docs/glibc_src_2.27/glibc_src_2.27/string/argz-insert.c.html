24/* Insert ENTRY into ARGZ & ARGZ_LEN before BEFORE, which should be an
25 existing entry in ARGZ; if BEFORE is NULL, ENTRY is appended to the end.
26 Since ARGZ's first entry is the same as ARGZ, argz_insert (ARGZ, ARGZ_LEN,
27 ARGZ, ENTRY) will insert ENTRY at the beginning of ARGZ. If BEFORE is not
28 in ARGZ, EINVAL is returned, else if memory can't be allocated for the new
29 ARGZ, ENOMEM is returned, else 0. */
30error_t
31__argz_insert (char **argz, size_t *argz_len, char *before, const char *entry)
32{
33 if (! before)
34 return __argz_add (argz, argz_len, entry);
35
36 if (before < *argz || before >= *argz + *argz_len)
37 return EINVAL;
38
39 if (before > *argz)
40 /* Make sure before is actually the beginning of an entry. */
41 while (before[-1])
42 before--;
43
44 {
45 size_t after_before = *argz_len - (before - *argz);
46 size_t entry_len = strlen (entry) + 1;
47 size_t new_argz_len = *argz_len + entry_len;
48 char *new_argz = realloc (*argz, new_argz_len);
49
50 if (new_argz)
51 {
52 before = new_argz + (before - *argz);
53 memmove (before + entry_len, before, after_before);
54 memmove (before, entry, entry_len);
55 *argz = new_argz;
56 *argz_len = new_argz_len;
57 return 0;
58 }
59 else
60 return ENOMEM;
61 }
62}
