37/* Duplicate S, returning an identical malloc'd string. */
38char *
39__strdup (const char *s)
40{
41 size_t len = strlen (s) + 1;
42 void *new = malloc (len);
43
44 if (new == NULL)
45 return NULL;
46
47 return (char *) memcpy (new, s, len);
48}
