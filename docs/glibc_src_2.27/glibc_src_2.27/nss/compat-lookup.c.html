25/* On i386, the function calling convention changed from the standard
26 ABI calling convention to three register parameters in glibc 2.8.
27 The following error-returning stub happens to be compatible with
28 glibc 2.7 and earlier and glibc 2.8 and later, even on i386. */
29int
30attribute_compat_text_section
31__nss_passwd_lookup (service_user **ni, const char *fct_name, void **fctp)
32{
33 __set_errno (ENOSYS);
34 return -1;
35}
36compat_symbol (libc, __nss_passwd_lookup, __nss_passwd_lookup, GLIBC_2_0);
37strong_alias (__nss_passwd_lookup, __nss_group_lookup)
38compat_symbol (libc, __nss_group_lookup, __nss_group_lookup, GLIBC_2_0);
39strong_alias (__nss_passwd_lookup, __nss_hosts_lookup)
40compat_symbol (libc, __nss_hosts_lookup, __nss_hosts_lookup, GLIBC_2_0);
