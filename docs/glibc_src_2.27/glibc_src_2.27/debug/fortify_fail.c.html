18#include <stdio.h>
19#include <stdlib.h>
20#include <stdbool.h>
21
22
23extern char **__libc_argv attribute_hidden;
24
25void
26__attribute__ ((noreturn))
27__fortify_fail_abort (_Bool need_backtrace, const char *msg)
28{
29 /* The loop is added only to keep gcc happy. Don't pass down
30 __libc_argv[0] if we aren't doing backtrace since __libc_argv[0]
31 may point to the corrupted stack. */
32 while (1)
33 __libc_message (need_backtrace ? (do_abort | do_backtrace) : do_abort,
34 "*** %s ***: %s terminated\n",
35 msg,
36 (need_backtrace && __libc_argv[0] != NULL
37 ? __libc_argv[0] : "<unknown>"));
38}
39
40void
41__attribute__ ((noreturn))
42__fortify_fail (const char *msg)
43{
44 __fortify_fail_abort (true, msg);
45}
46
47libc_hidden_def (__fortify_fail)
48libc_hidden_def (__fortify_fail_abort)
