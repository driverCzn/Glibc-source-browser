18#include <langinfo.h>
19#include <locale.h>

24
25size_t
26__wcrtomb_chk (char *s, wchar_t wchar, mbstate_t *ps, size_t buflen)
27{
28 /* We do not have to implement the full wctomb semantics since we
29 know that S cannot be NULL when we come here. */
30 if (buflen < MB_CUR_MAX)
31 __chk_fail ();
32
33 return __wcrtomb (s, wchar, ps);
34}
