18#include <libioP.h>
19#include <stdarg.h>
20#include <stdio.h>
21
22
23/* Write formatted output from FORMAT to a string which is
24 allocated with malloc and stored in *STRING_PTR. */
25int
26__asprintf_chk (char **result_ptr, int flags, const char *format, ...)
27{
28 va_list arg;
29 int done;
30
31 va_start (arg, format);
32 done = __vasprintf_chk (result_ptr, flags, format, arg);
33 va_end (arg);
34
35 return done;
36}
