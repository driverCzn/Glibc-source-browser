20#include <execinfo.h>
21#include <string.h>
22#include <sys/uio.h>
23
24#include <_itoa.h>
25#include <ldsodefs.h>
26
27#if __ELF_NATIVE_CLASS == 32
28# define WORD_WIDTH 8
29#else
30/* We assume 64bits. */
31# define WORD_WIDTH 16
32#endif
33
34
35void
36__backtrace_symbols_fd (void *const *array, int size, int fd)
37{
38 struct iovec iov[9];
39 int cnt;
40
41 for (cnt = 0; cnt < size; ++cnt)
42 {
43 char buf[WORD_WIDTH];
44 char buf2[WORD_WIDTH];
45 Dl_info info;
46 struct link_map *map;
47 size_t last = 0;
48
49 if (_dl_addr (array[cnt], &info, &map, NULL)
50 && info.dli_fname != NULL && info.dli_fname[0] != '\0')
51 {
52 /* Name of the file. */
53 iov[0].iov_base = (void *) info.dli_fname;
54 iov[0].iov_len = strlen (info.dli_fname);
55 last = 1;
56
57 if (info.dli_sname != NULL || map->l_addr != 0)
58 {
59 size_t diff;
60
61 iov[last].iov_base = (void *) "(";
62 iov[last].iov_len = 1;
63 ++last;
64
65 if (info.dli_sname != NULL)
66 {
67 /* We have a symbol name. */
68 iov[last].iov_base = (void *) info.dli_sname;
69 iov[last].iov_len = strlen (info.dli_sname);
70 ++last;
71 }
72 else
73 /* We have no symbol, so describe it as relative to the file.
74 The load bias is more useful to the user than the load
75 address. The use of these addresses is to calculate an
76 address in the ELF file, so its prelinked bias is not
77 something we want to subtract out. */
78 info.dli_saddr = (void *) map->l_addr;
79
80 if (array[cnt] >= (void *) info.dli_saddr)
81 {
82 iov[last].iov_base = (void *) "+0x";
83 diff = array[cnt] - info.dli_saddr;
84 }
85 else
86 {
87 iov[last].iov_base = (void *) "-0x";
88 diff = info.dli_saddr - array[cnt];
89 }
90 iov[last].iov_len = 3;
91 ++last;
92
93 iov[last].iov_base = _itoa_word ((unsigned long int) diff,
94 &buf2[WORD_WIDTH], 16, 0);
95 iov[last].iov_len = (&buf2[WORD_WIDTH]
96 - (char *) iov[last].iov_base);
97 ++last;
98
99 iov[last].iov_base = (void *) ")";
100 iov[last].iov_len = 1;
101 ++last;
102 }
103 }
104
105 iov[last].iov_base = (void *) "[0x";
106 iov[last].iov_len = 3;
107 ++last;
108
109 iov[last].iov_base = _itoa_word ((unsigned long int) array[cnt],
110 &buf[WORD_WIDTH], 16, 0);
111 iov[last].iov_len = &buf[WORD_WIDTH] - (char *) iov[last].iov_base;
112 ++last;
113
114 iov[last].iov_base = (void *) "]\n";
115 iov[last].iov_len = 2;
116 ++last;
117
118 __writev (fd, iov, last);
119 }
120}
121weak_alias (__backtrace_symbols_fd, backtrace_symbols_fd)
122libc_hidden_def (__backtrace_symbols_fd)
