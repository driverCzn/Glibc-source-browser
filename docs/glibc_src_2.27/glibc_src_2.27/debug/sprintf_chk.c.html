18#include <libioP.h>
19#include <stdarg.h>
20#include <stdio.h>
21
22/* Write formatted output into S, according to the format string FORMAT. */
23/* VARARGS4 */
24int
25___sprintf_chk (char *s, int flags, size_t slen, const char *format, ...)
26{
27 va_list arg;
28 int done;
29
30 va_start (arg, format);
31 done = __vsprintf_chk (s, flags, slen, format, arg);
32 va_end (arg);
33
34 return done;
35}
36ldbl_strong_alias (___sprintf_chk, __sprintf_chk)
