18#include "libioP.h"
19#include <wchar.h>
20#include <sys/param.h>
21
22wchar_t *
23__fgetws_chk (wchar_t *buf, size_t size, int n, _IO_FILE *fp)
24{
25 _IO_size_t count;
26 wchar_t *result;
27 int old_error;
28 CHECK_FILE (fp, NULL);
29 if (n <= 0)
30 return NULL;
31 _IO_acquire_lock (fp);
32 /* This is very tricky since a file descriptor may be in the
33 non-blocking mode. The error flag doesn't mean much in this
34 case. We return an error only when there is a new error. */
35 old_error = fp->_IO_file_flags & _IO_ERR_SEEN;
36 fp->_IO_file_flags &= ~_IO_ERR_SEEN;
37 count = _IO_getwline (fp, buf, MIN ((size_t) n - 1, size), L'\n', 1);
38 /* If we read in some bytes and errno is EAGAIN, that error will
39 be reported for next read. */
40 if (count == 0 || (_IO_ferror_unlocked (fp) && errno != EAGAIN))
41 result = NULL;
42 else if (count >= size)
43 __chk_fail ();
44 else
45 {
46 buf[count] = '\0';
47 result = buf;
48 }
49 fp->_IO_file_flags |= old_error;
50 _IO_release_lock (fp);
51 return result;
52}
