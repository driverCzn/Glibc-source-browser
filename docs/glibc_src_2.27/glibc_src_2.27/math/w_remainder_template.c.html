19/* Only build wrappers from the templates for the types that define the macro
20 below. This macro is set in math-type-macros-<type>.h in sysdeps/generic
21 for each floating-point type. */
22#if __USE_WRAPPER_TEMPLATE
23
24# include <errno.h>
25# include <fenv.h>
26# include <math.h>
27# include <math_private.h>
28
29FLOAT
30M_DECL_FUNC (__remainder) (FLOAT x, FLOAT y)
31{
32 if (__glibc_unlikely (isinf (x) || y == 0) && !isnan (y) && !isnan (x))
33 /* Domain error: remainder(+-Inf,y) or remainder(x,0).
34 If x or y are nan, these conditions should not be considered. */
35 __set_errno (EDOM);
36 return M_SUF (__ieee754_remainder) (x, y);
37}
38declare_mgen_alias (__remainder, remainder)
39
40#endif /* __USE_WRAPPER_TEMPLATE. */
