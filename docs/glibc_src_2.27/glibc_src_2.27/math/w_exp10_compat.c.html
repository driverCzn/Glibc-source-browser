20/*
21 * wrapper exp10(x)
22 */
23
24#include <math.h>
25#include <math_private.h>
26#include <math-svid-compat.h>
27#include <libm-alias-double.h>
28
29#if LIBM_SVID_COMPAT
30double
31__exp10 (double x)
32{
33 double z = __ieee754_exp10 (x);
34 if (__builtin_expect (!isfinite (z) || z == 0, 0)
35 && isfinite (x) && _LIB_VERSION != _IEEE_)
36 /* exp10 overflow (46) if x > 0, underflow (47) if x < 0. */
37 return __kernel_standard (x, x, 46 + !!signbit (x));
38
39 return z;
40}
41libm_alias_double (__exp10, exp10)
42# if SHLIB_COMPAT (libm, GLIBC_2_1, GLIBC_2_27)
43strong_alias (__exp10, __pow10)
44compat_symbol (libm, __pow10, pow10, GLIBC_2_1);
45# endif
46# ifdef NO_LONG_DOUBLE
47# if SHLIB_COMPAT (libm, GLIBC_2_1, GLIBC_2_27)
48strong_alias (__exp10l, __pow10l)
49compat_symbol (libm, __pow10l, pow10l, GLIBC_2_1);
50# endif
51# endif
52#endif
