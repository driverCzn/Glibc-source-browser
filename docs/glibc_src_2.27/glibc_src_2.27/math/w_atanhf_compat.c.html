25#if LIBM_SVID_COMPAT
26/* wrapper atanhf */
27float
28__atanhf (float x)
29{
30 if (__builtin_expect (isgreaterequal (fabsf (x), 1.0f), 0)
31 && _LIB_VERSION != _IEEE_)
32 return __kernel_standard_f (x, x,
33 fabsf (x) > 1.0f
34 ? 130 /* atanh(|x|>1) */
35 : 131); /* atanh(|x|==1) */
36
37 return __ieee754_atanhf (x);
38}
39libm_alias_float (__atanh, atanh)
