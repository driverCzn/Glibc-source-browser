25#if LIBM_SVID_COMPAT
26/* wrapper powl */
27long double
28__powl (long double x, long double y)
29{
30 long double z = __ieee754_powl (x, y);
31 if (__glibc_unlikely (!isfinite (z)))
32 {
33 if (_LIB_VERSION != _IEEE_)
34 {
35 if (isfinite (x) && isfinite (y))
36 {
37 if (isnan (z))
38 /* pow neg**non-int */
39 return __kernel_standard_l (x, y, 224);
40 else if (x == 0.0L && y < 0.0L)
41 {
42 if (signbit (x) && signbit (z))
43 /* pow(-0.0,negative) */
44 return __kernel_standard_l (x, y, 223);
45 else
46 /* pow(+0.0,negative) */
47 return __kernel_standard_l (x, y, 243);
48 }
49 else
50 /* pow overflow */
51 return __kernel_standard_l (x, y, 221);
52 }
53 }
54 }
55 else if (__builtin_expect (z == 0.0L, 0)
56 && isfinite (x) && x != 0 && isfinite (y)
57 && _LIB_VERSION != _IEEE_)
58 /* pow underflow */
59 return __kernel_standard_l (x, y, 222);
60
61 return z;
62}
63libm_alias_ldouble (__pow, pow)
64#endif
