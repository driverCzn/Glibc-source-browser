22#include <libm-alias-double.h>

24#if LIBM_SVID_COMPAT
25/* wrapper exp */
26double
27__exp (double x)
28{
29 double z = __ieee754_exp (x);
30 if (__builtin_expect (!isfinite (z) || z == 0, 0)
31 && isfinite (x) && _LIB_VERSION != _IEEE_)
32 return __kernel_standard (x, x, 6 + !!signbit (x));
33
34 return z;
35}
36hidden_def (__exp)
37libm_alias_double (__exp, exp)
