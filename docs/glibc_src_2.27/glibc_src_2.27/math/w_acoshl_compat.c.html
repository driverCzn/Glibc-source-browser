25#if LIBM_SVID_COMPAT
26/* wrapper acosl */
27long double
28__acoshl (long double x)
29{
30 if (__builtin_expect (isless (x, 1.0L), 0) && _LIB_VERSION != _IEEE_)
31 /* acosh(x<1) */
32 return __kernel_standard_l (x, x, 229);
33
34 return __ieee754_acoshl (x);
35}
