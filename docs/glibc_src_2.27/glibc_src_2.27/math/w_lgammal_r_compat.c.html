19 */

21#include <math.h>
22#include <math_private.h>
23#include <math-svid-compat.h>
24#include <libm-alias-ldouble.h>

27#if LIBM_SVID_COMPAT
28long double
29__lgammal_r(long double x, int *signgamp)
30{
31 long double y = __ieee754_lgammal_r(x,signgamp);
32 if(__builtin_expect(!isfinite(y), 0)
33 && isfinite(x) && _LIB_VERSION != _IEEE_)
34 return __kernel_standard(x, x,
35 __floorl(x)==x&&x<=0.0
36 ? 215 /* lgamma pole */
37 : 214); /* lgamma overflow */
38
39 return y;
40}
41libm_alias_ldouble_r (__lgamma, lgamma, _r)
42#endif
