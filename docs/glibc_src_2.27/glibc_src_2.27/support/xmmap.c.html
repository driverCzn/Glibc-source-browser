23void *
24xmmap (void *addr, size_t length, int prot, int flags, int fd)
25{
26 void *result = mmap (addr, length, prot, flags, fd, 0);
27 if (result == MAP_FAILED)
28 FAIL_EXIT1 ("mmap of %zu bytes, prot=0x%x, flags=0x%x: %m",
29 length, prot, flags);
30 return result;
31}
