37(,
38 STRING_FIELD (result->sp_namp, ISCOLON, 0);
39 if (line[0] == '\0'
40 && (result->sp_namp[0] == '+' || result->sp_namp[0] == '-'))
41 {
42 result->sp_pwdp = NULL;

88/* Read one shadow entry from the given stream. */
89int
90__sgetspent_r (const char *string, struct spwd *resbuf, char *buffer,
91 size_t buflen, struct spwd **result)
92{
93 buffer[buflen - 1] = '\0';
94 char *sp = strncpy (buffer, string, buflen);
95 if (buffer[buflen - 1] != '\0')
96 return ERANGE;
97
98 int parse_result = parse_line (sp, resbuf, NULL, 0, &errno);
99 *result = parse_result > 0 ? resbuf : NULL;
100
101 return *result == NULL ? errno : 0;
102}
