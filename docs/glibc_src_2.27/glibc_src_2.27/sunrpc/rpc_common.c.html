40/*
41 * This file should only contain common data (global data) that is exported
42 * by public interfaces
43 */
44/* We are very tricky here. We want to have _null_auth in a read-only
45 section but we cannot add const to the type because this isn't how
46 the variable is declared. So we use the section attribute. */
47struct opaque_auth _null_auth __attribute__ ((nocommon));
48libc_hidden_nolink_sunrpc (_null_auth, GLIBC_2_0)
49
50/* The variables need the nocommon attribute, so that it is possible
51 to create aliases and specify symbol versions. */
52fd_set svc_fdset __attribute__ ((nocommon));
53libc_hidden_nolink_sunrpc (svc_fdset, GLIBC_2_0)
54struct rpc_createerr rpc_createerr __attribute__ ((nocommon));
55libc_hidden_nolink_sunrpc (rpc_createerr, GLIBC_2_0)
56struct pollfd *svc_pollfd __attribute__ ((nocommon));
57libc_hidden_nolink_sunrpc (svc_pollfd, GLIBC_2_2)
58int svc_max_pollfd __attribute__ ((nocommon));
59libc_hidden_nolink_sunrpc (svc_max_pollfd, GLIBC_2_2)
