23/* Write formatted output to STREAM from the format string FORMAT. */
24/* VARARGS2 */
25int
26__fprintf (FILE *stream, const char *format, ...)
27{
28 va_list arg;
29 int done;
30
31 va_start (arg, format);
32 done = vfprintf (stream, format, arg);
33 va_end (arg);
34
35 return done;
36}

38ldbl_strong_alias (__fprintf, fprintf)

40/* We define the function with the real name here. But deep down in
41 libio the original function _IO_fprintf is also needed. So make
42 an alias. */
