18/* We need to avoid the header declaration of versionsort64, because
19 the types don't match versionsort and then the compiler will
20 complain about the mismatch when we do the alias below. */
21#define versionsort64 __renamed_versionsort64
22
23#include <dirent.h>
24
25#undef versionsort64
26
27#include <string.h>
28
29int
30versionsort (const struct dirent **a, const struct dirent **b)
31{
32 return __strverscmp ((*a)->d_name, (*b)->d_name);
33}
34
35#ifdef _DIRENT_MATCHES_DIRENT64
36weak_alias (versionsort, versionsort64)
37#endif
