19#include <stdlib.h>
20#include <limits.h>
21
22int
23__srand48_r (long int seedval, struct drand48_data *buffer)
24{
25 /* The standards say we only have 32 bits. */
26 if (sizeof (long int) > 4)
27 seedval &= 0xffffffffl;
28
29 buffer->__x[2] = seedval >> 16;
30 buffer->__x[1] = seedval & 0xffffl;
31 buffer->__x[0] = 0x330e;
32
33 buffer->__a = 0x5deece66dull;
34 buffer->__c = 0xb;
35 buffer->__init = 1;
36
37 return 0;
38}
39weak_alias (__srand48_r, srand48_r)
