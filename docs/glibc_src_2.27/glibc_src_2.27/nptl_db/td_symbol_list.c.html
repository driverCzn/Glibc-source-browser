20#include <assert.h>
21#include "thread_dbP.h"
22
23static const char *symbol_list_arr[] =
24{
25# define DB_LOOKUP_NAME(idx, name) [idx] = STRINGIFY (name),
26# define DB_LOOKUP_NAME_TH_UNIQUE(idx, name) [idx] = STRINGIFY (name),
27# include "db-symbols.h"
28# undef DB_LOOKUP_NAME
29# undef DB_LOOKUP_NAME_TH_UNIQUE
30
31 [SYM_NUM_MESSAGES] = NULL
32};
33
34
35const char **
36td_symbol_list (void)
37{
38 return symbol_list_arr;
39}
40
41
42ps_err_e
43td_mod_lookup (struct ps_prochandle *ps, const char *mod,
44 int idx, psaddr_t *sym_addr)
45{
46 ps_err_e result;
47 assert (idx >= 0 && idx < SYM_NUM_MESSAGES);
48 result = ps_pglobal_lookup (ps, mod, symbol_list_arr[idx], sym_addr);
49
50 return result;
51}
