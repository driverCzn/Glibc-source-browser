18#include <errno.h>
19#include <signal.h>
20#include <string.h>
21
22/* Clear all signals from SET. */
23int
24sigemptyset (sigset_t *set)
25{
26 if (set == NULL)
27 {
28 __set_errno (EINVAL);
29 return -1;
30 }
31
32 memset (set, 0, sizeof (sigset_t));
33
34 return 0;
35}
