18/* Put the name of the current YP domain in no more than LEN bytes of NAME.
19 The result is null-terminated if LEN is large enough for the full
20 name and the terminator. */
21
22#include <errno.h>
23#include <unistd.h>
24#include <sys/param.h>
25#include <sys/utsname.h>
26#include <string.h>
27
28#if _UTSNAME_DOMAIN_LENGTH
29/* The `uname' information includes the domain name. */
30
31int
32getdomainname (char *name, size_t len)
33{
34 struct utsname u;
35 size_t u_len;
36
37 if (uname (&u) < 0)
38 return -1;
39
40 u_len = strlen (u.domainname);
41 memcpy (name, u.domainname, MIN (u_len + 1, len));
42 return 0;
43}
44
45#else
46
47int
48getdomainname (char *name, size_t len)
49{
50 __set_errno (ENOSYS);
51 return -1;
52}
53
54stub_warning (getdomainname)
55
56#endif
57
58libc_hidden_def (getdomainname)
