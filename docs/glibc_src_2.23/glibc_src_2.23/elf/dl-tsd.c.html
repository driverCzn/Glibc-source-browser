26/* _dl_error_catch_tsd points to this for the single-threaded case.
27 It's reset by the thread library for multithreaded programs
28 if we're not using __thread. */
29void ** __attribute__ ((const))
30_dl_initial_error_catch_tsd (void)
31{
32 static __thread void *data;
33 return &data;
34}
35void **(*_dl_error_catch_tsd) (void) __attribute__ ((const))
36 = &_dl_initial_error_catch_tsd;
37
38# else
39
40/* libpthread sets _dl_error_catch_tsd to point to this function.
41 We define it here instead of in libpthread so that it doesn't
42 need to have a TLS segment of its own just for this one pointer. */
43
44void ** __attribute__ ((const))
45__libc_dl_error_tsd (void)
46{
47 static __thread void *data attribute_tls_model_ie;
48 return &data;
49}
50
51# endif /* SHARED */

53#endif /* _LIBC_REENTRANT */
