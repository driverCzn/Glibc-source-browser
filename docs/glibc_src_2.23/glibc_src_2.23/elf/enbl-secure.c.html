19/* This file is used in the static libc. For the shared library,
20 dl-sysdep.c defines and initializes __libc_enable_secure. */
21
22#include <unistd.h>
23#include <libc-internal.h>
24
25/* If nonzero __libc_enable_secure is already set. */
26int __libc_enable_secure_decided;
27/* Safest assumption, if somehow the initializer isn't run. */
28int __libc_enable_secure = 1;
29
30void
31__libc_init_secure (void)
32{
33 if (__libc_enable_secure_decided == 0)
34 __libc_enable_secure = (__geteuid () != __getuid ()
35 || __getegid () != __getgid ());
36}
