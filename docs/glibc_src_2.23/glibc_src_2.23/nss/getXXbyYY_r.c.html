36|* *|
37|* LOOKUP_TYPE - the return type of the function *|
38|* *|
39|* FUNCTION_NAME - name of the non-reentrant function *|
40|* *|
41|* DATABASE_NAME - name of the database the function accesses *|
42|* (e.g., host, services, ...) *|
43|* *|
44|* ADD_PARAMS - additional parameters, can vary in number *|
45|* *|
46|* ADD_VARIABLES - names of additional parameters *|
47|* *|
48|* Optionally the following vars can be defined: *|
49|* *|
50|* EXTRA_PARAMS - optional parameters, can vary in number *|

139/* The lookup function for the first entry of this service. */
140extern int DB_LOOKUP_FCT (service_user **nip, const char *name,
141 const char *name2, void **fctp)
142 internal_function;
143libc_hidden_proto (DB_LOOKUP_FCT)
144
145
146int
147INTERNAL (REENTRANT_NAME) (ADD_PARAMS, LOOKUP_TYPE *resbuf, char *buffer,
148 size_t buflen, LOOKUP_TYPE **result H_ERRNO_PARM
149 EXTRA_PARAMS)
150{
151 static bool startp_initialized;
152 static service_user *startp;
153 static lookup_function start_fct;
154 service_user *nip;
155 union
156 {
157 lookup_function l;
158 void *ptr;
159 } fct;
160
161 int no_more;
162 enum nss_status status = NSS_STATUS_UNAVAIL;
163#ifdef USE_NSCD
164 int nscd_status;
165#endif
166#ifdef NEED_H_ERRNO
167 bool any_service = false;
168#endif
169
170#ifdef PREPROCESS
171 PREPROCESS;
172#endif
173
174#ifdef HANDLE_DIGITS_DOTS
175 switch (__nss_hostname_digits_dots (name, resbuf, &buffer, NULL,
176 buflen, result, &status, AF_VAL,
177 H_ERRNO_VAR_P))
178 {
179 case -1:
180 return errno;
181 case 1:
182#ifdef NEED_H_ERRNO
183 any_service = true;
184#endif
185 goto done;
186 }
187#endif
188
189#ifdef USE_NSCD
190 if (NOT_USENSCD_NAME > 0 && ++NOT_USENSCD_NAME > NSS_NSCD_RETRY)
191 NOT_USENSCD_NAME = 0;
192
193 if (!NOT_USENSCD_NAME
194 && !__nss_database_custom[CONCAT2 (NSS_DBSIDX_, DATABASE_NAME)])
195 {
196 nscd_status = NSCD_NAME (ADD_VARIABLES, resbuf, buffer, buflen, result
197 H_ERRNO_VAR);
198 if (nscd_status >= 0)
199 return nscd_status;
200 }
201#endif
202
203 if (! startp_initialized)
204 {
205 no_more = DB_LOOKUP_FCT (&nip, REENTRANT_NAME_STRING,
206 REENTRANT2_NAME_STRING, &fct.ptr);
207 if (no_more)
208 {
209 void *tmp_ptr = (service_user *) -1l;
210#ifdef PTR_MANGLE
211 PTR_MANGLE (tmp_ptr);
212#endif
213 startp = tmp_ptr;
214 }
215 else
216 {
217#ifdef NEED__RES
218 /* The resolver code will really be used so we have to
219 initialize it. */
220 if (__res_maybe_init (&_res, 0) == -1)
221 {
222 *h_errnop = NETDB_INTERNAL;
223 *result = NULL;
224 return errno;
225 }
226#endif /* need _res */
227#ifdef NEED__RES_HCONF
228 if (!_res_hconf.initialized)
229 _res_hconf_init ();
230#endif /* need _res_hconf */
231
232 void *tmp_ptr = fct.l;
233#ifdef PTR_MANGLE
234 PTR_MANGLE (tmp_ptr);
235#endif
236 start_fct = tmp_ptr;
237 tmp_ptr = nip;
238#ifdef PTR_MANGLE
239 PTR_MANGLE (tmp_ptr);
240#endif
241 startp = tmp_ptr;
242 }
243
244 /* Make sure start_fct and startp are written before
245 startp_initialized. */
246 atomic_write_barrier ();
247 startp_initialized = true;
248 }
249 else
250 {
251 fct.l = start_fct;
252 nip = startp;
253#ifdef PTR_DEMANGLE
254 PTR_DEMANGLE (fct.l);
255 PTR_DEMANGLE (nip);
256#endif
257 no_more = nip == (service_user *) -1l;
258 }
259
260 while (no_more == 0)
261 {
262#ifdef NEED_H_ERRNO
263 any_service = true;
264#endif
265
266 status = DL_CALL_FCT (fct.l, (ADD_VARIABLES, resbuf, buffer, buflen,
267 &errno H_ERRNO_VAR EXTRA_VARIABLES));
268
269 /* The status is NSS_STATUS_TRYAGAIN and errno is ERANGE the
270 provided buffer is too small. In this case we should give
271 the user the possibility to enlarge the buffer and we should
272 not simply go on with the next service (even if the TRYAGAIN
273 action tells us so). */
274 if (status == NSS_STATUS_TRYAGAIN
275#ifdef NEED_H_ERRNO
276 && *h_errnop == NETDB_INTERNAL
277#endif
278 && errno == ERANGE)
279 break;
280
281 no_more = __nss_next2 (&nip, REENTRANT_NAME_STRING,
282 REENTRANT2_NAME_STRING, &fct.ptr, status, 0);
283 }
284
285#ifdef HANDLE_DIGITS_DOTS
286done:
287#endif
288 *result = status == NSS_STATUS_SUCCESS ? resbuf : NULL;
289#ifdef NEED_H_ERRNO
290 if (status == NSS_STATUS_UNAVAIL && !any_service && errno != ENOENT)
291 /* This happens when we weren't able to use a service for reasons other
292 than the module not being found. In such a case, we'd want to tell the
293 caller that errno has the real reason for failure. */
294 *h_errnop = NETDB_INTERNAL;
295 else if (status != NSS_STATUS_SUCCESS && !any_service)
296 /* We were not able to use any service. */
297 *h_errnop = NO_RECOVERY;
298#endif
299#ifdef POSTPROCESS
300 POSTPROCESS;
301#endif
302
303 int res;
304 if (status == NSS_STATUS_SUCCESS || status == NSS_STATUS_NOTFOUND)
305 res = 0;
306 /* Don't pass back ERANGE if this is not for a too-small buffer. */
307 else if (errno == ERANGE && status != NSS_STATUS_TRYAGAIN)
308 res = EINVAL;
309#ifdef NEED_H_ERRNO
310 /* These functions only set errno if h_errno is NETDB_INTERNAL. */
311 else if (status == NSS_STATUS_TRYAGAIN && *h_errnop != NETDB_INTERNAL)
312 res = EAGAIN;
313#endif
314 else
315 return errno;
316
317 __set_errno (res);
318 return res;
319}
