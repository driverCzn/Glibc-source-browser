18#include <nss.h>
19#include <stdbool.h>
20#include <string.h>
21
22static const char invalid_characters[] = NSS_INVALID_FIELD_CHARACTERS ",";
23
24/* Check that all list members match the field syntax requirements and
25 do not contain the character ','. */
26_Bool
27internal_function
28__nss_valid_list_field (char **list)
29{
30 if (list == NULL)
31 return true;
32 for (; *list != NULL; ++list)
33 if (strpbrk (*list, invalid_characters) != NULL)
34 return false;
35 return true;
36}
