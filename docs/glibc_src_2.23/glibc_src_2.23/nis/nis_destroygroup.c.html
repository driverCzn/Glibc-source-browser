19#include <string.h>
20#include <rpcsvc/nis.h>
21
22nis_error
23nis_destroygroup (const_nis_name group)
24{
25 if (group != NULL && group[0] != '\0')
26 {
27 size_t grouplen = strlen (group);
28 char buf[grouplen + 50];
29 char leafbuf[grouplen + 3];
30 char domainbuf[grouplen + 3];
31 nis_error status;
32 nis_result *res;
33 char *cp, *cp2;
34
35 cp = stpcpy (buf, nis_leaf_of_r (group, leafbuf, sizeof (leafbuf) - 1));
36 cp = stpcpy (cp, ".groups_dir");
37 cp2 = nis_domain_of_r (group, domainbuf, sizeof (domainbuf) - 1);
38 if (cp2 != NULL && cp2[0] != '\0')
39 {
40 *cp++ = '.';
41 stpcpy (cp, cp2);
42 }
43 res = nis_remove (buf, NULL);
44 status = NIS_RES_STATUS (res);
45 nis_freeresult (res);
46 return status;
47 }
48 else
49 return NIS_FAIL;
50
51}
