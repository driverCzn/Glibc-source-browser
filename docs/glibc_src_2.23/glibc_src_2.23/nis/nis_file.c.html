25typedef bool_t (*iofct_t) (XDR *, void *);
26typedef void (*freefct_t) (void *);
27
28
29static void *
30read_nis_obj (const char *name, iofct_t readfct, freefct_t freefct,
31 size_t objsize)
32{
33 FILE *in = fopen (name, "rce");
34 if (in == NULL)
35 return NULL;
36
37 void *obj = calloc (1, objsize);
38
39 if (obj != NULL)
40 {
41 XDR xdrs;
42 xdrstdio_create (&xdrs, in, XDR_DECODE);
43 bool_t status = readfct (&xdrs, obj);
44 xdr_destroy (&xdrs);
45
46 if (!status)
47 {
48 freefct (obj);
49 obj = NULL;
50 }
51 }
52
53 fclose (in);
54
55 return obj;
56}
57
58static bool_t
59write_nis_obj (const char *name, const void *obj, iofct_t writefct)
60{
61 FILE *out = fopen (name, "wce");
62 if (out == NULL)
63 return FALSE;
64
65 XDR xdrs;
66 xdrstdio_create (&xdrs, out, XDR_ENCODE);
67 bool_t status = writefct (&xdrs, (void *) obj);
68 xdr_destroy (&xdrs);
69 fclose (out);
70
71 return status;
72}

75static const char cold_start_file[] = "/var/nis/NIS_COLD_START";
76
77directory_obj *
78readColdStartFile (void)
79{
80 return read_nis_obj (cold_start_file, (iofct_t) _xdr_directory_obj,
81 (freefct_t) nis_free_directory, sizeof (directory_obj));
82}
83libnsl_hidden_def (readColdStartFile)
84
85bool_t
86writeColdStartFile (const directory_obj *obj)
87{
88 return write_nis_obj (cold_start_file, obj, (iofct_t) _xdr_directory_obj);
89}
90
91nis_object *
92nis_read_obj (const char *name)
93{
94 return read_nis_obj (name, (iofct_t) _xdr_nis_object,
95 (freefct_t) nis_free_object, sizeof (nis_object));
96}
97
98bool_t
99nis_write_obj (const char *name, const nis_object *obj)
100{
101 return write_nis_obj (name, obj, (iofct_t) _xdr_nis_object);
102}
