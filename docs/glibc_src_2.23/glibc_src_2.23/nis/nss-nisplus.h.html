27/* Convert NIS+ error number to NSS error number. */
28extern const enum nss_status __niserr2nss_tab[] attribute_hidden;
29extern const unsigned int __niserr2nss_count attribute_hidden;
30
31static enum nss_status
32__attribute__ ((unused))
33niserr2nss (int errval)
34{
35 if ((unsigned int) errval >= __niserr2nss_count)
36 return NSS_STATUS_UNAVAIL;
37 return __niserr2nss_tab[(unsigned int) errval];
38}
