23/* Write formatted output to stdout from the format string FORMAT. */
24int
25___vprintf_chk (int flag, const char *format, va_list ap)
26{
27 int done;
28
29 _IO_acquire_lock_clear_flags2 (stdout);
30 if (flag > 0)
31 stdout->_flags2 |= _IO_FLAGS2_FORTIFY;
32
33 done = vfprintf (stdout, format, ap);
34
35 if (flag > 0)
36 stdout->_flags2 &= ~_IO_FLAGS2_FORTIFY;
37 _IO_release_lock (stdout);
38
39 return done;
40}
41ldbl_strong_alias (___vprintf_chk, __vprintf_chk)
