18#include <stdarg.h>
19#include <wchar.h>
20
21/* Write formatted output into S, according to the format string FORMAT. */
22/* VARARGS5 */
23int
24__swprintf_chk (wchar_t *s, size_t n, int flag, size_t s_len,
25 const wchar_t *format, ...)
26{
27 va_list arg;
28 int done;
29
30 va_start (arg, format);
31 done = __vswprintf_chk (s, n, flag, s_len, format, arg);
32 va_end (arg);
33
34 return done;
35}
