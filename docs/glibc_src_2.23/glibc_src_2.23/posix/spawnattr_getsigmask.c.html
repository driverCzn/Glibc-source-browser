21/* Store signal mask for the new process from ATTR in SIGMASK. */
22int
23posix_spawnattr_getsigmask (const posix_spawnattr_t *attr,
24 sigset_t *sigmask)
25{
26 /* Copy the sigset_t data to the user buffer. */
27 memcpy (sigmask, &attr->__ss, sizeof (sigset_t));
28
29 return 0;
30}
