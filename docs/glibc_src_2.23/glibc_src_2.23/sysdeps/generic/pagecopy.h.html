19/* The macro PAGE_COPY_FWD_MAYBE defined in memcopy.h is used in memmove if the
20 PAGE_COPY_THRESHOLD macro is set to a non-zero value. The default is 0,
21 that is copying by pages is not implemented.
22
23 System-specific pagecopy.h files that want to support page copying should
24 define these macros:
25
26 PAGE_COPY_THRESHOLD
27 -- A non-zero minimum size for which virtual copying by pages is worthwhile.
28
29 PAGE_SIZE
30 -- Size of a page.
31
32 PAGE_COPY_FWD (dstp, srcp, nbytes_left, nbytes)
33 -- Macro to perform the virtual copy operation.
34 The pointers will be aligned to PAGE_SIZE bytes.
35*/
36
37#define PAGE_COPY_THRESHOLD 0
