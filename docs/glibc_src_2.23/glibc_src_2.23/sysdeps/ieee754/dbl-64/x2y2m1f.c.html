23/* Return X^2 + Y^2 - 1, computed without large cancellation error.
24 It is given that 1 > X >= Y >= epsilon / 2, and that X^2 + Y^2 >=
25 0.5. */
26
27float
28__x2y2m1f (float x, float y)
29{
30 double dx = x, dy = y;
31 return (float) ((dx - 1) * (dx + 1) + dy * dy);
32}
