25double
26__trunc (double x)
27{
28 int64_t i0, j0;
29 int64_t sx;
30
31 EXTRACT_WORDS64 (i0, x);
32 sx = i0 & UINT64_C(0x8000000000000000);
33 j0 = ((i0 >> 52) & 0x7ff) - 0x3ff;
34 if (j0 < 52)
35 {
36 if (j0 < 0)
37 /* The magnitude of the number is < 1 so the result is +-0. */
38 INSERT_WORDS64 (x, sx);
39 else
40 INSERT_WORDS64 (x, sx | (i0 & ~(UINT64_C(0x000fffffffffffff) >> j0)));
41 }
42 else
43 {
44 if (j0 == 0x400)
45 /* x is inf or NaN. */
46 return x + x;
47 }
48
49 return x;
50}

54weak_alias (__trunc, truncl)
