13/* asinh(x)
14 * Method :
15 * Based on
16 * asinh(x) = sign(x) * log [ |x| + sqrt(x*x+1) ]
17 * we have
18 * asinh(x) := x if 1+x*x=1,
19 * := sign(x)*(log(x)+ln2)) for large |x|, else
20 * := sign(x)*log(2|x|+1/(|x|+sqrt(x*x+1))) if|x|>2, else
21 * := sign(x)*log1p(|x| + x^2/(1 + sqrt(1+x^2)))
22 */
23
24#include <float.h>
25#include <math.h>
26#include <math_private.h>
27
28static const double
29 one = 1.00000000000000000000e+00, /* 0x3FF00000, 0x00000000 */
30 ln2 = 6.93147180559945286227e-01, /* 0x3FE62E42, 0xFEFA39EF */
31 huge = 1.00000000000000000000e+300;
32
33double
34__asinh (double x)
35{
36 double w;
37 int32_t hx, ix;
38 GET_HIGH_WORD (hx, x);
39 ix = hx & 0x7fffffff;
40 if (__glibc_unlikely (ix < 0x3e300000)) /* |x|<2**-28 */
41 {
42 math_check_force_underflow (x);
43 if (huge + x > one)
44 return x; /* return x inexact except 0 */
45 }
46 if (__glibc_unlikely (ix > 0x41b00000)) /* |x| > 2**28 */
47 {
48 if (ix >= 0x7ff00000)
49 return x + x; /* x is inf or NaN */
50 w = __ieee754_log (fabs (x)) + ln2;
51 }
52 else
53 {
54 double xa = fabs (x);
55 if (ix > 0x40000000) /* 2**28 > |x| > 2.0 */
56 {
57 w = __ieee754_log (2.0 * xa + one / (__ieee754_sqrt (xa * xa + one) +
58 xa));
59 }
60 else /* 2.0 > |x| > 2**-28 */
61 {
62 double t = xa * xa;
63 w = __log1p (xa + t / (one + __ieee754_sqrt (one + t)));
64 }
65 }
66 return __copysign (w, x);
67}
68weak_alias (__asinh, asinh)
69#ifdef NO_LONG_DOUBLE
70strong_alias (__asinh, __asinhl)
71weak_alias (__asinh, asinhl)
72#endif
