17/* Tanh(x)
18 * Return the Hyperbolic Tangent of x
19 *
20 * Method :
21 * x -x
22 * e - e
23 * 0. tanh(x) is defined to be -----------
24 * x -x
25 * e + e
26 * 1. reduce x to non-negative by tanh(-x) = -tanh(x).
27 * 2. 0 <= x <= 2**-55 : tanh(x) := x*(one+x)
28 * -t
29 * 2**-55 < x <= 1 : tanh(x) := -----; t = expm1(-2x)
30 * t + 2
31 * 2
32 * 1 <= x <= 22.0 : tanh(x) := 1- ----- ; t=expm1(2x)
33 * t + 2
34 * 22.0 < x <= INF : tanh(x) := 1.
35 *
36 * Special cases:
37 * tanh(NaN) is NaN;
38 * only tanh(0)=0 is exact for finite argument.
39 */
40
41#include <float.h>
42#include <math.h>
43#include <math_private.h>
44
45static const double one = 1.0, two = 2.0, tiny = 1.0e-300;
46
47double
48__tanh (double x)
49{
50 double t, z;
51 int32_t jx, ix, lx;
52
53 /* High word of |x|. */
54 EXTRACT_WORDS (jx, lx, x);
55 ix = jx & 0x7fffffff;
56
57 /* x is INF or NaN */
58 if (ix >= 0x7ff00000)
59 {
60 if (jx >= 0)
61 return one / x + one; /* tanh(+-inf)=+-1 */
62 else
63 return one / x - one; /* tanh(NaN) = NaN */
64 }
65
66 /* |x| < 22 */
67 if (ix < 0x40360000) /* |x|<22 */
68 {
69 if ((ix | lx) == 0)
70 return x; /* x == +-0 */
71 if (ix < 0x3c800000) /* |x|<2**-55 */
72 {
73 math_check_force_underflow (x);
74 return x * (one + x); /* tanh(small) = small */
75 }
76 if (ix >= 0x3ff00000) /* |x|>=1 */
77 {
78 t = __expm1 (two * fabs (x));
79 z = one - two / (t + two);
80 }
81 else
82 {
83 t = __expm1 (-two * fabs (x));
84 z = -t / (t + two);
85 }
86 /* |x| > 22, return +-1 */
87 }
88 else
89 {
90 z = one - tiny; /* raised inexact flag */
91 }
92 return (jx >= 0) ? z : -z;
93}

97weak_alias (__tanh, tanhl)
98#endif
