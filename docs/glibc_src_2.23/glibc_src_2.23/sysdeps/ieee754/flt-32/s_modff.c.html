16#include <math.h>
17#include <math_private.h>
18
19static const float one = 1.0;
20
21float
22__modff(float x, float *iptr)
23{
24 int32_t i0,j0;
25 u_int32_t i;
26 GET_FLOAT_WORD(i0,x);
27 j0 = ((i0>>23)&0xff)-0x7f; /* exponent of x */
28 if(__builtin_expect(j0<23, 1)) { /* integer part in x */
29 if(j0<0) { /* |x|<1 */
30 SET_FLOAT_WORD(*iptr,i0&0x80000000); /* *iptr = +-0 */
31 return x;
32 } else {
33 i = (0x007fffff)>>j0;
34 if((i0&i)==0) { /* x is integral */
35 u_int32_t ix;
36 *iptr = x;
37 GET_FLOAT_WORD(ix,x);
38 SET_FLOAT_WORD(x,ix&0x80000000); /* return +-0 */
39 return x;
40 } else {
41 SET_FLOAT_WORD(*iptr,i0&(~i));
42 return x - *iptr;
43 }
44 }
45 } else { /* no fraction part */
46 *iptr = x*one;
47 /* We must handle NaNs separately. */
48 if (j0 == 0x80 && (i0 & 0x7fffff))
49 return x*one;
50 SET_FLOAT_WORD(x,i0&0x80000000); /* return +-0 */
51 return x;
52 }
53}
