18#include <stddef.h>
19#include <signal.h>
20#include <errno.h>
21
22/* If INTERRUPT is nonzero, make signal SIG interrupt system calls
23 (causing them to fail with EINTR); if INTERRUPT is zero, make system
24 calls be restarted after signal SIG. */
25int
26siginterrupt (int sig, int interrupt)
27{
28#ifdef SA_RESTART
29 extern sigset_t _sigintr attribute_hidden; /* Defined in signal.c. */
30 struct sigaction action;
31
32 if (__sigaction (sig, (struct sigaction *) NULL, &action) < 0)
33 return -1;
34
35 if (interrupt)
36 {
37 __sigaddset (&_sigintr, sig);
38 action.sa_flags &= ~SA_RESTART;
39 }
40 else
41 {
42 __sigdelset (&_sigintr, sig);
43 action.sa_flags |= SA_RESTART;
44 }
45
46 if (__sigaction (sig, &action, (struct sigaction *) NULL) < 0)
47 return -1;
48
49 return 0;
50#else
51 __set_errno (ENOSYS);
52 return -1;
53#endif
54}
