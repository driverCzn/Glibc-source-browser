19#ifndef LONG_DOUBLE_DENORM_BIAS
20# define LONG_DOUBLE_DENORM_BIAS (IEEE854_LONG_DOUBLE_BIAS - 1)
21#endif
22
23#define PRINT_FPHEX_LONG_DOUBLE \
24do { \
25 /* The "strange" 80 bit format on ix86 and m68k has an explicit \
26 leading digit in the 64 bit mantissa. */ \
27 unsigned long long int num; \
28 union ieee854_long_double u; \
29 u.d = fpnum.ldbl; \
30 \
31 num = (((unsigned long long int) u.ieee.mantissa0) << 32 \
32 | u.ieee.mantissa1); \
33 \
34 zero_mantissa = num == 0; \
35 \
36 if (sizeof (unsigned long int) > 6) \
37 { \
38 numstr = _itoa_word (num, numbuf + sizeof numbuf, 16, \
39 info->spec == 'A'); \
40 wnumstr = _itowa_word (num, \
41 wnumbuf + sizeof (wnumbuf) / sizeof (wchar_t),\
42 16, info->spec == 'A'); \
43 } \
44 else \
45 { \
46 numstr = _itoa (num, numbuf + sizeof numbuf, 16, info->spec == 'A');\
47 wnumstr = _itowa (num, \
48 wnumbuf + sizeof (wnumbuf) / sizeof (wchar_t), \
49 16, info->spec == 'A'); \
50 } \
51 \
52 /* Fill with zeroes. */ \
53 while (numstr > numbuf + (sizeof numbuf - 64 / 4)) \
54 { \
55 *--numstr = '0'; \
56 *--wnumstr = L'0'; \
57 } \
58 \
59 /* We use a full nibble for the leading digit. */ \
60 leading = *numstr++; \
61 wnumstr++; \
62 \
63 /* We have 3 bits from the mantissa in the leading nibble. \
64 Therefore we are here using `IEEE854_LONG_DOUBLE_BIAS + 3'. */ \
65 exponent = u.ieee.exponent; \
66 \
67 if (exponent == 0) \
68 { \
69 if (zero_mantissa) \
70 expnegative = 0; \
71 else \
72 { \
73 /* This is a denormalized number. */ \
74 expnegative = 1; \
75 /* This is a hook for the m68k long double format, where the \
76 exponent bias is the same for normalized and denormalized \
77 numbers. */ \
78 exponent = LONG_DOUBLE_DENORM_BIAS + 3; \
79 } \
80 } \
81 else if (exponent >= IEEE854_LONG_DOUBLE_BIAS + 3) \
82 { \
83 expnegative = 0; \
84 exponent -= IEEE854_LONG_DOUBLE_BIAS + 3; \
85 } \
86 else \
87 { \
88 expnegative = 1; \
89 exponent = -(exponent - (IEEE854_LONG_DOUBLE_BIAS + 3)); \
90 } \
91} while (0)
