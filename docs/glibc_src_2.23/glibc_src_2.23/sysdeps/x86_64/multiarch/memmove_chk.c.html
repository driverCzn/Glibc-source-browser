20#include <string.h>
21#include "init-arch.h"
22
23#define MEMMOVE_CHK __memmove_chk_sse2
24
25extern __typeof (__memmove_chk) __memmove_chk_sse2 attribute_hidden;
26extern __typeof (__memmove_chk) __memmove_chk_ssse3 attribute_hidden;
27extern __typeof (__memmove_chk) __memmove_chk_ssse3_back attribute_hidden;
28extern __typeof (__memmove_chk) __memmove_chk_avx_unaligned attribute_hidden;
29# ifdef HAVE_AVX512_ASM_SUPPORT
30 extern __typeof (__memmove_chk) __memmove_chk_avx512_no_vzeroupper attribute_hidden;
31# endif
32
33#include "debug/memmove_chk.c"
34
35libc_ifunc (__memmove_chk,
36#ifdef HAVE_AVX512_ASM_SUPPORT
37 HAS_ARCH_FEATURE (AVX512F_Usable)
38 && HAS_ARCH_FEATURE (Prefer_No_VZEROUPPER)
39 ? __memmove_chk_avx512_no_vzeroupper
40 :
41#endif
42 HAS_ARCH_FEATURE (AVX_Fast_Unaligned_Load) ? __memmove_chk_avx_unaligned :
43 (HAS_CPU_FEATURE (SSSE3)
44 ? (HAS_ARCH_FEATURE (Fast_Copy_Backward)
45 ? __memmove_chk_ssse3_back : __memmove_chk_ssse3)
46 : __memmove_chk_sse2));
