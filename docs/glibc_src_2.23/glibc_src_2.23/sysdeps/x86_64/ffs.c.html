21#include <string.h>
22
23#undef ffs
24
25int
26__ffs (int x)
27{
28 int cnt;
29 int tmp;
30
31 asm ("bsfl %2,%0\n" /* Count low bits in X and store in %1. */
32 "cmovel %1,%0\n" /* If number was zero, use -1 as result. */
33 : "=&r" (cnt), "=r" (tmp) : "rm" (x), "1" (-1));
34
35 return cnt + 1;
36}
37weak_alias (__ffs, ffs)
