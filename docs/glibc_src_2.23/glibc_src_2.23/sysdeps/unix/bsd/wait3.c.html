22/* Wait for a child to exit. When one does, put its status in *STAT_LOC and
23 return its process ID. For errors return (pid_t) -1. If USAGE is not nil,
24 store information about the child's resource usage (as a `struct rusage')
25 there. If the WUNTRACED bit is set in OPTIONS, return status for stopped
26 children; otherwise don't. */
27pid_t
28__wait3 (__WAIT_STATUS stat_loc, int options, struct rusage *usage)
29{
30 return __wait4 (WAIT_ANY, stat_loc, options, usage);
31}

33weak_alias (__wait3, wait3)
