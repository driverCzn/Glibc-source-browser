18#include <sys/timeb.h>
19#include <sys/time.h>
20
21int
22ftime (struct timeb *timebuf)
23{
24 struct timeval tv;
25 struct timezone tz;
26
27 if (__gettimeofday (&tv, &tz) < 0)
28 return -1;
29
30 timebuf->time = tv.tv_sec;
31 timebuf->millitm = (tv.tv_usec + 500) / 1000;
32 if (timebuf->millitm == 1000)
33 {
34 ++timebuf->time;
35 timebuf->millitm = 0;
36 }
37 timebuf->timezone = tz.tz_minuteswest;
38 timebuf->dstflag = tz.tz_dsttime;
39 return 0;
40}
