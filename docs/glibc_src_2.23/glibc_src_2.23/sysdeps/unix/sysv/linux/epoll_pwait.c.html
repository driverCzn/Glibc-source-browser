26#ifdef __NR_epoll_pwait
27
28/* Wait for events on an epoll instance "epfd". Returns the number of
29 triggered events returned in "events" buffer. Or -1 in case of
30 error with the "errno" variable set to the specific error code. The
31 "events" parameter is a buffer that will contain triggered
32 events. The "maxevents" is the maximum number of events to be
33 returned ( usually size of "events" ). The "timeout" parameter
34 specifies the maximum wait time in milliseconds (-1 == infinite).
35 The thread's signal mask is temporarily and atomically replaced with
36 the one provided as parameter. */
37
38int epoll_pwait (int epfd, struct epoll_event *events,
39 int maxevents, int timeout,
40 const sigset_t *set)
41{
42 return SYSCALL_CANCEL (epoll_pwait, epfd, events, maxevents,
43 timeout, set, _NSIG / 8);
44}
45
46#else
47
48int epoll_pwait (int epfd, struct epoll_event *events,
49 int maxevents, int timeout,
50 const sigset_t *set)
51{
52 __set_errno (ENOSYS);
53 return -1;
54}
55stub_warning (epoll_pwait)
56
57#endif
