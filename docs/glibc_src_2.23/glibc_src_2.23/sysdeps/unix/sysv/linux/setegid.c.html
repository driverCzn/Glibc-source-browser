18#include <errno.h>
19#include <unistd.h>
20#include <setxid.h>
21
22
23int
24setegid (gid_t gid)
25{
26 int result;
27
28 if (gid == (gid_t) ~0)
29 {
30 __set_errno (EINVAL);
31 return -1;
32 }
33
34#ifdef __NR_setresgid32
35 result = INLINE_SETXID_SYSCALL (setresgid32, 3, -1, gid, -1);
36#else
37 result = INLINE_SETXID_SYSCALL (setresgid, 3, -1, gid, -1);
38#endif
39
40 return result;
41}
42#ifndef setegid
43libc_hidden_def (setegid)
44#endif
