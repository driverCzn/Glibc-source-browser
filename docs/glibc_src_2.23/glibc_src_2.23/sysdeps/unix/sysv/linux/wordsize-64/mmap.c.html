25/* An architecture may override this. */
26#ifndef MMAP_PREPARE
27# define MMAP_PREPARE(addr, len, prot, flags, fd, offset)
28#endif
29
30__ptr_t
31__mmap (__ptr_t addr, size_t len, int prot, int flags, int fd, off_t offset)
32{
33 MMAP_PREPARE (addr, len, prot, flags, fd, offset);
34 return (__ptr_t) INLINE_SYSCALL (mmap, 6, addr, len, prot, flags,
35 fd, offset);
36}
