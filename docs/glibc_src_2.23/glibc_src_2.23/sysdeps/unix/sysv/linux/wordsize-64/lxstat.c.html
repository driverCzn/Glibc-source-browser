19/* Ho hum, since xstat == xstat64 we must get rid of the prototype or gcc
20 will complain since they don't strictly match. */
21#define __lxstat64 __lxstat64_disable
22
23#include <errno.h>
24#include <stddef.h>
25#include <sys/stat.h>
26
27#include <sysdep.h>
28#include <sys/syscall.h>

30/* Get information about the file FD in BUF. */
31int
32__lxstat (int vers, const char *name, struct stat *buf)
33{
34 if (vers == _STAT_VER_KERNEL || vers == _STAT_VER_LINUX)
35 return INLINE_SYSCALL (lstat, 2, name, buf);
36
37 __set_errno (EINVAL);
38 return -1;
39}

42weak_alias (__lxstat, _lxstat);
43#undef __lxstat64
