18#include <sys/timex.h>
19
20#ifndef MOD_OFFSET
21# define modes mode
22#endif
23
24
25int
26ntp_gettimex (struct ntptimeval *ntv)
27{
28 struct timex tntx;
29 int result;
30
31 tntx.modes = 0;
32 result = __adjtimex (&tntx);
33 ntv->time = tntx.time;
34 ntv->maxerror = tntx.maxerror;
35 ntv->esterror = tntx.esterror;
36 ntv->tai = tntx.tai;
37 ntv->__glibc_reserved1 = 0;
38 ntv->__glibc_reserved2 = 0;
39 ntv->__glibc_reserved3 = 0;
40 ntv->__glibc_reserved4 = 0;
41 return result;
42}
