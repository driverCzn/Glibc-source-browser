22/* Send N bytes of BUF to socket FD. Returns the number sent or -1. */
23ssize_t
24__libc_send (int fd, const void *buf, size_t n, int flags)
25{
26 return SYSCALL_CANCEL (sendto, fd, buf, n, flags, NULL, (size_t) 0);
27}
