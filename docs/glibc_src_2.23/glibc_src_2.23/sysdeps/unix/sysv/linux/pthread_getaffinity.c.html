29int
30__pthread_getaffinity_new (pthread_t th, size_t cpusetsize, cpu_set_t *cpuset)
31{
32 const struct pthread *pd = (const struct pthread *) th;
33
34 INTERNAL_SYSCALL_DECL (err);
35 int res = INTERNAL_SYSCALL (sched_getaffinity, err, 3, pd->tid,
36 MIN (INT_MAX, cpusetsize), cpuset);
37 if (INTERNAL_SYSCALL_ERROR_P (res, err))
38 return INTERNAL_SYSCALL_ERRNO (res, err);
39
40 /* Clean the rest of the memory the kernel didn't do. */
41 memset ((char *) cpuset + res, '\0', cpusetsize - res);
42
43 return 0;
44}
45strong_alias (__pthread_getaffinity_new, __pthread_getaffinity_np)
46versioned_symbol (libpthread, __pthread_getaffinity_new,
47 pthread_getaffinity_np, GLIBC_2_3_4);
48
49
50#if SHLIB_COMPAT (libpthread, GLIBC_2_3_3, GLIBC_2_3_4)
51int
52__pthread_getaffinity_old (pthread_t th, cpu_set_t *cpuset)
53{
54 /* The old interface by default assumed a 1024 processor bitmap. */
55 return __pthread_getaffinity_new (th, 128, cpuset);
56}
57compat_symbol (libpthread, __pthread_getaffinity_old, pthread_getaffinity_np,
58 GLIBC_2_3_3);
59#endif
