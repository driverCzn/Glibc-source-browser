19#include <stddef.h>
20#include <sys/stat.h>
21#include <sys/statfs.h>
22#include <sys/statvfs.h>
23
24extern void __internal_statvfs (const char *name, struct statvfs *buf,
25 struct statfs *fsbuf, int fd);
26
27
28int
29__fstatvfs (int fd, struct statvfs *buf)
30{
31 struct statfs fsbuf;
32
33 /* Get as much information as possible from the system. */
34 if (__fstatfs (fd, &fsbuf) < 0)
35 return -1;
36
37 /* Convert the result. */
38 __internal_statvfs (NULL, buf, &fsbuf, fd);
39
40 /* We signal success if the statfs call succeeded. */
41 return 0;
42}
43weak_alias (__fstatvfs, fstatvfs)
44libc_hidden_weak (fstatvfs)
