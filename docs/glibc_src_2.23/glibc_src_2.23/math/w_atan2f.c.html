19/*
20 * wrapper atan2f(y,x)
21 */
22
23#include <errno.h>
24#include <math.h>
25#include <math_private.h>
26
27
28float
29__atan2f (float y, float x)
30{
31 float z;
32
33 if (__builtin_expect (x == 0.0f && y == 0.0f, 0) && _LIB_VERSION == _SVID_)
34 return __kernel_standard_f (y, x, 103); /* atan2(+-0,+-0) */
35
36 z = __ieee754_atan2f (y, x);
37 if (__glibc_unlikely (z == 0.0f && y != 0.0f && isfinite (x)))
38 __set_errno (ERANGE);
39 return z;
40}
41weak_alias (__atan2f, atan2f)
