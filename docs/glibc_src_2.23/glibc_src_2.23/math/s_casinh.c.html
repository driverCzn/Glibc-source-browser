20#include <complex.h>
21#include <math.h>
22#include <math_private.h>
23
24__complex__ double
25__casinh (__complex__ double x)
26{
27 __complex__ double res;
28 int rcls = fpclassify (__real__ x);
29 int icls = fpclassify (__imag__ x);
30
31 if (rcls <= FP_INFINITE || icls <= FP_INFINITE)
32 {
33 if (icls == FP_INFINITE)
34 {
35 __real__ res = __copysign (HUGE_VAL, __real__ x);
36
37 if (rcls == FP_NAN)
38 __imag__ res = __nan ("");
39 else
40 __imag__ res = __copysign (rcls >= FP_ZERO ? M_PI_2 : M_PI_4,
41 __imag__ x);
42 }
43 else if (rcls <= FP_INFINITE)
44 {
45 __real__ res = __real__ x;
46 if ((rcls == FP_INFINITE && icls >= FP_ZERO)
47 || (rcls == FP_NAN && icls == FP_ZERO))
48 __imag__ res = __copysign (0.0, __imag__ x);
49 else
50 __imag__ res = __nan ("");
51 }
52 else
53 {
54 __real__ res = __nan ("");
55 __imag__ res = __nan ("");
56 }
57 }
58 else if (rcls == FP_ZERO && icls == FP_ZERO)
59 {
60 res = x;
61 }
62 else
63 {
64 res = __kernel_casinh (x, 0);
65 }
66
67 return res;
68}

72weak_alias (__casinh, casinhl)
