36#include <sys/types.h>
37#include <sys/gmon.h>
38#include <stdlib.h>
39#include <unistd.h>
40#define __ASSEMBLY__
41#include <entry.h>
42
43/* Beginning and end of our code segment. We cannot declare them
44 as the external functions since we want the addresses of those
45 labels. Taking the address of a function may have different
46 meanings on different platforms. */
47#ifdef ENTRY_POINT_DECL
48ENTRY_POINT_DECL(extern)
49#else
50extern char ENTRY_POINT[];
51#endif
52extern char etext[];
53
54#ifndef TEXT_START
55# ifdef ENTRY_POINT_DECL
56# define TEXT_START ENTRY_POINT
57# else
58# define TEXT_START &ENTRY_POINT
59# endif
60#endif
61
62#ifdef GMON_START_ARRAY_SECTION
63static void __gmon_start__ (void);
64static void (*const gmon_start_initializer) (void)
65 __attribute__ ((used, section (GMON_START_ARRAY_SECTION))) = &__gmon_start__;
66static
67#else
68/* We cannot use the normal constructor mechanism to call
69 __gmon_start__ because gcrt1.o appears before crtbegin.o in the link.
70 Instead crti.o calls it specially. */
71extern void __gmon_start__ (void);
72#endif
73
74void
75__gmon_start__ (void)
76{
77 /* Protect from being called more than once. Since crti.o is linked
78 into every shared library, each of their init functions will call us. */
79 static int called;
80
81 if (called)
82 return;
83
84 called = 1;
85
86 /* Start keeping profiling records. */
87 __monstartup ((u_long) TEXT_START, (u_long) &etext);
88
89 /* Call _mcleanup before exiting; it will write out gmon.out from the
90 collected data. */
91 atexit (&_mcleanup);
92}
