18#include <errno.h>
19#include <signal.h>
20#define __need_NULL
21#include <stddef.h>
22
23/* Test whether SET is empty. */
24int
25sigisemptyset (const sigset_t *set)
26{
27 if (set == NULL)
28 {
29 __set_errno (EINVAL);
30 return -1;
31 }
32
33 return __sigisemptyset (set);
34}
