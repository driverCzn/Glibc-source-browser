22/* Read formatted input from STREAM according to the format string FORMAT. */
23/* VARARGS2 */
24int
25__fscanf (FILE *stream, const char *format, ...)
26{
27 va_list arg;
28 int done;
29
30 va_start (arg, format);
31 done = __vfscanf (stream, format, arg);
32 va_end (arg);
33
34 return done;
35}
36ldbl_strong_alias (__fscanf, fscanf)
