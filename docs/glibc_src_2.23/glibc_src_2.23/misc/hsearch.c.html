19#include <search.h>
20
21/* The non-reentrant version use a global space for storing the table. */
22static struct hsearch_data htab;
23
24
25/* Define the non-reentrant function using the reentrant counterparts. */
26ENTRY *
27hsearch (ENTRY item, ACTION action)
28{
29 ENTRY *result;
30
31 (void) __hsearch_r (item, action, &result, &htab);
32
33 return result;
34}
35
36
37int
38hcreate (size_t nel)
39{
40 return __hcreate_r (nel, &htab);
41}
42
43
44void
45__hdestroy (void)
46{
47 __hdestroy_r (&htab);
48}
49weak_alias (__hdestroy, hdestroy)
50
51/* Make sure the table is freed if we want to free everything before
52 exiting. */
53text_set_element (__libc_subfreeres, __hdestroy);
