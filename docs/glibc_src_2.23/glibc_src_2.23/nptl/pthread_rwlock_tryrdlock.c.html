19#include <errno.h>
20#include "pthreadP.h"
21#include <lowlevellock.h>
22#include <futex-internal.h>
23#include <elide.h>
24#include <stdbool.h>
25
26
27int
28__pthread_rwlock_tryrdlock (pthread_rwlock_t *rwlock)
29{
30 int result = EBUSY;
31 bool wake = false;
32 int futex_shared =
33 rwlock->__data.__shared == LLL_PRIVATE ? FUTEX_PRIVATE : FUTEX_SHARED;
34
35 if (ELIDE_TRYLOCK (rwlock->__data.__rwelision,
36 rwlock->__data.__lock == 0
37 && rwlock->__data.__nr_readers == 0
38 && rwlock->__data.__writer, 0))
39 return 0;
40
41 lll_lock (rwlock->__data.__lock, rwlock->__data.__shared);
42
43 if (rwlock->__data.__writer == 0
44 && (rwlock->__data.__nr_writers_queued == 0
45 || PTHREAD_RWLOCK_PREFER_READER_P (rwlock)))
46 {
47 if (__glibc_unlikely (++rwlock->__data.__nr_readers == 0))
48 {
49 --rwlock->__data.__nr_readers;
50 result = EAGAIN;
51 }
52 else
53 {
54 result = 0;
55 /* See pthread_rwlock_rdlock. */
56 if (rwlock->__data.__nr_readers == 1
57 && rwlock->__data.__nr_readers_queued > 0
58 && rwlock->__data.__nr_writers_queued > 0)
59 {
60 ++rwlock->__data.__readers_wakeup;
61 wake = true;
62 }
63 }
64 }
65
66 lll_unlock (rwlock->__data.__lock, rwlock->__data.__shared);
67
68 if (wake)
69 futex_wake (&rwlock->__data.__readers_wakeup, INT_MAX, futex_shared);
70
71 return result;
72}
73strong_alias (__pthread_rwlock_tryrdlock, pthread_rwlock_tryrdlock)
