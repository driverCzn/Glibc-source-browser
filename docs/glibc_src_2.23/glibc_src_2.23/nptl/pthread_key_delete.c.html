19#include <errno.h>
20#include "pthreadP.h"
21#include <atomic.h>
22
23
24int
25pthread_key_delete (pthread_key_t key)
26{
27 int result = EINVAL;
28
29 if (__glibc_likely (key < PTHREAD_KEYS_MAX))
30 {
31 unsigned int seq = __pthread_keys[key].seq;
32
33 if (__builtin_expect (! KEY_UNUSED (seq), 1)
34 && ! atomic_compare_and_exchange_bool_acq (&__pthread_keys[key].seq,
35 seq + 1, seq))
36 /* We deleted a valid key. */
37 result = 0;
38 }
39
40 return result;
41}
