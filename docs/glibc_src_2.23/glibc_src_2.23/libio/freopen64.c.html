37FILE *
38freopen64 (const char *filename, const char *mode, FILE *fp)
39{
40 FILE *result;
41 CHECK_FILE (fp, NULL);
42 if (!(fp->_flags & _IO_IS_FILEBUF))
43 return NULL;
44 _IO_acquire_lock (fp);
45 int fd = _IO_fileno (fp);
46 const char *gfilename = (filename == NULL && fd >= 0
47 ? fd_to_filename (fd) : filename);
48 fp->_flags2 |= _IO_FLAGS2_NOCLOSE;
49 _IO_file_close_it (fp);
50 _IO_JUMPS_FILE_plus (fp) = &_IO_file_jumps;
51 if (_IO_vtable_offset (fp) == 0 && fp->_wide_data != NULL)
52 fp->_wide_data->_wide_vtable = &_IO_wfile_jumps;
53 result = _IO_file_fopen (fp, gfilename, mode, 0);
54 fp->_flags2 &= ~_IO_FLAGS2_NOCLOSE;
55 if (result != NULL)
56 result = __fopen_maybe_mmap (result);
57 if (result != NULL)
58 {
59 /* unbound stream orientation */
60 result->_mode = 0;
61
62 if (fd != -1)
63 {
64#ifdef O_CLOEXEC
65# ifndef __ASSUME_DUP3
66 int newfd;
67 if (__have_dup3 < 0)
68 newfd = -1;
69 else
70 newfd =
71# endif
72 __dup3 (_IO_fileno (result), fd,
73 (result->_flags2 & _IO_FLAGS2_CLOEXEC) != 0
74 ? O_CLOEXEC : 0);
75#else
76# define newfd 1
77#endif
78
79#ifndef __ASSUME_DUP3
80 if (newfd < 0)
81 {
82 if (errno == ENOSYS)
83 __have_dup3 = -1;
84
85 __dup2 (_IO_fileno (result), fd);
86 if ((result->_flags2 & _IO_FLAGS2_CLOEXEC) != 0)
87 __fcntl (fd, F_SETFD, FD_CLOEXEC);
88 }
89#endif
90 __close (_IO_fileno (result));
91 _IO_fileno (result) = fd;
92 }
93 }
94 else if (fd != -1)
95 __close (fd);
96 if (filename == NULL)
97 free ((char *) gfilename);
98 _IO_release_lock (fp);
99 return result;
100}
