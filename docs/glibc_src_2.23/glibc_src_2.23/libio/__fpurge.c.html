18#include <stdio_ext.h>
19#include "libioP.h"
20
21void
22__fpurge (FILE *fp)
23{
24 if (fp->_mode > 0)
25 {
26 /* Wide-char stream. */
27 if (_IO_in_backup (fp))
28 _IO_free_wbackup_area (fp);
29
30 fp->_wide_data->_IO_read_end = fp->_wide_data->_IO_read_ptr;
31 fp->_wide_data->_IO_write_ptr = fp->_wide_data->_IO_write_base;
32 }
33 else
34 {
35 /* Byte stream. */
36 if (_IO_in_backup (fp))
37 _IO_free_backup_area (fp);
38
39 fp->_IO_read_end = fp->_IO_read_ptr;
40 fp->_IO_write_ptr = fp->_IO_write_base;
41 }
42}
