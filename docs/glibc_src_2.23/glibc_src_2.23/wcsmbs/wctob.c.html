29int
30wctob (wint_t c)
31{
32 unsigned char buf[MB_LEN_MAX];
33 struct __gconv_step_data data;
34 wchar_t inbuf[1];
35 wchar_t *inptr = inbuf;
36 size_t dummy;
37 int status;
38 const struct gconv_fcts *fcts;
39
40 if (c == WEOF)
41 return EOF;
42
43 /* We know that only ASCII compatible encodings are used for the
44 locale and that the wide character encoding is ISO 10646. */
45 if (c >= L'\0' && c <= L'\x7f')
46 return (int) c;
47
48 /* Tell where we want the result. */
49 data.__outbuf = buf;
50 data.__outbufend = buf + MB_LEN_MAX;
51 data.__invocation_counter = 0;
52 data.__internal_use = 1;
53 data.__flags = __GCONV_IS_LAST;
54 data.__statep = &data.__state;
55
56 /* Make sure we start in the initial state. */
57 memset (&data.__state, '\0', sizeof (mbstate_t));
58
59 /* Get the conversion functions. */
60 fcts = get_gconv_fcts (_NL_CURRENT_DATA (LC_CTYPE));
61
62 /* Create the input string. */
63 inbuf[0] = c;
64
65 const unsigned char *argptr = (const unsigned char *) inptr;
66 __gconv_fct fct = fcts->tomb->__fct;
67#ifdef PTR_DEMANGLE
68 if (fcts->tomb->__shlib_handle != NULL)
69 PTR_DEMANGLE (fct);
70#endif
71 status = DL_CALL_FCT (fct,
72 (fcts->tomb, &data, &argptr,
73 argptr + sizeof (inbuf[0]), NULL, &dummy, 0, 1));
74
75 /* The conversion failed or the output is too long. */
76 if ((status != __GCONV_OK && status != __GCONV_FULL_OUTPUT
77 && status != __GCONV_EMPTY_INPUT)
78 || data.__outbuf != (unsigned char *) (buf + 1))
79 return EOF;
80
81 return buf[0];
82}
