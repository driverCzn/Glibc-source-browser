26ssize_t
27__strfmon (char *s, size_t maxsize, const char *format, ...)
28{
29 va_list ap;
30
31 va_start (ap, format);
32
33 ssize_t res = __vstrfmon_l (s, maxsize, _NL_CURRENT_LOCALE, format, ap);
34
35 va_end (ap);
36
37 return res;
38}
