156/*
157 * <------- NLA_HDRLEN ------> <-- NLA_ALIGN(payload)-->
158 * +---------------------+- - -+- - - - - - - - - -+- - -+
159 * | Header | Pad | Payload | Pad |
160 * | (struct nlattr) | ing | | ing |
161 * +---------------------+- - -+- - - - - - - - - -+- - -+
162 * <-------------- nlattr->nla_len -------------->
163 */
164
165struct nlattr {
166 __u16 nla_len;
167 __u16 nla_type;
168};
169
170/*
171 * nla_type (16 bits)
172 * +---+---+-------------------------------+
173 * | N | O | Attribute Type |
174 * +---+---+-------------------------------+
175 * N := Carries nested attributes
176 * O := Payload stored in network byte order
177 *
178 * Note: The N and O flag are mutually exclusive.
179 */
180#define NLA_F_NESTED (1 << 15)
181#define NLA_F_NET_BYTEORDER (1 << 14)
182#define NLA_TYPE_MASK ~(NLA_F_NESTED | NLA_F_NET_BYTEORDER)
183
184#define NLA_ALIGNTO 4
185#define NLA_ALIGN(len) (((len) + NLA_ALIGNTO - 1) & ~(NLA_ALIGNTO - 1))
186#define NLA_HDRLEN ((int) NLA_ALIGN(sizeof(struct nlattr)))
