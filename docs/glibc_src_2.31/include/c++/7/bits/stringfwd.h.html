34#ifndef _STRINGFWD_H
35#define _STRINGFWD_H 1
36
37#pragma GCC system_header
38
39#include <bits/c++config.h>
40#include <bits/memoryfwd.h>
41
42namespace std _GLIBCXX_VISIBILITY(default)
43{
44_GLIBCXX_BEGIN_NAMESPACE_VERSION
45
46 /**
47 * @defgroup strings Strings
48 *
49 * @{
50 */
51
52 template<class _CharT>
53 struct char_traits;
54
55 template<> struct char_traits<char>;
56
57#ifdef _GLIBCXX_USE_WCHAR_T
58 template<> struct char_traits<wchar_t>;
59#endif
60
61#if ((__cplusplus >= 201103L) \
62 && defined(_GLIBCXX_USE_C99_STDINT_TR1))
63 template<> struct char_traits<char16_t>;
64 template<> struct char_traits<char32_t>;
65#endif
66
67_GLIBCXX_BEGIN_NAMESPACE_CXX11
68
69 template<typename _CharT, typename _Traits = char_traits<_CharT>,
70 typename _Alloc = allocator<_CharT> >
71 class basic_string;
72
73 /// A string of @c char
74 typedef basic_string<char> string;
75
76#ifdef _GLIBCXX_USE_WCHAR_T
77 /// A string of @c wchar_t
78 typedef basic_string<wchar_t> wstring;
79#endif
80
81#if ((__cplusplus >= 201103L) \
82 && defined(_GLIBCXX_USE_C99_STDINT_TR1))
83 /// A string of @c char16_t
84 typedef basic_string<char16_t> u16string;
85
86 /// A string of @c char32_t
87 typedef basic_string<char32_t> u32string;
88#endif
89
90_GLIBCXX_END_NAMESPACE_CXX11
91
92 /** @} */
93
94_GLIBCXX_END_NAMESPACE_VERSION
95} // namespace std
96
97#endif // _STRINGFWD_H
