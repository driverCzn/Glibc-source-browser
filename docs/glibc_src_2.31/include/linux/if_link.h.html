8/* This struct should be in sync with struct rtnl_link_stats64 */
9struct rtnl_link_stats {
10 __u32 rx_packets; /* total packets received */
11 __u32 tx_packets; /* total packets transmitted */
12 __u32 rx_bytes; /* total bytes received */
13 __u32 tx_bytes; /* total bytes transmitted */
14 __u32 rx_errors; /* bad packets received */
15 __u32 tx_errors; /* packet transmit problems */
16 __u32 rx_dropped; /* no space in linux buffers */
17 __u32 tx_dropped; /* no space available in linux */
18 __u32 multicast; /* multicast packets received */
19 __u32 collisions;
20
21 /* detailed rx_errors: */
22 __u32 rx_length_errors;
23 __u32 rx_over_errors; /* receiver ring buff overflow */
24 __u32 rx_crc_errors; /* recved pkt with crc error */
25 __u32 rx_frame_errors; /* recv'd frame alignment error */
26 __u32 rx_fifo_errors; /* recv'r fifo overrun */
27 __u32 rx_missed_errors; /* receiver missed packet */
28
29 /* detailed tx_errors */
30 __u32 tx_aborted_errors;
31 __u32 tx_carrier_errors;
32 __u32 tx_fifo_errors;
33 __u32 tx_heartbeat_errors;
34 __u32 tx_window_errors;
35
36 /* for cslip etc */
37 __u32 rx_compressed;
38 __u32 tx_compressed;
39
40 __u32 rx_nohandler; /* dropped, no handler found */
41};

185/* ifi_flags.
186
187 IFF_* flags.
188
189 The only change is:
190 IFF_LOOPBACK, IFF_BROADCAST and IFF_POINTOPOINT are
191 more not changeable by user. They describe link media
192 characteristics and set by device driver.
193
194 Comments:
195 - Combination IFF_BROADCAST|IFF_POINTOPOINT is invalid
196 - If neither of these three flags are set;
197 the interface is NBMA.
198
199 - IFF_MULTICAST does not mean anything special:
200 multicasts can be used on all not-NBMA links.
201 IFF_MULTICAST means that this media uses special encapsulation
202 for multicast frames. Apparently, all IFF_POINTOPOINT and
203 IFF_BROADCAST devices are able to use multicasts too.
204 */
205
206/* IFLA_LINK.
207 For usual devices it is equal ifi_index.
208 If it is a "virtual interface" (f.e. tunnel), ifi_link
209 can point to real physical interface (f.e. for bandwidth calculations),
210 or maybe 0, what means, that real media is unknown (usual
211 for IPIP tunnels, when route to endpoint is allowed to change)
212 */
