18#include <locale.h>
19#include <string.h>
20#include <wctype.h>
21#include "../locale/localeinfo.h"
22
23wctrans_t
24__wctrans_l (const char *property, locale_t locale)
25{
26 const char *names;
27 size_t cnt;
28 size_t i;
29
30 names = locale->__locales[LC_CTYPE]->values[_NL_ITEM_INDEX (_NL_CTYPE_MAP_NAMES)].string;
31 cnt = 0;
32 while (names[0] != '\0')
33 {
34 if (strcmp (property, names) == 0)
35 break;
36
37 names = strchr (names, '\0') + 1;
38 ++cnt;
39 }
40
41 if (names[0] == '\0')
42 return 0;
43
44 i = locale->__locales[LC_CTYPE]->values[_NL_ITEM_INDEX (_NL_CTYPE_MAP_OFFSET)].word + cnt;
45 return (wctrans_t) locale->__locales[LC_CTYPE]->values[i].string;
46}
47weak_alias (__wctrans_l, wctrans_l)
