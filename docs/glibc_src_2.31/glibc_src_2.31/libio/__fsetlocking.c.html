18#include <stdio_ext.h>
19
20#undef __fsetlocking
21
22int
23__fsetlocking (FILE *fp, int type)
24{
25 int result = ((fp->_flags & _IO_USER_LOCK)
26 ? FSETLOCKING_BYCALLER : FSETLOCKING_INTERNAL);
27
28 if (type != FSETLOCKING_QUERY)
29 {
30 fp->_flags &= ~_IO_USER_LOCK;
31 if (type == FSETLOCKING_BYCALLER)
32 fp->_flags |= _IO_USER_LOCK;
33 }
34
35 return result;
36}
