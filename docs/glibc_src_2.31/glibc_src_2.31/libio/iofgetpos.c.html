27/* We need to avoid the header declarations of these, because
28 the types don't match _IO_fgetpos and then the compiler will
29 complain about the mismatch when we do the alias below. */
30#define _IO_new_fgetpos64 __renamed__IO_new_fgetpos64
31#define _IO_fgetpos64 __renamed__IO_fgetpos64
32#define fgetpos64 __renamed_fgetpos64
33
34#include "libioP.h"
35
36#undef _IO_new_fgetpos64
37#undef _IO_fgetpos64
38#undef fgetpos64
39
40#include <errno.h>
41#include <stdlib.h>
42#include <shlib-compat.h>
43
44int
45_IO_new_fgetpos (FILE *fp, __fpos_t *posp)
46{
47 off64_t pos;
48 int result = 0;
49 CHECK_FILE (fp, EOF);
50 _IO_acquire_lock (fp);
51 pos = _IO_seekoff_unlocked (fp, 0, _IO_seek_cur, 0);
52 if (_IO_in_backup (fp) && pos != _IO_pos_BAD)
53 {
54 if (fp->_mode <= 0)
55 pos -= fp->_IO_save_end - fp->_IO_save_base;
56 }
57 if (pos == _IO_pos_BAD)
58 {
59 /* ANSI explicitly requires setting errno to a positive value on
60 failure. */
61 if (errno == 0)
62 __set_errno (EIO);
63 result = EOF;
64 }
65 else if ((off64_t) (__typeof (posp->__pos)) pos != pos)
66 {
67 __set_errno (EOVERFLOW);
68 result = EOF;
69 }
70 else
71 {
72 posp->__pos = pos;
73 if (fp->_mode > 0 && __libio_codecvt_encoding (fp->_codecvt) < 0)
74 /* This is a stateful encoding, safe the state. */
75 posp->__state = fp->_wide_data->_IO_state;
76 }
77
78 _IO_release_lock (fp);
79 return result;
80}
