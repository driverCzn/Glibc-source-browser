22/* Copy no more than N wide-characters of SRC to DEST. */
23wchar_t *
24__wcsncpy_chk (wchar_t *dest, const wchar_t *src, size_t n, size_t destlen)
25{
26 if (__glibc_unlikely (destlen < n))
27 __chk_fail ();
28
29 /* This function is not often enough used to justify not using a
30 tail call. */
31 return __wcsncpy (dest, src, n);
32}
