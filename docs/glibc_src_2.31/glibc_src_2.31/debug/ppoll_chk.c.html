18#include <sys/poll.h>
19
20
21int
22__ppoll_chk (struct pollfd *fds, nfds_t nfds, const struct timespec *timeout,
23 const __sigset_t *ss, __SIZE_TYPE__ fdslen)
24{
25 if (fdslen / sizeof (*fds) < nfds)
26 __chk_fail ();
27
28 return ppoll (fds, nfds, timeout, ss);
29}
