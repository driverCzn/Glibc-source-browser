18#include <errno.h>
19#include <spawn.h>
20#include <unistd.h>
21
22#include "spawn_int.h"
23
24/* Add an action to FILE-ACTIONS which tells the implementation to call
25 `close' for the given file descriptor during the `spawn' call. */
26int
27__posix_spawn_file_actions_addclose (posix_spawn_file_actions_t *file_actions,
28 int fd)
29{
30 struct __spawn_action *rec;
31
32 if (!__spawn_valid_fd (fd))
33 return EBADF;
34
35 /* Allocate more memory if needed. */
36 if (file_actions->__used == file_actions->__allocated
37 && __posix_spawn_file_actions_realloc (file_actions) != 0)
38 /* This can only mean we ran out of memory. */
39 return ENOMEM;
40
41 /* Add the new value. */
42 rec = &file_actions->__actions[file_actions->__used];
43 rec->tag = spawn_do_close;
44 rec->action.open_action.fd = fd;
45
46 /* Account for the new entry. */
47 ++file_actions->__used;
48
49 return 0;
50}
51weak_alias (__posix_spawn_file_actions_addclose,
52 posix_spawn_file_actions_addclose)
