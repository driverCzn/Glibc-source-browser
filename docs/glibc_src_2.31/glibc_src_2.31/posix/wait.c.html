18#include <sys/wait.h>
19
20/* Wait for a child to die. When one does, put its status in *STAT_LOC
21 and return its process ID. For errors, return (pid_t) -1. */
22__pid_t
23__wait (int *stat_loc)
24{
25 return __waitpid (WAIT_ANY, stat_loc, 0);
26}
27weak_alias (__wait, wait)
