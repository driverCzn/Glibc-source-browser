20#include <string.h>

24/* Duplicate S, returning an identical malloc'd string. */
25wchar_t *
26wcsdup (const wchar_t *s)
27{
28 size_t len = (__wcslen (s) + 1) * sizeof (wchar_t);
29 void *new = malloc (len);
30
31 if (new == NULL)
32 return NULL;
33
34 return (wchar_t *) memcpy (new, (void *) s, len);
35}
