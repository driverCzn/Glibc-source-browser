19#include <string.h>
20#include <wchar.h>
21
22#undef mbsinit
23#undef __mbsinit
24
25/* In GNU libc the conversion functions only can convert between the
26 fixed wide character representation and the multibyte
27 representation of the same character set. Since we use ISO 10646
28 in UCS4 encoding for wide characters the best solution for
29 multibyte characters is the UTF8 encoding. I.e., the only state
30 information is a counter of the processed bytes so far and the
31 value collected so far. Especially, we don't have different shift
32 states. */
33int
34__mbsinit (const mbstate_t *ps)
35{
36 return ps == NULL || ps->__count == 0;
37}
38weak_alias (__mbsinit, mbsinit)
