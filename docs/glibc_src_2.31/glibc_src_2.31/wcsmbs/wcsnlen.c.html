19#include <wchar.h>

21#ifdef WCSNLEN
22# define __wcsnlen WCSNLEN
23#endif
24
25/* Return length of string S at most maxlen. */
26size_t
27__wcsnlen (const wchar_t *s, size_t maxlen)
28{
29 const wchar_t *ret = __wmemchr (s, L'\0', maxlen);
30 if (ret)
31 maxlen = ret - s;
32 return maxlen;
33}
34#ifndef WCSNLEN
35weak_alias (__wcsnlen, wcsnlen)
36#endif
