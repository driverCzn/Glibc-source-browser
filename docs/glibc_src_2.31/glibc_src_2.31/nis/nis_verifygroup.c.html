19#include <string.h>
20#include <rpcsvc/nis.h>
21#include <shlib-compat.h>

23nis_error
24nis_verifygroup (const_nis_name group)
25{
26 if (group != NULL && group[0] != '\0')
27 {
28 size_t grouplen = strlen (group);
29 char buf[grouplen + 50];
30 char leafbuf[grouplen + 2];
31 char domainbuf[grouplen + 2];
32 nis_result *res;
33 nis_error status;
34 char *cp, *cp2;
35
36 cp = stpcpy (buf, nis_leaf_of_r (group, leafbuf, sizeof (leafbuf) - 1));
37 cp = stpcpy (cp, ".groups_dir");
38 cp2 = nis_domain_of_r (group, domainbuf, sizeof (domainbuf) - 1);
39 if (cp2 != NULL && cp2[0] != '\0')
40 {
41 *cp++ = '.';
42 stpcpy (cp, cp2);
43 }
44 res = nis_lookup (buf, 0);
45 status = NIS_RES_STATUS (res);
46 nis_freeresult (res);
47 return status;
48 }
49 else
50 return NIS_FAIL;
51}
