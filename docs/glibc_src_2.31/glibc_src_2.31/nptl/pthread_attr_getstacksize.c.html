19#include "pthreadP.h"
20
21
22int
23__pthread_attr_getstacksize (const pthread_attr_t *attr, size_t *stacksize)
24{
25 struct pthread_attr *iattr;
26
27 iattr = (struct pthread_attr *) attr;
28
29 size_t size = iattr->stacksize;
30
31 /* If the user has not set a stack size we return what the system
32 will use as the default. */
33 if (size == 0)
34 {
35 lll_lock (__default_pthread_attr_lock, LLL_PRIVATE);
36 size = __default_pthread_attr.stacksize;
37 lll_unlock (__default_pthread_attr_lock, LLL_PRIVATE);
38 }
39 *stacksize = size;
40
41 return 0;
42}
43strong_alias (__pthread_attr_getstacksize, pthread_attr_getstacksize)
