22void
23_pthread_cleanup_push_defer (struct _pthread_cleanup_buffer *buffer,
24 void (*routine) (void *), void *arg)
25{
26 struct pthread *self = THREAD_SELF;
27
28 buffer->__routine = routine;
29 buffer->__arg = arg;
30 buffer->__prev = THREAD_GETMEM (self, cleanup);
31
32 int cancelhandling = THREAD_GETMEM (self, cancelhandling);
33
34 /* Disable asynchronous cancellation for now. */
35 if (__glibc_unlikely (cancelhandling & CANCELTYPE_BITMASK))
36 while (1)
37 {
38 int curval = THREAD_ATOMIC_CMPXCHG_VAL (self, cancelhandling,
39 cancelhandling
40 & ~CANCELTYPE_BITMASK,
41 cancelhandling);
42 if (__glibc_likely (curval == cancelhandling))
43 /* Successfully replaced the value. */
44 break;
45
46 /* Prepare for the next round. */
47 cancelhandling = curval;
48 }
49
50 buffer->__canceltype = (cancelhandling & CANCELTYPE_BITMASK
51 ? PTHREAD_CANCEL_ASYNCHRONOUS
52 : PTHREAD_CANCEL_DEFERRED);
53
54 THREAD_SETMEM (self, cleanup, buffer);
55}
56strong_alias (_pthread_cleanup_push_defer, __pthread_cleanup_push_defer)

59void
60_pthread_cleanup_pop_restore (struct _pthread_cleanup_buffer *buffer,
61 int execute)
62{
63 struct pthread *self = THREAD_SELF;
64
65 THREAD_SETMEM (self, cleanup, buffer->__prev);
66
67 int cancelhandling;
68 if (__builtin_expect (buffer->__canceltype != PTHREAD_CANCEL_DEFERRED, 0)
69 && ((cancelhandling = THREAD_GETMEM (self, cancelhandling))
70 & CANCELTYPE_BITMASK) == 0)
71 {
72 while (1)
73 {
74 int curval = THREAD_ATOMIC_CMPXCHG_VAL (self, cancelhandling,
75 cancelhandling
76 | CANCELTYPE_BITMASK,
77 cancelhandling);
78 if (__glibc_likely (curval == cancelhandling))
79 /* Successfully replaced the value. */
80 break;
81
82 /* Prepare for the next round. */
83 cancelhandling = curval;
84 }
85
86 CANCELLATION_P (self);
87 }
88
89 /* If necessary call the cleanup routine after we removed the
90 current cleanup block from the list. */
91 if (execute)
92 buffer->__routine (buffer->__arg);
93}
94strong_alias (_pthread_cleanup_pop_restore, __pthread_cleanup_pop_restore)
