19#include <time.h>
20#include <sysdep-cancel.h>
21
22#include "thrd_priv.h"
23
24int
25thrd_sleep (const struct timespec* time_point, struct timespec* remaining)
26{
27 int ret = __clock_nanosleep (CLOCK_REALTIME, 0, time_point, remaining);
28 /* C11 states thrd_sleep function returns -1 if it has been interrupted
29 by a signal, or a negative value if it fails. */
30 switch (ret)
31 {
32 case 0: return 0;
33 case EINTR: return -1;
34 default: return -2;
35 }
36}
