20#include <errno.h>
21#include <pthreadP.h>
22
23
24int
25pthread_mutex_getprioceiling (const pthread_mutex_t *mutex, int *prioceiling)
26{
27 /* See concurrency notes regarding __kind in struct __pthread_mutex_s
28 in sysdeps/nptl/bits/thread-shared-types.h. */
29 if (__builtin_expect ((atomic_load_relaxed (&(mutex->__data.__kind))
30 & PTHREAD_MUTEX_PRIO_PROTECT_NP) == 0, 0))
31 return EINVAL;
32
33 *prioceiling = (mutex->__data.__lock & PTHREAD_MUTEX_PRIO_CEILING_MASK)
34 >> PTHREAD_MUTEX_PRIO_CEILING_SHIFT;
35
36 return 0;
37}
