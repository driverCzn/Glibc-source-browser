21/* See pthread_rwlock_common.c. */
22int
23__pthread_rwlock_rdlock (pthread_rwlock_t *rwlock)
24{
25 LIBC_PROBE (rdlock_entry, 1, rwlock);
26
27 int result = __pthread_rwlock_rdlock_full (rwlock, CLOCK_REALTIME, NULL);
28 LIBC_PROBE (rdlock_acquire_read, 1, rwlock);
29 return result;
30}
