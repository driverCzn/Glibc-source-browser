22
23td_err_e
24td_ta_event_addr (const td_thragent_t *ta_arg,
25 td_event_e event, td_notify_t *addr)
26{
27 td_thragent_t *const ta = (td_thragent_t *) ta_arg;
28 td_err_e err;
29 psaddr_t taddr;
30
31 LOG ("td_ta_event_addr");
32
33 /* Test whether the TA parameter is ok. */
34 if (! ta_ok (ta))
35 return TD_BADTA;
36
37 switch (event)
38 {
39 case TD_CREATE:
40 err = DB_GET_SYMBOL (taddr, ta, __nptl_create_event);
41 break;
42
43 case TD_DEATH:
44 err = DB_GET_SYMBOL (taddr, ta, __nptl_death_event);
45 break;
46
47 default:
48 /* Event cannot be handled. */
49 return TD_NOEVENT;
50 }
51
52 if (err == TD_OK)
53 {
54 /* Success, we got the address. */
55 addr->type = NOTIFY_BPT;
56 addr->u.bptaddr = taddr;
57 }
58
59 return err;
60}
