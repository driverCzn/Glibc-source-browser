23/* Local buffer to store the result. */
24libc_freeres_ptr (static struct utmp *buffer);
25
26struct utmp *
27__getutid (const struct utmp *id)
28{
29 struct utmp *result;
30
31 if (buffer == NULL)
32 {
33 buffer = (struct utmp *) malloc (sizeof (struct utmp));
34 if (buffer == NULL)
35 return NULL;
36 }
37 if (__getutid_r (id, buffer, &result) < 0)
38 return NULL;
39
40 return result;
41}

43weak_alias (__getutid, getutid)
