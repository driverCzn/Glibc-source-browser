19#include <sys/timeb.h>
20#include <time.h>
21
22int
23ftime (struct timeb *timebuf)
24{
25 struct timespec ts;
26 __clock_gettime (CLOCK_REALTIME, &ts);
27
28 timebuf->time = ts.tv_sec;
29 timebuf->millitm = ts.tv_nsec / 1000000;
30 timebuf->timezone = 0;
31 timebuf->dstflag = 0;
32 return 0;
33}
