21/* Return the `struct tm' representation of *T in UTC,
22 using *TP to store the result. */
23struct tm *
24__gmtime64_r (const __time64_t *t, struct tm *tp)
25{
26 return __tz_convert (*t, 0, tp);
27}
28
29/* Provide a 32-bit variant if needed. */

31#if __TIMESIZE != 64

35struct tm *
36__gmtime_r (const time_t *t, struct tm *tp)

42#endif

44libc_hidden_def (__gmtime_r)
45weak_alias (__gmtime_r, gmtime_r)

47/* Return the `struct tm' representation of *T in UTC. */
48struct tm *
49__gmtime64 (const __time64_t *t)
50{
51 return __tz_convert (*t, 0, &_tmbuf);
52}
53
54/* Provide a 32-bit variant if needed. */
