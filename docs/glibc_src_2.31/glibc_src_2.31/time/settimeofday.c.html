18#include <errno.h>
19#include <time.h>
20#include <sys/time.h>
21
22/* Set the current time of day and timezone information.
23 This call is restricted to the super-user. */
24int
25__settimeofday (const struct timeval *tv, const struct timezone *tz)
26{
27 if (__glibc_unlikely (tz != 0))
28 {
29 if (tv != 0)
30 {
31 __set_errno (EINVAL);
32 return -1;
33 }
34 return __settimezone (tz);
35 }
36
37 struct timespec ts;
38 TIMEVAL_TO_TIMESPEC (tv, &ts);
39 return __clock_settime (CLOCK_REALTIME, &ts);
40}
41
42#ifdef VERSION_settimeofday
43weak_alias (__settimeofday, __settimeofday_w);
44default_symbol_version (__settimeofday_w, settimeofday, VERSION_settimeofday);
45#else
46weak_alias (__settimeofday, settimeofday);
47#endif
