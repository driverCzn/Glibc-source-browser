21/* Set TS to calendar time based in time base BASE. */
22int
23timespec_get (struct timespec *ts, int base)
24{
25 if (base == TIME_UTC)
26 {
27 __clock_gettime (CLOCK_REALTIME, ts);
28 return base;
29 }
30 return 0;
31}
