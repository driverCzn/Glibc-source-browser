18#include <unistd.h>
19
20void
21swab (const void *bfrom, void *bto, ssize_t n)
22{
23 const char *from = (const char *) bfrom;
24 char *to = (char *) bto;
25
26 n &= ~((ssize_t) 1);
27 while (n > 1)
28 {
29 const char b0 = from[--n], b1 = from[--n];
30 to[n] = b0;
31 to[n + 1] = b1;
32 }
33}
