32#undef __strndup
33#undef strndup

35#ifndef weak_alias
36# define __strndup strndup
37#endif
38
39char *
40__strndup (const char *s, size_t n)
41{
42 size_t len = __strnlen (s, n);
43 char *new = (char *) malloc (len + 1);
44
45 if (new == NULL)
46 return NULL;
47
48 new[len] = '\0';
49 return (char *) memcpy (new, s, len);
50}
