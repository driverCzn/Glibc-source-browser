397static inline void
398mul_splitf128 (_Float128 *hi, _Float128 *lo, _Float128 x, _Float128 y)
399{
400#ifdef __FP_FAST_FMAF128
401 /* Fast built-in fused multiply-add. */
402 *hi = x * y;
403 *lo = __builtin_fmal (x, y, -*hi);
404#else
405 /* Apply Dekker's algorithm. */
406 *hi = x * y;
407# define C ((1LL << (FLT128_MANT_DIG + 1) / 2) + 1)
408 _Float128 x1 = x * C;
409 _Float128 y1 = y * C;
410# undef C
411 x1 = (x - x1) + x1;
412 y1 = (y - y1) + y1;
413 _Float128 x2 = x - x1;
414 _Float128 y2 = y - y1;
415 *lo = (((x1 * y1 - *hi) + x1 * y2) + x2 * y1) + x2 * y2;
416#endif
417}
