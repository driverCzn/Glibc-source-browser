25int
26__fpclassifyf (float x)
27{
28 uint32_t wx;
29 int retval = FP_NORMAL;
30
31 GET_FLOAT_WORD (wx, x);
32 wx &= 0x7fffffff;
33 if (wx == 0)
34 retval = FP_ZERO;
35 else if (wx < 0x800000)
36 retval = FP_SUBNORMAL;
37 else if (wx >= 0x7f800000)
38 retval = wx > 0x7f800000 ? FP_NAN : FP_INFINITE;
39
40 return retval;
41}
