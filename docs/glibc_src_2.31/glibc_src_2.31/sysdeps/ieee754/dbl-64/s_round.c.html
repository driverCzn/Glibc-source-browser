29double
30__round (double x)
31{
32#if USE_ROUND_BUILTIN
33 return __builtin_round (x);
34#else
35 /* Use generic implementation. */
36 int64_t i0, j0;
37
38 EXTRACT_WORDS64 (i0, x);
39 j0 = ((i0 >> 52) & 0x7ff) - 0x3ff;
40 if (__glibc_likely (j0 < 52))
41 {
42 if (j0 < 0)
43 {
44 i0 &= UINT64_C (0x8000000000000000);
45 if (j0 == -1)
46 i0 |= UINT64_C (0x3ff0000000000000);
47 }
48 else
49 {
50 uint64_t i = UINT64_C (0x000fffffffffffff) >> j0;
51 if ((i0 & i) == 0)
52 /* X is integral. */
53 return x;
54
55 i0 += UINT64_C (0x0008000000000000) >> j0;
56 i0 &= ~i;
57 }
58 }
59 else
60 {
61 if (j0 == 0x400)
62 /* Inf or NaN. */
63 return x + x;
64 else
65 return x;
66 }
67
68 INSERT_WORDS64 (x, i0);
69 return x;
70#endif /* ! USE_ROUND_BUILTIN */
71}
