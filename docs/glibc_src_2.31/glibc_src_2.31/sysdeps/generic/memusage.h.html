28#ifndef GETTIME
29# define GETTIME(low,high) \
30 { \
31 struct timespec now; \
32 uint64_t usecs; \
33 clock_gettime (CLOCK_REALTIME, &now); \
34 usecs = (uint64_t)now.tv_nsec / 1000 + (uint64_t)now.tv_sec * 1000000; \
35 low = usecs & 0xffffffff; \
36 high = usecs >> 32; \
37 }
38#endif
