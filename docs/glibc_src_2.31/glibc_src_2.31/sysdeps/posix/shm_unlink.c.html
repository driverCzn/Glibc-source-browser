31/* Remove shared memory object. */
32int
33shm_unlink (const char *name)
34{
35 SHM_GET_NAME (ENOENT, -1, "");
36
37 int result = unlink (shm_name);
38 if (result < 0 && errno == EPERM)
39 __set_errno (EACCES);
40 return result;
41}
