25extern void REDIRECT_NAME (void);
26extern __typeof (REDIRECT_NAME) OPTIMIZE (avx2_wrapper) attribute_hidden;
27extern __typeof (REDIRECT_NAME) OPTIMIZE (knl) attribute_hidden;
28extern __typeof (REDIRECT_NAME) OPTIMIZE (skx) attribute_hidden;
29
30static inline void *
31IFUNC_SELECTOR (void)
32{
33 const struct cpu_features* cpu_features = __get_cpu_features ();
34
35 if (!CPU_FEATURES_ARCH_P (cpu_features, MathVec_Prefer_No_AVX512))
36 {
37 if (CPU_FEATURES_ARCH_P (cpu_features, AVX512DQ_Usable))
38 return OPTIMIZE (skx);
39
40 if (CPU_FEATURES_ARCH_P (cpu_features, AVX512F_Usable))
41 return OPTIMIZE (knl);
42 }
43
44 return OPTIMIZE (avx2_wrapper);
45}
