20#include <sysdep.h>
21#include <libm-alias-double.h>
22
23 .text
24ENTRY(__fmin)
25 ucomisd %xmm0, %xmm1
26 jp 1f // jump if unordered
27 minsd %xmm1, %xmm0
28 jmp 2f
29
301: ucomisd %xmm1, %xmm1 // Is xmm1 a NaN?
31 jp 3f
32 // xmm0 is a NaN; xmm1 is not. Test if xmm0 is signaling.
33 movsd %xmm0, -8(%rsp)
34 testb $0x8, -2(%rsp)
35 jz 4f
36 movsd %xmm1, %xmm0 // otherwise return xmm1
37 ret
38
393: // xmm1 is a NaN; xmm0 may or may not be.
40 ucomisd %xmm0, %xmm0
41 jp 4f
42 // xmm1 is a NaN; xmm0 is not. Test if xmm1 is signaling.
43 movsd %xmm1, -8(%rsp)
44 testb $0x8, -2(%rsp)
45 jz 4f
46 ret
47
484: // Both arguments are NaNs, or one is a signaling NaN.
49 addsd %xmm1, %xmm0
50
512: ret
52END(__fmin)
53libm_alias_double (__fmin, fmin)
