19#include <init-arch.h>

21extern __typeof (REDIRECT_NAME) OPTIMIZE (sse2_unaligned) attribute_hidden;
22extern __typeof (REDIRECT_NAME) OPTIMIZE (avx2_unaligned) attribute_hidden;
23extern __typeof (REDIRECT_NAME) OPTIMIZE (avx512_unaligned) attribute_hidden;
24
25static inline void *
26IFUNC_SELECTOR (void)
27{
28 const struct cpu_features* cpu_features = __get_cpu_features ();
29
30 if (!CPU_FEATURES_ARCH_P (cpu_features, Prefer_No_VZEROUPPER)
31 && CPU_FEATURES_ARCH_P (cpu_features, AVX2_Usable)
32 && CPU_FEATURES_ARCH_P (cpu_features, AVX_Fast_Unaligned_Load))
33 {
34 if (CPU_FEATURES_ARCH_P (cpu_features, AVX512F_Usable)
35 && !CPU_FEATURES_ARCH_P (cpu_features, Prefer_No_AVX512))
36 return OPTIMIZE (avx512_unaligned);
37 else
38 return OPTIMIZE (avx2_unaligned);
39 }
40
41 return OPTIMIZE (sse2_unaligned);
42}
