35# ifdef USE_AS_WMEMCMP
36 shl $2, %RDX_LP
37 test %RDX_LP, %RDX_LP
38 jz L(equal)
39# elif defined __ILP32__
40 /* Clear the upper 32 bits. */
41 mov %edx, %edx
42# endif
43 mov %rdx, %rcx
44 mov %rdi, %rdx
45 cmp $48, %rcx;

47
48 add %rcx, %rsi
49 add %rcx, %rdi

64 and $0xf, %edx
65 xor %rdx, %rdi
66 sub %rdx, %rsi
