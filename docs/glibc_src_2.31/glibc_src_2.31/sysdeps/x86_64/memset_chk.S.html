19#include <sysdep.h>
20#include "asm-syntax.h"
21
22#ifndef SHARED
23 /* For libc.so this is defined in memset.S.
24 For libc.a, this is a separate source to avoid
25 memset bringing in __chk_fail and all routines
26 it calls. */
27 .text
28ENTRY (__memset_chk)
29 cmpq %rdx, %rcx
30 jb __chk_fail
31 jmp memset
32END (__memset_chk)
33#endif
