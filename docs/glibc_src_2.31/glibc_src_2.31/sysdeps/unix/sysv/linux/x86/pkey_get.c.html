19#include <arch-pkey.h>
20#include <errno.h>
21
22int
23pkey_get (int key)
24{
25 if (key < 0 || key > 15)
26 {
27 __set_errno (EINVAL);
28 return -1;
29 }
30 unsigned int pkru = pkey_read ();
31 return (pkru >> (2 * key)) & 3;
32 return 0;
33}
