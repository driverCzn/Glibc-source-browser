18#include <errno.h>
19#include <unistd.h>
20#include <sys/eventfd.h>

22
23int
24eventfd_write (int fd, eventfd_t value)
25{
26 return __write (fd, &value,
27 sizeof (eventfd_t)) != sizeof (eventfd_t) ? -1 : 0;
28}
