28/* Change the access time of FILE to TVP[0] and
29 the modification time of FILE to TVP[1]. */
30int
31__utimes (const char *file, const struct timeval tvp[2])
32{
33 /* Avoid implicit array coercion in syscall macros. */
34 return INLINE_SYSCALL (utimes, 2, file, &tvp[0]);
35}
