19#include <sys/select.h>
20#include <sysdep-cancel.h>
21
22int
23__pselect (int nfds, fd_set *readfds, fd_set *writefds, fd_set *exceptfds,
24 const struct timespec *timeout, const sigset_t *sigmask)
25{
26 /* The Linux kernel can in some situations update the timeout value.
27 We do not want that so use a local variable. */
28 struct timespec tval;
29 if (timeout != NULL)
30 {
31 tval = *timeout;
32 timeout = &tval;
33 }
34
35 /* Note: the system call expects 7 values but on most architectures
36 we can only pass in 6 directly. If there is an architecture with
37 support for more parameters a new version of this file needs to
38 be created. */
39 struct
40 {
41 __syscall_ulong_t ss;
42 __syscall_ulong_t ss_len;
43 } data;
44
45 data.ss = (__syscall_ulong_t) (uintptr_t) sigmask;
46 data.ss_len = _NSIG / 8;
47
48 return SYSCALL_CANCEL (pselect6, nfds, readfds, writefds, exceptfds,
49 timeout, &data);
50}
51#ifndef __pselect
52weak_alias (__pselect, pselect)
53#endif
