24/* Set CLOCK to value TP. */
25int
26__clock_settime64 (clockid_t clock_id, const struct __timespec64 *tp)
27{
28 /* Make sure the time cvalue is OK. */
29 if (! valid_nanoseconds (tp->tv_nsec))
30 {
31 __set_errno (EINVAL);
32 return -1;
33 }
34
35#ifdef __ASSUME_TIME64_SYSCALLS
36# ifndef __NR_clock_settime64
37# define __NR_clock_settime64 __NR_clock_settime
38# endif
39 return INLINE_SYSCALL_CALL (clock_settime64, clock_id, tp);
40#else
41# ifdef __NR_clock_settime64
42 int ret = INLINE_SYSCALL_CALL (clock_settime64, clock_id, tp);
43 if (ret == 0 || errno != ENOSYS)
44 return ret;
45# endif
46 if (! in_time_t_range (tp->tv_sec))
47 {
48 __set_errno (EOVERFLOW);
49 return -1;
50 }
51
52 struct timespec ts32 = valid_timespec64_to_timespec (*tp);
53 return INLINE_SYSCALL_CALL (clock_settime, clock_id, &ts32);
54#endif
55}
56
57#if __TIMESIZE != 64
58int
59__clock_settime (clockid_t clock_id, const struct timespec *tp)
60{
61 struct __timespec64 ts64 = valid_timespec_to_timespec64 (*tp);
62
63 return __clock_settime64 (clock_id, &ts64);
64}
65#endif
