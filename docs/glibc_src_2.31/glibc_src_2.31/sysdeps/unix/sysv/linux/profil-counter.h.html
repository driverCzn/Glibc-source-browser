19#include <signal.h>
20#include <sigcontextinfo.h>
21
22/* sa_sigaction signature to use along SA_SIGINFO. */
23static void
24__profil_counter (int signo, siginfo_t *info, void *ctx)
25{
26 profil_count (sigcontext_get_pc (ctx));
27
28 /* This is a hack to prevent the compiler from implementing the
29 above function call as a sibcall. The sibcall would overwrite
30 the signal context. */
31 asm volatile ("");
32}
