20#include <alloca.h>
21#include <errno.h>
22#include <stdlib.h>
23#include <string.h>
24#include <stdint.h>
25#include <netinet/in.h>
26#include <sys/socket.h>

29
30int
31setsourcefilter (int s, uint32_t interface, const struct sockaddr *group,
32 socklen_t grouplen, uint32_t fmode, uint32_t numsrc,
33 const struct sockaddr_storage *slist)
34{
35 /* We have to create an struct ip_msfilter object which we can pass
36 to the kernel. */
37 size_t needed = GROUP_FILTER_SIZE (numsrc);
38 int use_alloca = __libc_use_alloca (needed);
39
40 struct group_filter *gf;
41 if (use_alloca)
42 gf = (struct group_filter *) alloca (needed);
43 else
44 {
45 gf = (struct group_filter *) malloc (needed);
46 if (gf == NULL)
47 return -1;
48 }
49
50 gf->gf_interface = interface;
51 memcpy (&gf->gf_group, group, grouplen);
52 gf->gf_fmode = fmode;
53 gf->gf_numsrc = numsrc;
54 memcpy (gf->gf_slist, slist, numsrc * sizeof (struct sockaddr_storage));
55
56 /* We need to provide the appropriate socket level value. */
57 int result;
58 int sol = __get_sol (group->sa_family, grouplen);
59 if (sol == -1)
60 {
61 __set_errno (EINVAL);
62 result = -1;
63 }
64 else
65 result = __setsockopt (s, sol, MCAST_MSFILTER, gf, needed);
66
67 if (! use_alloca)
68 {
69 int save_errno = errno;
70 free (gf);
71 __set_errno (save_errno);
72 }
73
74 return result;
75}
