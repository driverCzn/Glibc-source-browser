19#include <sys/uio.h>
20#include <sysdep-cancel.h>
21
22#if !defined(__NR_pwritev64v2) && defined(__NR_pwritev2)
23# define __NR_pwritev64v2 __NR_pwritev2
24#endif
25
26ssize_t
27pwritev64v2 (int fd, const struct iovec *vector, int count, off64_t offset,
28 int flags)
29{
30#ifdef __NR_pwritev64v2
31 ssize_t result = SYSCALL_CANCEL (pwritev64v2, fd, vector, count,
32 LO_HI_LONG (offset), flags);
33 if (result >= 0 || errno != ENOSYS)
34 return result;
35#endif
36 /* Trying to emulate the pwritev2 syscall flags is troublesome:
37
38 * We can not temporary change the file state of the O_DSYNC and O_SYNC
39 flags to emulate RWF_{D}SYNC (attempts to change the state of using
40 fcntl are silently ignored).
41
42 * IOCB_HIPRI requires the file opened in O_DIRECT and uses an internal
43 semantic not provided by any other flag (O_NONBLOCK for instance). */
44
45 if (flags != 0)
46 {
47 __set_errno (ENOTSUP);
48 return -1;
49 }
50 if (offset == -1)
51 return __writev (fd, vector, count);
52 else
53 return pwritev64 (fd, vector, count, offset);
54}
55
56#ifdef __OFF_T_MATCHES_OFF64_T
57strong_alias (pwritev64v2, pwritev2)
58#endif
