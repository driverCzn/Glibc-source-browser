20#ifndef _SYSDEP_CANCEL_H
21#define _SYSDEP_CANCEL_H
22
23#include <sysdep.h>
24#include <tls.h>
25#include <errno.h>
26
27/* The two functions are in libc.so and not exported. */
28extern int __libc_enable_asynccancel (void) attribute_hidden;
29extern void __libc_disable_asynccancel (int oldtype) attribute_hidden;
30
31/* The two functions are in librt.so and not exported. */
32extern int __librt_enable_asynccancel (void) attribute_hidden;
33extern void __librt_disable_asynccancel (int oldtype) attribute_hidden;
34
35/* The two functions are in libpthread.so and not exported. */
36extern int __pthread_enable_asynccancel (void) attribute_hidden;
37extern void __pthread_disable_asynccancel (int oldtype) attribute_hidden;
38
39/* Set cancellation mode to asynchronous. */
40#define CANCEL_ASYNC() \
41 __pthread_enable_asynccancel ()
42/* Reset to previous cancellation mode. */
43#define CANCEL_RESET(oldtype) \
44 __pthread_disable_asynccancel (oldtype)
45
46#if IS_IN (libc)
47/* Same as CANCEL_ASYNC, but for use in libc.so. */
48# define LIBC_CANCEL_ASYNC() \
49 __libc_enable_asynccancel ()
50/* Same as CANCEL_RESET, but for use in libc.so. */
51# define LIBC_CANCEL_RESET(oldtype) \
52 __libc_disable_asynccancel (oldtype)
53#elif IS_IN (libpthread)
54# define LIBC_CANCEL_ASYNC() CANCEL_ASYNC ()
55# define LIBC_CANCEL_RESET(val) CANCEL_RESET (val)
56#elif IS_IN (librt)
57# define LIBC_CANCEL_ASYNC() \
58 __librt_enable_asynccancel ()
59# define LIBC_CANCEL_RESET(val) \
60 __librt_disable_asynccancel (val)
61#else
62# define LIBC_CANCEL_ASYNC() 0 /* Just a dummy value. */
63# define LIBC_CANCEL_RESET(val) ((void)(val)) /* Nothing, but evaluate it. */
64#endif
65
66
67#endif
