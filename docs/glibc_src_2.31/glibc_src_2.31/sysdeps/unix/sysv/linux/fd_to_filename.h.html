19#include <sys/stat.h>
20#include <string.h>
21#include <_itoa.h>
22
23#define FD_TO_FILENAME_SIZE ((sizeof ("/proc/self/fd/") - 1) \
24 + (sizeof ("4294967295") - 1) + 1)
25
26static inline const char *
27fd_to_filename (unsigned int fd, char *buf)
28{
29 *_fitoa_word (fd, __stpcpy (buf, "/proc/self/fd/"), 10, 0) = '\0';
30
31 /* We must make sure the file exists. */
32 struct stat64 st;
33 if (__lxstat64 (_STAT_VER, buf, &st) < 0)
34 /* /proc is not mounted or something else happened. */
35 return NULL;
36 return buf;
37}
