18#include <signal.h>
19#include <nptl/pthreadP.h> /* SIGCANCEL, SIGSETXID */
20
21/* Get and/or change the set of blocked signals. */
22int
23__sigprocmask (int how, const sigset_t *set, sigset_t *oset)
24{
25 sigset_t local_newmask;
26
27 /* The only thing we have to make sure here is that SIGCANCEL and
28 SIGSETXID are not blocked. */
29 if (set != NULL
30 && __glibc_unlikely (__sigismember (set, SIGCANCEL)
31 || __glibc_unlikely (__sigismember (set, SIGSETXID))))
32 {
33 local_newmask = *set;
34 __sigdelset (&local_newmask, SIGCANCEL);
35 __sigdelset (&local_newmask, SIGSETXID);
36 set = &local_newmask;
37 }
38
39 return INLINE_SYSCALL_CALL (rt_sigprocmask, how, set, oset, _NSIG / 8);
40}
41libc_hidden_def (__sigprocmask)
42weak_alias (__sigprocmask, sigprocmask)
