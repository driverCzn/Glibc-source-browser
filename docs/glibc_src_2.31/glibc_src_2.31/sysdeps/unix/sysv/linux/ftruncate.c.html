18#include <unistd.h>
19#include <sysdep.h>
20#include <errno.h>
21
22#ifndef __OFF_T_MATCHES_OFF64_T
23/* Truncate the file FD refers to LENGTH bytes. */
24int
25__ftruncate (int fd, off_t length)
26{
27# ifndef __NR_ftruncate
28 return INLINE_SYSCALL_CALL (ftruncate64, fd,
29 __ALIGNMENT_ARG SYSCALL_LL (length));
30# else
31 return INLINE_SYSCALL_CALL (ftruncate, fd, length);
32# endif
33}
34weak_alias (__ftruncate, ftruncate)
35#endif
