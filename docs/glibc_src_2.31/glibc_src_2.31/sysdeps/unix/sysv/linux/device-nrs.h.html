19#ifndef _DEVICE_NRS_H
20#define _DEVICE_NRS_H 1
21
22#include <sys/sysmacros.h>
23
24/* /dev/null is (1,3). */
25#define DEV_NULL_MAJOR 1
26#define DEV_NULL_MINOR 3
27
28/* /dev/full is (1,7). */
29#define DEV_FULL_MAJOR 1
30#define DEV_FULL_MINOR 7
31
32/* Pseudo tty slaves. For Linux we use the Unix98 ttys. We could
33 also include the old BSD-style tty buts they should not be used and
34 the extra test would only slow down correctly set up systems. If a
35 system still uses those device the slower tests performed (using
36 isatty) will catch it. */
37#define DEV_TTY_LOW_MAJOR 136
38#define DEV_TTY_HIGH_MAJOR 143
39
40/* Test whether given device is a tty. */
41#define DEV_TTY_P(statp) \
42 ({ int __dev_major = __gnu_dev_major ((statp)->st_rdev); \
43 __dev_major >= DEV_TTY_LOW_MAJOR && __dev_major <= DEV_TTY_HIGH_MAJOR; })
44
45#endif /* device-nrs.h */
