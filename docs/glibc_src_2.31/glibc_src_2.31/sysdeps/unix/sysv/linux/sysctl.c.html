25int
26__sysctl (int *name, int nlen, void *oldval, size_t *oldlenp,
27 void *newval, size_t newlen)
28{
29 struct __sysctl_args args =
30 {
31 .name = name,
32 .nlen = nlen,
33 .oldval = oldval,
34 .oldlenp = oldlenp,
35 .newval = newval,
36 .newlen = newlen
37 };
38
39 return INLINE_SYSCALL (_sysctl, 1, &args);
40}
