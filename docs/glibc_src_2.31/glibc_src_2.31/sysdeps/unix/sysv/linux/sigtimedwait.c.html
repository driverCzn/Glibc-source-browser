18#include <errno.h>
19#include <signal.h>
20#include <string.h>
21#include <sysdep-cancel.h>
22
23int
24__sigtimedwait (const sigset_t *set, siginfo_t *info,
25 const struct timespec *timeout)
26{
27 /* XXX The size argument hopefully will have to be changed to the
28 real size of the user-level sigset_t. */
29 int result = SYSCALL_CANCEL (rt_sigtimedwait, set, info, timeout, _NSIG / 8);
30
31 /* The kernel generates a SI_TKILL code in si_code in case tkill is
32 used. tkill is transparently used in raise(). Since having
33 SI_TKILL as a code is useful in general we fold the results
34 here. */
35 if (result != -1 && info != NULL && info->si_code == SI_TKILL)
36 info->si_code = SI_USER;
37
38 return result;
39}
