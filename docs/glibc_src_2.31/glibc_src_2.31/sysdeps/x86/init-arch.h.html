27/* Due to the reordering and the other nifty extensions in i686, it is
28 not really good to use heavily i586 optimized code on an i686. It's
29 better to use i486 code if it isn't an i586. */

31# define USE_I586 0
32# define USE_I686 1
33# else
34# define USE_I586 (HAS_ARCH_FEATURE (I586) && !HAS_ARCH_FEATURE (I686))

36# endif
