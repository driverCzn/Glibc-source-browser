24FILE *
25xfopen (const char *path, const char *mode)
26{
27 FILE *fp = fopen (path, mode);
28 if (fp == NULL)
29 FAIL_EXIT1 ("could not open %s (mode \"%s\"): %m", path, mode);
30 return fp;
31}
