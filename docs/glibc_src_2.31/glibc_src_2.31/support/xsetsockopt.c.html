25void
26xsetsockopt (int fd, int level, int name, const void *val, socklen_t vallen)
27{
28 if (setsockopt (fd, level, name, val, vallen) != 0)
29 FAIL_EXIT1 ("setsockopt (%d, %d, %d), %zu bytes: %m",
30 fd, level, name, (size_t) vallen);
31}
