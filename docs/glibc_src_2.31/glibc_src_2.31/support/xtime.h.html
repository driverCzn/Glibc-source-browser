19#ifndef SUPPORT_TIME_H
20#define SUPPORT_TIME_H
21
22#include <time.h>
23
24__BEGIN_DECLS
25
26/* The following functions call the corresponding libc functions and
27 terminate the process on error. */
28
29void xclock_gettime (clockid_t clock, struct timespec *ts);
30
31/* This helper can often simplify tests by avoiding an explicit
32 variable declaration or allowing that declaration to be const. */
33
34static inline struct timespec xclock_now (clockid_t clock)
35{
36 struct timespec ts;
37 xclock_gettime (clock, &ts);
38 return ts;
39}
40
41__END_DECLS
42
43#endif /* SUPPORT_TIME_H */
