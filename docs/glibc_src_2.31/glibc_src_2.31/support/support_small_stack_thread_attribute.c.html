19#include <support/xthread.h>
20#include <allocate_once.h>
21
22static void *
23allocate (void *closure)
24{
25 pthread_attr_t *result = malloc (sizeof (*result));
26 xpthread_attr_init (result);
27 support_set_small_thread_stack_size (result);
28 return result;
29}
30
31static void
32deallocate (void *ptr, void *closure)
33{
34 xpthread_attr_destroy (ptr);
35 free (ptr);
36}
37
38static void *small_stack_attr;
39
40pthread_attr_t *
41support_small_stack_thread_attribute (void)
42{
43 return allocate_once (&small_stack_attr, allocate, deallocate, NULL);
44}
45
46static void __attribute__ ((destructor))
47fini (void)
48{
49 if (small_stack_attr != NULL)
50 {
51 deallocate (small_stack_attr, NULL);
52 small_stack_attr = NULL;
53 }
54}
