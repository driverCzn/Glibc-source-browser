19#include <support/check.h>
20#include <support/xunistd.h>
21
22void
23support_isolate_in_subprocess (void (*callback) (void *), void *closure)
24{
25 pid_t pid = xfork ();
26 if (pid == 0)
27 {
28 /* Child process. */
29 callback (closure);
30 _exit (0);
31 }
32
33 /* Parent process. */
34 int status;
35 xwaitpid (pid, &status, 0);
36 if (status != 0)
37 FAIL_EXIT1 ("child process exited with status %d", status);
38}
