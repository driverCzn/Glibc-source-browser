23int
24xopen (const char *path, int flags, mode_t mode)
25{
26 int ret = open64 (path, flags, mode);
27 if (ret < 0)
28 FAIL_EXIT1 ("open64 (\"%s\", 0x%x, 0%o): %m", path, flags, mode);
29 return ret;
30}
