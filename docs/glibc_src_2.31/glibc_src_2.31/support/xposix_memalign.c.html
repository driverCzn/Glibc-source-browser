23void *
24xposix_memalign (size_t alignment, size_t n)
25{
26 void *p = NULL;
27
28 int ret = posix_memalign (&p, alignment, n);
29 if (ret)
30 {
31 errno = ret;
32 oom_error ("posix_memalign", n);
33 }
34 return p;
35}
