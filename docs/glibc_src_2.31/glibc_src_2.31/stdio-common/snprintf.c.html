19#include <libioP.h>

21/* Write formatted output into S, according to the format
22 string FORMAT, writing no more than MAXLEN characters. */
23/* VARARGS3 */
24int
25__snprintf (char *s, size_t maxlen, const char *format, ...)
26{
27 va_list arg;
28 int done;
29
30 va_start (arg, format);
31 done = __vsnprintf_internal (s, maxlen, format, arg, 0);
32 va_end (arg);
33
34 return done;
35}
36libc_hidden_def (__snprintf)
37ldbl_weak_alias (__snprintf, snprintf)
