39#include <unistd.h>
40
41extern void __libc_print_version (void) attribute_hidden;
42void
43__libc_print_version (void)
44{
45 __write (STDOUT_FILENO, banner, sizeof banner - 1);
46}
47
48extern const char *__gnu_get_libc_release (void);
49const char *
50__gnu_get_libc_release (void)
51{
52 return __libc_release;
53}
54weak_alias (__gnu_get_libc_release, gnu_get_libc_release)
55
56extern const char *__gnu_get_libc_version (void);
57const char *
58__gnu_get_libc_version (void)
59{
60 return __libc_version;
61}
62weak_alias (__gnu_get_libc_version, gnu_get_libc_version)
63
64/* This function is the entry point for the shared object.
65 Running the library as a program will get here. */
66
67extern void __libc_main (void) __attribute__ ((noreturn));
68void
69__libc_main (void)
70{
71 __libc_print_version ();
72 _exit (0);
73}
