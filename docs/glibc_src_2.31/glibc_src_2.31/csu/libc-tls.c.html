35static struct dtv_slotinfo_list static_slotinfo =
36 {
37 /* Allocate an array of 2 + TLS_SLOTINFO_SURPLUS elements. */
38 .slotinfo = { [array_length (_dl_static_dtv) - 1] = { 0 } },
39 };
40
41/* Highest dtv index currently needed. */
42size_t _dl_tls_max_dtv_idx;
43/* Flag signalling whether there are gaps in the module ID allocation. */
44bool _dl_tls_dtv_gaps;
45/* Information about the dtv slots. */
46struct dtv_slotinfo_list *_dl_tls_dtv_slotinfo_list;
47/* Number of modules in the static TLS block. */
48size_t _dl_tls_static_nelem;
49/* Size of the static TLS block. Giving this initialized value
50 preallocates some surplus bytes in the static TLS area. */
51size_t _dl_tls_static_size = 2048;
52/* Size actually allocated in the static TLS block. */
53size_t _dl_tls_static_used;
54/* Alignment requirement of the static TLS block. */
55size_t _dl_tls_static_align;
56
57/* Generation counter for the dtv. */
58size_t _dl_tls_generation;

66static void
67init_slotinfo (void)
68{
69 /* Create the slotinfo list. Note that the type of static_slotinfo
70 has effectively a zero-length array, so we cannot use the size of
71 static_slotinfo to determine the array length. */
72 static_slotinfo.len = array_length (_dl_static_dtv);
73 /* static_slotinfo.next = NULL; -- Already zero. */
74
75 /* The slotinfo list. Will be extended by the code doing dynamic
76 linking. */
77 GL(dl_tls_max_dtv_idx) = 1;
78 GL(dl_tls_dtv_slotinfo_list) = &static_slotinfo;
79}

81static void
82init_static_tls (size_t memsz, size_t align)
83{
84 /* That is the size of the TLS memory for this object. The initialized
85 value of _dl_tls_static_size is provided by dl-open.c to request some
86 surplus that permits dynamic loading of modules with IE-model TLS. */
87 GL(dl_tls_static_size) = roundup (memsz + GL(dl_tls_static_size),
88 TLS_TCB_ALIGN);
89#if TLS_TCB_AT_TP
90 GL(dl_tls_static_size) += TLS_TCB_SIZE;
91#endif
92 GL(dl_tls_static_used) = memsz;
93 /* The alignment requirement for the static TLS block. */
94 GL(dl_tls_static_align) = align;
95 /* Number of elements in the static TLS block. */
96 GL(dl_tls_static_nelem) = GL(dl_tls_max_dtv_idx);
97}

99void
100__libc_setup_tls (void)
101{
102 void *tlsblock;
103 size_t memsz = 0;
104 size_t filesz = 0;
105 void *initimage = NULL;
106 size_t align = 0;
107 size_t max_align = TCB_ALIGNMENT;
108 size_t tcb_offset;
109 const ElfW(Phdr) *phdr;
110
111 struct link_map *main_map = GL(dl_ns)[LM_ID_BASE]._ns_loaded;
112
113 /* Look through the TLS segment if there is any. */
114 if (_dl_phdr != NULL)
115 for (phdr = _dl_phdr; phdr < &_dl_phdr[_dl_phnum]; ++phdr)
116 if (phdr->p_type == PT_TLS)
117 {
118 /* Remember the values we need. */
119 memsz = phdr->p_memsz;
120 filesz = phdr->p_filesz;
121 initimage = (void *) phdr->p_vaddr + main_map->l_addr;
122 align = phdr->p_align;
123 if (phdr->p_align > max_align)
124 max_align = phdr->p_align;
125 break;
126 }
127
128 /* We have to set up the TCB block which also (possibly) contains
129 'errno'. Therefore we avoid 'malloc' which might touch 'errno'.
130 Instead we use 'sbrk' which would only uses 'errno' if it fails.
131 In this case we are right away out of memory and the user gets
132 what she/he deserves.
133
134 The initialized value of _dl_tls_static_size is provided by dl-open.c
135 to request some surplus that permits dynamic loading of modules with
136 IE-model TLS. */
137#if TLS_TCB_AT_TP
138 /* Align the TCB offset to the maximum alignment, as
139 _dl_allocate_tls_storage (in elf/dl-tls.c) does using __libc_memalign
140 and dl_tls_static_align. */
141 tcb_offset = roundup (memsz + GL(dl_tls_static_size), max_align);
142 tlsblock = __sbrk (tcb_offset + TLS_INIT_TCB_SIZE + max_align);
143#elif TLS_DTV_AT_TP
144 tcb_offset = roundup (TLS_INIT_TCB_SIZE, align ?: 1);
145 tlsblock = __sbrk (tcb_offset + memsz + max_align
146 + TLS_PRE_TCB_SIZE + GL(dl_tls_static_size));
147 tlsblock += TLS_PRE_TCB_SIZE;
148#else
149 /* In case a model with a different layout for the TCB and DTV
150 is defined add another #elif here and in the following #ifs. */
151# error "Either TLS_TCB_AT_TP or TLS_DTV_AT_TP must be defined"
152#endif
153
154 /* Align the TLS block. */
155 tlsblock = (void *) (((uintptr_t) tlsblock + max_align - 1)
156 & ~(max_align - 1));
157
158 /* Initialize the dtv. [0] is the length, [1] the generation counter. */
159 _dl_static_dtv[0].counter = (sizeof (_dl_static_dtv) / sizeof (_dl_static_dtv[0])) - 2;
160 // _dl_static_dtv[1].counter = 0; would be needed if not already done
161
162 /* Initialize the TLS block. */
163#if TLS_TCB_AT_TP
164 _dl_static_dtv[2].pointer.val = ((char *) tlsblock + tcb_offset
165 - roundup (memsz, align ?: 1));
166 main_map->l_tls_offset = roundup (memsz, align ?: 1);
167#elif TLS_DTV_AT_TP
168 _dl_static_dtv[2].pointer.val = (char *) tlsblock + tcb_offset;
169 main_map->l_tls_offset = tcb_offset;
170#else
171# error "Either TLS_TCB_AT_TP or TLS_DTV_AT_TP must be defined"
172#endif
173 _dl_static_dtv[2].pointer.to_free = NULL;
174 /* sbrk gives us zero'd memory, so we don't need to clear the remainder. */
175 memcpy (_dl_static_dtv[2].pointer.val, initimage, filesz);
176
177 /* Install the pointer to the dtv. */
178
179 /* Initialize the thread pointer. */
180#if TLS_TCB_AT_TP
181 INSTALL_DTV ((char *) tlsblock + tcb_offset, _dl_static_dtv);
182
183 const char *lossage = TLS_INIT_TP ((char *) tlsblock + tcb_offset);
184#elif TLS_DTV_AT_TP
185 INSTALL_DTV (tlsblock, _dl_static_dtv);
186 const char *lossage = TLS_INIT_TP (tlsblock);
187#else
188# error "Either TLS_TCB_AT_TP or TLS_DTV_AT_TP must be defined"
189#endif
190 if (__builtin_expect (lossage != NULL, 0))
191 _startup_fatal (lossage);
192
193 /* Update the executable's link map with enough information to make
194 the TLS routines happy. */
195 main_map->l_tls_align = align;
196 main_map->l_tls_blocksize = memsz;
197 main_map->l_tls_initimage = initimage;
198 main_map->l_tls_initimage_size = filesz;
199 main_map->l_tls_modid = 1;
200
201 init_slotinfo ();
202 /* static_slotinfo.slotinfo[1].gen = 0; -- Already zero. */
203 static_slotinfo.slotinfo[1].map = main_map;
204
205 memsz = roundup (memsz, align ?: 1);
206
207#if TLS_DTV_AT_TP
208 memsz += tcb_offset;
209#endif
210
211 init_static_tls (memsz, MAX (TLS_TCB_ALIGN, max_align));
212}
