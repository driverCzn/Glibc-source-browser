13/*
14 * wrapper sinh(x)
15 */
16
17#include <math.h>
18#include <math_private.h>
19#include <math-svid-compat.h>
20#include <libm-alias-double.h>
21
22#if LIBM_SVID_COMPAT
23double
24__sinh (double x)
25{
26 double z = __ieee754_sinh (x);
27 if (__builtin_expect (!isfinite (z), 0) && isfinite (x)
28 && _LIB_VERSION != _IEEE_)
29 return __kernel_standard (x, x, 25); /* sinh overflow */
30
31 return z;
32}
33libm_alias_double (__sinh, sinh)
34#endif
