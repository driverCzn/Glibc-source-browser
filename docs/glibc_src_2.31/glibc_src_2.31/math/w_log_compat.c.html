19#include <fenv.h>
20#include <math.h>
21#include <math_private.h>
22#include <math-svid-compat.h>
23#include <libm-alias-double.h>

25
26#if LIBM_SVID_COMPAT && (SHLIB_COMPAT (libm, GLIBC_2_0, GLIBC_2_29) \
27 || defined NO_LONG_DOUBLE \
28 || defined LONG_DOUBLE_COMPAT)
29/* wrapper log(x) */
30double
31__log_compat (double x)
32{
33 if (__builtin_expect (islessequal (x, 0.0), 0) && _LIB_VERSION != _IEEE_)
34 {
35 if (x == 0.0)
36 {
37 __feraiseexcept (FE_DIVBYZERO);
38 return __kernel_standard (x, x, 16); /* log(0) */
39 }
40 else
41 {
42 __feraiseexcept (FE_INVALID);
43 return __kernel_standard (x, x, 17); /* log(x<0) */
44 }
45 }
46
47 return __ieee754_log (x);
48}
49# if SHLIB_COMPAT (libm, GLIBC_2_0, GLIBC_2_29)
50compat_symbol (libm, __log_compat, log, GLIBC_2_0);
51# endif
52# ifdef NO_LONG_DOUBLE
53weak_alias (__log_compat, logl)
54# endif
55# ifdef LONG_DOUBLE_COMPAT
56/* Work around gas bug "multiple versions for symbol". */
57weak_alias (__log_compat, __log_compat_alias)
58
59LONG_DOUBLE_COMPAT_CHOOSE_libm_logl (
60 compat_symbol (libm, __log_compat_alias, logl, FIRST_VERSION_libm_logl), );
61# endif
62#endif
