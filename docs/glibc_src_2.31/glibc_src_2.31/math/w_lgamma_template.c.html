19/* Only build wrappers from the templates for the types that define the macro
20 below. This macro is set in math-type-macros-<type>.h in sysdeps/generic
21 for each floating-point type. */
22#if __USE_WRAPPER_TEMPLATE
23
24# include <errno.h>
25# include <fenv.h>
26# include <math.h>
27# include <math_private.h>
28
29#define M_CALL_FUNC_X(x) x ## _r
30#define M_CALL_FUNC(x) M_CALL_FUNC_X (x)
31
32FLOAT
33M_DECL_FUNC (__lgamma) (FLOAT x)
34{
35 FLOAT y = M_CALL_FUNC (M_SUF (__ieee754_lgamma)) (x, &__signgam);
36 if (__glibc_unlikely (!isfinite (y)) && isfinite (x))
37 /* Pole error: lgamma(integer x<0). Or overflow. */
38 __set_errno (ERANGE);
39 return y;
40}
41declare_mgen_alias (__lgamma, lgamma)
42
43#endif /* __USE_WRAPPER_TEMPLATE. */
