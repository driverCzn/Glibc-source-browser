26#if LIBM_SVID_COMPAT
27/* wrapper log2l(x) */
28long double
29__log2l (long double x)
30{
31 if (__builtin_expect (islessequal (x, 0.0L), 0) && _LIB_VERSION != _IEEE_)
32 {
33 if (x == 0.0L)
34 {
35 feraiseexcept (FE_DIVBYZERO);
36 return __kernel_standard_l (x, x, 248); /* log2(0) */
37 }
38 else
39 {
40 feraiseexcept (FE_INVALID);
41 return __kernel_standard_l (x, x, 249); /* log2(x<0) */
42 }
43 }
44
45 return __ieee754_log2l (x);
46}
47libm_alias_ldouble (__log2, log2)
48#endif
