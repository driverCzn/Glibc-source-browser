18#include <errno.h>
19#include <stddef.h>
20#include <sys/stat.h>
21
22/* Change the flags of the file referenced by FD to FLAGS. */
23
24int fchflags (int fd, unsigned long int flags) __THROW;
25
26int
27fchflags (int fd, unsigned long int flags)
28{
29 if (fd < 0)
30 {
31 __set_errno (EINVAL);
32 return -1;
33 }
34
35 __set_errno (ENOSYS);
36 return -1;
37}
