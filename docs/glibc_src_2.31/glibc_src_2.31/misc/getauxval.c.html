18#include <sys/auxv.h>
19#include <errno.h>
20#include <ldsodefs.h>
21
22
23unsigned long int
24__getauxval (unsigned long int type)
25{
26#ifdef HAVE_AUX_VECTOR
27 ElfW(auxv_t) *p;
28#endif
29
30 if (type == AT_HWCAP)
31 return GLRO(dl_hwcap);
32 else if (type == AT_HWCAP2)
33 return GLRO(dl_hwcap2);
34
35#ifdef HAVE_AUX_VECTOR
36 for (p = GLRO(dl_auxv); p->a_type != AT_NULL; p++)
37 if (p->a_type == type)
38 return p->a_un.a_val;
39#endif
40
41 __set_errno (ENOENT);
42 return 0;
43}
44
45weak_alias (__getauxval, getauxval)
46libc_hidden_def (__getauxval)
