18#define scandirat __no_scandirat_decl
19#include <dirent.h>
20#undef scandirat
21
22int
23scandirat64 (int dfd, const char *dir, struct dirent64 ***namelist,
24 int (*select) (const struct dirent64 *),
25 int (*cmp) (const struct dirent64 **, const struct dirent64 **))
26{
27 return __scandir64_tail (__opendirat (dfd, dir), namelist, select, cmp);
28}
