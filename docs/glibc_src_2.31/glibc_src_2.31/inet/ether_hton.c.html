26/* Type of the lookup function we need here. */
27typedef int (*lookup_function) (const char *, struct etherent *, char *, int,
28 int *);
29
30int
31ether_hostton (const char *hostname, struct ether_addr *addr)
32{
33 static service_user *startp;
34 static lookup_function start_fct;
35 service_user *nip;
36 union
37 {
38 lookup_function f;
39 void *ptr;
40 } fct;
41 int no_more;
42 enum nss_status status = NSS_STATUS_UNAVAIL;
43 struct etherent etherent;
44
45 if (startp == NULL)
46 {
47 no_more = __nss_ethers_lookup2 (&nip, "gethostton_r", NULL, &fct.ptr);
48 if (no_more)
49 startp = (service_user *) -1;
50 else
51 {
52 startp = nip;
53 start_fct = fct.f;
54 }
55 }
56 else
57 {
58 fct.f = start_fct;
59 no_more = (nip = startp) == (service_user *) -1;
60 }
61
62 while (no_more == 0)
63 {
64 char buffer[1024];
65
66 status = (*fct.f) (hostname, &etherent, buffer, sizeof buffer, &errno);
67
68 no_more = __nss_next2 (&nip, "gethostton_r", NULL, &fct.ptr, status, 0);
69 }
70
71 if (status == NSS_STATUS_SUCCESS)
72 memcpy (addr, etherent.e_addr.ether_addr_octet,
73 sizeof (struct ether_addr));
74
75 return status == NSS_STATUS_SUCCESS ? 0 : -1;
76}
