52#include <arpa/nameser.h>
53#include <sys/socket.h>
54
55typedef union {
56 int32_t al;
57 char ac;
58} align;
59
60static int
61map_v4v6_hostent (struct hostent *hp, char **bpp, int *lenp)
62{
63 char **ap;
64
65 if (hp->h_addrtype != AF_INET || hp->h_length != INADDRSZ)
66 return 0;
67 hp->h_addrtype = AF_INET6;
68 hp->h_length = IN6ADDRSZ;
69 for (ap = hp->h_addr_list; *ap; ap++)
70 {
71 int i = sizeof (align) - ((u_long) *bpp % sizeof (align));
72
73 if (*lenp < (i + IN6ADDRSZ))
74 /* Out of memory. */
75 return 1;
76 *bpp += i;
77 *lenp -= i;
78 map_v4v6_address (*ap, *bpp);
79 *ap = *bpp;
80 *bpp += IN6ADDRSZ;
81 *lenp -= IN6ADDRSZ;
82 }
83 return 0;
84}
