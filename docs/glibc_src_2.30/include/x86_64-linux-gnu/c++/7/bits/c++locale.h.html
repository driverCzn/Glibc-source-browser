64 // Convert numeric value of type double and long double to string and
65 // return length of string. If vsnprintf is available use it, otherwise
66 // fall back to the unsafe vsprintf which, in general, can be dangerous
67 // and should be avoided.
68 inline int
69 __convert_from_v(const __c_locale& __cloc __attribute__ ((__unused__)),
70 char* __out,
71 const int __size __attribute__ ((__unused__)),
72 const char* __fmt, ...)
73 {
74#if __GLIBC__ > 2 || (__GLIBC__ == 2 && __GLIBC_MINOR__ > 2)
75 __c_locale __old = __gnu_cxx::__uselocale(__cloc);
76#else
77 char* __old = std::setlocale(LC_NUMERIC, 0);
78 char* __sav = 0;
79 if (__builtin_strcmp(__old, "C"))
80 {
81 const size_t __len = __builtin_strlen(__old) + 1;
82 __sav = new char[__len];
83 __builtin_memcpy(__sav, __old, __len);
84 std::setlocale(LC_NUMERIC, "C");
85 }
86#endif
87
88 __builtin_va_list __args;
89 __builtin_va_start(__args, __fmt);
90
91#if _GLIBCXX_USE_C99_STDIO
92 const int __ret = __builtin_vsnprintf(__out, __size, __fmt, __args);
93#else
94 const int __ret = __builtin_vsprintf(__out, __fmt, __args);
95#endif
96
97 __builtin_va_end(__args);
98
99#if __GLIBC__ > 2 || (__GLIBC__ == 2 && __GLIBC_MINOR__ > 2)
100 __gnu_cxx::__uselocale(__old);
101#else
102 if (__sav)
103 {
104 std::setlocale(LC_NUMERIC, __sav);
105 delete [] __sav;
106 }
107#endif
108 return __ret;
109 }
