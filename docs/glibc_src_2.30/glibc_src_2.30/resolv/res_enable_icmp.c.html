19#include <errno.h>
20#include <netinet/in.h>
21#include <sys/socket.h>
22
23int
24__res_enable_icmp (int family, int fd)
25{
26 int one = 1;
27 switch (family)
28 {
29 case AF_INET:
30 return setsockopt (fd, SOL_IP, IP_RECVERR, &one, sizeof (one));
31 case AF_INET6:
32 return setsockopt (fd, SOL_IPV6, IPV6_RECVERR, &one, sizeof (one));
33 default:
34 __set_errno (EAFNOSUPPORT);
35 return -1;
36 }
37}
