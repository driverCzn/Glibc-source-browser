18#include <stdarg.h>
19#include <libio/libioP.h>
20
21
22/* Write formatted output to D, according to the format string FORMAT. */
23int
24__dprintf_chk (int d, int flag, const char *format, ...)
25{
26 /* For flag > 0 (i.e. __USE_FORTIFY_LEVEL > 1) request that %n
27 can only come from read-only format strings. */
28 unsigned int mode = (flag > 0) ? PRINTF_FORTIFY : 0;
29 va_list ap;
30 int ret;
31
32 va_start (ap, format);
33 ret = __vdprintf_internal (d, format, ap, mode);
34 va_end (ap);
35
36 return ret;
37}
