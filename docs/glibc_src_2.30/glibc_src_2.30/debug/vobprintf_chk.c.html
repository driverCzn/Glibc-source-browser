22int
23__obstack_vprintf_chk (struct obstack *obstack, int flag, const char *format,
24 va_list ap)
25{
26 /* For flag > 0 (i.e. __USE_FORTIFY_LEVEL > 1) request that %n
27 can only come from read-only format strings. */
28 unsigned int mode = (flag > 0) ? PRINTF_FORTIFY : 0;
29
30 return __obstack_vprintf_internal (obstack, format, ap, mode);
31}
