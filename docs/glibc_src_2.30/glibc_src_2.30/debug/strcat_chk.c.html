18#include <string.h>
19#include <memcopy.h>
20
21
22/* Append SRC on the end of DEST. */
23char *
24__strcat_chk (char *dest, const char *src, size_t destlen)
25{
26 char *s1 = dest;
27 const char *s2 = src;
28 char c;
29
30 /* Find the end of the string. */
31 do
32 {
33 if (__glibc_unlikely (destlen-- == 0))
34 __chk_fail ();
35 c = *s1++;
36 }
37 while (c != '\0');
38
39 /* Make S1 point before the next character, so we can increment
40 it while memory is read (wins on pipelined cpus). */
41 ++destlen;
42 s1 -= 2;
43
44 do
45 {
46 if (__glibc_unlikely (destlen-- == 0))
47 __chk_fail ();
48 c = *s2++;
49 *++s1 = c;
50 }
51 while (c != '\0');
52
53 return dest;
54}
