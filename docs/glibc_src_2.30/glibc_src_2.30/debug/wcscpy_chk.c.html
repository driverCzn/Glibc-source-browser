19#include <stddef.h>
20#include <wchar.h>
21
22
23/* Copy SRC to DEST. */
24wchar_t *
25__wcscpy_chk (wchar_t *dest, const wchar_t *src, size_t n)
26{
27 wint_t c;
28 wchar_t *wcp;
29
30 if (__alignof__ (wchar_t) >= sizeof (wchar_t))
31 {
32 const ptrdiff_t off = dest - src - 1;
33
34 wcp = (wchar_t *) src;
35
36 do
37 {
38 if (__glibc_unlikely (n-- == 0))
39 __chk_fail ();
40 c = *wcp++;
41 wcp[off] = c;
42 }
43 while (c != L'\0');
44 }
45 else
46 {
47 wcp = dest;
48
49 do
50 {
51 if (__glibc_unlikely (n-- == 0))
52 __chk_fail ();
53 c = *src++;
54 *wcp++ = c;
55 }
56 while (c != L'\0');
57 }
58
59 return dest;
60}
