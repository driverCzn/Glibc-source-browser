18#include <string.h>
19
20#undef __strsep
21#undef strsep
22
23char *
24__strsep (char **stringp, const char *delim)
25{
26 char *begin, *end;
27
28 begin = *stringp;
29 if (begin == NULL)
30 return NULL;
31
32 /* Find the end of the token. */
33 end = begin + strcspn (begin, delim);
34
35 if (*end)
36 {
37 /* Terminate the token and set *STRINGP past NUL character. */
38 *end++ = '\0';
39 *stringp = end;
40 }
41 else
42 /* No more delimiters; this is the last token. */
43 *stringp = NULL;
44
45 return begin;
46}

48strong_alias (__strsep, __strsep_g)
