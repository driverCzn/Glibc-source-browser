18#include <errno.h>
19#include <signal.h>
20#include <sigsetops.h>
21#include <internal-signals.h>
22
23/* Add SIGNO to SET. */
24int
25sigaddset (sigset_t *set, int signo)
26{
27 if (set == NULL || signo <= 0 || signo >= NSIG
28 || __is_internal_signal (signo))
29 {
30 __set_errno (EINVAL);
31 return -1;
32 }
33
34 __sigaddset (set, signo);
35 return 0;
36}
