18#include <errno.h>
19#include <signal.h>
20#include <string.h>
21#include <internal-signals.h>
22
23/* Set all signals in SET. */
24int
25sigfillset (sigset_t *set)
26{
27 if (set == NULL)
28 {
29 __set_errno (EINVAL);
30 return -1;
31 }
32
33 memset (set, 0xff, sizeof (sigset_t));
34
35 __clear_internal_signals (set);
36
37 return 0;
38}
39libc_hidden_def (sigfillset)
