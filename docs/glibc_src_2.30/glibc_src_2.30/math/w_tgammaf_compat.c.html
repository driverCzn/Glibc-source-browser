22#if LIBM_SVID_COMPAT
23float
24__tgammaf(float x)
25{
26 int local_signgam;
27 float y = __ieee754_gammaf_r(x,&local_signgam);
28
29 if(__glibc_unlikely (!isfinite (y) || y == 0)
30 && (isfinite (x) || (isinf (x) && x < 0.0))
31 && _LIB_VERSION != _IEEE_) {
32 if (x == (float)0.0)
33 /* tgammaf pole */
34 return __kernel_standard_f(x, x, 150);
35 else if(floorf(x)==x&&x<0.0f)
36 /* tgammaf domain */
37 return __kernel_standard_f(x, x, 141);
38 else if (y == 0)
39 /* tgammaf underflow */
40 __set_errno (ERANGE);
41 else
42 /* tgammaf overflow */
43 return __kernel_standard_f(x, x, 140);
44 }
45 return local_signgam < 0 ? - y : y;
46}
47libm_alias_float (__tgamma, tgamma)
