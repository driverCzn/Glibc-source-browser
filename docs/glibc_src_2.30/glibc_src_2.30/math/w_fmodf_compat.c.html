19#include <math.h>
20#include <math_private.h>
21#include <math-svid-compat.h>
22#include <libm-alias-float.h>

24#if LIBM_SVID_COMPAT
25/* wrapper fmodf */
26float
27__fmodf (float x, float y)
28{
29 if (__builtin_expect (isinf (x) || y == 0.0f, 0)
30 && _LIB_VERSION != _IEEE_ && !isnan (y) && !isnan (x))
31 /* fmod(+-Inf,y) or fmod(x,0) */
32 return __kernel_standard_f (x, y, 127);
33
34 return __ieee754_fmodf (x, y);
35}
36libm_alias_float (__fmod, fmod)
37#endif
