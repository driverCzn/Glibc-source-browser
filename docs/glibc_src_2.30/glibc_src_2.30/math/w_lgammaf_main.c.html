23#if BUILD_LGAMMA
24float
25LGFUNC (__lgammaf) (float x)
26{
27 float y = CALL_LGAMMA (float, __ieee754_lgammaf_r, x);
28 if(__builtin_expect(!isfinite(y), 0)
29 && isfinite(x) && _LIB_VERSION != _IEEE_)
30 return __kernel_standard_f(x, x,
31 floorf(x)==x&&x<=0.0f
32 ? 115 /* lgamma pole */
33 : 114); /* lgamma overflow */
34
35 return y;
36}
37# if USE_AS_COMPAT
38compat_symbol (libm, __lgammaf_compat, lgammaf, LGAMMA_OLD_VER);
39# else
40versioned_symbol (libm, __lgammaf, lgammaf, LGAMMA_NEW_VER);
41libm_alias_float_other (__lgamma, lgamma)
42# endif
43# if GAMMA_ALIAS
44strong_alias (LGFUNC (__lgammaf), __gammaf)
45weak_alias (__gammaf, gammaf)
46# endif
47#endif
