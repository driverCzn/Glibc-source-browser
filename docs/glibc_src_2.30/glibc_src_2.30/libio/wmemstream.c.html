61/* Open a stream that writes into a malloc'd buffer that is expanded as
62 necessary. *BUFLOC and *SIZELOC are updated with the buffer's location
63 and the number of characters written on fflush or fclose. */
64FILE *
65open_wmemstream (wchar_t **bufloc, size_t *sizeloc)
66{
67 struct locked_FILE
68 {
69 struct _IO_FILE_wmemstream fp;
70#ifdef _IO_MTSAFE_IO
71 _IO_lock_t lock;
72#endif
73 struct _IO_wide_data wd;
74 } *new_f;
75 wchar_t *buf;
76
77 new_f = (struct locked_FILE *) malloc (sizeof (struct locked_FILE));
78 if (new_f == NULL)
79 return NULL;
80#ifdef _IO_MTSAFE_IO
81 new_f->fp._sf._sbf._f._lock = &new_f->lock;
82#endif
83
84 buf = calloc (1, BUFSIZ);
85 if (buf == NULL)
86 {
87 free (new_f);
88 return NULL;
89 }
90 _IO_no_init (&new_f->fp._sf._sbf._f, 0, 0, &new_f->wd, &_IO_wmem_jumps);
91 _IO_fwide (&new_f->fp._sf._sbf._f, 1);
92 _IO_wstr_init_static (&new_f->fp._sf._sbf._f, buf,
93 BUFSIZ / sizeof (wchar_t), buf);
94 new_f->fp._sf._sbf._f._flags2 &= ~_IO_FLAGS2_USER_WBUF;
95 new_f->fp._sf._s._allocate_buffer_unused = (_IO_alloc_type) malloc;
96 new_f->fp._sf._s._free_buffer_unused = (_IO_free_type) free;
97
98 new_f->fp.bufloc = bufloc;
99 new_f->fp.sizeloc = sizeloc;
100
101 /* Disable single thread optimization. BZ 21735. */
102 new_f->fp._sf._sbf._f._flags2 |= _IO_FLAGS2_NEED_LOCK;
103
104 return (FILE *) &new_f->fp._sf._sbf;
105}

108static int
109_IO_wmem_sync (FILE *fp)
110{
111 struct _IO_FILE_wmemstream *mp = (struct _IO_FILE_wmemstream *) fp;
112
113 if (fp->_wide_data->_IO_write_ptr == fp->_wide_data->_IO_write_end)
114 {
115 _IO_wstr_overflow (fp, '\0');
116 --fp->_wide_data->_IO_write_ptr;
117 }
118
119 *mp->bufloc = fp->_wide_data->_IO_write_base;
120 *mp->sizeloc = (fp->_wide_data->_IO_write_ptr
121 - fp->_wide_data->_IO_write_base);
122
123 return 0;
124}
125
126
127static void
128_IO_wmem_finish (FILE *fp, int dummy)
129{
130 struct _IO_FILE_wmemstream *mp = (struct _IO_FILE_wmemstream *) fp;
131
132 *mp->bufloc = (wchar_t *) realloc (fp->_wide_data->_IO_write_base,
133 (fp->_wide_data->_IO_write_ptr
134 - fp->_wide_data->_IO_write_base + 1)
135 * sizeof (wchar_t));
136 if (*mp->bufloc != NULL)
137 {
138 size_t len = (fp->_wide_data->_IO_write_ptr
139 - fp->_wide_data->_IO_write_base);
140 (*mp->bufloc)[len] = '\0';
141 *mp->sizeloc = len;
142
143 fp->_wide_data->_IO_buf_base = NULL;
144 }
145
146 _IO_wstr_finish (fp, 0);
147}
