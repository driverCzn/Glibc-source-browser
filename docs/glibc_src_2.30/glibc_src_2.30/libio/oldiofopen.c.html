28#if SHLIB_COMPAT (libc, GLIBC_2_0, GLIBC_2_1)

30#define _IO_USE_OLD_IO_FILE
31#include "libioP.h"
32#include <stdlib.h>
33
34
35FILE *
36attribute_compat_text_section
37_IO_old_fopen (const char *filename, const char *mode)
38{
39 struct locked_FILE
40 {
41 struct _IO_FILE_complete_plus fp;
42#ifdef _IO_MTSAFE_IO
43 _IO_lock_t lock;
44#endif
45 } *new_f = (struct locked_FILE *) malloc (sizeof (struct locked_FILE));
46
47 if (new_f == NULL)
48 return NULL;
49#ifdef _IO_MTSAFE_IO
50 new_f->fp.file._file._lock = &new_f->lock;
51#endif
52 _IO_old_init (&new_f->fp.file._file, 0);
53 _IO_JUMPS_FILE_plus (&new_f->fp) = &_IO_old_file_jumps;
54 _IO_old_file_init_internal ((struct _IO_FILE_plus *) &new_f->fp);
55 if (_IO_old_file_fopen ((FILE *) &new_f->fp, filename, mode) != NULL)
56 return (FILE *) &new_f->fp;
57 _IO_un_link ((struct _IO_FILE_plus *) &new_f->fp);
58 free (new_f);
59 return NULL;
60}

62strong_alias (_IO_old_fopen, __old_fopen)
63compat_symbol (libc, _IO_old_fopen, _IO_fopen, GLIBC_2_0);
64compat_symbol (libc, __old_fopen, fopen, GLIBC_2_0);
65#endif
