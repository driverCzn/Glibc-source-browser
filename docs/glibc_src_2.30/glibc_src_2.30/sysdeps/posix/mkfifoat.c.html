18#include <sys/stat.h>

21/* Create a new FIFO with permission bits MODE. But interpret
22 relative PATH names relative to the directory associated with FD. */
23int
24mkfifoat (int fd, const char *file, mode_t mode)
25{
26 dev_t dev = 0;
27 return __xmknodat (_MKNOD_VER, fd, file, mode | S_IFIFO, &dev);
28}
