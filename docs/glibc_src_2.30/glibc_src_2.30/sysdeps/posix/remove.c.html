19#include <errno.h>
20#include <stdio.h>
21#include <unistd.h>
22
23
24#ifndef IS_NO_DIRECTORY_ERROR
25# define IS_NO_DIRECTORY_ERROR errno != EPERM
26#endif
27
28
29int
30remove (const char *file)
31{
32 /* First try to unlink since this is more frequently the necessary action. */
33 if (__unlink (file) != 0
34 /* If it is indeed a directory... */
35 && (IS_NO_DIRECTORY_ERROR
36 /* ...try to remove it. */
37 || __rmdir (file) != 0))
38 /* Cannot remove the object for whatever reason. */
39 return -1;
40
41 return 0;
42}
