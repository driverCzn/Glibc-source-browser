18#include <limits.h>
19#include <unistd.h>
20#include <sys/resource.h>
21
22/* Return the maximum number of file descriptors
23 the current process could possibly have. */
24int
25__getdtablesize (void)
26{
27 struct rlimit ru;
28
29 /* This should even work if `getrlimit' is not implemented. POSIX.1
30 does not define this function but we will generate a stub which
31 returns -1. */
32 return __getrlimit (RLIMIT_NOFILE, &ru) < 0 ? OPEN_MAX : ru.rlim_cur;
33}
34
35weak_alias (__getdtablesize, getdtablesize)
