29long int
30__lroundf (float x)
31{
32 int32_t j0;
33 uint32_t i;
34 long int result;
35 int sign;
36
37 GET_FLOAT_WORD (i, x);
38 j0 = ((i >> 23) & 0xff) - 0x7f;
39 sign = (i & 0x80000000) != 0 ? -1 : 1;
40 i &= 0x7fffff;
41 i |= 0x800000;
42
43 if (j0 < (int32_t) (8 * sizeof (long int)) - 1)
44 {
45 if (j0 < 0)
46 return j0 < -1 ? 0 : sign;
47 else if (j0 >= 23)
48 result = (long int) i << (j0 - 23);
49 else
50 {
51 i += 0x400000 >> j0;
52
53 result = i >> (23 - j0);
54 }
55 }
56 else
57 {
58#ifdef FE_INVALID
59 /* The number is too large. Unless it rounds to LONG_MIN,
60 FE_INVALID must be raised and the return value is
61 unspecified. */
62 if (FIX_FLT_LONG_CONVERT_OVERFLOW && x != (float) LONG_MIN)
63 {
64 feraiseexcept (FE_INVALID);
65 return sign == 1 ? LONG_MAX : LONG_MIN;
66 }
67#endif
68 return (long int) x;
69 }
70
71 return sign * result;
72}
