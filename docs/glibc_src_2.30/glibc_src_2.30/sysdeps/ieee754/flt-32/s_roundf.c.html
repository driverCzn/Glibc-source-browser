21#include <math.h>

27float
28__roundf (float x)
29{
30 int32_t i0, j0;
31
32 GET_FLOAT_WORD (i0, x);
33 j0 = ((i0 >> 23) & 0xff) - 0x7f;
34 if (j0 < 23)
35 {
36 if (j0 < 0)
37 {
38 i0 &= 0x80000000;
39 if (j0 == -1)
40 i0 |= 0x3f800000;
41 }
42 else
43 {
44 uint32_t i = 0x007fffff >> j0;
45 if ((i0 & i) == 0)
46 /* X is integral. */
47 return x;
48
49 i0 += 0x00400000 >> j0;
50 i0 &= ~i;
51 }
52 }
53 else
54 {
55 if (j0 == 0x80)
56 /* Inf or NaN. */
57 return x + x;
58 else
59 return x;
60 }
61
62 SET_FLOAT_WORD (x, i0);
63 return x;
64}
