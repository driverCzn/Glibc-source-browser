39/*
40 * floor(x)
41 * Return x rounded toward -inf to integral value
42 * Method:
43 * Bit twiddling.
44 */
45
46
47double
48__floor (double x)
49{
50 int64_t i0;
51 EXTRACT_WORDS64(i0,x);
52 int32_t j0 = ((i0>>52)&0x7ff)-0x3ff;
53 if(__builtin_expect(j0<52, 1)) {
54 if(j0<0) {
55 /* return 0*sign(x) if |x|<1 */
56 if(i0>=0) {i0=0;}
57 else if((i0&0x7fffffffffffffffl)!=0)
58 { i0=0xbff0000000000000l;}
59 } else {
60 uint64_t i = (0x000fffffffffffffl)>>j0;
61 if((i0&i)==0) return x; /* x is integral */
62 if(i0<0) i0 += (0x0010000000000000l)>>j0;
63 i0 &= (~i);
64 }
65 INSERT_WORDS64(x,i0);
66 } else if (j0==0x400)
67 return x+x; /* inf or NaN */
68 return x;
69}

71libm_alias_double (__floor, floor)
72#endif
