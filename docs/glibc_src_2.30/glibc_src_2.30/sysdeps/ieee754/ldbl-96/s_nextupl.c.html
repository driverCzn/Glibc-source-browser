24/* Return the least floating-point number greater than X. */
25long double
26__nextupl (long double x)
27{
28 uint32_t hx, ix;
29 uint32_t lx;
30 int32_t esx;
31
32 GET_LDOUBLE_WORDS (esx, hx, lx, x);
33 ix = esx & 0x7fff;
34
35 if (((ix == 0x7fff) && (((hx & 0x7fffffff) | lx) != 0))) /* x is nan. */
36 return x + x;
37 if ((ix | hx | lx) == 0)
38 return LDBL_TRUE_MIN;
39 if (esx >= 0)
40 { /* x > 0. */
41 if (isinf (x))
42 return x;
43 lx += 1;
44 if (lx == 0)
45 {
46 hx += 1;
47#if LDBL_MIN_EXP == -16381
48 if (hx == 0 || (esx == 0 && hx == 0x80000000))
49#else
50 if (hx == 0)
51#endif
52 {
53 esx += 1;
54 hx |= 0x80000000;
55 }
56 }
57 }
58 else
59 { /* x < 0. */
60 if (lx == 0)
61 {
62#if LDBL_MIN_EXP == -16381
63 if (hx <= 0x80000000 && esx != 0xffff8000)
64 {
65 esx -= 1;
66 hx = hx - 1;
67 if ((esx & 0x7fff) > 0)
68 hx |= 0x80000000;
69 }
70 else
71 hx -= 1;
72#else
73 if (ix != 0 && hx == 0x80000000)
74 hx = 0;
75 if (hx == 0)
76 esx -= 1;
77 hx -= 1;
78#endif
79 }
80 lx -= 1;
81 }
82 SET_LDOUBLE_WORDS (x, esx, hx, lx);
83 return x;
84}
