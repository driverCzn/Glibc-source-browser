17/* __ieee754_hypotl(x,y)
18 *
19 * Method :
20 * If (assume round-to-nearest) z=x*x+y*y
21 * has error less than sqrt(2)/2 ulp, than
22 * sqrt(z) has error less than 1 ulp (exercise).
23 *
24 * So, compute sqrt(x*x+y*y) with some care as
25 * follows to get the error below 1 ulp:
26 *
27 * Assume x>y>0;
28 * (if possible, set rounding to round-to-nearest)
29 * 1. if x > 2y use
30 * x1*x1+(y*y+(x2*(x+x1))) for x*x+y*y
31 * where x1 = x with lower 32 bits cleared, x2 = x-x1; else
32 * 2. if x <= 2y use
33 * t1*y1+((x-y)*(x-y)+(t1*y2+t2*y))
34 * where t1 = 2x with lower 32 bits cleared, t2 = 2x-t1,
35 * y1= y with lower 32 bits chopped, y2 = y-y1.
36 *
37 * NOTE: scaling may be necessary if some argument is too
38 * large or too tiny
39 *
40 * Special cases:
41 * hypot(x,y) is INF if x or y is +INF or -INF; else
42 * hypot(x,y) is NAN if x or y is NAN.
43 *
44 * Accuracy:
45 * hypot(x,y) returns sqrt(x^2+y^2) with error less
46 * than 1 ulps (units in the last place)
47 */
48
49#include <math.h>
50#include <math_private.h>
51#include <math-underflow.h>
52
53long double __ieee754_hypotl(long double x, long double y)
54{
55 long double a,b,t1,t2,y1,y2,w;
56 uint32_t j,k,ea,eb;
57
58 GET_LDOUBLE_EXP(ea,x);
59 ea &= 0x7fff;
60 GET_LDOUBLE_EXP(eb,y);
61 eb &= 0x7fff;
62 if(eb > ea) {a=y;b=x;j=ea; ea=eb;eb=j;} else {a=x;b=y;}
63 SET_LDOUBLE_EXP(a,ea); /* a <- |a| */
64 SET_LDOUBLE_EXP(b,eb); /* b <- |b| */
65 if((ea-eb)>0x46) {return a+b;} /* x/y > 2**70 */
66 k=0;
67 if(__builtin_expect(ea > 0x5f3f,0)) { /* a>2**8000 */
68 if(ea == 0x7fff) { /* Inf or NaN */
69 uint32_t exp __attribute__ ((unused));
70 uint32_t high,low;
71 w = a+b; /* for sNaN */
72 if (issignaling (a) || issignaling (b))
73 return w;
74 GET_LDOUBLE_WORDS(exp,high,low,a);
75 if(((high&0x7fffffff)|low)==0) w = a;
76 GET_LDOUBLE_WORDS(exp,high,low,b);
77 if(((eb^0x7fff)|(high&0x7fffffff)|low)==0) w = b;
78 return w;
79 }
80 /* scale a and b by 2**-9600 */
81 ea -= 0x2580; eb -= 0x2580; k += 9600;
82 SET_LDOUBLE_EXP(a,ea);
83 SET_LDOUBLE_EXP(b,eb);
84 }
85 if(__builtin_expect(eb < 0x20bf, 0)) { /* b < 2**-8000 */
86 if(eb == 0) { /* subnormal b or 0 */
87 uint32_t exp __attribute__ ((unused));
88 uint32_t high,low;
89 GET_LDOUBLE_WORDS(exp,high,low,b);
90 if((high|low)==0) return a;
91 SET_LDOUBLE_WORDS(t1, 0x7ffd, 0x80000000, 0); /* t1=2^16382 */
92 b *= t1;
93 a *= t1;
94 k -= 16382;
95 GET_LDOUBLE_EXP (ea, a);
96 GET_LDOUBLE_EXP (eb, b);
97 if (eb > ea)
98 {
99 t1 = a;
100 a = b;
101 b = t1;
102 j = ea;
103 ea = eb;
104 eb = j;
105 }
106 } else { /* scale a and b by 2^9600 */
107 ea += 0x2580; /* a *= 2^9600 */
108 eb += 0x2580; /* b *= 2^9600 */
109 k -= 9600;
110 SET_LDOUBLE_EXP(a,ea);
111 SET_LDOUBLE_EXP(b,eb);
112 }
113 }
114 /* medium size a and b */
115 w = a-b;
116 if (w>b) {
117 uint32_t high;
118 GET_LDOUBLE_MSW(high,a);
119 SET_LDOUBLE_WORDS(t1,ea,high,0);
120 t2 = a-t1;
121 w = sqrtl(t1*t1-(b*(-b)-t2*(a+t1)));
122 } else {
123 uint32_t high;
124 GET_LDOUBLE_MSW(high,b);
125 a = a+a;
126 SET_LDOUBLE_WORDS(y1,eb,high,0);
127 y2 = b - y1;
128 GET_LDOUBLE_MSW(high,a);
129 SET_LDOUBLE_WORDS(t1,ea+1,high,0);
130 t2 = a - t1;
131 w = sqrtl(t1*y1-(w*(-w)-(t1*y2+t2*b)));
132 }
133 if(k!=0) {
134 uint32_t exp;
135 t1 = 1.0;
136 GET_LDOUBLE_EXP(exp,t1);
137 SET_LDOUBLE_EXP(t1,exp+k);
138 w *= t1;
139 math_check_force_underflow_nonneg (w);
140 return w;
141 } else return w;
142}
