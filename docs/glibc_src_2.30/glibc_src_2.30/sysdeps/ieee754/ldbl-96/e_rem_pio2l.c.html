184int32_t
185__ieee754_rem_pio2l (long double x, long double *y)
186{
187 double tx[3], ty[3];
188 int32_t se, j0;
189 uint32_t i0, i1;
190 int sx;
191 int n, exp;
192
193 GET_LDOUBLE_WORDS (se, i0, i1, x);
194 sx = (se >> 15) & 1;
195 j0 = (se & 0x7fff) - 0x3fff;
196
197 if (j0 < -1)
198 {
199 /* |x| < pi/4. */
200 y[0] = x;
201 y[1] = 0;
202 return 0;
203 }
204
205 if (j0 >= 0x8000)
206 {
207 /* x is infinite or NaN. */
208 y[0] = x - x;
209 y[1] = y[0];
210 return 0;
211 }
212
213 /* Split the 64 bits of the mantissa into three 24-bit integers
214 stored in a double array. */
215 exp = j0 - 23;
216 tx[0] = (double) (i0 >> 8);
217 tx[1] = (double) (((i0 << 16) | (i1 >> 16)) & 0xffffff);
218 tx[2] = (double) ((i1 << 8) & 0xffffff);
219
220 n = __kernel_rem_pio2 (tx, ty, exp, 3, 2, two_over_pi);
221
222 /* The result is now stored in two double values, we need to convert
223 it into two long double values. */
224 if (sx == 0)
225 {
226 y[0] = (long double) ty[0] + (long double) ty[1];
227 y[1] = ty[1] - (y[0] - ty[0]);
228 return n;
229 }
230 else
231 {
232 y[0] = -((long double) ty[0] + (long double) ty[1]);
233 y[1] = -ty[1] - (y[0] + ty[0]);
234 return -n;
235 }
