20#include <tmmintrin.h>

22extern const int8_t ___m128i_shift_right[31] attribute_hidden;
23
24static __inline__ __m128i
25__m128i_shift_right (__m128i value, unsigned long int offset)
26{
27 return _mm_shuffle_epi8 (value,
28 _mm_loadu_si128 ((__m128i *) (___m128i_shift_right
29 + offset)));
30}
