23/* If defined to 1, enable __finitel, __isinfl, and __isnanl function
24 aliases for binary compatibility when built without long double
25 support. If defined to 0, or if long double does not have the same
26 format as double, there are no such aliases. New ports should use
27 the default definition of this as 0, as such
28 implementation-namespace functions should only have one exported
29 name per floating-point format, not one per floating-point
30 type. */
31#define LDBL_CLASSIFY_COMPAT 0
