18#include <sys/ioctl.h>
19#include <errno.h>
20#include <unistd.h>
21#include <sys/types.h>
22
23/* Return the foreground process group ID of FD. */
24pid_t
25tcgetpgrp (int fd)
26{
27 int pgrp;
28
29 if (__ioctl (fd, TIOCGPGRP, &pgrp) < 0)
30 return (pid_t) -1;
31 return (pid_t) pgrp;
32}
33libc_hidden_def (tcgetpgrp)
