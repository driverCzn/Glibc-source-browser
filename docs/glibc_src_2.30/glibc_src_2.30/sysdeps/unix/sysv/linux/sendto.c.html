18#include <sys/socket.h>
19#include <sysdep-cancel.h>
20#include <socketcall.h>
21
22ssize_t
23__libc_sendto (int fd, const void *buf, size_t len, int flags,
24 __CONST_SOCKADDR_ARG addr, socklen_t addrlen)
25{
26#ifdef __ASSUME_SENDTO_SYSCALL
27 return SYSCALL_CANCEL (sendto, fd, buf, len, flags, addr.__sockaddr__,
28 addrlen);
29#else
30 return SOCKETCALL_CANCEL (sendto, fd, buf, len, flags, addr.__sockaddr__,
31 addrlen);
32#endif
33}
34weak_alias (__libc_sendto, sendto)
