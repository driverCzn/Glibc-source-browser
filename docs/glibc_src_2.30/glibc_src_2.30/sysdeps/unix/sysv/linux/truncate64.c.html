18#include <unistd.h>
19#include <sysdep.h>
20#include <errno.h>
21
22#ifndef __NR_truncate64
23# define __NR_truncate64 __NR_truncate
24#endif
25
26/* Truncate PATH to LENGTH bytes. */
27int
28__truncate64 (const char *path, off64_t length)
29{
30 return INLINE_SYSCALL_CALL (truncate64, path,
31 __ALIGNMENT_ARG SYSCALL_LL64 (length));
32}
33weak_alias (__truncate64, truncate64)
34
35#ifdef __OFF_T_MATCHES_OFF64_T
36weak_alias (__truncate64, truncate);
37#endif
