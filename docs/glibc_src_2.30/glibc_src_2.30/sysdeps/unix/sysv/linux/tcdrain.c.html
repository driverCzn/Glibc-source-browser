18#include <errno.h>
19#include <termios.h>
20#include <sys/ioctl.h>
21#include <sysdep-cancel.h>

23/* Wait for pending output to be written on FD. */
24int
25__libc_tcdrain (int fd)
26{
27 /* With an argument of 1, TCSBRK for output to be drain. */
28 return SYSCALL_CANCEL (ioctl, fd, TCSBRK, 1);
29}
30weak_alias (__libc_tcdrain, tcdrain)
