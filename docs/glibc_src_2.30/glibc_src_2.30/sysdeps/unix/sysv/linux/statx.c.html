19#include <errno.h>
20#include <sys/stat.h>
21
22#include "statx_generic.c"
23
24int
25statx (int fd, const char *path, int flags,
26 unsigned int mask, struct statx *buf)
27{
28#ifdef __NR_statx
29 int ret = INLINE_SYSCALL_CALL (statx, fd, path, flags, mask, buf);
30# ifdef __ASSUME_STATX
31 return ret;
32# else
33 if (ret == 0 || errno != ENOSYS)
34 /* Preserve non-error/non-ENOSYS return values. */
35 return ret;
36# endif
37#endif
38#ifndef __ASSUME_STATX
39 return statx_generic (fd, path, flags, mask, buf);
40#endif
41}
