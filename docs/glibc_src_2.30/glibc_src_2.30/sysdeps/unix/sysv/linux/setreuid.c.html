18#include <errno.h>
19#include <unistd.h>
20#include <setxid.h>
21
22
23int
24__setreuid (uid_t ruid, uid_t euid)
25{
26#ifdef __NR_setreuid32
27 return INLINE_SETXID_SYSCALL (setreuid32, 2, ruid, euid);
28#else
29 return INLINE_SETXID_SYSCALL (setreuid, 2, ruid, euid);
30#endif
31}
32#ifndef __setreuid
33weak_alias (__setreuid, setreuid)
34#endif
