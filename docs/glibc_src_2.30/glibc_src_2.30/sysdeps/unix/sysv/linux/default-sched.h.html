21/* This should fill in PD->schedpolicy if PD->flags does not contain
22 ATTR_FLAG_POLICY_SET, and set it; and PD->schedparam if PD->flags does
23 not contain ATTR_FLAG_SCHED_SET, and set it. It won't be called at all
24 if both bits are already set. */
25
26static void
27collect_default_sched (struct pthread *pd)
28{
29 INTERNAL_SYSCALL_DECL (scerr);
30
31 if ((pd->flags & ATTR_FLAG_POLICY_SET) == 0)
32 {
33 pd->schedpolicy = INTERNAL_SYSCALL (sched_getscheduler, scerr, 1, 0);
34 pd->flags |= ATTR_FLAG_POLICY_SET;
35 }
36
37 if ((pd->flags & ATTR_FLAG_SCHED_SET) == 0)
38 {
39 INTERNAL_SYSCALL (sched_getparam, scerr, 2, 0, &pd->schedparam);
40 pd->flags |= ATTR_FLAG_SCHED_SET;
41 }
42}
