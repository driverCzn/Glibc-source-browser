18#include <sys/socket.h>
19#include <sysdep-cancel.h>
20#include <socketcall.h>
21
22int
23__libc_accept (int fd, __SOCKADDR_ARG addr, socklen_t *len)
24{
25#ifdef __ASSUME_ACCEPT_SYSCALL
26 return SYSCALL_CANCEL (accept, fd, addr.__sockaddr__, len);
27#elif defined __ASSUME_ACCEPT4_SYSCALL
28 return SYSCALL_CANCEL (accept4, fd, addr.__sockaddr__, len, 0);
29#else
30 return SOCKETCALL_CANCEL (accept, fd, addr.__sockaddr__, len);
31#endif
32}
33weak_alias (__libc_accept, accept)
34libc_hidden_def (accept)
