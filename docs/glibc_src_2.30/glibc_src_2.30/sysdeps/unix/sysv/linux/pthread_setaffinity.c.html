19#include <errno.h>
20#include <pthreadP.h>
21#include <sysdep.h>
22#include <sys/types.h>
23#include <shlib-compat.h>
24
25
26int
27__pthread_setaffinity_new (pthread_t th, size_t cpusetsize,
28 const cpu_set_t *cpuset)
29{
30 const struct pthread *pd = (const struct pthread *) th;
31 INTERNAL_SYSCALL_DECL (err);
32 int res;
33
34 res = INTERNAL_SYSCALL (sched_setaffinity, err, 3, pd->tid, cpusetsize,
35 cpuset);
36
37#ifdef RESET_VGETCPU_CACHE
38 if (!INTERNAL_SYSCALL_ERROR_P (res, err))
39 RESET_VGETCPU_CACHE ();
40#endif
41
42 return (INTERNAL_SYSCALL_ERROR_P (res, err)
43 ? INTERNAL_SYSCALL_ERRNO (res, err)
44 : 0);
45}
46versioned_symbol (libpthread, __pthread_setaffinity_new,
47 pthread_setaffinity_np, GLIBC_2_3_4);
48
49
50#if SHLIB_COMPAT (libpthread, GLIBC_2_3_3, GLIBC_2_3_4)
51int
52__pthread_setaffinity_old (pthread_t th, cpu_set_t *cpuset)
53{
54 /* The old interface by default assumed a 1024 processor bitmap. */
55 return __pthread_setaffinity_new (th, 128, cpuset);
56}
57compat_symbol (libpthread, __pthread_setaffinity_old, pthread_setaffinity_np,
58 GLIBC_2_3_3);
59#endif
