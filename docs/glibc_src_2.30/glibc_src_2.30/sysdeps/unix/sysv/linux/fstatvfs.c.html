19#include <stddef.h>
20#include <sys/stat.h>
21#include <sys/statfs.h>
22#include "internal_statvfs.h"
23
24int
25__fstatvfs (int fd, struct statvfs *buf)
26{
27 struct statfs fsbuf;
28
29 /* Get as much information as possible from the system. */
30 if (__fstatfs (fd, &fsbuf) < 0)
31 return -1;
32
33 /* Convert the result. */
34 __internal_statvfs (NULL, buf, &fsbuf, fd);
35
36 /* We signal success if the statfs call succeeded. */
37 return 0;
38}
39weak_alias (__fstatvfs, fstatvfs)
40libc_hidden_weak (fstatvfs)
