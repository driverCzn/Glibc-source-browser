22#include <libc-vdso.h>

24long int (*VDSO_SYMBOL(clock_gettime)) (clockid_t, struct timespec *)
25 attribute_hidden;
26long int (*VDSO_SYMBOL(getcpu)) (unsigned *, unsigned *, void *)
27 attribute_hidden;
28
29extern __typeof (clock_gettime) __syscall_clock_gettime attribute_hidden;
30
31
32static inline void
33__vdso_platform_setup (void)
34{
35 PREPARE_VERSION_KNOWN (linux26, LINUX_2_6);
36
37 void *p = _dl_vdso_vsym ("__vdso_clock_gettime", &linux26);
38 if (p == NULL)
39 p = __syscall_clock_gettime;
40 PTR_MANGLE (p);
41 VDSO_SYMBOL(clock_gettime) = p;
42
43 p = _dl_vdso_vsym ("__vdso_getcpu", &linux26);
44 PTR_MANGLE (p);
45 VDSO_SYMBOL(getcpu) = p;
46}
47
48#define VDSO_SETUP __vdso_platform_setup
