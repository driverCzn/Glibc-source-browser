19#include <wchar.h>
20
21#ifdef WCSCAT
22# define __wcscat WCSCAT
23#endif
24
25/* Append SRC on the end of DEST. */
26wchar_t *
27__wcscat (wchar_t *dest, const wchar_t *src)
28{
29 __wcscpy (dest + __wcslen (dest), src);
30 return dest;
31}
32#ifndef WCSCAT
33weak_alias (__wcscat, wcscat)
34#endif
