31wint_t
32__btowc (int c)
33{
34 const struct gconv_fcts *fcts;
35
36 /* If the parameter does not fit into one byte or it is the EOF value
37 we can give the answer now. */
38 if (c < SCHAR_MIN || c > UCHAR_MAX || c == EOF)
39 return WEOF;
40
41 /* We know that only ASCII compatible encodings are used for the
42 locale and that the wide character encoding is ISO 10646. */
43 if (isascii (c))
44 return (wint_t) c;
45
46 /* Get the conversion functions. */
47 fcts = get_gconv_fcts (_NL_CURRENT_DATA (LC_CTYPE));
48 __gconv_btowc_fct btowc_fct = fcts->towc->__btowc_fct;
49#ifdef PTR_DEMANGLE
50 if (fcts->towc->__shlib_handle != NULL)
51 PTR_DEMANGLE (btowc_fct);
52#endif
53
54 if (__builtin_expect (fcts->towc_nsteps == 1, 1)
55 && __builtin_expect (btowc_fct != NULL, 1))
56 {
57 /* Use the shortcut function. */
58 return DL_CALL_FCT (btowc_fct, (fcts->towc, (unsigned char) c));
59 }
60 else
61 {
62 /* Fall back to the slow but generic method. */
63 wchar_t result;
64 struct __gconv_step_data data;
65 unsigned char inbuf[1];
66 const unsigned char *inptr = inbuf;
67 size_t dummy;
68 int status;
69
70 /* Tell where we want the result. */
71 data.__outbuf = (unsigned char *) &result;
72 data.__outbufend = data.__outbuf + sizeof (wchar_t);
73 data.__invocation_counter = 0;
74 data.__internal_use = 1;
75 data.__flags = __GCONV_IS_LAST;
76 data.__statep = &data.__state;
77
78 /* Make sure we start in the initial state. */
79 memset (&data.__state, '\0', sizeof (mbstate_t));
80
81 /* Create the input string. */
82 inbuf[0] = c;
83
84 __gconv_fct fct = fcts->towc->__fct;
85#ifdef PTR_DEMANGLE
86 if (fcts->towc->__shlib_handle != NULL)
87 PTR_DEMANGLE (fct);
88#endif
89 status = DL_CALL_FCT (fct, (fcts->towc, &data, &inptr, inptr + 1,
90 NULL, &dummy, 0, 1));
91
92 if (status != __GCONV_OK && status != __GCONV_FULL_OUTPUT
93 && status != __GCONV_EMPTY_INPUT)
94 /* The conversion failed. */
95 result = WEOF;
96
97 return result;
98 }
99}
