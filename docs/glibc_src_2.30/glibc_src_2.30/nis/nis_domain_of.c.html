22nis_name
23nis_domain_of (const_nis_name name)
24{
25 static char result[NIS_MAXNAMELEN + 1];
26
27 return nis_domain_of_r (name, result, NIS_MAXNAMELEN);
28}
29libnsl_hidden_nolink_def (nis_domain_of, GLIBC_2_1)
30
31const_nis_name
32__nis_domain_of (const_nis_name name)
33{
34 const_nis_name cptr = strchr (name, '.');
35
36 if (cptr == NULL)
37 return "";
38
39 if (*++cptr == '\0')
40 return ".";
41
42 return cptr;
43}
