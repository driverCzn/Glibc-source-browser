21/* Lookup in a table of int32_t, with default value 0. */
22int32_t
23__collidx_table_lookup (const char *table, uint32_t wc)
24{
25 uint32_t shift1 = ((const uint32_t *) table)[0];
26 uint32_t index1 = wc >> shift1;
27 uint32_t bound = ((const uint32_t *) table)[1];
28 if (index1 < bound)
29 {
30 uint32_t lookup1 = ((const uint32_t *) table)[5 + index1];
31 if (lookup1 != 0)
32 {
33 uint32_t shift2 = ((const uint32_t *) table)[2];
34 uint32_t mask2 = ((const uint32_t *) table)[3];
35 uint32_t index2 = (wc >> shift2) & mask2;
36 uint32_t lookup2 = ((const uint32_t *)(table + lookup1))[index2];
37 if (lookup2 != 0)
38 {
39 uint32_t mask3 = ((const uint32_t *) table)[4];
40 uint32_t index3 = wc & mask3;
41 int32_t lookup3 = ((const int32_t *)(table + lookup2))[index3];
42
43 return lookup3;
44 }
45 }
46 }
47 return 0;
48}

51/* Lookup in a table of uint32_t, with default value 0xffffffff. */
52uint32_t
53__collseq_table_lookup (const char *table, uint32_t wc)
54{
55 uint32_t shift1 = ((const uint32_t *) table)[0];
56 uint32_t index1 = wc >> shift1;
57 uint32_t bound = ((const uint32_t *) table)[1];
58 if (index1 < bound)
59 {
60 uint32_t lookup1 = ((const uint32_t *) table)[5 + index1];
61 if (lookup1 != 0)
62 {
63 uint32_t shift2 = ((const uint32_t *) table)[2];
64 uint32_t mask2 = ((const uint32_t *) table)[3];
65 uint32_t index2 = (wc >> shift2) & mask2;
66 uint32_t lookup2 = ((const uint32_t *)(table + lookup1))[index2];
67 if (lookup2 != 0)
68 {
69 uint32_t mask3 = ((const uint32_t *) table)[4];
70 uint32_t index3 = wc & mask3;
71 uint32_t lookup3 = ((const uint32_t *)(table + lookup2))[index3];
72
73 return lookup3;
74 }
75 }
76 }
77 return ~((uint32_t) 0);
78}
