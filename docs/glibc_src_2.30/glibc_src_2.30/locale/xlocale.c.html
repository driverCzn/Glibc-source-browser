20#include <locale.h>
21#include "localeinfo.h"
22
23#define DEFINE_CATEGORY(category, category_name, items, a) \
24extern struct __locale_data _nl_C_##category;
25#include "categories.def"
26#undef DEFINE_CATEGORY
27
28/* Defined in locale/C-ctype.c. */
29extern const char _nl_C_LC_CTYPE_class[] attribute_hidden;
30extern const char _nl_C_LC_CTYPE_toupper[] attribute_hidden;
31extern const char _nl_C_LC_CTYPE_tolower[] attribute_hidden;
32
33
34const struct __locale_struct _nl_C_locobj attribute_hidden =
35 {
36 .__locales =
37 {
38#define DEFINE_CATEGORY(category, category_name, items, a) \
39 [category] = &_nl_C_##category,
40#include "categories.def"
41#undef DEFINE_CATEGORY
42 },
43 .__names =
44 {
45 [LC_ALL] = _nl_C_name,
46#define DEFINE_CATEGORY(category, category_name, items, a) \
47 [category] = _nl_C_name,
48#include "categories.def"
49#undef DEFINE_CATEGORY
50 },
51 .__ctype_b = (const unsigned short int *) _nl_C_LC_CTYPE_class + 128,
52 .__ctype_tolower = (const int *) _nl_C_LC_CTYPE_tolower + 128,
53 .__ctype_toupper = (const int *) _nl_C_LC_CTYPE_toupper + 128
54 };
