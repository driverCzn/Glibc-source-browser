20/* The hashing function used for the table with collation symbols. */
21static int32_t __attribute__ ((pure, unused))
22elem_hash (const char *str, int_fast32_t n)
23{
24 int32_t result = n;
25
26 while (n-- > 0)
27 {
28 result <<= 3;
29 result += *str++;
30 }
31
32 return result;
33}
