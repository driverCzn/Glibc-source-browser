18#include <stdio.h>

20#include <libio/iolibio.h>
21#define fread(p, m, n, s) _IO_fread (p, m, n, s)
22
23/* Read a word (int) from STREAM. */
24int
25getw (FILE *stream)
26{
27 int w;
28
29 /* Is there a better way? */
30 if (fread ((void *) &w, sizeof (w), 1, stream) != 1)
31 return EOF;
32 return w;
33}
