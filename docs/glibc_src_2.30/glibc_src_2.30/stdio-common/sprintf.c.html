19#include <libioP.h>

21/* Write formatted output into S, according to the format string FORMAT. */
22/* VARARGS2 */
23int
24__sprintf (char *s, const char *format, ...)
25{
26 va_list arg;
27 int done;
28
29 va_start (arg, format);
30 done = __vsprintf_internal (s, -1, format, arg, 0);
31 va_end (arg);
32
33 return done;
34}
