22int
23__pthread_attr_getinheritsched (const pthread_attr_t *attr, int *inherit)
24{
25 struct pthread_attr *iattr;
26
27 iattr = (struct pthread_attr *) attr;
28
29 /* Store the current values. */
30 *inherit = (iattr->flags & ATTR_FLAG_NOTINHERITSCHED
31 ? PTHREAD_EXPLICIT_SCHED : PTHREAD_INHERIT_SCHED);
32
33 return 0;
34}
