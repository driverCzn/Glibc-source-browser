19#include <stdalign.h>
20
21#include "thrd_priv.h"
22
23int
24cnd_init (cnd_t *cond)
25{
26 _Static_assert (sizeof (cnd_t) == sizeof (pthread_cond_t),
27 "(sizeof (cnd_t) != sizeof (pthread_cond_t)");
28 _Static_assert (alignof (cnd_t) == alignof (pthread_cond_t),
29 "alignof (cnd_t) != alignof (pthread_cond_t)");
30
31 int err_code = __pthread_cond_init ((pthread_cond_t *)cond, NULL);
32 return thrd_err_map (err_code);
33}
