20#include <atomic.h>
21#include <errno.h>
22#include <lowlevellock.h>
23#include <sys/time.h>
24
25
26int
27__lll_clocklock_wait (int *futex, clockid_t clockid,
28 const struct timespec *abstime, int private)
29{
30 /* Reject invalid timeouts. */
31 if (abstime->tv_nsec < 0 || abstime->tv_nsec >= 1000000000)
32 return EINVAL;
33
34 /* Try locking. */
35 while (atomic_exchange_acq (futex, 2) != 0)
36 {
37 struct timespec ts;
38
39 /* Get the current time. This can only fail if clockid is not
40 valid. */
41 if (__glibc_unlikely (__clock_gettime (clockid, &ts) != 0))
42 return EINVAL;
43
44 /* Compute relative timeout. */
45 struct timespec rt;
46 rt.tv_sec = abstime->tv_sec - ts.tv_sec;
47 rt.tv_nsec = abstime->tv_nsec - ts.tv_nsec;
48 if (rt.tv_nsec < 0)
49 {
50 rt.tv_nsec += 1000000000;
51 --rt.tv_sec;
52 }
53
54 if (rt.tv_sec < 0)
55 return ETIMEDOUT;
56
57 /* If *futex == 2, wait until woken or timeout. */
58 lll_futex_timed_wait (futex, 2, &rt, private);
59 }
60
61 return 0;
62}
