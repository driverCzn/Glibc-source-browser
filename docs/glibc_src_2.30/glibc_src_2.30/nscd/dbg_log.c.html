18#include <stdarg.h>

26/* if in debug mode and we have a debug file, we write the messages to it,
27 if in debug mode and no debug file, we write the messages to stderr,
28 else to syslog. */
29
30static char *logfilename;
31FILE *dbgout;
32int debug_level;
33
34void
35set_logfile (const char *logfile)
36{
37 logfilename = strdup (logfile);
38}
39
40int
41init_logfile (void)
42{
43 if (logfilename)
44 {
45 dbgout = fopen64 (logfilename, "a");
46 return dbgout == NULL ? 0 : 1;
47 }
48 return 1;
49}
50
51void
52dbg_log (const char *fmt,...)
53{
54 va_list ap;
55 char msg2[512];
56
57 va_start (ap, fmt);
58 vsnprintf (msg2, sizeof (msg2), fmt, ap);
59
60 if (debug_level > 0)
61 {
62 time_t t = time (NULL);
63
64 struct tm now;
65 localtime_r (&t, &now);
66
67 char buf[256];
68 strftime (buf, sizeof (buf), "%c", &now);
69
70 char msg[1024];
71 snprintf (msg, sizeof (msg), "%s - %d: %s%s", buf, getpid (), msg2,
72 msg2[strlen (msg2) - 1] == '\n' ? "" : "\n");
73 if (dbgout)
74 {
75 fputs (msg, dbgout);
76 fflush (dbgout);
77 }
78 else
79 fputs (msg, stderr);
80 }
81 else
82 syslog (LOG_NOTICE, "%d %s", getpid (), msg2);
83
84 va_end (ap);
85}
