23td_err_e
24td_thr_setfpregs (const td_thrhandle_t *th, const prfpregset_t *fpregs)
25{
26 psaddr_t cancelhandling, tid;
27 td_err_e err;
28
29 LOG ("td_thr_setfpregs");
30
31 if (th->th_unique == 0)
32 /* Special case for the main thread before initialization. */
33 return ps_lsetfpregs (th->th_ta_p->ph, ps_getpid (th->th_ta_p->ph),
34 fpregs) != PS_OK ? TD_ERR : TD_OK;
35
36 /* We have to get the state and the PID for this thread. */
37 err = DB_GET_FIELD (cancelhandling, th->th_ta_p, th->th_unique, pthread,
38 cancelhandling, 0);
39 if (err != TD_OK)
40 return err;
41
42 /* Only set the registers if the thread hasn't yet terminated. */
43 if ((((int) (uintptr_t) cancelhandling) & TERMINATED_BITMASK) == 0)
44 {
45 err = DB_GET_FIELD (tid, th->th_ta_p, th->th_unique, pthread, tid, 0);
46 if (err != TD_OK)
47 return err;
48
49 if (ps_lsetfpregs (th->th_ta_p->ph, (uintptr_t) tid, fpregs) != PS_OK)
50 return TD_ERR;
51 }
52
53 return TD_OK;
54}
