22td_err_e
23td_ta_get_nthreads (const td_thragent_t *ta_arg, int *np)
24{
25 td_thragent_t *const ta = (td_thragent_t *) ta_arg;
26 td_err_e err;
27 psaddr_t n;
28
29 LOG ("td_ta_get_nthreads");
30
31 /* Test whether the TA parameter is ok. */
32 if (! ta_ok (ta))
33 return TD_BADTA;
34
35 /* Access the variable in the inferior that tells us. */
36 err = DB_GET_VALUE (n, ta, __nptl_nthreads, 0);
37 if (err == TD_OK)
38 *np = (uintptr_t) n;
39
40 return err;
41}
