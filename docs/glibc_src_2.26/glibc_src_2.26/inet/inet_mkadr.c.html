30#include <sys/param.h>
31#include <netinet/in.h>
32#include <arpa/inet.h>
33
34/*
35 * Formulate an Internet address from network + host. Used in
36 * building addresses stored in the ifnet structure.
37 */
38struct in_addr
39__inet_makeaddr (in_addr_t net, in_addr_t host)
40{
41 struct in_addr in;
42
43 if (net < 128)
44 in.s_addr = (net << IN_CLASSA_NSHIFT) | (host & IN_CLASSA_HOST);
45 else if (net < 65536)
46 in.s_addr = (net << IN_CLASSB_NSHIFT) | (host & IN_CLASSB_HOST);
47 else if (net < 16777216L)
48 in.s_addr = (net << IN_CLASSC_NSHIFT) | (host & IN_CLASSC_HOST);
49 else
50 in.s_addr = net | host;
51 in.s_addr = htonl(in.s_addr);
52 return in;
53}
54libc_hidden_def (__inet_makeaddr)
55weak_alias (__inet_makeaddr, inet_makeaddr)
