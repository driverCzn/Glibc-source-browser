23struct alloc_buffer
24__libc_alloc_buffer_allocate (size_t size, void **pptr)
25{
26 *pptr = malloc (size);
27 if (*pptr == NULL)
28 return (struct alloc_buffer)
29 {
30 .__alloc_buffer_current = __ALLOC_BUFFER_INVALID_POINTER,
31 .__alloc_buffer_end = __ALLOC_BUFFER_INVALID_POINTER
32 };
33 else
34 return alloc_buffer_create (*pptr, size);
35}
