19#include <scratch_buffer.h>
20#include <errno.h>
21#include <limits.h>
22
23bool
24__libc_scratch_buffer_set_array_size (struct scratch_buffer *buffer,
25 size_t nelem, size_t size)
26{
27 size_t new_length = nelem * size;
28
29 /* Avoid overflow check if both values are small. */
30 if ((nelem | size) >> (sizeof (size_t) * CHAR_BIT / 2) != 0
31 && nelem != 0 && size != new_length / nelem)
32 {
33 /* Overflow. Discard the old buffer, but it must remain valid
34 to free. */
35 scratch_buffer_free (buffer);
36 scratch_buffer_init (buffer);
37 __set_errno (ENOMEM);
38 return false;
39 }
40
41 if (new_length <= buffer->length)
42 return true;
43
44 /* Discard old buffer. */
45 scratch_buffer_free (buffer);
46
47 char *new_ptr = malloc (new_length);
48 if (new_ptr == NULL)
49 {
50 /* Buffer must remain valid to free. */
51 scratch_buffer_init (buffer);
52 return false;
53 }
54
55 /* Install new heap-based buffer. */
56 buffer->data = new_ptr;
57 buffer->length = new_length;
58 return true;
59}
60libc_hidden_def (__libc_scratch_buffer_set_array_size);
