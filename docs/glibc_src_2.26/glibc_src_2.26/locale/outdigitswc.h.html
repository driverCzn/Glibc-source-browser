23/* Look up the value of the next multibyte character and return its numerical
24 value if it is one of the digits known in the locale. If *DECIDED is
25 -1 this means it is not yet decided which form it is and we have to
26 search through all available digits. Otherwise we know which script
27 the digits are from. */
28static inline wchar_t
29outdigitwc_value (int n)
30{
31 assert (0 <= n && n <= 9);
32
33 return _NL_CURRENT_WORD (LC_CTYPE, _NL_CTYPE_OUTDIGIT0_WC + n);
34}
