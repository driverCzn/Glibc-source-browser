21#include <unistd.h>
22
23bool
24internal_function
25__spawn_valid_fd (int fd)
26{
27 long maxfd = __sysconf (_SC_OPEN_MAX);
28 return __glibc_likely (fd >= 0)
29 && (__glibc_unlikely (maxfd < 0) /* No limit set. */
30 || __glibc_likely (fd < maxfd));
31}
