18#include <spawn.h>
19#include "spawn_int.h"
20#include <shlib-compat.h>
21
22/* Spawn a new process executing FILE with the attributes describes in *ATTRP.
23 Before running the process perform the actions described in FILE-ACTIONS. */
24int
25__posix_spawnp (pid_t *pid, const char *file,
26 const posix_spawn_file_actions_t *file_actions,
27 const posix_spawnattr_t *attrp, char *const argv[],
28 char *const envp[])
29{
30 return __spawni (pid, file, file_actions, attrp, argv, envp,
31 SPAWN_XFLAGS_USE_PATH);
32}
33versioned_symbol (libc, __posix_spawnp, posix_spawnp, GLIBC_2_15);
34
35
36#if SHLIB_COMPAT (libc, GLIBC_2_2, GLIBC_2_15)
37int
38attribute_compat_text_section
39__posix_spawnp_compat (pid_t *pid, const char *file,
40 const posix_spawn_file_actions_t *file_actions,
41 const posix_spawnattr_t *attrp, char *const argv[],
42 char *const envp[])
43{
44 return __spawni (pid, file, file_actions, attrp, argv, envp,
45 SPAWN_XFLAGS_USE_PATH | SPAWN_XFLAGS_TRY_SHELL);
46}
47compat_symbol (libc, __posix_spawnp_compat, posix_spawnp, GLIBC_2_2);
48#endif
