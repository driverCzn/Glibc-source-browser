29 /* Read the ethernet address: 6 x 8bit hexadecimal number. */
30 {
31 size_t cnt;
32
33 for (cnt = 0; cnt < 6; ++cnt)
34 {
35 unsigned int number;
36
37 if (cnt < 5)
38 INT_FIELD (number, ISCOLON , 0, 16, (unsigned int))
39 else
40 INT_FIELD (number, isspace, 1, 16, (unsigned int))
41
42 if (number > 0xff)
43 return 0;
44 result->e_addr.ether_addr_octet[cnt] = number;
45 }
46 };
47 STRING_FIELD (result->e_name, isspace, 1);
