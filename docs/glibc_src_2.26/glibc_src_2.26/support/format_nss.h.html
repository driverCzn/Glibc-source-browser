27/* The following functions format their arguments as human-readable
28 strings (which can span multiple lines). The caller must free the
29 returned buffer. For NULL pointers or failure status arguments,
30 error variables such as h_errno and errno are included in the
31 result. */

33char *support_format_addrinfo (struct addrinfo *, int ret);
34char *support_format_dns_packet (const unsigned char *buffer, size_t length);
