22long long
23xlseek (int fd, long long offset, int whence)
24{
25 long long result = lseek64 (fd, offset, whence);
26 if (result < 0)
27 FAIL_EXIT1 ("lseek64 (%d, %lld, %d): %m", fd, offset, whence);
28 return result;
29}
