25int
26xaccept (int fd, struct sockaddr *sa, socklen_t *salen)
27{
28 int clientfd = accept (fd, sa, salen);
29 if (clientfd < 0)
30 FAIL_EXIT1 ("accept (%d): %m", fd);
31 return clientfd;
32}
