36/*
37 * Cache the result of getdtablesize(), so we don't have to do an
38 * expensive system call every time.
39 */
40int
41_rpc_dtablesize (void)
42{
43 static int size;
44
45 if (size == 0)
46 size = __getdtablesize ();
47
48 return size;
49}
