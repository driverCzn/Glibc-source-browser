43/*
44 * Create a socket that is locally bound to a non-reserve port. For
45 * any failures, -1 is returned which will cause the RPC code to
46 * create the socket.
47 */
48int
49internal_function
50__get_socket (struct sockaddr_in *saddr)
51{
52 int so = __socket (AF_INET, SOCK_STREAM, IPPROTO_TCP);
53 if (so < 0)
54 return -1;
55
56 struct sockaddr_in laddr;
57 socklen_t namelen = sizeof (laddr);
58 laddr.sin_family = AF_INET;
59 laddr.sin_port = 0;
60 laddr.sin_addr.s_addr = htonl (INADDR_ANY);
61
62 int cc = __bind (so, (struct sockaddr *) &laddr, namelen);
63 if (__glibc_unlikely (cc < 0))
64 {
65 fail:
66 __close (so);
67 return -1;
68 }
69
70 cc = __connect (so, (struct sockaddr *) saddr, namelen);
71 if (__glibc_unlikely (cc < 0))
72 goto fail;
73
74 return so;
75}

78/*
79 * Find the mapped port for program,version.
80 * Internal version with additional parameters.
81 * Calls the pmap service remotely to do the lookup.
82 * Returns 0 if no map exists.
83 */
84u_short
85internal_function
86__libc_rpc_getport (struct sockaddr_in *address, u_long program,
87 u_long version, u_int protocol, time_t timeout_sec,
88 time_t tottimeout_sec)
89{
90 const struct timeval timeout = {timeout_sec, 0};
91 const struct timeval tottimeout = {tottimeout_sec, 0};
92
93 u_short port = 0;
94 int socket = -1;
95 CLIENT *client;
96 struct pmap parms;
97 bool closeit = false;
98
99 address->sin_port = htons (PMAPPORT);
100 if (protocol == IPPROTO_TCP)
101 {
102 /* Don't need a reserved port to get ports from the portmapper. */
103 socket = __get_socket(address);
104 if (socket != -1)
105 closeit = true;
106 client = clnttcp_create (address, PMAPPROG, PMAPVERS, &socket,
107 RPCSMALLMSGSIZE, RPCSMALLMSGSIZE);
108 }
109 else
110 client = clntudp_bufcreate (address, PMAPPROG, PMAPVERS, timeout,
111 &socket, RPCSMALLMSGSIZE, RPCSMALLMSGSIZE);
112 if (client != (CLIENT *) NULL)
113 {
114 struct rpc_createerr *ce = &get_rpc_createerr ();
115 parms.pm_prog = program;
116 parms.pm_vers = version;
117 parms.pm_prot = protocol;
118 parms.pm_port = 0; /* not needed or used */
119 if (CLNT_CALL (client, PMAPPROC_GETPORT, (xdrproc_t)xdr_pmap,
120 (caddr_t)&parms, (xdrproc_t)xdr_u_short,
121 (caddr_t)&port, tottimeout) != RPC_SUCCESS)
122 {
123 ce->cf_stat = RPC_PMAPFAILURE;
124 clnt_geterr (client, &ce->cf_error);
125 }
126 else if (port == 0)
127 {
128 ce->cf_stat = RPC_PROGNOTREGISTERED;
129 }
130 CLNT_DESTROY (client);
131 }
132 /* We only need to close the socket here if we opened it. */
133 if (closeit)
134 (void) __close (socket);
135 address->sin_port = 0;
136 return port;
137}

145/*
146 * Find the mapped port for program,version.
147 * Calls the pmap service remotely to do the lookup.
148 * Returns 0 if no map exists.
149 */
150u_short
151pmap_getport (struct sockaddr_in *address, u_long program, u_long version,
152 u_int protocol)
153{
154 return __libc_rpc_getport (address, program, version, protocol, 5, 60);
155}
