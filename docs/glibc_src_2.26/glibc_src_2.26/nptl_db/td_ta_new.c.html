28/* Datatype for the list of known thread agents. Normally there will
29 be exactly one so we don't spend much though on making it fast. */
30LIST_HEAD (__td_agent_list);
31
32
33td_err_e
34td_ta_new (struct ps_prochandle *ps, td_thragent_t **ta)
35{
36 psaddr_t versaddr;
37 char versbuf[sizeof (VERSION)];
38
39 LOG ("td_ta_new");
40
41 /* Check whether the versions match. */
42 if (td_lookup (ps, SYM_nptl_version, &versaddr) != PS_OK)
43 return TD_NOLIBTHREAD;
44 if (ps_pdread (ps, versaddr, versbuf, sizeof (versbuf)) != PS_OK)
45 return TD_ERR;
46
47 if (memcmp (versbuf, VERSION, sizeof VERSION) != 0)
48 /* Not the right version. */
49 return TD_VERSION;
50
51 /* Fill in the appropriate information. */
52 *ta = (td_thragent_t *) calloc (1, sizeof (td_thragent_t));
53 if (*ta == NULL)
54 return TD_MALLOC;
55
56 /* Store the proc handle which we will pass to the callback functions
57 back into the debugger. */
58 (*ta)->ph = ps;
59
60 /* Now add the new agent descriptor to the list. */
61 list_add (&(*ta)->list, &__td_agent_list);
62
63 return TD_OK;
64}
