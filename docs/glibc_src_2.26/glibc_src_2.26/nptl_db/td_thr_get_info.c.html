20#include <stddef.h>
21#include <string.h>
22#include <stdint.h>
23#include "thread_dbP.h"

25
26td_err_e
27td_thr_get_info (const td_thrhandle_t *th, td_thrinfo_t *infop)
28{
29 td_err_e err;
30 void *copy;
31 psaddr_t tls, schedpolicy, schedprio, cancelhandling, tid, report_events;
32
33 LOG ("td_thr_get_info");
34
35 if (th->th_unique == 0)
36 {
37 /* Special case for the main thread before initialization. */
38 copy = NULL;
39 tls = 0;
40 cancelhandling = 0;
41 schedpolicy = SCHED_OTHER;
42 schedprio = 0;
43 tid = 0;
44 err = DB_GET_VALUE (report_events, th->th_ta_p,
45 __nptl_initial_report_events, 0);
46 }
47 else
48 {
49 /* Copy the whole descriptor in once so we can access the several
50 fields locally. Excess copying in one go is much better than
51 multiple ps_pdread calls. */
52 err = DB_GET_STRUCT (copy, th->th_ta_p, th->th_unique, pthread);
53 if (err != TD_OK)
54 return err;
55
56 err = DB_GET_FIELD_ADDRESS (tls, th->th_ta_p, th->th_unique,
57 pthread, specific, 0);
58 if (err != TD_OK)
59 return err;
60
61 err = DB_GET_FIELD_LOCAL (schedpolicy, th->th_ta_p, copy, pthread,
62 schedpolicy, 0);
63 if (err != TD_OK)
64 return err;
65 err = DB_GET_FIELD_LOCAL (schedprio, th->th_ta_p, copy, pthread,
66 schedparam_sched_priority, 0);
67 if (err != TD_OK)
68 return err;
69 err = DB_GET_FIELD_LOCAL (tid, th->th_ta_p, copy, pthread, tid, 0);
70 if (err != TD_OK)
71 return err;
72 err = DB_GET_FIELD_LOCAL (cancelhandling, th->th_ta_p, copy, pthread,
73 cancelhandling, 0);
74 if (err != TD_OK)
75 return err;
76 err = DB_GET_FIELD_LOCAL (report_events, th->th_ta_p, copy, pthread,
77 report_events, 0);
78 }
79 if (err != TD_OK)
80 return err;
81
82 /* Fill in information. Clear first to provide reproducable
83 results for the fields we do not fill in. */
84 memset (infop, '\0', sizeof (td_thrinfo_t));
85
86 infop->ti_tid = (thread_t) th->th_unique;
87 infop->ti_tls = (char *) tls;
88 infop->ti_pri = ((uintptr_t) schedpolicy == SCHED_OTHER
89 ? 0 : (uintptr_t) schedprio);
90 infop->ti_type = TD_THR_USER;
91
92 if ((((int) (uintptr_t) cancelhandling) & EXITING_BITMASK) == 0)
93 /* XXX For now there is no way to get more information. */
94 infop->ti_state = TD_THR_ACTIVE;
95 else if ((((int) (uintptr_t) cancelhandling) & TERMINATED_BITMASK) == 0)
96 infop->ti_state = TD_THR_ZOMBIE;
97 else
98 infop->ti_state = TD_THR_UNKNOWN;
99
100 /* Initialization which are the same in both cases. */
101 infop->ti_ta_p = th->th_ta_p;
102 infop->ti_lid = tid == 0 ? ps_getpid (th->th_ta_p->ph) : (uintptr_t) tid;
103 infop->ti_traceme = report_events != 0;
104
105 if (copy != NULL)
106 err = DB_GET_FIELD_LOCAL (infop->ti_startfunc, th->th_ta_p, copy, pthread,
107 start_routine, 0);
108 if (copy != NULL && err == TD_OK)
109 {
110 uint32_t idx;
111 for (idx = 0; idx < TD_EVENTSIZE; ++idx)
112 {
113 psaddr_t word;
114 err = DB_GET_FIELD_LOCAL (word, th->th_ta_p, copy, pthread,
115 eventbuf_eventmask_event_bits, idx);
116 if (err != TD_OK)
117 break;
118 infop->ti_events.event_bits[idx] = (uintptr_t) word;
119 }
120 if (err == TD_NOAPLIC)
121 memset (&infop->ti_events.event_bits[idx], 0,
122 (TD_EVENTSIZE - idx) * sizeof infop->ti_events.event_bits[0]);
123 }
124
125 return err;
126}
