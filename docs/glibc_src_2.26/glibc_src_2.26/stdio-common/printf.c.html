18#include <libioP.h>
19#include <stdarg.h>
20#include <stdio.h>
21
22#undef printf
23
24/* Write formatted output to stdout from the format string FORMAT. */
25/* VARARGS1 */
26int
27__printf (const char *format, ...)
28{
29 va_list arg;
30 int done;
31
32 va_start (arg, format);
33 done = vfprintf (stdout, format, arg);
34 va_end (arg);
35
36 return done;
37}
38
39#undef _IO_printf
40ldbl_strong_alias (__printf, printf);
41/* This is for libg++. */
42ldbl_strong_alias (__printf, _IO_printf);
