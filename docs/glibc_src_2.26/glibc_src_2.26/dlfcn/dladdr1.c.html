21#if !defined SHARED && IS_IN (libdl)
22
23int
24dladdr1 (const void *address, Dl_info *info, void **extra, int flags)
25{
26 return __dladdr1 (address, info, extra, flags);
27}
28
29#else
30
31int
32__dladdr1 (const void *address, Dl_info *info, void **extra, int flags)
33{
34# ifdef SHARED
35 if (__glibc_unlikely (_dlfcn_hook != NULL))
36 return _dlfcn_hook->dladdr1 (address, info, extra, flags);
37# endif
38
39 switch (flags)
40 {
41 default: /* Make this an error? */
42 case 0:
43 return _dl_addr (address, info, NULL, NULL);
44 case RTLD_DL_SYMENT:
45 return _dl_addr (address, info, NULL, (const ElfW(Sym) **) extra);
46 case RTLD_DL_LINKMAP:
47 return _dl_addr (address, info, (struct link_map **) extra, NULL);
48 }
49}
50# ifdef SHARED
51strong_alias (__dladdr1, dladdr1)
52# endif
53#endif
