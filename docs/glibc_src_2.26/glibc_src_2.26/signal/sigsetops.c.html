19#include <signal.h>
20#include <sigsetops.h>

23/* These were formerly defined by <signal.h> as inline functions,
24 so they require out-of-line compatibility definitions. */
25#if SHLIB_COMPAT (libc, GLIBC_2_0, GLIBC_2_26)
26
27int
28attribute_compat_text_section
29(__sigismember) (const __sigset_t *set, int sig)
30{
31 return __sigismember (set, sig);
32}
33compat_symbol (libc, __sigismember, __sigismember, GLIBC_2_0);
34
35int
36attribute_compat_text_section
37(__sigaddset) (__sigset_t *set, int sig)
38{
39 __sigaddset (set, sig);
40 return 0;
41}
42compat_symbol (libc, __sigaddset, __sigaddset, GLIBC_2_0);
43
44int
45attribute_compat_text_section
46(__sigdelset) (__sigset_t *set, int sig)
47{
48 __sigdelset (set, sig);
49 return 0;
50}
51compat_symbol (libc, __sigdelset, __sigdelset, GLIBC_2_0);
52
53#endif
