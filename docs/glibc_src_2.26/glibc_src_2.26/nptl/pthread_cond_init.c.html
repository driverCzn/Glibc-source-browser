25/* See __pthread_cond_wait for details. */
26int
27__pthread_cond_init (pthread_cond_t *cond, const pthread_condattr_t *cond_attr)
28{
29 struct pthread_condattr *icond_attr = (struct pthread_condattr *) cond_attr;
30
31 memset (cond, 0, sizeof (pthread_cond_t));
32
33 /* Update the pretty printers if the internal representation of icond_attr
34 is changed. */
35
36 /* Iff not equal to ~0l, this is a PTHREAD_PROCESS_PRIVATE condvar. */
37 if (icond_attr != NULL && (icond_attr->value & 1) != 0)
38 cond->__data.__wrefs |= __PTHREAD_COND_SHARED_MASK;
39 int clockid = (icond_attr != NULL
40 ? ((icond_attr->value >> 1) & ((1 << COND_CLOCK_BITS) - 1))
41 : CLOCK_REALTIME);
42 /* If 0, CLOCK_REALTIME is used; CLOCK_MONOTONIC otherwise. */
43 if (clockid != CLOCK_REALTIME)
44 cond->__data.__wrefs |= __PTHREAD_COND_CLOCK_MONOTONIC_MASK;
45
46 LIBC_PROBE (cond_init, 2, cond, cond_attr);
47
48 return 0;
49}
50versioned_symbol (libpthread, __pthread_cond_init,
51 pthread_cond_init, GLIBC_2_3_2);
