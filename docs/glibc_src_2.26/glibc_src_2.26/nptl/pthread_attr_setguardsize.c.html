19#include <assert.h>
20#include "pthreadP.h"
21
22
23int
24pthread_attr_setguardsize (pthread_attr_t *attr, size_t guardsize)
25{
26 struct pthread_attr *iattr;
27
28 assert (sizeof (*attr) >= sizeof (struct pthread_attr));
29 iattr = (struct pthread_attr *) attr;
30
31 /* Note that we don't round the value here. The standard requires
32 that subsequent pthread_attr_getguardsize calls return the value
33 set by the user. */
34 iattr->guardsize = guardsize;
35
36 return 0;
37}
