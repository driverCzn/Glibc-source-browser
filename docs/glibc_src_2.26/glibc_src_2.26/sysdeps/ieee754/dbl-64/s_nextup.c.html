22/* Return the least floating-point number greater than X. */
23double
24__nextup (double x)
25{
26 int32_t hx, ix;
27 u_int32_t lx;
28
29 EXTRACT_WORDS (hx, lx, x);
30 ix = hx & 0x7fffffff;
31
32 if (((ix >= 0x7ff00000) && ((ix - 0x7ff00000) | lx) != 0)) /* x is nan. */
33 return x + x;
34 if ((ix | lx) == 0)
35 return DBL_TRUE_MIN;
36 if (hx >= 0)
37 { /* x > 0. */
38 if (isinf (x))
39 return x;
40 lx += 1;
41 if (lx == 0)
42 hx += 1;
43 }
44 else
45 { /* x < 0. */
46 if (lx == 0)
47 hx -= 1;
48 lx -= 1;
49 }
50 INSERT_WORDS (x, hx, lx);
51 return x;
52}
