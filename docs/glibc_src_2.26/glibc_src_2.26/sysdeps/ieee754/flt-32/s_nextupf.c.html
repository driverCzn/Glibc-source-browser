22/* Return the least floating-point number greater than X. */
23float
24__nextupf (float x)
25{
26 int32_t hx, ix;
27
28 GET_FLOAT_WORD (hx, x);
29 ix = hx & 0x7fffffff;
30 if (ix == 0)
31 return FLT_TRUE_MIN;
32 if (ix > 0x7f800000) /* x is nan. */
33 return x + x;
34 if (hx >= 0)
35 { /* x > 0. */
36 if (isinf (x))
37 return x;
38 hx += 1;
39 }
40 else
41 hx -= 1;
42 SET_FLOAT_WORD (x, hx);
43 return x;
44}
