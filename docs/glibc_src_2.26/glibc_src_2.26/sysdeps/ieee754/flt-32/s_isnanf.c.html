20/*
21 * isnanf(x) returns 1 is x is nan, else 0;
22 * no branching!
23 */
24
25#include <math.h>
26#include <math_private.h>
27
28#undef __isnanf
29int __isnanf(float x)
30{
31 int32_t ix;
32 GET_FLOAT_WORD(ix,x);
33 ix &= 0x7fffffff;
34 ix = 0x7f800000 - ix;
35 return (int)(((u_int32_t)(ix))>>31);
36}
37hidden_def (__isnanf)
38weak_alias (__isnanf, isnanf)
