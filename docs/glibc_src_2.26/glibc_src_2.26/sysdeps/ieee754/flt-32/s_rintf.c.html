16#include <math.h>
17#include <math_private.h>
18
19static const float
20TWO23[2]={
21 8.3886080000e+06, /* 0x4b000000 */
22 -8.3886080000e+06, /* 0xcb000000 */
23};
24
25float
26__rintf(float x)
27{
28 int32_t i0,j0,sx;
29 float w,t;
30 GET_FLOAT_WORD(i0,x);
31 sx = (i0>>31)&1;
32 j0 = ((i0>>23)&0xff)-0x7f;
33 if(j0<23) {
34 if(j0<0) {
35 w = TWO23[sx]+x;
36 t = w-TWO23[sx];
37 GET_FLOAT_WORD(i0,t);
38 SET_FLOAT_WORD(t,(i0&0x7fffffff)|(sx<<31));
39 return t;
40 }
41 } else {
42 if(j0==0x80) return x+x; /* inf or NaN */
43 else return x; /* x is integral */
44 }
45 w = TWO23[sx]+x;
46 return w-TWO23[sx];
47}

49weak_alias (__rintf, rintf)
50#endif
