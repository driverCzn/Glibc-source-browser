20/*
21 * fabsl(x) returns the absolute value of x.
22 */
23
24#include <math.h>
25#include <math_private.h>
26
27_Float128 __fabsl(_Float128 x)
28{
29 u_int64_t hx;
30 GET_LDOUBLE_MSW64(hx,x);
31 SET_LDOUBLE_MSW64(x,hx&0x7fffffffffffffffLL);
32 return x;
33}
34weak_alias (__fabsl, fabsl)
