16/* __ieee754_acoshl(x)
17 * Method :
18 * Based on
19 * acoshl(x) = logl [ x + sqrtl(x*x-1) ]
20 * we have
21 * acoshl(x) := logl(x)+ln2, if x is large; else
22 * acoshl(x) := logl(2x-1/(sqrtl(x*x-1)+x)) if x>2; else
23 * acoshl(x) := log1pl(t+sqrtl(2.0*t+t*t)); where t=x-1.
24 *
25 * Special cases:
26 * acoshl(x) is NaN with signal if x<1.
27 * acoshl(NaN) is NaN without signal.
28 */
29
30#include <math.h>
31#include <math_private.h>
32
33static const _Float128
34one = 1.0,
35ln2 = L(0.6931471805599453094172321214581766);
36
37_Float128
38__ieee754_acoshl(_Float128 x)
39{
40 _Float128 t;
41 u_int64_t lx;
42 int64_t hx;
43 GET_LDOUBLE_WORDS64(hx,lx,x);
44 if(hx<0x3fff000000000000LL) { /* x < 1 */
45 return (x-x)/(x-x);
46 } else if(hx >=0x4035000000000000LL) { /* x > 2**54 */
47 if(hx >=0x7fff000000000000LL) { /* x is inf of NaN */
48 return x+x;
49 } else
50 return __ieee754_logl(x)+ln2; /* acoshl(huge)=logl(2x) */
51 } else if(((hx-0x3fff000000000000LL)|lx)==0) {
52 return 0; /* acosh(1) = 0 */
53 } else if (hx > 0x4000000000000000LL) { /* 2**28 > x > 2 */
54 t=x*x;
55 return __ieee754_logl(2*x-one/(x+__ieee754_sqrtl(t-one)));
56 } else { /* 1<x<2 */
57 t = x-one;
58 return __log1pl(t+__sqrtl(2*t+t*t));
59 }
60}
61strong_alias (__ieee754_acoshl, __acoshl_finite)
