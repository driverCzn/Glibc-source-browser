18#include <math.h>
19#include <math_private.h>
20#include <float.h>
21
22static const _Float128 log10_high = L(0x2.4d763776aaa2bp0);
23static const _Float128 log10_low = L(0x5.ba95b58ae0b4c28a38a3fb3e7698p-60);
24
25_Float128
26__ieee754_exp10l (_Float128 arg)
27{
28 ieee854_long_double_shape_type u;
29 _Float128 arg_high, arg_low;
30 _Float128 exp_high, exp_low;
31
32 if (!isfinite (arg))
33 return __ieee754_expl (arg);
34 if (arg < LDBL_MIN_10_EXP - LDBL_DIG - 10)
35 return LDBL_MIN * LDBL_MIN;
36 else if (arg > LDBL_MAX_10_EXP + 1)
37 return LDBL_MAX * LDBL_MAX;
38 else if (fabsl (arg) < L(0x1p-116))
39 return 1;
40
41 u.value = arg;
42 u.parts64.lsw &= 0xfe00000000000000LL;
43 arg_high = u.value;
44 arg_low = arg - arg_high;
45 exp_high = arg_high * log10_high;
46 exp_low = arg_high * log10_low + arg_low * M_LN10l;
47 return __ieee754_expl (exp_high) * __ieee754_expl (exp_low);
48}
