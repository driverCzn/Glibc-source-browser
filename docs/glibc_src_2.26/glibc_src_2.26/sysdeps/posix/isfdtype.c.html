19#include <errno.h>
20#include <sys/stat.h>
21#include <sys/socket.h>
22#include <sys/types.h>
23
24int
25isfdtype (int fildes, int fdtype)
26{
27 struct stat64 st;
28 int result;
29
30 {
31 int save_error = errno;
32 result = fstat64 (fildes, &st);
33 __set_errno (save_error);
34 }
35
36 return result ?: (st.st_mode & S_IFMT) == (mode_t) fdtype;
37}
