32__float128
33__ieee754_sqrtf128 (__float128 a)
34{
35 FP_DECL_EX;
36 FP_DECL_Q (A);
37 FP_DECL_Q (R);
38 __float128 r;
39
40 FP_INIT_ROUNDMODE;
41 FP_UNPACK_Q (A, a);
42 FP_SQRT_Q (R, A);
43 FP_PACK_Q (r, R);
44 FP_HANDLE_EXCEPTIONS;
45 return r;
46}
