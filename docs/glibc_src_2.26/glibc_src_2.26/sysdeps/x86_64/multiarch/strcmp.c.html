20/* Define multiple versions only for the definition in libc. */
21#if IS_IN (libc)
22# define strcmp __redirect_strcmp
23# include <string.h>
24# undef strcmp
25
26# define SYMBOL_NAME strcmp
27# include <init-arch.h>
28
29extern __typeof (REDIRECT_NAME) OPTIMIZE (sse2) attribute_hidden;
30extern __typeof (REDIRECT_NAME) OPTIMIZE (sse2_unaligned) attribute_hidden;
31extern __typeof (REDIRECT_NAME) OPTIMIZE (ssse3) attribute_hidden;
32
33static inline void *
34IFUNC_SELECTOR (void)
35{
36 const struct cpu_features* cpu_features = __get_cpu_features ();
37
38 if (CPU_FEATURES_ARCH_P (cpu_features, Fast_Unaligned_Load))
39 return OPTIMIZE (sse2_unaligned);
40
41 if (CPU_FEATURES_CPU_P (cpu_features, SSSE3))
42 return OPTIMIZE (ssse3);
43
44 return OPTIMIZE (sse2);
45}
46
47libc_ifunc_redirected (__redirect_strcmp, strcmp, IFUNC_SELECTOR ());
48

50__hidden_ver1 (strcmp, __GI_strcmp, __redirect_strcmp)
51 __attribute__ ((visibility ("hidden")));
52# endif
53#endif
