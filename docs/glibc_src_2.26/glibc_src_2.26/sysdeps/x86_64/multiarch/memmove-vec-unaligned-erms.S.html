19/* memmove/memcpy/mempcpy is implemented as:
20 1. Use overlapping load and store to avoid branch.
21 2. Load all sources into registers and store them together to avoid
22 possible address overlap between source and destination.
23 3. If size is 8 * VEC_SIZE or less, load all sources into registers
24 and store them together.
25 4. If address of destination > address of source, backward copy
26 4 * VEC_SIZE at a time with unaligned load and aligned store.
27 Load the first 4 * VEC and last VEC before the loop and store
28 them after the loop to support overlapping addresses.
29 5. Otherwise, forward copy 4 * VEC_SIZE at a time with unaligned
30 load and aligned store. Load the last 4 * VEC and first VEC
31 before the loop and store them after the loop to support
32 overlapping addresses.
33 6. If size >= __x86_shared_non_temporal_threshold and there is no
34 overlap between destination and source, use non-temporal store
35 instead of aligned store. */
36

59/* Threshold to use Enhanced REP MOVSB. Since there is overhead to set
60 up REP MOVSB operation, REP MOVSB isn't faster on short data. The
61 memcpy micro benchmark in glibc shows that 2KB is the approximate
62 value above which REP MOVSB becomes faster than SSE2 optimization
63 on processors with Enhanced REP MOVSB. Since larger register size
64 can move more data with a single load and store, the threshold is
65 higher with larger register size. */
66#ifndef REP_MOVSB_THRESHOLD
67# define REP_MOVSB_THRESHOLD (2048 * (VEC_SIZE / 16))
68#endif

71# define PREFETCH(addr) prefetcht0 addr

74/* Assume 64-byte prefetch size. */
75#ifndef PREFETCH_SIZE
76# define PREFETCH_SIZE 64
77#endif

79#define PREFETCHED_LOAD_SIZE (VEC_SIZE * 4)
80
81#if PREFETCH_SIZE == 64
82# if PREFETCHED_LOAD_SIZE == PREFETCH_SIZE
83# define PREFETCH_ONE_SET(dir, base, offset) \
84 PREFETCH ((offset)base)
85# elif PREFETCHED_LOAD_SIZE == 2 * PREFETCH_SIZE
86# define PREFETCH_ONE_SET(dir, base, offset) \
87 PREFETCH ((offset)base); \
88 PREFETCH ((offset + dir * PREFETCH_SIZE)base)
89# elif PREFETCHED_LOAD_SIZE == 4 * PREFETCH_SIZE
90# define PREFETCH_ONE_SET(dir, base, offset) \
91 PREFETCH ((offset)base); \
92 PREFETCH ((offset + dir * PREFETCH_SIZE)base); \
93 PREFETCH ((offset + dir * PREFETCH_SIZE)base); \
94 PREFETCH ((offset + dir * PREFETCH_SIZE * 2)base); \
95 PREFETCH ((offset + dir * PREFETCH_SIZE * 3)base)
96# else
97# error Unsupported PREFETCHED_LOAD_SIZE!
98# endif

130ENTRY (MEMMOVE_SYMBOL (__memmove, unaligned))
131 movq %rdi, %rax
132L(start):
133# ifdef __ILP32__
134 /* Clear the upper 32 bits. */
135 movl %edx, %edx
136# endif
137 cmp $VEC_SIZE, %RDX_LP
138 jb L(less_vec)
139 cmp $(VEC_SIZE * 2), %RDX_LP
140 ja L(more_2x_vec)
141#if !defined USE_MULTIARCH || !IS_IN (libc)
142L(last_2x_vec):
143#endif
144 /* From VEC and to 2 * VEC. No branch when size == VEC_SIZE. */
145 VMOVU (%rsi), %VEC(0)
146 VMOVU -VEC_SIZE(%rsi,%rdx), %VEC(1)
147 VMOVU %VEC(0), (%rdi)
148 VMOVU %VEC(1), -VEC_SIZE(%rdi,%rdx)
149 VZEROUPPER
150#if !defined USE_MULTIARCH || !IS_IN (libc)
151L(nop):
152#endif
153 ret
154#if defined USE_MULTIARCH && IS_IN (libc)
155END (MEMMOVE_SYMBOL (__memmove, unaligned))

157# if VEC_SIZE == 16
158# if defined SHARED
159ENTRY (__mempcpy_chk_erms)
160 cmp %RDX_LP, %RCX_LP
161 jb HIDDEN_JUMPTARGET (__chk_fail)
162END (__mempcpy_chk_erms)

164/* Only used to measure performance of REP MOVSB. */
165ENTRY (__mempcpy_erms)
166 mov %RDI_LP, %RAX_LP
167 add %RDX_LP, %RAX_LP
168 jmp L(start_movsb)
169END (__mempcpy_erms)
170# endif
171
172ENTRY (__memmove_chk_erms)
173 cmp %RDX_LP, %RCX_LP
174 jb HIDDEN_JUMPTARGET (__chk_fail)
175END (__memmove_chk_erms)
176
177ENTRY (__memmove_erms)
178 movq %rdi, %rax
179L(start_movsb):
180 mov %RDX_LP, %RCX_LP
181 cmp %RSI_LP, %RDI_LP
182 jb 1f
183 /* Source == destination is less common. */
184 je 2f
185 lea (%rsi,%rcx), %RDX_LP
186 cmp %RDX_LP, %RDI_LP
187 jb L(movsb_backward)
1881:
189 rep movsb
1902:
191 ret
192L(movsb_backward):
193 leaq -1(%rdi,%rcx), %rdi
194 leaq -1(%rsi,%rcx), %rsi
195 std
196 rep movsb
197 cld
198 ret
199END (__memmove_erms)

201strong_alias (__memmove_erms, __memcpy_erms)
202strong_alias (__memmove_chk_erms, __memcpy_chk_erms)
203# endif
204# endif

206# ifdef SHARED

209 jb HIDDEN_JUMPTARGET (__chk_fail)

212ENTRY (MEMMOVE_SYMBOL (__mempcpy, unaligned_erms))
213 mov %RDI_LP, %RAX_LP
214 add %RDX_LP, %RAX_LP
215 jmp L(start_erms)
216END (MEMMOVE_SYMBOL (__mempcpy, unaligned_erms))

218ENTRY (MEMMOVE_CHK_SYMBOL (__memmove_chk, unaligned_erms))
219 cmp %RDX_LP, %RCX_LP
220 jb HIDDEN_JUMPTARGET (__chk_fail)
221END (MEMMOVE_CHK_SYMBOL (__memmove_chk, unaligned_erms))

224ENTRY (MEMMOVE_SYMBOL (__memmove, unaligned_erms))
225 movq %rdi, %rax
226L(start_erms):
227# ifdef __ILP32__
228 /* Clear the upper 32 bits. */
229 movl %edx, %edx
230# endif
231 cmp $VEC_SIZE, %RDX_LP
232 jb L(less_vec)
233 cmp $(VEC_SIZE * 2), %RDX_LP
234 ja L(movsb_more_2x_vec)
235L(last_2x_vec):
236 /* From VEC and to 2 * VEC. No branch when size == VEC_SIZE. */
237 VMOVU (%rsi), %VEC(0)
238 VMOVU -VEC_SIZE(%rsi,%rdx), %VEC(1)
239 VMOVU %VEC(0), (%rdi)
240 VMOVU %VEC(1), -VEC_SIZE(%rdi,%rdx)
241L(return):
242 VZEROUPPER
243 ret
244
245L(movsb):
246 cmpq __x86_shared_non_temporal_threshold(%rip), %rdx
247 jae L(more_8x_vec)
248 cmpq %rsi, %rdi
249 jb 1f
250 /* Source == destination is less common. */
251 je L(nop)
252 leaq (%rsi,%rdx), %r9
253 cmpq %r9, %rdi
254 /* Avoid slow backward REP MOVSB. */
255# if REP_MOVSB_THRESHOLD <= (VEC_SIZE * 8)
256# error Unsupported REP_MOVSB_THRESHOLD and VEC_SIZE!
257# endif
258 jb L(more_8x_vec_backward)
2591:
260 mov %RDX_LP, %RCX_LP
261 rep movsb
262L(nop):
263 ret
264#endif
265
266L(less_vec):
267 /* Less than 1 VEC. */
268#if VEC_SIZE != 16 && VEC_SIZE != 32 && VEC_SIZE != 64
269# error Unsupported VEC_SIZE!
270#endif
271#if VEC_SIZE > 32
272 cmpb $32, %dl
273 jae L(between_32_63)
274#endif
275#if VEC_SIZE > 16
276 cmpb $16, %dl
277 jae L(between_16_31)
278#endif
279 cmpb $8, %dl
280 jae L(between_8_15)
281 cmpb $4, %dl
282 jae L(between_4_7)
283 cmpb $1, %dl
284 ja L(between_2_3)
285 jb 1f
286 movzbl (%rsi), %ecx
287 movb %cl, (%rdi)
2881:
289 ret
290#if VEC_SIZE > 32
291L(between_32_63):
292 /* From 32 to 63. No branch when size == 32. */
293 vmovdqu (%rsi), %ymm0
294 vmovdqu -32(%rsi,%rdx), %ymm1
295 vmovdqu %ymm0, (%rdi)
296 vmovdqu %ymm1, -32(%rdi,%rdx)
297 VZEROUPPER
298 ret
299#endif
300#if VEC_SIZE > 16
301 /* From 16 to 31. No branch when size == 16. */
302L(between_16_31):
303 vmovdqu (%rsi), %xmm0
304 vmovdqu -16(%rsi,%rdx), %xmm1
305 vmovdqu %xmm0, (%rdi)
306 vmovdqu %xmm1, -16(%rdi,%rdx)
307 ret
308#endif
309L(between_8_15):
310 /* From 8 to 15. No branch when size == 8. */
311 movq -8(%rsi,%rdx), %rcx
312 movq (%rsi), %rsi
313 movq %rcx, -8(%rdi,%rdx)
314 movq %rsi, (%rdi)
315 ret
316L(between_4_7):
317 /* From 4 to 7. No branch when size == 4. */
318 movl -4(%rsi,%rdx), %ecx
319 movl (%rsi), %esi
320 movl %ecx, -4(%rdi,%rdx)
321 movl %esi, (%rdi)
322 ret
323L(between_2_3):
324 /* From 2 to 3. No branch when size == 2. */
325 movzwl -2(%rsi,%rdx), %ecx
326 movzwl (%rsi), %esi
327 movw %cx, -2(%rdi,%rdx)
328 movw %si, (%rdi)
329 ret
330
331#if defined USE_MULTIARCH && IS_IN (libc)
332L(movsb_more_2x_vec):
333 cmpq $REP_MOVSB_THRESHOLD, %rdx
334 ja L(movsb)
335#endif
336L(more_2x_vec):
337 /* More than 2 * VEC and there may be overlap between destination
338 and source. */
339 cmpq $(VEC_SIZE * 8), %rdx
340 ja L(more_8x_vec)
341 cmpq $(VEC_SIZE * 4), %rdx
342 jb L(last_4x_vec)
343 /* Copy from 4 * VEC to 8 * VEC, inclusively. */
344 VMOVU (%rsi), %VEC(0)
345 VMOVU VEC_SIZE(%rsi), %VEC(1)
346 VMOVU (VEC_SIZE * 2)(%rsi), %VEC(2)
347 VMOVU (VEC_SIZE * 3)(%rsi), %VEC(3)
348 VMOVU -VEC_SIZE(%rsi,%rdx), %VEC(4)
349 VMOVU -(VEC_SIZE * 2)(%rsi,%rdx), %VEC(5)
350 VMOVU -(VEC_SIZE * 3)(%rsi,%rdx), %VEC(6)
351 VMOVU -(VEC_SIZE * 4)(%rsi,%rdx), %VEC(7)
352 VMOVU %VEC(0), (%rdi)
353 VMOVU %VEC(1), VEC_SIZE(%rdi)
354 VMOVU %VEC(2), (VEC_SIZE * 2)(%rdi)
355 VMOVU %VEC(3), (VEC_SIZE * 3)(%rdi)
356 VMOVU %VEC(4), -VEC_SIZE(%rdi,%rdx)
357 VMOVU %VEC(5), -(VEC_SIZE * 2)(%rdi,%rdx)
358 VMOVU %VEC(6), -(VEC_SIZE * 3)(%rdi,%rdx)
359 VMOVU %VEC(7), -(VEC_SIZE * 4)(%rdi,%rdx)
360 VZEROUPPER
361 ret
362L(last_4x_vec):
363 /* Copy from 2 * VEC to 4 * VEC. */
364 VMOVU (%rsi), %VEC(0)
365 VMOVU VEC_SIZE(%rsi), %VEC(1)
366 VMOVU -VEC_SIZE(%rsi,%rdx), %VEC(2)
367 VMOVU -(VEC_SIZE * 2)(%rsi,%rdx), %VEC(3)
368 VMOVU %VEC(0), (%rdi)
369 VMOVU %VEC(1), VEC_SIZE(%rdi)
370 VMOVU %VEC(2), -VEC_SIZE(%rdi,%rdx)
371 VMOVU %VEC(3), -(VEC_SIZE * 2)(%rdi,%rdx)
372 VZEROUPPER
373 ret
374
375L(more_8x_vec):
376 cmpq %rsi, %rdi
377 ja L(more_8x_vec_backward)
378 /* Source == destination is less common. */
379 je L(nop)
380 /* Load the first VEC and last 4 * VEC to support overlapping
381 addresses. */
382 VMOVU (%rsi), %VEC(4)
383 VMOVU -VEC_SIZE(%rsi, %rdx), %VEC(5)
384 VMOVU -(VEC_SIZE * 2)(%rsi, %rdx), %VEC(6)
385 VMOVU -(VEC_SIZE * 3)(%rsi, %rdx), %VEC(7)
386 VMOVU -(VEC_SIZE * 4)(%rsi, %rdx), %VEC(8)
387 /* Save start and stop of the destination buffer. */
388 movq %rdi, %r11
389 leaq -VEC_SIZE(%rdi, %rdx), %rcx
390 /* Align destination for aligned stores in the loop. Compute
391 how much destination is misaligned. */
392 movq %rdi, %r8
393 andq $(VEC_SIZE - 1), %r8
394 /* Get the negative of offset for alignment. */
395 subq $VEC_SIZE, %r8
396 /* Adjust source. */
397 subq %r8, %rsi
398 /* Adjust destination which should be aligned now. */
399 subq %r8, %rdi
400 /* Adjust length. */
401 addq %r8, %rdx
402#if (defined USE_MULTIARCH || VEC_SIZE == 16) && IS_IN (libc)
403 /* Check non-temporal store threshold. */
404 cmpq __x86_shared_non_temporal_threshold(%rip), %rdx
405 ja L(large_forward)
406#endif
407L(loop_4x_vec_forward):
408 /* Copy 4 * VEC a time forward. */
409 VMOVU (%rsi), %VEC(0)
410 VMOVU VEC_SIZE(%rsi), %VEC(1)
411 VMOVU (VEC_SIZE * 2)(%rsi), %VEC(2)
412 VMOVU (VEC_SIZE * 3)(%rsi), %VEC(3)
413 addq $(VEC_SIZE * 4), %rsi
414 subq $(VEC_SIZE * 4), %rdx
415 VMOVA %VEC(0), (%rdi)
416 VMOVA %VEC(1), VEC_SIZE(%rdi)
417 VMOVA %VEC(2), (VEC_SIZE * 2)(%rdi)
418 VMOVA %VEC(3), (VEC_SIZE * 3)(%rdi)
419 addq $(VEC_SIZE * 4), %rdi
420 cmpq $(VEC_SIZE * 4), %rdx
421 ja L(loop_4x_vec_forward)
422 /* Store the last 4 * VEC. */
423 VMOVU %VEC(5), (%rcx)
424 VMOVU %VEC(6), -VEC_SIZE(%rcx)
425 VMOVU %VEC(7), -(VEC_SIZE * 2)(%rcx)
426 VMOVU %VEC(8), -(VEC_SIZE * 3)(%rcx)
427 /* Store the first VEC. */
428 VMOVU %VEC(4), (%r11)
429 VZEROUPPER
430 ret
431
432L(more_8x_vec_backward):
433 /* Load the first 4 * VEC and last VEC to support overlapping
434 addresses. */
435 VMOVU (%rsi), %VEC(4)
436 VMOVU VEC_SIZE(%rsi), %VEC(5)
437 VMOVU (VEC_SIZE * 2)(%rsi), %VEC(6)
438 VMOVU (VEC_SIZE * 3)(%rsi), %VEC(7)
439 VMOVU -VEC_SIZE(%rsi,%rdx), %VEC(8)
440 /* Save stop of the destination buffer. */
441 leaq -VEC_SIZE(%rdi, %rdx), %r11
442 /* Align destination end for aligned stores in the loop. Compute
443 how much destination end is misaligned. */
444 leaq -VEC_SIZE(%rsi, %rdx), %rcx
445 movq %r11, %r9
446 movq %r11, %r8
447 andq $(VEC_SIZE - 1), %r8
448 /* Adjust source. */
449 subq %r8, %rcx
450 /* Adjust the end of destination which should be aligned now. */
451 subq %r8, %r9
452 /* Adjust length. */
453 subq %r8, %rdx
454#if (defined USE_MULTIARCH || VEC_SIZE == 16) && IS_IN (libc)
455 /* Check non-temporal store threshold. */
456 cmpq __x86_shared_non_temporal_threshold(%rip), %rdx
457 ja L(large_backward)
458#endif
459L(loop_4x_vec_backward):
460 /* Copy 4 * VEC a time backward. */
461 VMOVU (%rcx), %VEC(0)
462 VMOVU -VEC_SIZE(%rcx), %VEC(1)
463 VMOVU -(VEC_SIZE * 2)(%rcx), %VEC(2)
464 VMOVU -(VEC_SIZE * 3)(%rcx), %VEC(3)
465 subq $(VEC_SIZE * 4), %rcx
466 subq $(VEC_SIZE * 4), %rdx
467 VMOVA %VEC(0), (%r9)
468 VMOVA %VEC(1), -VEC_SIZE(%r9)
469 VMOVA %VEC(2), -(VEC_SIZE * 2)(%r9)
470 VMOVA %VEC(3), -(VEC_SIZE * 3)(%r9)
471 subq $(VEC_SIZE * 4), %r9
472 cmpq $(VEC_SIZE * 4), %rdx
473 ja L(loop_4x_vec_backward)
474 /* Store the first 4 * VEC. */
475 VMOVU %VEC(4), (%rdi)
476 VMOVU %VEC(5), VEC_SIZE(%rdi)
477 VMOVU %VEC(6), (VEC_SIZE * 2)(%rdi)
478 VMOVU %VEC(7), (VEC_SIZE * 3)(%rdi)
479 /* Store the last VEC. */
480 VMOVU %VEC(8), (%r11)
481 VZEROUPPER
482 ret
483
484#if (defined USE_MULTIARCH || VEC_SIZE == 16) && IS_IN (libc)
485L(large_forward):
486 /* Don't use non-temporal store if there is overlap between
487 destination and source since destination may be in cache
488 when source is loaded. */
489 leaq (%rdi, %rdx), %r10
490 cmpq %r10, %rsi
491 jb L(loop_4x_vec_forward)
492L(loop_large_forward):
493 /* Copy 4 * VEC a time forward with non-temporal stores. */
494 PREFETCH_ONE_SET (1, (%rsi), PREFETCHED_LOAD_SIZE * 2)
495 PREFETCH_ONE_SET (1, (%rsi), PREFETCHED_LOAD_SIZE * 3)
496 VMOVU (%rsi), %VEC(0)
497 VMOVU VEC_SIZE(%rsi), %VEC(1)
498 VMOVU (VEC_SIZE * 2)(%rsi), %VEC(2)
499 VMOVU (VEC_SIZE * 3)(%rsi), %VEC(3)
500 addq $PREFETCHED_LOAD_SIZE, %rsi
501 subq $PREFETCHED_LOAD_SIZE, %rdx
502 VMOVNT %VEC(0), (%rdi)
503 VMOVNT %VEC(1), VEC_SIZE(%rdi)
504 VMOVNT %VEC(2), (VEC_SIZE * 2)(%rdi)
505 VMOVNT %VEC(3), (VEC_SIZE * 3)(%rdi)
506 addq $PREFETCHED_LOAD_SIZE, %rdi
507 cmpq $PREFETCHED_LOAD_SIZE, %rdx
508 ja L(loop_large_forward)
509 sfence
510 /* Store the last 4 * VEC. */
511 VMOVU %VEC(5), (%rcx)
512 VMOVU %VEC(6), -VEC_SIZE(%rcx)
513 VMOVU %VEC(7), -(VEC_SIZE * 2)(%rcx)
514 VMOVU %VEC(8), -(VEC_SIZE * 3)(%rcx)
515 /* Store the first VEC. */
516 VMOVU %VEC(4), (%r11)
517 VZEROUPPER
518 ret
519
520L(large_backward):
521 /* Don't use non-temporal store if there is overlap between
522 destination and source since destination may be in cache
523 when source is loaded. */
524 leaq (%rcx, %rdx), %r10
525 cmpq %r10, %r9
526 jb L(loop_4x_vec_backward)
527L(loop_large_backward):
528 /* Copy 4 * VEC a time backward with non-temporal stores. */
529 PREFETCH_ONE_SET (-1, (%rcx), -PREFETCHED_LOAD_SIZE * 2)
530 PREFETCH_ONE_SET (-1, (%rcx), -PREFETCHED_LOAD_SIZE * 3)
531 VMOVU (%rcx), %VEC(0)
532 VMOVU -VEC_SIZE(%rcx), %VEC(1)
533 VMOVU -(VEC_SIZE * 2)(%rcx), %VEC(2)
534 VMOVU -(VEC_SIZE * 3)(%rcx), %VEC(3)
535 subq $PREFETCHED_LOAD_SIZE, %rcx
536 subq $PREFETCHED_LOAD_SIZE, %rdx
537 VMOVNT %VEC(0), (%r9)
538 VMOVNT %VEC(1), -VEC_SIZE(%r9)
539 VMOVNT %VEC(2), -(VEC_SIZE * 2)(%r9)
540 VMOVNT %VEC(3), -(VEC_SIZE * 3)(%r9)
541 subq $PREFETCHED_LOAD_SIZE, %r9
542 cmpq $PREFETCHED_LOAD_SIZE, %rdx
543 ja L(loop_large_backward)
544 sfence
545 /* Store the first 4 * VEC. */
546 VMOVU %VEC(4), (%rdi)
547 VMOVU %VEC(5), VEC_SIZE(%rdi)
548 VMOVU %VEC(6), (VEC_SIZE * 2)(%rdi)
549 VMOVU %VEC(7), (VEC_SIZE * 3)(%rdi)
550 /* Store the last VEC. */
551 VMOVU %VEC(8), (%r11)
552 VZEROUPPER
553 ret
554#endif
555END (MEMMOVE_SYMBOL (__memmove, unaligned_erms))
