23ENTRY(__ceilf)
24 .type __ceilf, @gnu_indirect_function
25 LOAD_RTLD_GLOBAL_RO_RDX
26 leaq __ceilf_sse41(%rip), %rax
27 HAS_CPU_FEATURE (SSE4_1)
28 jnz 2f
29 leaq __ceilf_c(%rip), %rax
302: ret
31END(__ceilf)
