21int
22fegetexceptflag (fexcept_t *flagp, int excepts)
23{
24 fexcept_t temp;
25 unsigned int mxscr;
26
27 /* Get the current exceptions for the x87 FPU and SSE unit. */
28 __asm__ ("fnstsw %0\n"
29 "stmxcsr %1" : "=m" (*&temp), "=m" (*&mxscr));
30
31 *flagp = (temp | mxscr) & FE_ALL_EXCEPT & excepts;
32
33 /* Success. */
34 return 0;
35}
