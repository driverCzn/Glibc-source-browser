20#ifndef _DL_I386_PROCINFO_H
21#define _DL_I386_PROCINFO_H 1
22#include <sysdeps/x86/dl-procinfo.h>
23
24#undef _dl_procinfo
25static inline int
26__attribute__ ((unused))
27_dl_procinfo (unsigned int type, unsigned long int word)
28{
29 /* This table should match the information from arch/i386/kernel/setup.c
30 in the kernel sources. */
31 int i;
32
33 /* Fallback to unknown output mechanism. */
34 if (type == AT_HWCAP2)
35 return -1;
36
37 _dl_printf ("AT_HWCAP: ");
38
39 for (i = 0; i < 32; ++i)
40 if (word & (1 << i))
41 _dl_printf (" %s", GLRO(dl_x86_cap_flags)[i]);
42
43 _dl_printf ("\n");
44
45 return 0;
46}
47#endif
