24/* Detach shared memory segment starting at address specified by SHMADDR
25 from the caller's data segment. */
26
27int
28shmdt (const void *shmaddr)
29{
30#ifdef __ASSUME_DIRECT_SYSVIPC_SYSCALLS
31 return INLINE_SYSCALL_CALL (shmdt, shmaddr);
32#else
33 return INLINE_SYSCALL_CALL (ipc, IPCOP_shmdt, 0, 0, 0, shmaddr);
34#endif
35}
