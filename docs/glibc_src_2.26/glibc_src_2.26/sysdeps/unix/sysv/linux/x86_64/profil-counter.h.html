19#include <signal.h>
20#include <sigcontextinfo.h>
21
22static void
23__profil_counter (int signo, SIGCONTEXT scp)
24{
25 profil_count ((void *) GET_PC (scp));
26
27 /* This is a hack to prevent the compiler from implementing the
28 above function call as a sibcall. The sibcall would overwrite
29 the signal context. */
30 asm volatile ("");
31}
