24/* Wait for a child to die. When one does, put its status in *STAT_LOC
25 and return its process ID. For errors, return (pid_t) -1. */
26pid_t
27__libc_wait (int *stat_loc)
28{
29 pid_t result = SYSCALL_CANCEL (wait4, WAIT_ANY, stat_loc, 0,
30 (struct rusage *) NULL);
31 return result;
32}
