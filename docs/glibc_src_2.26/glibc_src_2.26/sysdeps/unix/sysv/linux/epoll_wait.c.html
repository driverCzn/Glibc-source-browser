20#include <unistd.h>
21#include <sys/types.h>
22#include <sys/epoll.h>

24int
25epoll_wait (int epfd, struct epoll_event *events, int maxevents, int timeout)
26{
27#ifdef __NR_epoll_wait
28 return SYSCALL_CANCEL (epoll_wait, epfd, events, maxevents, timeout);
29#else
30 return epoll_pwait (epfd, events, maxevents, timeout, NULL);
31#endif
32}
