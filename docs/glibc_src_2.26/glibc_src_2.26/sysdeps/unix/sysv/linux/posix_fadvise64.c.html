22int __posix_fadvise64_l64 (int fd, off64_t offset, off64_t len, int advise);

24/* Both arm and powerpc implements fadvise64_64 with last 'advise' argument
25 just after 'fd' to avoid the requirement of implementing 7-arg syscalls.
26 ARM also defines __NR_fadvise64_64 as __NR_arm_fadvise64_64.
27
28 tile requires __ASSUME_ALIGNED_REGISTER_PAIRS but implements the 32-bit
29 fadvise64_64 without the padding 0 after fd.
30
31 s390 implements fadvice64_64 using a specific struct with arguments
32 packed inside. This is the only implementation handled in arch-specific
33 code. */
34
35#ifdef __ASSUME_FADVISE64_64_NO_ALIGN
36# undef __ALIGNMENT_ARG
37# define __ALIGNMENT_ARG
38#endif
39
40#ifndef __NR_fadvise64_64
41# define __NR_fadvise64_64 __NR_fadvise64
42#endif
43
44/* Advice the system about the expected behaviour of the application with
45 respect to the file associated with FD. */
46
47int
48__posix_fadvise64_l64 (int fd, off64_t offset, off64_t len, int advise)
49{
50 INTERNAL_SYSCALL_DECL (err);
51#ifdef __ASSUME_FADVISE64_64_6ARG
52 int ret = INTERNAL_SYSCALL_CALL (fadvise64_64, err, fd, advise,
53 SYSCALL_LL64 (offset), SYSCALL_LL64 (len));
54#else
55 int ret = INTERNAL_SYSCALL_CALL (fadvise64_64, err, fd,
56 __ALIGNMENT_ARG SYSCALL_LL64 (offset),
57 SYSCALL_LL64 (len), advise);
58#endif
59 if (!INTERNAL_SYSCALL_ERROR_P (ret, err))
60 return 0;
61 return INTERNAL_SYSCALL_ERRNO (ret, err);
62}
63
64/* The type of the len argument was changed from size_t to off_t in
65 POSIX.1-2003 TC1. */
66#ifndef __OFF_T_MATCHES_OFF64_T
67# if SHLIB_COMPAT(libc, GLIBC_2_2, GLIBC_2_3_3)
68int __posix_fadvise64_l32 (int fd, off64_t offset, size_t len, int advise);

70int
71attribute_compat_text_section
72__posix_fadvise64_l32 (int fd, off64_t offset, size_t len, int advise)
73{
74 return __posix_fadvise64_l64 (fd, offset, len, advise);
75}
