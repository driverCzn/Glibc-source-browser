18#include <errno.h>
19#include <unistd.h>
20#include <setxid.h>
21
22
23int
24seteuid (uid_t uid)
25{
26 int result;
27
28 if (uid == (uid_t) ~0)
29 return INLINE_SYSCALL_ERROR_RETURN_VALUE (EINVAL);
30
31#ifdef __NR_setresuid32
32 result = INLINE_SETXID_SYSCALL (setresuid32, 3, -1, uid, -1);
33#else
34 result = INLINE_SETXID_SYSCALL (setresuid, 3, -1, uid, -1);
35#endif
36
37 return result;
38}
39#ifndef seteuid
40libc_hidden_def (seteuid)
41#endif
