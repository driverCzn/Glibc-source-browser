19#include <errno.h>
20#include <fcntl.h>
21#include <pthreadP.h>
22#include <stdio.h>
23#include <string.h>
24#include <unistd.h>
25#include <sys/prctl.h>
26
27#include <not-cancel.h>
28
29
30int
31pthread_setname_np (pthread_t th, const char *name)
32{
33 const struct pthread *pd = (const struct pthread *) th;
34
35 /* Unfortunately the kernel headers do not export the TASK_COMM_LEN
36 macro. So we have to define it here. */
37#define TASK_COMM_LEN 16
38 size_t name_len = strlen (name);
39 if (name_len >= TASK_COMM_LEN)
40 return ERANGE;
41
42 if (pd == THREAD_SELF)
43 return prctl (PR_SET_NAME, name) ? errno : 0;
44
45#define FMT "/proc/self/task/%u/comm"
46 char fname[sizeof (FMT) + 8];
47 sprintf (fname, FMT, (unsigned int) pd->tid);
48
49 int fd = open_not_cancel_2 (fname, O_RDWR);
50 if (fd == -1)
51 return errno;
52
53 int res = 0;
54 ssize_t n = TEMP_FAILURE_RETRY (write_not_cancel (fd, name, name_len));
55 if (n < 0)
56 res = errno;
57 else if (n != name_len)
58 res = EIO;
59
60 close_not_cancel_no_status (fd);
61
62 return res;
63}
