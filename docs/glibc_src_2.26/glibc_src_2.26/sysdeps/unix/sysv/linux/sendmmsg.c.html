27int
28__sendmmsg (int fd, struct mmsghdr *vmessages, unsigned int vlen, int flags)
29{
30#ifdef __ASSUME_SENDMMSG_SYSCALL
31 return SYSCALL_CANCEL (sendmmsg, fd, vmessages, vlen, flags);
32#else
33 return SOCKETCALL_CANCEL (sendmmsg, fd, vmessages, vlen, flags);
34#endif
35}
