27static bool
28writev_for_fatal (int fd, const struct iovec *iov, size_t niov, size_t total)
29{
30 INTERNAL_SYSCALL_DECL (err);
31 ssize_t cnt;
32 do
33 cnt = INTERNAL_SYSCALL (writev, err, 3, fd, iov, niov);
34 while (INTERNAL_SYSCALL_ERROR_P (cnt, err)
35 && INTERNAL_SYSCALL_ERRNO (cnt, err) == EINTR);
36 return cnt == total;
37}
38#define WRITEV_FOR_FATAL writev_for_fatal

40static void
41backtrace_and_maps (int do_abort, bool written, int fd)
42{
43 if (do_abort > 1 && written)
44 {
45 void *addrs[64];
46#define naddrs (sizeof (addrs) / sizeof (addrs[0]))
47 int n = __backtrace (addrs, naddrs);
48 if (n > 2)
49 {
50#define strnsize(str) str, strlen (str)
51#define writestr(str) write_not_cancel (fd, str)
52 writestr (strnsize ("======= Backtrace: =========\n"));
53 __backtrace_symbols_fd (addrs + 1, n - 1, fd);
54
55 writestr (strnsize ("======= Memory map: ========\n"));
56 int fd2 = open_not_cancel_2 ("/proc/self/maps", O_RDONLY);
57 char buf[1024];
58 ssize_t n2;
59 while ((n2 = read_not_cancel (fd2, buf, sizeof (buf))) > 0)
60 if (write_not_cancel (fd, buf, n2) != n2)
61 break;
62 close_not_cancel_no_status (fd2);
63 }
64 }
65}
66#define BEFORE_ABORT backtrace_and_maps
