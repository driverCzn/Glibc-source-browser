23/* Create FILE with protections MODE. */
24int
25__creat64 (const char *file, mode_t mode)
26{
27#if defined __OFF_T_MATCHES_OFF64_T && defined __NR_creat
28 return SYSCALL_CANCEL (creat, file, mode);
29#else
30 /* We need to pass O_LARGEFILE. */
31 return __open64 (file, O_WRONLY | O_CREAT | O_TRUNC, mode);
32#endif
33}
34weak_alias (__creat64, creat64)
35
36#ifdef __OFF_T_MATCHES_OFF64_T
37strong_alias (__creat64, __creat)
38weak_alias (__creat64, creat)
39#endif

41LIBC_CANCEL_HANDLED ();
