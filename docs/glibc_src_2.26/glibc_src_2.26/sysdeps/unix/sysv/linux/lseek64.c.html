21#include <sys/types.h>
22#include <sysdep.h>

25off64_t
26__lseek64 (int fd, off64_t offset, int whence)
27{
28#ifdef __NR__llseek
29 loff_t res;
30 int rc = INLINE_SYSCALL_CALL (_llseek, fd,
31 (long) (((uint64_t) (offset)) >> 32),
32 (long) offset, &res, whence);
33 return rc ?: res;
34#else
35 return INLINE_SYSCALL_CALL (lseek, fd, offset, whence);
36#endif
37}

39#ifdef __OFF_T_MATCHES_OFF64_T
40weak_alias (__lseek64, lseek)
41weak_alias (__lseek64, __lseek)
42strong_alias (__lseek64, __libc_lseek)
43libc_hidden_def (__lseek)
44#endif

47weak_alias (__lseek64, lseek64)

49/* llseek doesn't have a prototype. Since the second parameter is a
50 64bit type, this results in wrong behaviour if no prototype is
51 provided. */
52weak_alias (__lseek64, llseek)
53link_warning (llseek, "\
54the `llseek' function may be dangerous; use `lseek64' instead.")
