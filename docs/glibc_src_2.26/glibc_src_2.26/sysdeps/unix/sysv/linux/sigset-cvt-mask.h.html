21static inline int __attribute__ ((unused))
22sigset_set_old_mask (sigset_t *set, int mask)
23{
24 unsigned long int *ptr;
25 int cnt;
26
27 ptr = &set->__val[0];
28
29 *ptr++ = (unsigned int) mask;
30
31 cnt = _SIGSET_NWORDS - 2;
32 do
33 *ptr++ = 0ul;
34 while (--cnt >= 0);
35
36 return 0;
37}
38
39static inline int __attribute__ ((unused))
40sigset_get_old_mask (const sigset_t *set)
41{
42 return (unsigned int) set->__val[0];
43}
