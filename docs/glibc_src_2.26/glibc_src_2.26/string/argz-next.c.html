21#include <string.h>

23char *
24__argz_next (const char *argz, size_t argz_len, const char *entry)
25{
26 if (entry)
27 {
28 if (entry < argz + argz_len)
29 entry = strchr (entry, '\0') + 1;
30
31 return entry >= argz + argz_len ? NULL : (char *) entry;
32 }
33 else
34 if (argz_len > 0)
35 return (char *) argz;
36 else
37 return NULL;
38}
39weak_alias (__argz_next, argz_next)
