18#include <libintl.h>
19#include <stdio.h>
20#include <string.h>
21#include <errno.h>
22
23/* Return a string describing the errno code in ERRNUM.
24 The storage is good only until the next call to strerror.
25 Writing to the storage causes undefined behavior. */
26libc_freeres_ptr (static char *buf);
27
28char *
29strerror (int errnum)
30{
31 char *ret = __strerror_r (errnum, NULL, 0);
32 int saved_errno;
33
34 if (__glibc_likely (ret != NULL))
35 return ret;
36 saved_errno = errno;
37 if (buf == NULL)
38 buf = malloc (1024);
39 __set_errno (saved_errno);
40 if (buf == NULL)
41 return _("Unknown error");
42 return __strerror_r (errnum, buf, 1024);
43}
