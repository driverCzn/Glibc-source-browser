17/*
18 * wrapper hypotl(x,y)
19 */
20
21#include <math.h>
22#include <math_private.h>
23
24
25long double
26__hypotl(long double x, long double y)
27{
28 long double z;
29 z = __ieee754_hypotl(x,y);
30 if(__builtin_expect(!isfinite(z), 0)
31 && isfinite(x) && isfinite(y) && _LIB_VERSION != _IEEE_)
32 return __kernel_standard_l(x, y, 204); /* hypot overflow */
33
34 return z;
35}
36weak_alias (__hypotl, hypotl)
