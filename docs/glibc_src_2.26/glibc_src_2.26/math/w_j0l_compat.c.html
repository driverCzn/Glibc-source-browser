24/* wrapper j0l */
25long double
26__j0l (long double x)
27{
28 if (__builtin_expect (isgreater (fabsl (x), X_TLOSS), 0)
29 && _LIB_VERSION != _IEEE_ && _LIB_VERSION != _POSIX_)
30 /* j0(|x|>X_TLOSS) */
31 return __kernel_standard_l (x, x, 234);
32
33 return __ieee754_j0l (x);
34}
35weak_alias (__j0l, j0l)

38/* wrapper y0l */
39long double
40__y0l (long double x)
41{
42 if (__builtin_expect (islessequal (x, 0.0L) || isgreater (x, X_TLOSS), 0)
43 && _LIB_VERSION != _IEEE_)
44 {
45 if (x < 0.0L)
46 {
47 /* d = zero/(x-x) */
48 feraiseexcept (FE_INVALID);
49 return __kernel_standard_l (x, x, 209);
50 }
51 else if (x == 0.0L)
52 {
53 /* d = -one/(x-x) */
54 feraiseexcept (FE_DIVBYZERO);
55 return __kernel_standard_l (x, x, 208);
56 }
57 else if (_LIB_VERSION != _POSIX_)
58 /* y0(x>X_TLOSS) */
59 return __kernel_standard_l (x, x, 235);
60 }
61
62 return __ieee754_y0l (x);
63}
64weak_alias (__y0l, y0l)
