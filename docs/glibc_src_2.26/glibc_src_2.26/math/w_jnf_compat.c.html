24/* wrapper jnf */
25float
26jnf (int n, float x)
27{
28 if (__builtin_expect (isgreater (fabsf (x), (float) X_TLOSS), 0)
29 && _LIB_VERSION != _IEEE_ && _LIB_VERSION != _POSIX_)
30 /* jn(n,|x|>X_TLOSS) */
31 return __kernel_standard_f (n, x, 138);
32
33 return __ieee754_jnf (n, x);
34}
35
36
37/* wrapper ynf */
38float
39ynf (int n, float x)
40{
41 if (__builtin_expect (islessequal (x, 0.0f)
42 || isgreater (x, (float) X_TLOSS), 0)
43 && _LIB_VERSION != _IEEE_)
44 {
45 if (x < 0.0f)
46 {
47 /* d = zero/(x-x) */
48 feraiseexcept (FE_INVALID);
49 return __kernel_standard_f (n, x, 113);
50 }
51 else if (x == 0.0)
52 {
53 /* d = -one/(x-x) */
54 feraiseexcept (FE_DIVBYZERO);
55 return __kernel_standard_f (n, x, 112);
56 }
57 else if (_LIB_VERSION != _POSIX_)
58 /* yn(n,x>X_TLOSS) */
59 return __kernel_standard_f (n, x, 139);
60 }
61
62 return __ieee754_ynf (n, x);
63}
