13/*
14 * wrapper double lgamma_r(double x, int *signgamp)
15 */
16
17#include <math.h>
18#include <math_private.h>
19
20
21double
22__lgamma_r(double x, int *signgamp)
23{
24 double y = __ieee754_lgamma_r(x,signgamp);
25 if(__builtin_expect(!isfinite(y), 0)
26 && isfinite(x) && _LIB_VERSION != _IEEE_)
27 return __kernel_standard(x, x,
28 __floor(x)==x&&x<=0.0
29 ? 15 /* lgamma pole */
30 : 14); /* lgamma overflow */
31
32 return y;
33}
34weak_alias (__lgamma_r, lgamma_r)
35#ifdef NO_LONG_DOUBLE
36strong_alias (__lgamma_r, __lgammal_r)
37weak_alias (__lgamma_r, lgammal_r)
38#endif
