19#include <math.h>
20#include <math_private.h>

22/* wrapper fmod */
23double
24__fmod (double x, double y)
25{
26 if (__builtin_expect (isinf (x) || y == 0.0, 0)
27 && _LIB_VERSION != _IEEE_ && !isnan (y) && !isnan (x))
28 /* fmod(+-Inf,y) or fmod(x,0) */
29 return __kernel_standard (x, y, 27);
30
31 return __ieee754_fmod (x, y);
32}
33weak_alias (__fmod, fmod)
34#ifdef NO_LONG_DOUBLE
35strong_alias (__fmod, __fmodl)
36weak_alias (__fmod, fmodl)
37#endif
