23/* wrapper sqrtf */
24float
25__sqrtf (float x)
26{
27 if (__builtin_expect (isless (x, 0.0f), 0) && _LIB_VERSION != _IEEE_)
28 return __kernel_standard_f (x, x, 126); /* sqrt(negative) */
29
30 return __ieee754_sqrtf (x);
31}
32weak_alias (__sqrtf, sqrtf)
