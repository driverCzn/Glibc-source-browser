20/*
21 * wrapper exp10l(x)
22 */
23
24#include <math.h>
25#include <math_private.h>
26
27long double
28__exp10l (long double x)
29{
30 long double z = __ieee754_exp10l (x);
31 if (__builtin_expect (!isfinite (z) || z == 0, 0)
32 && isfinite (x) && _LIB_VERSION != _IEEE_)
33 /* exp10l overflow (246) if x > 0, underflow (247) if x < 0. */
34 return __kernel_standard_l (x, x, 246 + !!signbit (x));
35
36 return z;
37}
38weak_alias (__exp10l, exp10l)
39strong_alias (__exp10l, __pow10l)
40weak_alias (__pow10l, pow10l)
