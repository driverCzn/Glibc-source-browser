19#include <stdlib.h>

21/* Conversion table. */
22static const char conv_table[64] =
23{
24 '.', '/', '0', '1', '2', '3', '4', '5',
25 '6', '7', '8', '9', 'A', 'B', 'C', 'D',
26 'E', 'F', 'G', 'H', 'I', 'J', 'K', 'L',
27 'M', 'N', 'O', 'P', 'Q', 'R', 'S', 'T',
28 'U', 'V', 'W', 'X', 'Y', 'Z', 'a', 'b',
29 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j',
30 'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r',
31 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'
32};
33
34char *
35l64a (long int n)
36{
37 unsigned long int m = (unsigned long int) n;
38 static char result[7];
39 int cnt;
40
41 /* The standard says that only 32 bits are used. */
42 m &= 0xffffffff;
43
44 if (m == 0ul)
45 /* The value for N == 0 is defined to be the empty string. */
46 return (char *) "";
47
48 for (cnt = 0; m > 0ul; ++cnt)
49 {
50 result[cnt] = conv_table[m & 0x3f];
51 m >>= 6;
52 }
53 result[cnt] = '\0';
54
55 return result;
56}
