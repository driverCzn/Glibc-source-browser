116_IO_FILE *
117attribute_compat_text_section
118_IO_old_proc_open (_IO_FILE *fp, const char *command, const char *mode)
119{
120 volatile int read_or_write;
121 volatile int parent_end, child_end;
122 int pipe_fds[2];
123 _IO_pid_t child_pid;
124 if (_IO_file_is_open (fp))
125 return NULL;
126 if (_IO_pipe (pipe_fds) < 0)
127 return NULL;
128 if (mode[0] == 'r' && mode[1] == '\0')
129 {
130 parent_end = pipe_fds[0];
131 child_end = pipe_fds[1];
132 read_or_write = _IO_NO_WRITES;
133 }
134 else if (mode[0] == 'w' && mode[1] == '\0')
135 {
136 parent_end = pipe_fds[1];
137 child_end = pipe_fds[0];
138 read_or_write = _IO_NO_READS;
139 }
140 else
141 {
142 _IO_close (pipe_fds[0]);
143 _IO_close (pipe_fds[1]);
144 __set_errno (EINVAL);
145 return NULL;
146 }
147 ((_IO_proc_file *) fp)->pid = child_pid = _IO_fork ();
148 if (child_pid == 0)
149 {
150 int child_std_end = mode[0] == 'r' ? 1 : 0;
151 struct _IO_proc_file *p;
152
153 _IO_close (parent_end);
154 if (child_end != child_std_end)
155 {
156 _IO_dup2 (child_end, child_std_end);
157 _IO_close (child_end);
158 }
159 /* POSIX.2: "popen() shall ensure that any streams from previous
160 popen() calls that remain open in the parent process are closed
161 in the new child process." */
162 for (p = old_proc_file_chain; p; p = p->next)
163 _IO_close (_IO_fileno ((_IO_FILE *) p));
164
165 _IO_execl ("/bin/sh", "sh", "-c", command, (char *) 0);
166 _IO__exit (127);
167 }
168 _IO_close (child_end);
169 if (child_pid < 0)
170 {
171 _IO_close (parent_end);
172 return NULL;
173 }
174 _IO_fileno (fp) = parent_end;
175
176 /* Link into old_proc_file_chain. */
177#ifdef _IO_MTSAFE_IO
178 _IO_cleanup_region_start_noarg (unlock);
179 _IO_lock_lock (proc_file_chain_lock);
180#endif
181 ((_IO_proc_file *) fp)->next = old_proc_file_chain;
182 old_proc_file_chain = (_IO_proc_file *) fp;
183#ifdef _IO_MTSAFE_IO
184 _IO_lock_unlock (proc_file_chain_lock);
185 _IO_cleanup_region_end (0);
186#endif
187
188 _IO_mask_flags (fp, read_or_write, _IO_NO_READS|_IO_NO_WRITES);
189 return fp;
190}

192_IO_FILE *
193attribute_compat_text_section
194_IO_old_popen (const char *command, const char *mode)
195{
196 struct locked_FILE
197 {
198 struct _IO_proc_file fpx;
199#ifdef _IO_MTSAFE_IO
200 _IO_lock_t lock;
201#endif
202 } *new_f;
203 _IO_FILE *fp;
204
205 new_f = (struct locked_FILE *) malloc (sizeof (struct locked_FILE));
206 if (new_f == NULL)
207 return NULL;
208#ifdef _IO_MTSAFE_IO
209 new_f->fpx.file.file._file._lock = &new_f->lock;
210#endif
211 fp = &new_f->fpx.file.file._file;
212 _IO_old_init (fp, 0);
213 _IO_JUMPS_FILE_plus (&new_f->fpx.file) = &_IO_old_proc_jumps;
214 _IO_old_file_init_internal ((struct _IO_FILE_plus *) &new_f->fpx.file);
215#if !_IO_UNIFIED_JUMPTABLES
216 new_f->fpx.file.vtable = NULL;
217#endif
218 if (_IO_old_proc_open (fp, command, mode) != NULL)
219 return fp;
220 _IO_un_link ((struct _IO_FILE_plus *) &new_f->fpx.file);
221 free (new_f);
222 return NULL;
223}

225int
226attribute_compat_text_section
227_IO_old_proc_close (_IO_FILE *fp)
228{
229 /* This is not name-space clean. FIXME! */
230 int wstatus;
231 _IO_proc_file **ptr = &old_proc_file_chain;
232 _IO_pid_t wait_pid;
233 int status = -1;
234
235 /* Unlink from old_proc_file_chain. */
236#ifdef _IO_MTSAFE_IO
237 _IO_cleanup_region_start_noarg (unlock);
238 _IO_lock_lock (proc_file_chain_lock);
239#endif
240 for ( ; *ptr != NULL; ptr = &(*ptr)->next)
241 {
242 if (*ptr == (_IO_proc_file *) fp)
243 {
244 *ptr = (*ptr)->next;
245 status = 0;
246 break;
247 }
248 }
249#ifdef _IO_MTSAFE_IO
250 _IO_lock_unlock (proc_file_chain_lock);
251 _IO_cleanup_region_end (0);
252#endif
253
254 if (status < 0 || _IO_close (_IO_fileno(fp)) < 0)
255 return -1;
256 /* POSIX.2 Rationale: "Some historical implementations either block
257 or ignore the signals SIGINT, SIGQUIT, and SIGHUP while waiting
258 for the child process to terminate. Since this behavior is not
259 described in POSIX.2, such implementations are not conforming." */
260 do
261 {
262 wait_pid = _IO_waitpid (((_IO_proc_file *) fp)->pid, &wstatus, 0);
263 }
264 while (wait_pid == -1 && errno == EINTR);
265 if (wait_pid == -1)
266 return -1;
267 return wstatus;
268}
