90/* Close all open sockets. If FREE_ADDR is true, deallocate any
91 separately allocated name server addresses. */
92void
93__res_iclose (res_state statp, bool free_addr)
94{
95 if (statp->_vcsock >= 0)
96 {
97 close_not_cancel_no_status (statp->_vcsock);
98 statp->_vcsock = -1;
99 statp->_flags &= ~(RES_F_VC | RES_F_CONN);
100 }
101 for (int ns = 0; ns < statp->nscount; ns++)
102 if (statp->_u._ext.nsaddrs[ns] != NULL)
103 {
104 if (statp->_u._ext.nssocks[ns] != -1)
105 {
106 close_not_cancel_no_status (statp->_u._ext.nssocks[ns]);
107 statp->_u._ext.nssocks[ns] = -1;
108 }
109 if (free_addr)
110 {
111 free (statp->_u._ext.nsaddrs[ns]);
112 statp->_u._ext.nsaddrs[ns] = NULL;
113 }
114 }
115 if (free_addr)
116 __resolv_conf_detach (statp);
117}
118libc_hidden_def (__res_iclose)
119
120void
121res_nclose (res_state statp)
122{
123 __res_iclose (statp, true);
124}
125libc_hidden_def (__res_nclose)

127/* This is called when a thread is exiting to free resources held in _res. */
128static void __attribute__ ((section ("__libc_thread_freeres_fn")))
129res_thread_freeres (void)
130{
131 __resolv_context_freeres ();
132
133 if (_res.nscount == 0)
134 /* Never called res_ninit. */
135 return;
136
137 __res_iclose (&_res, true); /* Close any VC sockets. */
138
139 /* Make sure we do a full re-initialization the next time. */
140 _res.options = 0;
141}
142text_set_element (__libc_thread_subfreeres, res_thread_freeres);
143text_set_element (__libc_subfreeres, res_thread_freeres);
