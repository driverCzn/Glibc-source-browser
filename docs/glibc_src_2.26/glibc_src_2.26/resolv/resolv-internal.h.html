25/* Resolver flags. Used for _flags in struct __res_state. */

27#define RES_F_CONN 0x00000002 /* Socket is connected. */
28#define RES_F_EDNS0ERR 0x00000004 /* EDNS0 caused errors. */

31/* Internal version of RES_USE_INET6 which does not trigger a
32 deprecation warning. */
33#define DEPRECATED_RES_USE_INET6 0x00002000
34
35static inline bool
36res_use_inet6 (void)
37{
38 return _res.options & DEPRECATED_RES_USE_INET6;
39}

41enum
42 {
43 /* The advertized EDNS buffer size. The value 1200 is derived
44 from the IPv6 minimum MTU (1280 bytes) minus some arbitrary
45 space for tunneling overhead. If the DNS server does not react
46 to ICMP Fragmentation Needed But DF Set messages, this should
47 avoid all UDP fragments on current networks. Avoiding UDP
48 fragments is desirable because it prevents fragmentation-based
49 spoofing attacks because the randomness in a DNS packet is
50 concentrated in the first fragment (with the headers) and does
51 not protect subsequent fragments. */
52 RESOLV_EDNS_BUFFER_SIZE = 1200,
53 };

91/* Convert from presentation format (which usually means ASCII
92 printable) to network format (which is usually some kind of binary
93 format). The input is in the range [SRC, SRC + SRCLEN). The
94 output is written to DST (which has to be 4 or 16 bytes long,
95 depending on AF). Return 0 for invalid input, 1 for success, -1
96 for an invalid address family. */
97int __inet_pton_length (int af, const char *src, size_t srclen, void *);
98libc_hidden_proto (__inet_pton_length)
