52#include <string.h>
53#include <arpa/nameser.h>
54
55static void
56map_v4v6_address (const char *src, char *dst)
57{
58 u_char *p = (u_char *) dst;
59 int i;
60
61 /* Move the IPv4 part to the right position. */
62 memcpy (dst + 12, src, INADDRSZ);
63
64 /* Mark this ipv6 addr as a mapped ipv4. */
65 for (i = 0; i < 10; i++)
66 *p++ = 0x00;
67 *p++ = 0xff;
68 *p = 0xff;
69}
