19#include <wchar.h>
20
21#ifndef WCSNCMP
22# define WCSNCMP wcsncmp
23#endif

25/* Compare no more than N characters of S1 and S2,
26 returning less than, equal to or greater than zero
27 if S1 is lexicographically less than, equal to or
28 greater than S2. */
29int
30WCSNCMP (const wchar_t *s1, const wchar_t *s2, size_t n)
31{
32 wchar_t c1 = L'\0';
33 wchar_t c2 = L'\0';
34
35 if (n >= 4)
36 {
37 size_t n4 = n >> 2;
38 do
39 {
40 c1 = *s1++;
41 c2 = *s2++;
42 if (c1 == L'\0' || c1 != c2)
43 return c1 > c2 ? 1 : (c1 < c2 ? -1 : 0);
44 c1 = *s1++;
45 c2 = *s2++;
46 if (c1 == L'\0' || c1 != c2)
47 return c1 > c2 ? 1 : (c1 < c2 ? -1 : 0);
48 c1 = *s1++;
49 c2 = *s2++;
50 if (c1 == L'\0' || c1 != c2)
51 return c1 > c2 ? 1 : (c1 < c2 ? -1 : 0);
52 c1 = *s1++;
53 c2 = *s2++;
54 if (c1 == L'\0' || c1 != c2)
55 return c1 > c2 ? 1 : (c1 < c2 ? -1 : 0);
56 } while (--n4 > 0);
57 n &= 3;
58 }
59
60 while (n > 0)
61 {
62 c1 = *s1++;
63 c2 = *s2++;
64 if (c1 == L'\0' || c1 != c2)
65 return c1 > c2 ? 1 : (c1 < c2 ? -1 : 0);
66 n--;
67 }
68
69 return 0;
70}
