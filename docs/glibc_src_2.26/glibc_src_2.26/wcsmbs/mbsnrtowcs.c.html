34/* This is the private state used if PS is NULL. */
35static mbstate_t state;
36
37/* This is a non-standard function but it is very useful in the
38 implementation of stdio because we have to deal with unterminated
39 buffers. At most NMC bytes will be converted. */
40size_t
41__mbsnrtowcs (wchar_t *dst, const char **src, size_t nmc, size_t len,
42 mbstate_t *ps)
43{
44 const unsigned char *srcend;
45 struct __gconv_step_data data;
46 size_t result;
47 int status;
48 struct __gconv_step *towc;
49 size_t dummy;
50 const struct gconv_fcts *fcts;
51
52 /* Tell where we want the result. */
53 data.__invocation_counter = 0;
54 data.__internal_use = 1;
55 data.__flags = __GCONV_IS_LAST;
56 data.__statep = ps ?: &state;
57
58 if (nmc == 0)
59 return 0;
60 srcend = (const unsigned char *) *src + __strnlen (*src, nmc - 1) + 1;
61
62 /* Get the conversion functions. */
63 fcts = get_gconv_fcts (_NL_CURRENT_DATA (LC_CTYPE));
64
65 /* Get the structure with the function pointers. */
66 towc = fcts->towc;
67 __gconv_fct fct = towc->__fct;
68#ifdef PTR_DEMANGLE
69 if (towc->__shlib_handle != NULL)
70 PTR_DEMANGLE (fct);
71#endif
72
73 /* We have to handle DST == NULL special. */
74 if (dst == NULL)
75 {
76 mbstate_t temp_state;
77 wchar_t buf[64]; /* Just an arbitrary size. */
78 const unsigned char *inbuf = (const unsigned char *) *src;
79
80 temp_state = *data.__statep;
81 data.__statep = &temp_state;
82
83 result = 0;
84 data.__outbufend = (unsigned char *) buf + sizeof (buf);
85 do
86 {
87 data.__outbuf = (unsigned char *) buf;
88
89 status = DL_CALL_FCT (fct, (towc, &data, &inbuf, srcend, NULL,
90 &dummy, 0, 1));
91
92 result += (wchar_t *) data.__outbuf - buf;
93 }
94 while (status == __GCONV_FULL_OUTPUT);
95
96 if ((status == __GCONV_OK || status == __GCONV_EMPTY_INPUT)
97 && ((wchar_t *) data.__outbuf)[-1] == L'\0')
98 /* Don't count the NUL character in. */
99 --result;
100 }
101 else
102 {
103 /* This code is based on the safe assumption that all internal
104 multi-byte encodings use the NUL byte only to mark the end
105 of the string. */
106 data.__outbuf = (unsigned char *) dst;
107 data.__outbufend = data.__outbuf + len * sizeof (wchar_t);
108
109 status = DL_CALL_FCT (fct,
110 (towc, &data, (const unsigned char **) src, srcend,
111 NULL, &dummy, 0, 1));
112
113 result = (wchar_t *) data.__outbuf - dst;
114
115 /* We have to determine whether the last character converted
116 is the NUL character. */
117 if ((status == __GCONV_OK || status == __GCONV_EMPTY_INPUT)
118 && (assert (result > 0),
119 ((wchar_t *) dst)[result - 1] == L'\0'))
120 {
121 assert (__mbsinit (data.__statep));
122 *src = NULL;
123 --result;
124 }
125 }
126
127 /* There must not be any problems with the conversion but illegal input
128 characters. */
129 assert (status == __GCONV_OK || status == __GCONV_EMPTY_INPUT
130 || status == __GCONV_ILLEGAL_INPUT
131 || status == __GCONV_INCOMPLETE_INPUT
132 || status == __GCONV_FULL_OUTPUT);
133
134 if (status != __GCONV_OK && status != __GCONV_FULL_OUTPUT
135 && status != __GCONV_EMPTY_INPUT && status != __GCONV_INCOMPLETE_INPUT)
136 {
137 result = (size_t) -1;
138 __set_errno (EILSEQ);
139 }
140
141 return result;
142}
