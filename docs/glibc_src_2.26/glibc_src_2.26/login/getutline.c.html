24libc_freeres_ptr (static struct utmp *buffer);

27struct utmp *
28__getutline (const struct utmp *line)
29{
30 struct utmp *result;
31
32 if (buffer == NULL)
33 {
34 buffer = (struct utmp *) malloc (sizeof (struct utmp));
35 if (buffer == NULL)
36 return NULL;
37 }
38 if (__getutline_r (line, buffer, &result) < 0)
39 return NULL;
40
41 return result;
42}
43weak_alias (__getutline, getutline)
