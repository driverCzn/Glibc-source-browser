19#include <string.h>
20#include <sys/time.h>
21#include <time.h>
22#include <unistd.h>
23#include <utmp.h>
24
25
26void
27logwtmp (const char *line, const char *name, const char *host)
28{
29 struct utmp ut;
30
31 /* Set information in new entry. */
32 memset (&ut, 0, sizeof (ut));
33#if _HAVE_UT_PID - 0
34 ut.ut_pid = getpid ();
35#endif
36#if _HAVE_UT_TYPE - 0
37 ut.ut_type = name[0] ? USER_PROCESS : DEAD_PROCESS;
38#endif
39 strncpy (ut.ut_line, line, sizeof ut.ut_line);
40 strncpy (ut.ut_name, name, sizeof ut.ut_name);
41#if _HAVE_UT_HOST - 0
42 strncpy (ut.ut_host, host, sizeof ut.ut_host);
43#endif
44
45#if _HAVE_UT_TV - 0
46 struct timeval tv;
47 __gettimeofday (&tv, NULL);
48 ut.ut_tv.tv_sec = tv.tv_sec;
49 ut.ut_tv.tv_usec = tv.tv_usec;
50#else
51 ut.ut_time = time (NULL);
52#endif
53
54 updwtmp (_PATH_WTMP, &ut);
55}
