19#include <string.h>
20#include <libc-internal.h>
21
22char *__progname_full = (char *) "";
23char *__progname = (char *) "";
24weak_alias (__progname_full, program_invocation_name)
25weak_alias (__progname, program_invocation_short_name)
26
27
28void
29__init_misc (int argc, char **argv, char **envp)
30{
31 if (argv && argv[0])
32 {
33 char *p = strrchr (argv[0], '/');
34 if (p == NULL)
35 __progname = argv[0];
36 else
37 __progname = p + 1;
38 __progname_full = argv[0];
39 }
40}
