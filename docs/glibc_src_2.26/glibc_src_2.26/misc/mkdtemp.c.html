21/* Generate a unique temporary directory from TEMPLATE.
22 The last six characters of TEMPLATE must be "XXXXXX";
23 they are replaced with a string that makes the filename unique.
24 The directory is created, mode 700, and its name is returned.
25 (This function comes from OpenBSD.) */
26char *
27mkdtemp (char *template)
28{
29 if (__gen_tempname (template, 0, 0, __GT_DIR))
30 return NULL;
31 else
32 return template;
33}
