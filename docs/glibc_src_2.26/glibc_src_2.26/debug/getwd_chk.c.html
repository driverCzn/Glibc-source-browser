18#include <errno.h>
19#include <unistd.h>
20#include <sys/param.h>
21
22
23char *
24__getwd_chk (char *buf, size_t buflen)
25{
26 char *res = __getcwd (buf, buflen);
27 if (res == NULL && errno == ERANGE)
28 __chk_fail ();
29 return res;
30}
31
32link_warning (getwd,
33 "the `getwd' function is dangerous and should not be used.")
