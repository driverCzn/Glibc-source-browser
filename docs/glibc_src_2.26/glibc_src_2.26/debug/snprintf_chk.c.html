23/* Write formatted output into S, according to the format
24 string FORMAT, writing no more than MAXLEN characters. */
25/* VARARGS5 */
26int
27___snprintf_chk (char *s, size_t maxlen, int flags, size_t slen,
28 const char *format, ...)
29{
30 va_list arg;
31 int done;
32
33 va_start (arg, format);
34 done = __vsnprintf_chk (s, maxlen, flags, slen, format, arg);
35 va_end (arg);
36
37 return done;
38}
39ldbl_strong_alias (___snprintf_chk, __snprintf_chk)
