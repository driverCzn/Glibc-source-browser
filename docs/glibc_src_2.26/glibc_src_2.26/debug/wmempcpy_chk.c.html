19#include <wchar.h>
20#include <string.h>

22
23wchar_t *
24__wmempcpy_chk (wchar_t *s1, const wchar_t *s2, size_t n, size_t ns1)
25{
26 if (__glibc_unlikely (ns1 < n))
27 __chk_fail ();
28 return (wchar_t *) __mempcpy ((char *) s1, (char *) s2,
29 n * sizeof (wchar_t));
30}
