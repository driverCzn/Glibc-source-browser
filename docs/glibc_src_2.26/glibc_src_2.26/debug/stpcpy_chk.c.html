25/* Copy SRC to DEST, returning the address of the terminating '\0' in DEST. */
26char *
27__stpcpy_chk (char *dest, const char *src, size_t destlen)
28{
29 size_t len = strlen (src);
30 if (len >= destlen)
31 __chk_fail ();
32
33 return memcpy (dest, src, len + 1) + len;
34}
