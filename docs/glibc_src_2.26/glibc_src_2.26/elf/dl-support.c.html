73/* A dummy link map for the executable, used by dlopen to access the global
74 scope. We don't export any symbols ourselves, so this can be minimal. */
75static struct link_map _dl_main_map =
76 {
77 .l_name = (char *) "",
78 .l_real = &_dl_main_map,
79 .l_ns = LM_ID_BASE,
80 .l_libname = &(struct libname_list) { .name = "", .dont_free = 1 },
81 .l_searchlist =
82 {
83 .r_list = &(struct link_map *) { &_dl_main_map },
84 .r_nlist = 1,
85 },
86 .l_symbolic_searchlist = { .r_list = &(struct link_map *) { NULL } },
87 .l_type = lt_executable,
88 .l_scope_mem = { &_dl_main_map.l_searchlist },
89 .l_scope_max = (sizeof (_dl_main_map.l_scope_mem)
90 / sizeof (_dl_main_map.l_scope_mem[0])),
91 .l_scope = _dl_main_map.l_scope_mem,
92 .l_local_scope = { &_dl_main_map.l_searchlist },
93 .l_used = 1,
94 .l_tls_offset = NO_TLS_OFFSET,
95 .l_serial = 1,
96 };

207/* During the program run we must not modify the global data of
208 loaded shared object simultanously in two threads. Therefore we
209 protect `_dl_open' and `_dl_close' in dl-close.c.
210
211 This must be a recursive lock since the initializer function of
212 the loaded object might as well require a call to this function.
213 At this time it is not anymore a problem to modify the tables. */
214__rtld_lock_define_initialized_recursive (, _dl_load_lock)
215/* This lock is used to keep __dl_iterate_phdr from inspecting the
216 list of loaded objects while an object is added to or removed from
217 that list. */
218__rtld_lock_define_initialized_recursive (, _dl_load_write_lock)

224void
225internal_function
226_dl_aux_init (ElfW(auxv_t) *av)
227{
228 int seen = 0;
229 uid_t uid = 0;
230 gid_t gid = 0;
231
232 _dl_auxv = av;
233 for (; av->a_type != AT_NULL; ++av)
234 switch (av->a_type)
235 {
236 case AT_PAGESZ:
237 if (av->a_un.a_val != 0)
238 GLRO(dl_pagesize) = av->a_un.a_val;
239 break;
240 case AT_CLKTCK:
241 GLRO(dl_clktck) = av->a_un.a_val;
242 break;
243 case AT_PHDR:
244 GL(dl_phdr) = (const void *) av->a_un.a_val;
245 break;
246 case AT_PHNUM:
247 GL(dl_phnum) = av->a_un.a_val;
248 break;
249 case AT_PLATFORM:
250 GLRO(dl_platform) = (void *) av->a_un.a_val;
251 break;
252 case AT_HWCAP:
253 GLRO(dl_hwcap) = (unsigned long int) av->a_un.a_val;
254 break;
255 case AT_HWCAP2:
256 GLRO(dl_hwcap2) = (unsigned long int) av->a_un.a_val;
257 break;
258 case AT_FPUCW:
259 GLRO(dl_fpu_control) = av->a_un.a_val;
260 break;
261#ifdef NEED_DL_SYSINFO
262 case AT_SYSINFO:
263 GL(dl_sysinfo) = av->a_un.a_val;
264 break;
265#endif
266#ifdef NEED_DL_SYSINFO_DSO
267 case AT_SYSINFO_EHDR:
268 GL(dl_sysinfo_dso) = (void *) av->a_un.a_val;
269 break;
270#endif
271 case AT_UID:
272 uid ^= av->a_un.a_val;
273 seen |= 1;
274 break;
275 case AT_EUID:
276 uid ^= av->a_un.a_val;
277 seen |= 2;
278 break;
279 case AT_GID:
280 gid ^= av->a_un.a_val;
281 seen |= 4;
282 break;
283 case AT_EGID:
284 gid ^= av->a_un.a_val;
285 seen |= 8;
286 break;
287 case AT_SECURE:
288 seen = -1;
289 __libc_enable_secure = av->a_un.a_val;
290 __libc_enable_secure_decided = 1;
291 break;
292 case AT_RANDOM:
293 _dl_random = (void *) av->a_un.a_val;
294 break;
295# ifdef DL_PLATFORM_AUXV
296 DL_PLATFORM_AUXV
297# endif
298 }
299 if (seen == 0xf)
300 {
301 __libc_enable_secure = uid != 0 || gid != 0;
302 __libc_enable_secure_decided = 1;
303 }
304}
305#endif

308void
309internal_function
310_dl_non_dynamic_init (void)
311{
312 _dl_main_map.l_origin = _dl_get_origin ();
313 _dl_main_map.l_phdr = GL(dl_phdr);
314 _dl_main_map.l_phnum = GL(dl_phnum);
315
316 if (HP_SMALL_TIMING_AVAIL)
317 HP_TIMING_NOW (_dl_cpuclock_offset);
318
319 _dl_verbose = *(getenv ("LD_WARN") ?: "") == '\0' ? 0 : 1;
320
321 /* Set up the data structures for the system-supplied DSO early,
322 so they can influence _dl_init_paths. */
323 setup_vdso (NULL, NULL);
324
325 /* Initialize the data structures for the search paths for shared
326 objects. */
327 _dl_init_paths (getenv ("LD_LIBRARY_PATH"));
328
329 /* Remember the last search directory added at startup. */
330 _dl_init_all_dirs = GL(dl_all_dirs);
331
332 _dl_lazy = *(getenv ("LD_BIND_NOW") ?: "") == '\0';
333
334 _dl_bind_not = *(getenv ("LD_BIND_NOT") ?: "") != '\0';
335
336 _dl_dynamic_weak = *(getenv ("LD_DYNAMIC_WEAK") ?: "") == '\0';
337
338 _dl_profile_output = getenv ("LD_PROFILE_OUTPUT");
339 if (_dl_profile_output == NULL || _dl_profile_output[0] == '\0')
340 _dl_profile_output
341 = &"/var/tmp\0/var/profile"[__libc_enable_secure ? 9 : 0];
342
343 if (__libc_enable_secure)
344 {
345 static const char unsecure_envvars[] =
346 UNSECURE_ENVVARS
347#ifdef EXTRA_UNSECURE_ENVVARS
348 EXTRA_UNSECURE_ENVVARS
349#endif
350 ;
351 const char *cp = unsecure_envvars;
352
353 while (cp < unsecure_envvars + sizeof (unsecure_envvars))
354 {
355 __unsetenv (cp);
356 cp = (const char *) __rawmemchr (cp, '\0') + 1;
357 }
358
359#if !HAVE_TUNABLES
360 if (__access ("/etc/suid-debug", F_OK) != 0)
361 __unsetenv ("MALLOC_CHECK_");
362#endif
363 }
364
365#ifdef DL_PLATFORM_INIT
366 DL_PLATFORM_INIT;
367#endif
368
369#ifdef DL_OSVERSION_INIT
370 DL_OSVERSION_INIT;
371#endif
372
373 /* Now determine the length of the platform string. */
374 if (_dl_platform != NULL)
375 _dl_platformlen = strlen (_dl_platform);
376
377 /* Scan for a program header telling us the stack is nonexecutable. */
378 if (_dl_phdr != NULL)
379 for (uint_fast16_t i = 0; i < _dl_phnum; ++i)
380 if (_dl_phdr[i].p_type == PT_GNU_STACK)
381 {
382 _dl_stack_flags = _dl_phdr[i].p_flags;
383 break;
384 }
385}
