59/*
60 * Add odd parity to low bit of 8 byte key
61 */
62void
63des_setparity (char *p)
64{
65 int i;
66
67 for (i = 0; i < 8; i++)
68 {
69 *p = partab[*p & 0x7f];
70 p++;
71 }
72}
