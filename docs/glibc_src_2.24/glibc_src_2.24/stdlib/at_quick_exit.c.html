35#include <stdlib.h>
36#include "exit.h"
37
38
39/* This is defined by newer gcc version unique for each module. */
40extern void *__dso_handle __attribute__ ((__weak__));
41
42
43/* Register FUNC to be executed by `quick_exit'. */
44int
45attribute_hidden
46at_quick_exit (void (*func) (void))
47{
48 return __cxa_at_quick_exit ((void (*) (void *)) func,
49 &__dso_handle == NULL ? NULL : __dso_handle);
50}
