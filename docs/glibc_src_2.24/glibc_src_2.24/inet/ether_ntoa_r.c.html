24char *
25ether_ntoa_r (const struct ether_addr *addr, char *buf)
26{
27 sprintf (buf, "%x:%x:%x:%x:%x:%x",
28 addr->ether_addr_octet[0], addr->ether_addr_octet[1],
29 addr->ether_addr_octet[2], addr->ether_addr_octet[3],
30 addr->ether_addr_octet[4], addr->ether_addr_octet[5]);
31 return buf;
32}
