19#include <ctype.h>
20#include <stdlib.h>
21#include <netinet/ether.h>
22#include <netinet/if_ether.h>

25struct ether_addr *
26ether_aton_r (const char *asc, struct ether_addr *addr)
27{
28 size_t cnt;
29
30 for (cnt = 0; cnt < 6; ++cnt)
31 {
32 unsigned int number;
33 char ch;
34
35 ch = _tolower (*asc++);
36 if ((ch < '0' || ch > '9') && (ch < 'a' || ch > 'f'))
37 return NULL;
38 number = isdigit (ch) ? (ch - '0') : (ch - 'a' + 10);
39
40 ch = _tolower (*asc);
41 if ((cnt < 5 && ch != ':') || (cnt == 5 && ch != '\0' && !isspace (ch)))
42 {
43 ++asc;
44 if ((ch < '0' || ch > '9') && (ch < 'a' || ch > 'f'))
45 return NULL;
46 number <<= 4;
47 number += isdigit (ch) ? (ch - '0') : (ch - 'a' + 10);
48
49 ch = *asc;
50 if (cnt < 5 && ch != ':')
51 return NULL;
52 }
53
54 /* Store result. */
55 addr->ether_addr_octet[cnt] = (unsigned char) number;
56
57 /* Skip ':'. */
58 ++asc;
59 }
60
61 return addr;
62}
63libc_hidden_def (ether_aton_r)
