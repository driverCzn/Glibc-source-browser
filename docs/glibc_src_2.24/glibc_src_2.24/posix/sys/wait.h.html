58/* The following values are used by the `waitid' function. */
59#if defined __USE_XOPEN || defined __USE_XOPEN2K8
60typedef enum
61{
62 P_ALL, /* Wait for any child. */
63 P_PID, /* Wait for specified process. */
64 P_PGID /* Wait for members of process group. */
65} idtype_t;
66#endif

68
69/* Wait for a child to die. When one does, put its status in *STAT_LOC
70 and return its process ID. For errors, return (pid_t) -1.
71
72 This function is a cancellation point and therefore not marked with
73 __THROW. */
74extern __pid_t wait (int *__stat_loc);
75
76#ifdef __USE_MISC
77/* Special values for the PID argument to `waitpid' and `wait4'. */
78# define WAIT_ANY (-1) /* Any process. */
79# define WAIT_MYPGRP 0 /* Any process in my process group. */
80#endif
81
82/* Wait for a child matching PID to die.
83 If PID is greater than 0, match any process whose process ID is PID.
84 If PID is (pid_t) -1, match any process.
85 If PID is (pid_t) 0, match any process with the
86 same process group as the current process.
87 If PID is less than -1, match any process whose
88 process group is the absolute value of PID.
89 If the WNOHANG bit is set in OPTIONS, and that child
90 is not already dead, return (pid_t) 0. If successful,
91 return PID and store the dead child's status in STAT_LOC.
92 Return (pid_t) -1 for errors. If the WUNTRACED bit is
93 set in OPTIONS, return status for stopped children; otherwise don't.
94
95 This function is a cancellation point and therefore not marked with
96 __THROW. */
97extern __pid_t waitpid (__pid_t __pid, int *__stat_loc, int __options);
98
99#if defined __USE_XOPEN || defined __USE_XOPEN2K8

103# define __id_t_defined
104# endif

106# define __need_siginfo_t
107# include <bits/siginfo.h>
108
109/* Wait for a childing matching IDTYPE and ID to change the status and
110 place appropriate information in *INFOP.
111 If IDTYPE is P_PID, match any process whose process ID is ID.
112 If IDTYPE is P_PGID, match any process whose process group is ID.
113 If IDTYPE is P_ALL, match any process.
114 If the WNOHANG bit is set in OPTIONS, and that child
115 is not already dead, clear *INFOP and return 0. If successful, store
116 exit code and status in *INFOP.
117
118 This function is a cancellation point and therefore not marked with
119 __THROW. */
120extern int waitid (idtype_t __idtype, __id_t __id, siginfo_t *__infop,
121 int __options);
122#endif

124#if defined __USE_MISC || defined __USE_XOPEN_EXTENDED
125/* This being here makes the prototypes valid whether or not
126 we have already included <sys/resource.h> to define `struct rusage'. */
127struct rusage;
128
129/* Wait for a child to exit. When one does, put its status in *STAT_LOC and
130 return its process ID. For errors return (pid_t) -1. If USAGE is not
131 nil, store information about the child's resource usage there. If the
132 WUNTRACED bit is set in OPTIONS, return status for stopped children;
133 otherwise don't. */
134extern __pid_t wait3 (int *__stat_loc, int __options,
135 struct rusage * __usage) __THROWNL;
136#endif
