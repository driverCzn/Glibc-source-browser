26/* Function used to increase the size of the allocated array. This
27 function is called from the `add'-functions. */
28int
29__posix_spawn_file_actions_realloc (posix_spawn_file_actions_t *file_actions)
30{
31 int newalloc = file_actions->__allocated + 8;
32 void *newmem = realloc (file_actions->__actions,
33 newalloc * sizeof (struct __spawn_action));
34
35 if (newmem == NULL)
36 /* Not enough memory. */
37 return ENOMEM;
38
39 file_actions->__actions = (struct __spawn_action *) newmem;
40 file_actions->__allocated = newalloc;
41
42 return 0;
43}

45
46/* Initialize data structure for file attribute for `spawn' call. */
47int
48posix_spawn_file_actions_init (posix_spawn_file_actions_t *file_actions)
49{
50 /* Simply clear all the elements. */
51 memset (file_actions, '\0', sizeof (*file_actions));
52 return 0;
53}
