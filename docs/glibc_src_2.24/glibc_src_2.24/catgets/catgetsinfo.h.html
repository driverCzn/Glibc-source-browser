19#include <sys/types.h>

21
22struct catalog_obj
23{
24 u_int32_t magic;
25 u_int32_t plane_size;
26 u_int32_t plane_depth;
27 /* This is in fact two arrays in one: always a pair of name and
28 pointer into the data area. */
29 u_int32_t name_ptr[0];
30};
31
32
33/* This structure will be filled after loading the catalog. */
34typedef struct catalog_info
35{
36 enum { mmapped, malloced } status;
37
38 size_t plane_size;
39 size_t plane_depth;
40 u_int32_t *name_ptr;
41 const char *strings;
42
43 struct catalog_obj *file_ptr;
44 size_t file_size;
45} *__nl_catd;
