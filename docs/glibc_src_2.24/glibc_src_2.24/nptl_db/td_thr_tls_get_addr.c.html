20#include <link.h>
21#include "thread_dbP.h"
22
23td_err_e
24td_thr_tls_get_addr (const td_thrhandle_t *th,
25 psaddr_t map_address, size_t offset, psaddr_t *address)
26{
27 td_err_e err;
28 psaddr_t modid;
29
30 /* Get the TLS module ID from the `struct link_map' in the inferior. */
31 err = DB_GET_FIELD (modid, th->th_ta_p, map_address, link_map,
32 l_tls_modid, 0);
33 if (err == TD_NOCAPAB)
34 return TD_NOAPLIC;
35 if (err == TD_OK)
36 {
37 err = td_thr_tlsbase (th, (uintptr_t) modid, address);
38 if (err == TD_OK)
39 *address += offset;
40 }
41 return err;
42}
