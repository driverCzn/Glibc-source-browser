18/*
19 * POSIX Standard: 5.6.6 Set File Access and Modification Times <utime.h>
20 */
21
22#ifndef _UTIME_H
23#define _UTIME_H 1
24
25#include <features.h>
26
27__BEGIN_DECLS
28
29#include <bits/types.h>
30
31#if defined __USE_XOPEN || defined __USE_XOPEN2K
32# define __need_time_t
33# include <time.h>
34#endif
35
36/* Structure describing file times. */
37struct utimbuf
38 {
39 __time_t actime; /* Access time. */
40 __time_t modtime; /* Modification time. */
41 };
42
43/* Set the access and modification times of FILE to those given in
44 *FILE_TIMES. If FILE_TIMES is NULL, set them to the current time. */
45extern int utime (const char *__file,
46 const struct utimbuf *__file_times)
47 __THROW __nonnull ((1));
48
49__END_DECLS
50
51#endif /* utime.h */
