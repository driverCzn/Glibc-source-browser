27#include "libioP.h"
28#include <stdio.h>
29
30size_t
31__fread_chk (void *__restrict ptr, size_t ptrlen,
32 size_t size, size_t n, FILE *__restrict stream)
33{
34 size_t bytes_requested = size * n;
35 if (__builtin_expect ((n | size)
36 >= (((size_t) 1) << (8 * sizeof (size_t) / 2)), 0))
37 {
38 if (size != 0 && bytes_requested / size != n)
39 __chk_fail ();
40 }
41
42 if (__glibc_unlikely (bytes_requested > ptrlen))
43 __chk_fail ();
44
45 CHECK_FILE (stream, 0);
46 if (bytes_requested == 0)
47 return 0;
48
49 size_t bytes_read;
50 _IO_acquire_lock (stream);
51 bytes_read = _IO_sgetn (stream, (char *) ptr, bytes_requested);
52 _IO_release_lock (stream);
53 return bytes_requested == bytes_read ? n : bytes_read / size;
54}
