18#include <stdio.h>
19#include <libio/iolibio.h>
20#define fwrite(p, n, m, s) _IO_fwrite (p, n, m, s)
21
22/* Write the word (int) W to STREAM. */
23int
24putw (int w, FILE *stream)
25{
26 /* Is there a better way? */
27 if (fwrite ((const void *) &w, sizeof (w), 1, stream) < 1)
28 return EOF;
29 return 0;
30}
