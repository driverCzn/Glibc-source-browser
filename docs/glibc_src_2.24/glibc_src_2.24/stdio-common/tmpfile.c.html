30/* This returns a new stream opened on a temporary file (generated
31 by tmpnam). The file is opened with mode "w+b" (binary read/write).
32 If we couldn't generate a unique filename or the file couldn't
33 be opened, NULL is returned. */
34FILE *
35tmpfile (void)
36{
37 char buf[FILENAME_MAX];
38 int fd;
39 FILE *f;
40
41 if (__path_search (buf, FILENAME_MAX, NULL, "tmpf", 0))
42 return NULL;
43 int flags = 0;
44#ifdef FLAGS
45 flags = FLAGS;
46#endif
47 fd = __gen_tempname (buf, 0, flags, __GT_FILE);
48 if (fd < 0)
49 return NULL;
50
51 /* Note that this relies on the Unix semantics that
52 a file is not really removed until it is closed. */
53 (void) __unlink (buf);
54
55 if ((f = __fdopen (fd, "w+b")) == NULL)
56 __close (fd);
57
58 return f;
59}
