26/* Convert YP error number to NSS error number. */
27extern const enum nss_status __yperr2nss_tab[] attribute_hidden;
28extern const unsigned int __yperr2nss_count attribute_hidden;
29
30static inline enum nss_status
31yperr2nss (int errval)
32{
33 if ((unsigned int) errval >= __yperr2nss_count)
34 return NSS_STATUS_UNAVAIL;
35 return __yperr2nss_tab[(unsigned int) errval];
36}
