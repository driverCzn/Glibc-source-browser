21#include <sched.h>
22#include <stdio.h>
23#include <unistd.h>

25bool
26support_become_root (void)
27{
28#ifdef CLONE_NEWUSER
29 if (unshare (CLONE_NEWUSER | CLONE_NEWNS) == 0)
30 /* Even if we do not have UID zero, we have extended privileges at
31 this point. */
32 return true;
33#endif
34 if (setuid (0) != 0)
35 {
36 printf ("warning: could not become root outside namespace (%m)\n");
37 return false;
38 }
39 return true;
40}
