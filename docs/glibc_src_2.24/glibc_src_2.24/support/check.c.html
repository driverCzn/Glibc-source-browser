21#include <stdarg.h>

26static void
27print_failure (const char *file, int line, const char *format, va_list ap)
28{
29 printf ("error: %s:%d: ", file, line);
30 vprintf (format, ap);
31 puts ("");
32}
33
34int
35support_print_failure_impl (const char *file, int line,
36 const char *format, ...)
37{
38 support_record_failure ();
39 va_list ap;
40 va_start (ap, format);
41 print_failure (file, line, format, ap);
42 va_end (ap);
43 return 1;
44}
45
46void
47support_exit_failure_impl (int status, const char *file, int line,
48 const char *format, ...)
49{
50 if (status != EXIT_SUCCESS && status != EXIT_UNSUPPORTED)
51 support_record_failure ();
52 va_list ap;
53 va_start (ap, format);
54 print_failure (file, line, format, ap);
55 va_end (ap);
56 exit (status);
57}
