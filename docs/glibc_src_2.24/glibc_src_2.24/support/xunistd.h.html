19/* These wrapper functions use POSIX types and therefore cannot be
20 declared in <support/support.h>. */
21
22#ifndef SUPPORT_XUNISTD_H
23#define SUPPORT_XUNISTD_H
24
25#include <sys/cdefs.h>
26#include <sys/types.h>
27#include <unistd.h>
28
29__BEGIN_DECLS
30
31struct stat64;
32
33pid_t xfork (void);
34pid_t xwaitpid (pid_t, int *status, int flags);
35void xpipe (int[2]);
36void xdup2 (int, int);
37int xopen (const char *path, int flags, mode_t);
38void xstat (const char *path, struct stat64 *);
39void xmkdir (const char *path, mode_t);
40void xchroot (const char *path);
41
42/* Close the file descriptor. Ignore EINTR errors, but terminate the
43 process on other errors. */
44void xclose (int);
45
46/* Write the buffer. Retry on short writes. */
47void xwrite (int, const void *, size_t);
48
49/* Invoke mmap with a zero file offset. */
50void *xmmap (void *addr, size_t length, int prot, int flags, int fd);
51
52void xmunmap (void *addr, size_t length);
53
54__END_DECLS
55
56#endif /* SUPPORT_XUNISTD_H */
