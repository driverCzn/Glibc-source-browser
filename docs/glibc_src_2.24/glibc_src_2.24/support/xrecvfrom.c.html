25size_t
26xrecvfrom (int fd, void *buf, size_t buflen, int flags,
27 struct sockaddr *sa, socklen_t *salen)
28{
29 ssize_t ret = recvfrom (fd, buf, buflen, flags, sa, salen);
30 if (ret < 0)
31 FAIL_EXIT1 ("error: recvfrom (%d), %zu bytes buffer: %m", fd, buflen);
32 return ret;
33}
