18#include <assert.h>
19#include <string.h>
20#include <utmp.h>
21#ifndef _UTMPX_H
22/* This is an ugly hack but we must not see the getutmpx declaration. */
23# define getutmpx XXXgetutmpx
24# include <utmpx.h>
25# undef getutmpx
26#endif
27
28void
29getutmp (const struct utmpx *utmpx, struct utmp *utmp)
30{
31 assert (sizeof (struct utmp) == sizeof (struct utmpx));
32 memcpy (utmp, utmpx, sizeof (struct utmp));
33}
34strong_alias (getutmp, getutmpx)
