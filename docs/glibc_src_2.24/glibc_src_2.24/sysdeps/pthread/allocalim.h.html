19#include <alloca.h>
20#include <limits.h>
21
22
23extern __always_inline
24int
25__libc_use_alloca (size_t size)
26{
27 return (__builtin_expect (size <= PTHREAD_STACK_MIN / 4, 1)
28 || __builtin_expect (__libc_alloca_cutoff (size), 1));
29}
