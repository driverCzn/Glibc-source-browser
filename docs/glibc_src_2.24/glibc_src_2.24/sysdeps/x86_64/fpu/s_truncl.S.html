20#include <machine/asm.h>
21
22ENTRY(__truncl)
23 fldt 8(%rsp)
24 fnstenv -28(%rsp)
25 movl $0xc00, %edx
26 orl -28(%rsp), %edx
27 movl %edx, -32(%rsp)
28 fldcw -32(%rsp)
29 frndint
30 fnstsw
31 andl $0x1, %eax
32 orl %eax, -24(%rsp)
33 fldenv -28(%rsp)
34 ret
35END(__truncl)
36weak_alias (__truncl, truncl)
