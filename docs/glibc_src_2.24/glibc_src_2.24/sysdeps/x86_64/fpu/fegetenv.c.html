19#include <fenv.h>
20
21int
22__fegetenv (fenv_t *envp)
23{
24 __asm__ ("fnstenv %0\n"
25 /* fnstenv changes the exception mask, so load back the
26 stored environment. */
27 "fldenv %0\n"
28 "stmxcsr %1" : "=m" (*envp), "=m" (envp->__mxcsr));
29
30 /* Success. */
31 return 0;
32}
