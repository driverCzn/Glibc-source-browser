23ENTRY(__fminf)
24 ucomiss %xmm0, %xmm1
25 jp 1f // jump if unordered
26 minss %xmm1, %xmm0
27 jmp 2f
28
291: ucomiss %xmm1, %xmm1 // Is xmm1 a NaN?
30 jp 2f // then return xmm0
31 movss %xmm1, %xmm0 // otherwise return xmm1
32
332: ret
34END(__fminf)
