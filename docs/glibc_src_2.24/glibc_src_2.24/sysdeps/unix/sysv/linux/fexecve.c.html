18#include <errno.h>
19#include <stddef.h>
20#include <stdio.h>
21#include <unistd.h>
22#include <sys/stat.h>

25/* Execute the file FD refers to, overlaying the running program image.
26 ARGV and ENVP are passed to the new program, as for `execve'. */
27int
28fexecve (int fd, char *const argv[], char *const envp[])
29{
30 if (fd < 0 || argv == NULL || envp == NULL)
31 {
32 __set_errno (EINVAL);
33 return -1;
34 }
35
36 /* We use the /proc filesystem to get the information. If it is not
37 mounted we fail. */
38 char buf[sizeof "/proc/self/fd/" + sizeof (int) * 3];
39 __snprintf (buf, sizeof (buf), "/proc/self/fd/%d", fd);
40
41 /* We do not need the return value. */
42 __execve (buf, argv, envp);
43
44 int save = errno;
45
46 /* We come here only if the 'execve' call fails. Determine whether
47 /proc is mounted. If not we return ENOSYS. */
48 struct stat st;
49 if (stat ("/proc/self/fd", &st) != 0 && errno == ENOENT)
50 save = ENOSYS;
51
52 __set_errno (save);
53
54 return -1;
55}
