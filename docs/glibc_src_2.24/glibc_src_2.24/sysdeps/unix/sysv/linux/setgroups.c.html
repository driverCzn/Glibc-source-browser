24/* Set the group set for the current user to GROUPS (N of them). For
25 Linux we must convert the array of groups into the format that the
26 kernel expects. */
27int
28setgroups (size_t n, const gid_t *groups)
29{
30#ifdef __NR_setgroups32
31# error "wrong setgroups.c file used"
32#endif
33 return INLINE_SETXID_SYSCALL (setgroups, 2, n, groups);
34}
