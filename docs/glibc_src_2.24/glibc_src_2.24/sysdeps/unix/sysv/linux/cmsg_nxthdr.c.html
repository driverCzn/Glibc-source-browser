22
23struct cmsghdr *
24__cmsg_nxthdr (struct msghdr *mhdr, struct cmsghdr *cmsg)
25{
26 if ((size_t) cmsg->cmsg_len < sizeof (struct cmsghdr))
27 /* The kernel header does this so there may be a reason. */
28 return NULL;
29
30 cmsg = (struct cmsghdr *) ((unsigned char *) cmsg
31 + CMSG_ALIGN (cmsg->cmsg_len));
32 if ((unsigned char *) (cmsg + 1) > ((unsigned char *) mhdr->msg_control
33 + mhdr->msg_controllen)
34 || ((unsigned char *) cmsg + CMSG_ALIGN (cmsg->cmsg_len)
35 > ((unsigned char *) mhdr->msg_control + mhdr->msg_controllen)))
36 /* No more entries. */
37 return NULL;
38 return cmsg;
39}
