18#include <errno.h>
19#include <sysdep-cancel.h>
20#include <stdlib.h>
21#include <sys/wait.h>

23__pid_t
24__waitpid (__pid_t pid, int *stat_loc, int options)
25{
26#ifdef __NR_waitpid
27 return SYSCALL_CANCEL (waitpid, pid, stat_loc, options);
28#else
29 return SYSCALL_CANCEL (wait4, pid, stat_loc, options, NULL);
30#endif
31}
