19#include <endian.h>
20#include <sys/sysmacros.h>
21
22unsigned int
23gnu_dev_major (unsigned long long int dev)
24{
25 return ((dev >> 8) & 0xfff) | ((unsigned int) (dev >> 32) & ~0xfff);
26}
27
28unsigned int
29gnu_dev_minor (unsigned long long int dev)
30{
31 return (dev & 0xff) | ((unsigned int) (dev >> 12) & ~0xff);
32}
33
34unsigned long long int
35gnu_dev_makedev (unsigned int major, unsigned int minor)
36{
37 return ((minor & 0xff) | ((major & 0xfff) << 8)
38 | (((unsigned long long int) (minor & ~0xff)) << 12)
39 | (((unsigned long long int) (major & ~0xfff)) << 32));
40}
