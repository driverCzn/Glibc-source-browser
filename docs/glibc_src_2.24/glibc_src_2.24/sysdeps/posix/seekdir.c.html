24/* Seek to position POS in DIRP. */
25/* XXX should be __seekdir ? */
26void
27seekdir (DIR *dirp, long int pos)
28{
29 __libc_lock_lock (dirp->lock);
30 (void) __lseek (dirp->fd, pos, SEEK_SET);
31 dirp->size = 0;
32 dirp->offset = 0;
33 dirp->filepos = pos;
34 __libc_lock_unlock (dirp->lock);
35}
