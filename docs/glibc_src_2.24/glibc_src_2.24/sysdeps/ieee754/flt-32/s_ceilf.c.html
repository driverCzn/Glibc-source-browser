16#include <math.h>
17#include <math_private.h>
18
19
20float
21__ceilf(float x)
22{
23 int32_t i0,j0;
24 u_int32_t i;
25
26 GET_FLOAT_WORD(i0,x);
27 j0 = ((i0>>23)&0xff)-0x7f;
28 if(j0<23) {
29 if(j0<0) {
30 /* return 0*sign(x) if |x|<1 */
31 if(i0<0) {i0=0x80000000;}
32 else if(i0!=0) { i0=0x3f800000;}
33 } else {
34 i = (0x007fffff)>>j0;
35 if((i0&i)==0) return x; /* x is integral */
36 if(i0>0) i0 += (0x00800000)>>j0;
37 i0 &= (~i);
38 }
39 } else {
40 if(__builtin_expect(j0==0x80, 0)) return x+x; /* inf or NaN */
41 else return x; /* x is integral */
42 }
43 SET_FLOAT_WORD(x,i0);
44 return x;
45}
46#ifndef __ceilf
47weak_alias (__ceilf, ceilf)
48#endif
