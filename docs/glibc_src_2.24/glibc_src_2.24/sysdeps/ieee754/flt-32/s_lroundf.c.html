28long int
29__lroundf (float x)
30{
31 int32_t j0;
32 u_int32_t i;
33 long int result;
34 int sign;
35
36 GET_FLOAT_WORD (i, x);
37 j0 = ((i >> 23) & 0xff) - 0x7f;
38 sign = (i & 0x80000000) != 0 ? -1 : 1;
39 i &= 0x7fffff;
40 i |= 0x800000;
41
42 if (j0 < (int32_t) (8 * sizeof (long int)) - 1)
43 {
44 if (j0 < 0)
45 return j0 < -1 ? 0 : sign;
46 else if (j0 >= 23)
47 result = (long int) i << (j0 - 23);
48 else
49 {
50 i += 0x400000 >> j0;
51
52 result = i >> (23 - j0);
53 }
54 }
55 else
56 {
57#ifdef FE_INVALID
58 /* The number is too large. Unless it rounds to LONG_MIN,
59 FE_INVALID must be raised and the return value is
60 unspecified. */
61 if (FIX_FLT_LONG_CONVERT_OVERFLOW && x != (float) LONG_MIN)
62 {
63 feraiseexcept (FE_INVALID);
64 return sign == 1 ? LONG_MAX : LONG_MIN;
65 }
66#endif
67 return (long int) x;
68 }
69
70 return sign * result;
71}
