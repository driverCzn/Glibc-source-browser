19#include <math.h>
20#include <math_private.h>
21
22int
23__issignalingl (long double x)
24{
25 u_int32_t exi, hxi, lxi;
26 GET_LDOUBLE_WORDS (exi, hxi, lxi, x);
27#ifdef HIGH_ORDER_BIT_IS_SET_FOR_SNAN
28# error not implemented
29#else
30 /* To keep the following comparison simple, toggle the quiet/signaling bit,
31 so that it is set for sNaNs. This is inverse to IEEE 754-2008 (as well as
32 common practice for IEEE 754-1985). */
33 hxi ^= 0x40000000;
34 /* If lxi != 0, then set any suitable bit of the significand in hxi. */
35 hxi |= (lxi | -lxi) >> 31;
36 /* We do not recognize a pseudo NaN as sNaN; they're invalid on 80387 and
37 later. */
38 /* We have to compare for greater (instead of greater or equal), because x's
39 significand being all-zero designates infinity not NaN. */
40 return ((exi & 0x7fff) == 0x7fff) && (hxi > 0xc0000000);
41#endif
42}
