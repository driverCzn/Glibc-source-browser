13/*
14 * isnan(x) returns 1 is x is nan, else 0;
15 * no branching!
16 */
17
18#include <math.h>
19#include <math_private.h>
20#include <shlib-compat.h>
21#include <stdint.h>
22
23#undef __isnan
24int __isnan(double x)
25{
26 int64_t hx;
27 EXTRACT_WORDS64(hx,x);
28 hx &= UINT64_C(0x7fffffffffffffff);
29 hx = UINT64_C(0x7ff0000000000000) - hx;
30 return (int)(((uint64_t)hx)>>63);
31}
32hidden_def (__isnan)
33weak_alias (__isnan, isnan)
34#ifdef NO_LONG_DOUBLE
35# if defined LDBL_CLASSIFY_COMPAT && SHLIB_COMPAT (libc, GLIBC_2_0, GLIBC_2_23)
36compat_symbol (libc, __isnan, __isnanl, GLIBC_2_0);
37# endif
38weak_alias (__isnan, isnanl)
39#endif
