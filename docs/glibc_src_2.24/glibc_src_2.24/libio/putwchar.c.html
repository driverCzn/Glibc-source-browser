18#include "libioP.h"
19#include <wchar.h>
20
21wint_t
22putwchar (wchar_t wc)
23{
24 wint_t result;
25 _IO_acquire_lock (_IO_stdout);
26 result = _IO_putwc_unlocked (wc, _IO_stdout);
27 _IO_release_lock (_IO_stdout);
28 return result;
29}
