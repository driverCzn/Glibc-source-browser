329/* Functions to do I/O and file management for a stream. */

331/* Read NBYTES bytes from COOKIE into a buffer pointed to by BUF.
332 Return number of bytes read. */
333typedef __ssize_t __io_read_fn (void *__cookie, char *__buf, size_t __nbytes);
334
335/* Write N bytes pointed to by BUF to COOKIE. Write all N bytes
336 unless there is an error. Return number of bytes written. If
337 there is an error, return 0 and do not write anything. If the file
338 has been opened for append (__mode.__append set), then set the file
339 pointer to the end of the file and then do the write; if not, just
340 write at the current file pointer. */
341typedef __ssize_t __io_write_fn (void *__cookie, const char *__buf,
342 size_t __n);
343
344/* Move COOKIE's file position to *POS bytes from the
345 beginning of the file (if W is SEEK_SET),
346 the current position (if W is SEEK_CUR),
347 or the end of the file (if W is SEEK_END).
348 Set *POS to the new file position.
349 Returns zero if successful, nonzero if not. */
350typedef int __io_seek_fn (void *__cookie, _IO_off64_t *__pos, int __w);
351
352/* Close COOKIE. */
353typedef int __io_close_fn (void *__cookie);
354
355
356#ifdef _GNU_SOURCE
357/* User-visible names for the above. */
358typedef __io_read_fn cookie_read_function_t;
359typedef __io_write_fn cookie_write_function_t;
360typedef __io_seek_fn cookie_seek_function_t;
361typedef __io_close_fn cookie_close_function_t;
362
363/* The structure with the cookie function pointers. */

365{
366 __io_read_fn *read; /* Read bytes. */

400#define _IO_getc_unlocked(_fp) \
401 (_IO_BE ((_fp)->_IO_read_ptr >= (_fp)->_IO_read_end, 0) \
402 ? __uflow (_fp) : *(unsigned char *) (_fp)->_IO_read_ptr++)
403#define _IO_peekc_unlocked(_fp) \
404 (_IO_BE ((_fp)->_IO_read_ptr >= (_fp)->_IO_read_end, 0) \
405 && __underflow (_fp) == EOF ? EOF \
406 : *(unsigned char *) (_fp)->_IO_read_ptr)
407#define _IO_putc_unlocked(_ch, _fp) \
408 (_IO_BE ((_fp)->_IO_write_ptr >= (_fp)->_IO_write_end, 0) \
409 ? __overflow (_fp, (unsigned char) (_ch)) \
410 : (unsigned char) (*(_fp)->_IO_write_ptr++ = (_ch)))
411
412#if defined _LIBC || defined _GLIBCPP_USE_WCHAR_T
413# define _IO_getwc_unlocked(_fp) \
414 (_IO_BE ((_fp)->_wide_data == NULL \
415 || ((_fp)->_wide_data->_IO_read_ptr \
416 >= (_fp)->_wide_data->_IO_read_end), 0) \
417 ? __wuflow (_fp) : (_IO_wint_t) *(_fp)->_wide_data->_IO_read_ptr++)
418# define _IO_putwc_unlocked(_wch, _fp) \
419 (_IO_BE ((_fp)->_wide_data == NULL \
420 || ((_fp)->_wide_data->_IO_write_ptr \
421 >= (_fp)->_wide_data->_IO_write_end), 0) \
422 ? __woverflow (_fp, _wch) \
423 : (_IO_wint_t) (*(_fp)->_wide_data->_IO_write_ptr++ = (_wch)))
424#endif
