18#include "libioP.h"
19#include "stdio.h"
20
21#undef _IO_putc
22
23int
24_IO_putc (int c, _IO_FILE *fp)
25{
26 int result;
27 CHECK_FILE (fp, EOF);
28 _IO_acquire_lock (fp);
29 result = _IO_putc_unlocked (c, fp);
30 _IO_release_lock (fp);
31 return result;
32}
33libc_hidden_def (_IO_putc)
34
35#undef putc

40#ifndef _IO_MTSAFE_IO
41#undef putc_unlocked

43#endif
44#endif
