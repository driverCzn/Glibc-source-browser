21#include <math.h>
22#include <math_private.h>

24
25__complex__ long double
26__casinl (__complex__ long double x)
27{
28 __complex__ long double res;
29
30 if (isnan (__real__ x) || isnan (__imag__ x))
31 {
32 if (__real__ x == 0.0)
33 {
34 res = x;
35 }
36 else if (isinf (__real__ x) || isinf (__imag__ x))
37 {
38 __real__ res = __nanl ("");
39 __imag__ res = __copysignl (HUGE_VALL, __imag__ x);
40 }
41 else
42 {
43 __real__ res = __nanl ("");
44 __imag__ res = __nanl ("");
45 }
46 }
47 else
48 {
49 __complex__ long double y;
50
51 __real__ y = -__imag__ x;
52 __imag__ y = __real__ x;
53
54 y = __casinhl (y);
55
56 __real__ res = __imag__ y;
57 __imag__ res = -__real__ y;
58 }
59
60 return res;
61}
