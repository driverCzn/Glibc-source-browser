20#include <complex.h>
21#include <fenv.h>
22#include <math.h>
23#include <math_private.h>
24
25
26__complex__ float
27__ccosf (__complex__ float x)
28{
29 __complex__ float y;
30
31 __real__ y = -__imag__ x;
32 __imag__ y = __real__ x;
33
34 return __ccoshf (y);
35}
36#ifndef __ccosf
37weak_alias (__ccosf, ccosf)
38#endif
