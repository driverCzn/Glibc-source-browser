19#include <errno.h>
20#include <math.h>
21#include <math_private.h>
22
23long double
24__w_scalblnl (long double x, long int n)
25{
26 if (!isfinite (x) || x == 0.0L)
27 return x + x;
28
29 x = __scalblnl (x, n);
30
31 if (!isfinite (x) || x == 0.0L)
32 __set_errno (ERANGE);
33
34 return x;
35}
36weak_alias (__w_scalblnl, scalblnl)
