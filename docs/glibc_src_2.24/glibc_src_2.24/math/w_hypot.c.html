13/*
14 * wrapper hypot(x,y)
15 */
16
17#include <math.h>
18#include <math_private.h>
19
20
21double
22__hypot (double x, double y)
23{
24 double z = __ieee754_hypot(x,y);
25 if(__builtin_expect(!isfinite(z), 0)
26 && isfinite(x) && isfinite(y) && _LIB_VERSION != _IEEE_)
27 return __kernel_standard(x, y, 4); /* hypot overflow */
28
29 return z;
30}
31weak_alias (__hypot, hypot)
32#ifdef NO_LONG_DOUBLE
33strong_alias (__hypot, __hypotl)
34weak_alias (__hypot, hypotl)
35#endif
