20#include <complex.h>
21#include <fenv.h>
22#include <math.h>
23#include <math_private.h>
24#include <float.h>

26__complex__ long double
27__csinl (__complex__ long double x)
28{
29 __complex__ long double retval;
30 int negate = signbit (__real__ x);
31 int rcls = fpclassify (__real__ x);
32 int icls = fpclassify (__imag__ x);
33
34 __real__ x = fabsl (__real__ x);
35
36 if (__glibc_likely (icls >= FP_ZERO))
37 {
38 /* Imaginary part is finite. */
39 if (__glibc_likely (rcls >= FP_ZERO))
40 {
41 /* Real part is finite. */
42 const int t = (int) ((LDBL_MAX_EXP - 1) * M_LN2l);
43 long double sinix, cosix;
44
45 if (__glibc_likely (__real__ x > LDBL_MIN))
46 {
47 __sincosl (__real__ x, &sinix, &cosix);
48 }
49 else
50 {
51 sinix = __real__ x;
52 cosix = 1.0;
53 }
54
55 if (negate)
56 sinix = -sinix;
57
58 if (fabsl (__imag__ x) > t)
59 {
60 long double exp_t = __ieee754_expl (t);
61 long double ix = fabsl (__imag__ x);
62 if (signbit (__imag__ x))
63 cosix = -cosix;
64 ix -= t;
65 sinix *= exp_t / 2.0L;
66 cosix *= exp_t / 2.0L;
67 if (ix > t)
68 {
69 ix -= t;
70 sinix *= exp_t;
71 cosix *= exp_t;
72 }
73 if (ix > t)
74 {
75 /* Overflow (original imaginary part of x > 3t). */
76 __real__ retval = LDBL_MAX * sinix;
77 __imag__ retval = LDBL_MAX * cosix;
78 }
79 else
80 {
81 long double exp_val = __ieee754_expl (ix);
82 __real__ retval = exp_val * sinix;
83 __imag__ retval = exp_val * cosix;
84 }
85 }
86 else
87 {
88 __real__ retval = __ieee754_coshl (__imag__ x) * sinix;
89 __imag__ retval = __ieee754_sinhl (__imag__ x) * cosix;
90 }
91
92 math_check_force_underflow_complex (retval);
93 }
94 else
95 {
96 if (icls == FP_ZERO)
97 {
98 /* Imaginary part is 0.0. */
99 __real__ retval = __nanl ("");
100 __imag__ retval = __imag__ x;
101
102 if (rcls == FP_INFINITE)
103 feraiseexcept (FE_INVALID);
104 }
105 else
106 {
107 __real__ retval = __nanl ("");
108 __imag__ retval = __nanl ("");
109
110 feraiseexcept (FE_INVALID);
111 }
112 }
113 }
114 else if (icls == FP_INFINITE)
115 {
116 /* Imaginary part is infinite. */
117 if (rcls == FP_ZERO)
118 {
119 /* Real part is 0.0. */
120 __real__ retval = __copysignl (0.0, negate ? -1.0 : 1.0);
121 __imag__ retval = __imag__ x;
122 }
123 else if (rcls > FP_ZERO)
124 {
125 /* Real part is finite. */
126 long double sinix, cosix;
127
128 if (__glibc_likely (__real__ x > LDBL_MIN))
129 {
130 __sincosl (__real__ x, &sinix, &cosix);
131 }
132 else
133 {
134 sinix = __real__ x;
135 cosix = 1.0;
136 }
137
138 __real__ retval = __copysignl (HUGE_VALL, sinix);
139 __imag__ retval = __copysignl (HUGE_VALL, cosix);
140
141 if (negate)
142 __real__ retval = -__real__ retval;
143 if (signbit (__imag__ x))
144 __imag__ retval = -__imag__ retval;
145 }
146 else
147 {
148 /* The addition raises the invalid exception. */
149 __real__ retval = __nanl ("");
150 __imag__ retval = HUGE_VALL;
151
152 if (rcls == FP_INFINITE)
153 feraiseexcept (FE_INVALID);
154 }
155 }
156 else
157 {
158 if (rcls == FP_ZERO)
159 __real__ retval = __copysignl (0.0, negate ? -1.0 : 1.0);
160 else
161 __real__ retval = __nanl ("");
162 __imag__ retval = __nanl ("");
163 }
164
165 return retval;
166}
