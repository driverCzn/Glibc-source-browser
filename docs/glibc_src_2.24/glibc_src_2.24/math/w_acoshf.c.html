23/* wrapper acoshf */
24float
25__acoshf (float x)
26{
27 if (__builtin_expect (isless (x, 1.0f), 0) && _LIB_VERSION != _IEEE_)
28 /* acosh(x<1) */
29 return __kernel_standard_f (x, x, 129);
30
31 return __ieee754_acoshf (x);
32}
33weak_alias (__acoshf, acoshf)
