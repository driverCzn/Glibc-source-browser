20#include <complex.h>
21#include <math.h>
22#include <math_private.h>
23
24
25__complex__ float
26__cprojf (__complex__ float x)
27{
28 if (isinf (__real__ x) || isinf (__imag__ x))
29 {
30 __complex__ float res;
31
32 __real__ res = INFINITY;
33 __imag__ res = __copysignf (0.0, __imag__ x);
34
35 return res;
36 }
37
38 return x;
39}
40#ifndef __cprojf
41weak_alias (__cprojf, cprojf)
42#endif
