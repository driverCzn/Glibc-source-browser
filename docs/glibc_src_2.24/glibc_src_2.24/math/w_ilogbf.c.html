24/* wrapper ilogbf */
25int
26__ilogbf (float x)
27{
28 int r = __ieee754_ilogbf (x);
29 if (__builtin_expect (r == FP_ILOGB0, 0)
30 || __builtin_expect (r == FP_ILOGBNAN, 0)
31 || __builtin_expect (r == INT_MAX, 0))
32 {
33 __set_errno (EDOM);
34 feraiseexcept (FE_INVALID);
35 }
36 return r;
37}
38weak_alias (__ilogbf, ilogbf)
