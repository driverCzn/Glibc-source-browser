17/*
18 * wrapper sinhl(x)
19 */
20
21#include <math.h>
22#include <math_private.h>
23
24long double
25__sinhl (long double x)
26{
27 long double z = __ieee754_sinhl (x);
28 if (__builtin_expect (!isfinite (z), 0) && isfinite (x)
29 && _LIB_VERSION != _IEEE_)
30 return __kernel_standard_l (x, x, 225); /* sinh overflow */
31
32 return z;
33}
34weak_alias (__sinhl, sinhl)
