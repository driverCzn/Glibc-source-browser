25long double __ldexpl(long double value, int exp)
26{
27 if(!isfinite(value)||value==0.0) return value + value;
28 value = __scalbnl(value,exp);
29 if(!isfinite(value)||value==0.0) __set_errno (ERANGE);
30 return value;
31}
32weak_alias (__ldexpl, ldexpl)
33weak_alias (__ldexpl, scalbnl)
