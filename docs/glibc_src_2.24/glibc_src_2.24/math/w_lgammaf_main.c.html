18
19#include <lgamma-compat.h>

21#if BUILD_LGAMMA
22float
23LGFUNC (__lgammaf) (float x)
24{
25 float y = CALL_LGAMMA (float, __ieee754_lgammaf_r, x);
26 if(__builtin_expect(!isfinite(y), 0)
27 && isfinite(x) && _LIB_VERSION != _IEEE_)
28 return __kernel_standard_f(x, x,
29 __floorf(x)==x&&x<=0.0f
30 ? 115 /* lgamma pole */
31 : 114); /* lgamma overflow */
32
33 return y;
34}
35# if USE_AS_COMPAT
36compat_symbol (libm, __lgammaf_compat, lgammaf, LGAMMA_OLD_VER);
37# else
38versioned_symbol (libm, __lgammaf, lgammaf, LGAMMA_NEW_VER);
39# endif
40# if GAMMA_ALIAS
41strong_alias (LGFUNC (__lgammaf), __gammaf)
42weak_alias (__gammaf, gammaf)
43# endif
44#endif
