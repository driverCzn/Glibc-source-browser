20#include <errno.h>
21#include <math.h>
22
23long double
24__fdiml (long double x, long double y)
25{
26 if (islessequal (x, y))
27 return 0.0f;
28
29 long double r = x - y;
30 if (isinf (r) && !isinf (x) && !isinf (y))
31 __set_errno (ERANGE);
32
33 return r;
34}
35weak_alias (__fdiml, fdiml)
