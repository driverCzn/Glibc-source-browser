19#include <errno.h>
20#include "pthreadP.h"
21
22#include <stap-probe.h>
23
24
25int
26__pthread_mutex_destroy (pthread_mutex_t *mutex)
27{
28 LIBC_PROBE (mutex_destroy, 1, mutex);
29
30 if ((mutex->__data.__kind & PTHREAD_MUTEX_ROBUST_NORMAL_NP) == 0
31 && mutex->__data.__nusers != 0)
32 return EBUSY;
33
34 /* Set to an invalid value. */
35 mutex->__data.__kind = -1;
36
37 return 0;
38}
39strong_alias (__pthread_mutex_destroy, pthread_mutex_destroy)
40hidden_def (__pthread_mutex_destroy)
