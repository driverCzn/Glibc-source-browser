19#include <stdlib.h>
20#include "pthreadP.h"

23void
24_pthread_cleanup_push (struct _pthread_cleanup_buffer *buffer,
25 void (*routine) (void *), void *arg)
26{
27 struct pthread *self = THREAD_SELF;
28
29 buffer->__routine = routine;
30 buffer->__arg = arg;
31 buffer->__prev = THREAD_GETMEM (self, cleanup);
32
33 THREAD_SETMEM (self, cleanup, buffer);
34}
35strong_alias (_pthread_cleanup_push, __pthread_cleanup_push)
36
37
38void
39_pthread_cleanup_pop (struct _pthread_cleanup_buffer *buffer, int execute)
40{
41 struct pthread *self __attribute ((unused)) = THREAD_SELF;
42
43 THREAD_SETMEM (self, cleanup, buffer->__prev);
44
45 /* If necessary call the cleanup routine after we removed the
46 current cleanup block from the list. */
47 if (execute)
48 buffer->__routine (buffer->__arg);
49}
50strong_alias (_pthread_cleanup_pop, __pthread_cleanup_pop)
