36#include "pthreadP.h"
37#include <fork.h>
38
39/* This is defined by newer gcc version unique for each module. */
40extern void *__dso_handle __attribute__ ((__weak__,
41 __visibility__ ("hidden")));
42
43
44/* Hide the symbol so that no definition but the one locally in the
45 executable or DSO is used. */
46int
47#ifndef __pthread_atfork
48/* Don't mark the compatibility function as hidden. */
49attribute_hidden
50#endif
51__pthread_atfork (void (*prepare) (void), void (*parent) (void),
52 void (*child) (void))
53{
54 return __register_atfork (prepare, parent, child,
55 &__dso_handle == NULL ? NULL : __dso_handle);
56}
57#ifndef __pthread_atfork
58extern int pthread_atfork (void (*prepare) (void), void (*parent) (void),
59 void (*child) (void)) attribute_hidden;
60strong_alias (__pthread_atfork, pthread_atfork)
61#endif
