19#include <assert.h>
20#include "pthreadP.h"
21
22
23int
24__pthread_attr_getstack (const pthread_attr_t *attr, void **stackaddr,
25 size_t *stacksize)
26{
27 struct pthread_attr *iattr;
28
29 assert (sizeof (*attr) >= sizeof (struct pthread_attr));
30 iattr = (struct pthread_attr *) attr;
31
32 /* Store the result. */
33#if _STACK_GROWS_DOWN
34 *stackaddr = (char *) iattr->stackaddr - iattr->stacksize;
35#else
36 *stackaddr = (char *) iattr->stackaddr;
37#endif
38 *stacksize = iattr->stacksize;
39
40 return 0;
41}
42strong_alias (__pthread_attr_getstack, pthread_attr_getstack)
