19#include <assert.h>
20#include "pthreadP.h"
21
22
23int
24__pthread_attr_getdetachstate (const pthread_attr_t *attr, int *detachstate)
25{
26 struct pthread_attr *iattr;
27
28 assert (sizeof (*attr) >= sizeof (struct pthread_attr));
29 iattr = (struct pthread_attr *) attr;
30
31 *detachstate = (iattr->flags & ATTR_FLAG_DETACHSTATE
32 ? PTHREAD_CREATE_DETACHED : PTHREAD_CREATE_JOINABLE);
33
34 return 0;
35}
36strong_alias (__pthread_attr_getdetachstate, pthread_attr_getdetachstate)
