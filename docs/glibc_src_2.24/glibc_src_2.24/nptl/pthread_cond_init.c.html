19#include <shlib-compat.h>
20#include "pthreadP.h"
21#include <stap-probe.h>
22
23
24int
25__pthread_cond_init (pthread_cond_t *cond, const pthread_condattr_t *cond_attr)
26{
27 struct pthread_condattr *icond_attr = (struct pthread_condattr *) cond_attr;
28
29 cond->__data.__lock = LLL_LOCK_INITIALIZER;
30 cond->__data.__futex = 0;
31 cond->__data.__nwaiters = (icond_attr != NULL
32 ? ((icond_attr->value >> 1)
33 & ((1 << COND_NWAITERS_SHIFT) - 1))
34 : CLOCK_REALTIME);
35 cond->__data.__total_seq = 0;
36 cond->__data.__wakeup_seq = 0;
37 cond->__data.__woken_seq = 0;
38 cond->__data.__mutex = (icond_attr == NULL || (icond_attr->value & 1) == 0
39 ? NULL : (void *) ~0l);
40 cond->__data.__broadcast_seq = 0;
41
42 LIBC_PROBE (cond_init, 2, cond, cond_attr);
43
44 return 0;
45}
46versioned_symbol (libpthread, __pthread_cond_init,
47 pthread_cond_init, GLIBC_2_3_2);
