19#include <errno.h>
20#include <string.h>
21#include "pthreadP.h"
22#include <lowlevellock.h>
23
24
25int
26__pthread_getschedparam (pthread_t threadid, int *policy,
27 struct sched_param *param)
28{
29 struct pthread *pd = (struct pthread *) threadid;
30
31 /* Make sure the descriptor is valid. */
32 if (INVALID_TD_P (pd))
33 /* Not a valid thread handle. */
34 return ESRCH;
35
36 int result = 0;
37
38 /* See CREATE THREAD NOTES in nptl/pthread_create.c. */
39 lll_lock (pd->lock, LLL_PRIVATE);
40
41 /* The library is responsible for maintaining the values at all
42 times. If the user uses an interface other than
43 pthread_setschedparam to modify the scheduler setting it is not
44 the library's problem. In case the descriptor's values have
45 not yet been retrieved do it now. */
46 if ((pd->flags & ATTR_FLAG_SCHED_SET) == 0)
47 {
48 if (__sched_getparam (pd->tid, &pd->schedparam) != 0)
49 result = 1;
50 else
51 pd->flags |= ATTR_FLAG_SCHED_SET;
52 }
53
54 if ((pd->flags & ATTR_FLAG_POLICY_SET) == 0)
55 {
56 pd->schedpolicy = __sched_getscheduler (pd->tid);
57 if (pd->schedpolicy == -1)
58 result = 1;
59 else
60 pd->flags |= ATTR_FLAG_POLICY_SET;
61 }
62
63 if (result == 0)
64 {
65 *policy = pd->schedpolicy;
66 memcpy (param, &pd->schedparam, sizeof (struct sched_param));
67 }
68
69 lll_unlock (pd->lock, LLL_PRIVATE);
70
71 return result;
72}
73strong_alias (__pthread_getschedparam, pthread_getschedparam)
