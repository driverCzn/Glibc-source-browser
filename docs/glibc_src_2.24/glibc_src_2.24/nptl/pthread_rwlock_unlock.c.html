29/* Unlock RWLOCK. */
30int
31__pthread_rwlock_unlock (pthread_rwlock_t *rwlock)
32{
33 int futex_shared =
34 rwlock->__data.__shared == LLL_PRIVATE ? FUTEX_PRIVATE : FUTEX_SHARED;
35
36 LIBC_PROBE (rwlock_unlock, 1, rwlock);
37
38 if (ELIDE_UNLOCK (rwlock->__data.__writer == 0
39 && rwlock->__data.__nr_readers == 0))
40 return 0;
41
42 lll_lock (rwlock->__data.__lock, rwlock->__data.__shared);
43 if (rwlock->__data.__writer)
44 rwlock->__data.__writer = 0;
45 else
46 --rwlock->__data.__nr_readers;
47 /* If there are still readers present, we do not yet need to wake writers
48 nor are responsible to wake any readers. */
49 if (rwlock->__data.__nr_readers == 0)
50 {
51 /* Note that if there is a blocked writer, we effectively make it
52 responsible for waking any readers because we don't wake readers in
53 this case. */
54 if (rwlock->__data.__nr_writers_queued)
55 {
56 ++rwlock->__data.__writer_wakeup;
57 lll_unlock (rwlock->__data.__lock, rwlock->__data.__shared);
58 futex_wake (&rwlock->__data.__writer_wakeup, 1, futex_shared);
59 return 0;
60 }
61 else if (rwlock->__data.__nr_readers_queued)
62 {
63 ++rwlock->__data.__readers_wakeup;
64 lll_unlock (rwlock->__data.__lock, rwlock->__data.__shared);
65 futex_wake (&rwlock->__data.__readers_wakeup, INT_MAX,
66 futex_shared);
67 return 0;
68 }
69 }
70 lll_unlock (rwlock->__data.__lock, rwlock->__data.__shared);
71 return 0;
72}
