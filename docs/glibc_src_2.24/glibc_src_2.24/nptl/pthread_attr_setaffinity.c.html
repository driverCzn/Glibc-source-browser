28int
29__pthread_attr_setaffinity_new (pthread_attr_t *attr, size_t cpusetsize,
30 const cpu_set_t *cpuset)
31{
32 struct pthread_attr *iattr;
33
34 assert (sizeof (*attr) >= sizeof (struct pthread_attr));
35 iattr = (struct pthread_attr *) attr;
36
37 if (cpuset == NULL || cpusetsize == 0)
38 {
39 free (iattr->cpuset);
40 iattr->cpuset = NULL;
41 iattr->cpusetsize = 0;
42 }
43 else
44 {
45 if (iattr->cpusetsize != cpusetsize)
46 {
47 void *newp = (cpu_set_t *) realloc (iattr->cpuset, cpusetsize);
48 if (newp == NULL)
49 return ENOMEM;
50
51 iattr->cpuset = newp;
52 iattr->cpusetsize = cpusetsize;
53 }
54
55 memcpy (iattr->cpuset, cpuset, cpusetsize);
56 }
57
58 return 0;
59}
60versioned_symbol (libpthread, __pthread_attr_setaffinity_new,
61 pthread_attr_setaffinity_np, GLIBC_2_3_4);

64#if SHLIB_COMPAT (libpthread, GLIBC_2_3_3, GLIBC_2_3_4)
65int
66__pthread_attr_setaffinity_old (pthread_attr_t *attr, cpu_set_t *cpuset)
67{
68 /* The old interface by default assumed a 1024 processor bitmap. */
69 return __pthread_attr_setaffinity_new (attr, 128, cpuset);
70}
71compat_symbol (libpthread, __pthread_attr_setaffinity_old,
72 pthread_attr_setaffinity_np, GLIBC_2_3_3);
