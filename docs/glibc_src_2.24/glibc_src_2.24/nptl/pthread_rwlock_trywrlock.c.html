19#include <errno.h>
20#include "pthreadP.h"
21#include <lowlevellock.h>
22#include <elide.h>
23
24
25int
26__pthread_rwlock_trywrlock (pthread_rwlock_t *rwlock)
27{
28 int result = EBUSY;
29
30 if (ELIDE_TRYLOCK (rwlock->__data.__rwelision,
31 rwlock->__data.__lock == 0
32 && rwlock->__data.__nr_readers == 0
33 && rwlock->__data.__writer, 1))
34 return 0;
35
36 lll_lock (rwlock->__data.__lock, rwlock->__data.__shared);
37
38 if (rwlock->__data.__writer == 0 && rwlock->__data.__nr_readers == 0)
39 {
40 rwlock->__data.__writer = THREAD_GETMEM (THREAD_SELF, tid);
41 result = 0;
42 }
43
44 lll_unlock (rwlock->__data.__lock, rwlock->__data.__shared);
45
46 return result;
47}
48strong_alias (__pthread_rwlock_trywrlock, pthread_rwlock_trywrlock)
